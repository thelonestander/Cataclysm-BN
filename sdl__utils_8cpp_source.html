47std::vector<SDL_Color> color_linear_interpolate( const SDL_Color &start_color,
48 const SDL_Color &end_color, const unsigned additional_steps )
49{
50 const unsigned steps = additional_steps + 2;
51 std::vector<SDL_Color> colors;
52 colors.reserve( steps );
53 colors.push_back( start_color );
54
55 const float step = 1.0f / ( steps - 1 );
56 for( unsigned i = 0; i < steps - 2; i++ ) {
57 const float percent = ( i + 1 ) * step;
58 const Uint8 r = static_cast<Uint8>( start_color.r + static_cast<Uint8>( percent *
59 ( end_color.r - start_color.r ) ) );
60 const Uint8 g = static_cast<Uint8>( start_color.g + static_cast<Uint8>( percent *
61 ( end_color.g - start_color.g ) ) );
62 const Uint8 b = static_cast<Uint8>( start_color.b + static_cast<Uint8>( percent *
63 ( end_color.b - start_color.b ) ) );
64 colors.push_back( { r, g, b, 255 } );
65 }
66 colors.push_back( end_color );
67 return colors;
68}

80{
81 const float inner_ratio = static_cast<float>( inner.w ) / inner.h;
82 const float outer_ratio = static_cast<float>( outer.w ) / outer.h;

86
87 const int w = factor * inner.w;
