391 } else {
392 loc.remove_item();
393 }
394
395 u.moves -= moves_taken;
396 }
397
398 return picked_up || !did_prompt;

480std::vector<stacked_items> stack_for_pickup_ui( const
481 std::vector<item_stack::iterator> &unstacked )
482{
483 const std::pair<time_point, int> no_parent = std::make_pair(
484 calendar::before_time_starts, 0 );
485 std::map<std::pair<time_point, int>, parent_child_check_t> parent_child_check;
486 // First, we need to check which parent-child groups exist
487 for( item_stack::iterator it : unstacked ) {
488 const auto &token = *it->drop_token;
489 if( token.drop_number > 0 ) {
490 std::pair<time_point, int> turn_and_drop = std::make_pair( token.turn, token.drop_number );
491 parent_child_check[turn_and_drop].parent_exists = true;
492 }
493 if( token.parent_number != token.drop_number && token.parent_number > 0 ) {
494 std::pair<time_point, int> turn_and_parent = std::make_pair( token.turn, token.parent_number );
495 parent_child_check[turn_and_parent].child_exists = true;
496 }
497 }
498
499 // Second pass: we group children and parents together, but only if both sides are known to exist
500 std::map<std::pair<time_point, int>, unstacked_items> children_by_parent;
501 for( item_stack::iterator it : unstacked ) {
502 const auto &token = *it->drop_token;
503 std::pair<time_point, int> turn_and_drop = std::make_pair( token.turn, token.drop_number );
504 if( token.drop_number > 0 && parent_child_check[turn_and_drop].child_exists ) {
505 children_by_parent[turn_and_drop].parent = it;
506 continue;
507 }
508
509 std::pair<time_point, int> turn_and_parent = std::make_pair( token.turn, token.parent_number );
510 if( token.parent_number > 0 && token.parent_number != token.drop_number &&
511 parent_child_check[turn_and_parent].parent_exists ) {
512 children_by_parent[turn_and_parent].unstacked_children.push_back( it );
513 } else {
514 children_by_parent[no_parent].unstacked_children.push_back( it );
515 }
516 }
517
518 std::vector<stacked_items> restacked_with_parents;
519 for( const auto &pr : children_by_parent ) {
520 std::vector<std::list<item_stack::iterator>> restacked_children;
521 for( item_stack::iterator it : pr.second.unstacked_children ) {
522 bool found_stack = false;
523 for( std::list<item_stack::iterator> &stack : restacked_children ) {
524 const item &stack_top = *stack.front();
525 if( stack_top.display_stacked_with( *it ) ) {
526 stack.push_back( it );
527 found_stack = true;
528 break;
529 }
530 }
531 if( !found_stack ) {
532 restacked_children.emplace_back( std::list<item_stack::iterator>( { it } ) );
533 }
534 }
535
536 // Each sub-stack has to be sorted separately
537 std::sort( restacked_children.begin(), restacked_children.end(),
538 []( const std::list<item_stack::iterator> &lhs, const std::list<item_stack::iterator> &rhs ) {
539 return *lhs.front() < *rhs.front();
540 } );
541 restacked_with_parents.emplace_back( stacked_items{ pr.second.parent, restacked_children } );
542 }
543
544 // Sorting by parent is a bit arbitrary (parent-less go last) - sort by count?
545 std::sort( restacked_with_parents.begin(), restacked_with_parents.end(),
546 []( const stacked_items & lhs, stacked_items & rhs ) {
547 return lhs.parent.has_value() && ( !rhs.parent.has_value() || *lhs.parent < *rhs.parent );
548 } );
549
550
551 return restacked_with_parents;
552}
