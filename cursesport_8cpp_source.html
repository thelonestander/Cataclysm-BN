14/**
15 * Whoever cares, btw. not my base design, but this is how it works:
16 * In absent of a native curses library, this is a simple implementation to
17 * store the data that would be given to the curses system.
18 * It is later displayed using code in sdltiles.cpp. This file only contains
19 * the curses interface.
20 *
21 * The struct WINDOW is the base. It acts as the normal curses window, having
22 * width/height, current cursor location, current coloring (background/foreground)
23 * and the actual text.
24 * The text is split into lines (curseline), which contains cells (cursecell).
25 * Each cell has individual foreground and background, and a character. The
26 * character is an UTF-8 encoded string. It should be one or two console cells
27 * width. If it's two cells width, the next cell in the line must be completely
28 * empty (the string must not contain anything). Also the last cell of a line
29 * must not contain a two cell width string.
30 */

221// Get a sequence of Unicode code points, store them in target
222// return the display width of the extracted string.
223inline int fill( const char *&fmt, int &len, std::string &target )
224{
225 const char *const start = fmt;
226 int dlen = 0; // display width
227 const char *tmpptr = fmt; // pointer for UTF8_getch, which increments it
228 int tmplen = len;
229 while( tmplen > 0 ) {
230 const uint32_t ch = UTF8_getch( &tmpptr, &tmplen );
231 // UNKNOWN_UNICODE is most likely a (vertical/horizontal) line or similar
232 const int cw = ch == UNKNOWN_UNICODE ? 1 : mk_wcwidth( ch );
233 if( cw > 0 && dlen > 0 ) {
234 // Stop at the *second* non-zero-width character
235 break;
236 } else if( cw == -1 && start == fmt ) {
237 // First char is a control character: they only disturb the screen,
238 // so replace it with a single space (e.g. instead of a '\t').
239 // Newlines at the begin of a sequence are handled in printstring
240 target.assign( " ", 1 );
241 len = tmplen;
242 fmt = tmpptr;
243 return 1; // the space
244 } else if( cw == -1 ) {
245 // Control character but behind some other characters, finish the sequence.
246 // The character will either by handled by printstring (if it's a newline),
247 // or by the next call to this function (replaced with a space).
248 break;
249 }
250 fmt = tmpptr;
251 dlen += cw;
252 }
253 target.assign( start, fmt - start );
254 len -= target.length();
255 return dlen;
256}

258// The current cell of the window, pointed to by the cursor. The next character
259// written to that window should go in this cell.
260// Returns nullptr if the cursor is invalid (outside the window).
261inline cata_cursesport::cursecell *cur_cell( cata_cursesport::WINDOW *win )
262{
263 if( win->cursor.y >= win->height || win->cursor.x >= win->width ) {
264 return nullptr;
265 }
266 return &win->line[win->cursor.y].chars[win->cursor.x];

269//The core printing function, prints characters to the array, and sets colors
270inline void printstring( cata_cursesport::WINDOW *win, const std::string &text )
271{
272 using cata_cursesport::cursecell;
273 win->draw = true;
274 int len = text.length();
275 if( len == 0 ) {
276 return;
277 }
278 const char *fmt = text.c_str();
279 // avoid having an invalid cursorx, so that cur_cell will only return nullptr
280 // when the bottom of the window has been reached.
281 if( win->cursor.x >= win->width ) {
282 if( newline( win ) == 0 ) {
283 return;
284 }
285 }
286 if( win->cursor.y >= win->height || win->cursor.x >= win->width ) {
287 return;
288 }
289 if( win->cursor.x > 0 && win->line[win->cursor.y].chars[win->cursor.x].ch.empty() ) {
290 // start inside a wide character, erase it for good
291 win->line[win->cursor.y].chars[win->cursor.x - 1].ch.assign( " " );
292 }
293 while( len > 0 ) {
294 if( *fmt == '\n' ) {
295 if( newline( win ) == 0 ) {
296 return;
297 }
298 fmt++;
299 len--;
300 continue;
301 }
302 cursecell *curcell = cur_cell( win );
303 if( curcell == nullptr ) {
304 return;
305 }
306 const int dlen = fill( fmt, len, curcell->ch );
307 if( dlen >= 1 ) {
308 curcell->FG = win->FG;
309 curcell->BG = win->BG;
310 addedchar( win );
311 }
312 if( dlen == 1 ) {
313 // a wide character was converted to a narrow character leaving a null in the
314 // following cell ~> clear it
315 cursecell *seccell = cur_cell( win );
316 if( seccell && seccell->ch.empty() ) {
317 seccell->ch.assign( ' ', 1 );
318 }
319 } else if( dlen == 2 ) {
320 // the second cell, per definition must be empty
321 cursecell *seccell = cur_cell( win );
322 if( seccell == nullptr ) {
323 // the previous cell was valid, this one is outside of the window
324 // --> the previous was the last cell of the last line
325 // --> there should not be a two-cell width character in the last cell
326 curcell->ch.assign( ' ', 1 );
327 return;
328 }
329 seccell->FG = win->FG;
330 seccell->BG = win->BG;
331 seccell->ch.erase();
332 addedchar( win );
333 // Have just written a wide-character into the last cell, it would not
334 // display correctly if it was the last *cell* of a line
335 if( win->cursor.x == 1 ) {
336 // So make that last cell a space, move the width
337 // character in the first cell of the line
338 seccell->ch = curcell->ch;
339 curcell->ch.assign( 1, ' ' );
340 // and make the second cell on the new line empty.
341 addedchar( win );
342 cursecell *thicell = cur_cell( win );
343 if( thicell != nullptr ) {
344 thicell->ch.erase();
345 }
346 }
347 }
348 if( win->cursor.y >= win->height ) {
349 return;
350 }
351 }
