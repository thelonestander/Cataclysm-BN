156/**
157 * @anchor color_tags
158 * @name color tags
159 *
160 * Most print function have only one color parameter (or none at all), therefor they would
161 * print the whole text in one color. To print some parts of a text in a different color,
162 * one would have to write separate calls to the print functions.
163 *
164 * Color tags allow embedding coloring instructions directly into the text, which allows
165 * a single call to the print function with the full text. The color tags are removed
166 * (not printed), but the text inside the tags is printed with a specific color.
167 *
168 * Example: Print "You need a tool to do this." in white and the word "tool" in red.
169 * \code
170 * wprintz( w, c_white, "You need a " );
171 * wprintz( w, c_red, "tool" );
172 * wprintz( w, c_white, " to do this." );
173 * \endcode
174 * Same code with color tags:
175 * \code
176 * print_colored_text( w, c_white, "You need a <color_red>tool</color> to do this." );
177 * \endcode
178 *
179 * Color tags must appear in pairs, first `<color_XXX>`, followed by text, followed by `</color>`.
180 * The text in between is colored according to the color `XXX`. `XXX` must be a valid color name,
181 * see @ref color_from_string. Color tags do *not* stack, the text inside the color tags should
182 * not contain any other color tags.
183 *
184 * Functions that handle color tags should contain a note about this in their documentation. If
185 * they have no such note, they probably don't handle color tags (which means they just print the
186 * string as is).
187 *
188 * Note: use @ref colorize to add color tags to a string.
189 * \code
190 * nc_color color = ...;
191 * text = colorize( "some text", color );
192 * \endcode
193 *
194 * One can use @ref utf8_width with the second parameter set to `true` to determine the printed
195 * length of a string containing color tags. The parameter instructs `utf8_width` to ignore the
196 * color tags. For example `utf8_width("<color_red>text</color>")` would return 23, but
197 * `utf8_width("<color_red>text</color>", true)` returns 4 (the length of "text").
198 */
199/*@{*/

469/**
470 * @name Popup windows
471 *
472 * Each function displays a popup (above all other windows) with the given (formatted)
473 * text. The popup function with the flags parameters does all the work, the other functions
474 * call it with specific flags. The function can be called with a bitwise combination of flags.
475 *
476 * The functions return the key (taken from @ref getch) that was entered by the user.
477 *
478 * The message is a printf-like string. It may contain @ref color_tags, which are used while printing.
479 *
480 * - PF_GET_KEY (ignored when combined with PF_NO_WAIT) cancels the popup on *any* user input.
481 * Without the flag the popup is only canceled when the user enters new-line, Space and Escape.
482 * This flag is passed by @ref popup_getkey.
483 * - PF_ON_TOP makes the window appear on the top of the screen (at the upper most row). Without
484 * this flag, the popup is centered on the screen.
485 * The flag is passed by @ref popup_top.
486 * - PF_FULLSCREEN makes the popup window as big as the whole screen.
487 * This flag is passed by @ref full_screen_popup.
488 * - PF_NONE is a placeholder for none of the above flags.
489 *
490 */

493 PF_NONE = 0,
494 PF_GET_KEY = 1 << 0,
495 PF_ON_TOP = 1 << 2,
496 PF_FULLSCREEN = 1 << 3,
497};

673/**
674 * @return String containing enumerated elements in format: "a, b, c, ..., and z". Uses the Oxford comma.
675 * @param values A vector of strings
676 * @param conj Choose how to separate the last elements.
677 */
678template<typename _Container>
679std::string enumerate_as_string( const _Container &values,
680 enumeration_conjunction conj = enumeration_conjunction::and_ )
681{
682 const std::string final_separator = [&]() {
683 switch( conj ) {
684 case enumeration_conjunction::none:
685 return _( ", " );
686 case enumeration_conjunction::and_:
687 return ( values.size() > 2 ? _( ", and " ) : _( " and " ) );
688 case enumeration_conjunction::or_:
689 return ( values.size() > 2 ? _( ", or " ) : _( " or " ) );
690 case enumeration_conjunction::newline:
691 return "\n";
692 }
693 debugmsg( "Unexpected conjunction" );
694 return _( ", " );
695 }
696 ();
697 std::string res;
698 for( auto iter = values.begin(); iter != values.end(); ++iter ) {
699 if( iter != values.begin() ) {
700 if( conj == enumeration_conjunction::newline ) {
701 res += "\n";
702 } else if( std::next( iter ) == values.end() ) {
703 res += final_separator;
704 } else {
705 res += _( ", " );
706 }
707 }
708 res += *iter;
709 }
710 return res;
711}

715 * @param first Iterator pointing to the first element.
716 * @param last Iterator pointing to the last element.
717 * @param string_for Function that accepts an element and returns a representing string.
718 * May return an empty string to omit the element.
719 * @param conj Choose how to separate the last elements.
720 */
721template<typename _FIter, typename F>
722std::string enumerate_as_string( _FIter first, _FIter last, F string_for,

724{

728 const std::string str( string_for( *iter ) );

731 }
732 }
