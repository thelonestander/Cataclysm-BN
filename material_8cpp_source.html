116void material_type::check() const
117{
118 if( name().empty() ) {
119 debugmsg( "material %s has no name.", id.c_str() );
120 }
121 if( _dmg_adj.size() < 4 ) {
122 debugmsg( "material %s specifies insufficient damaged adjectives.", id.c_str() );
123 }
124 if( _salvaged_into && ( !_salvaged_into->is_valid() || _salvaged_into->is_null() ) ) {
125 debugmsg( "invalid \"salvaged_into\" %s for %s.", _salvaged_into->c_str(), id.c_str() );
126 }
127 if( !_repaired_with.is_valid() ) {
128 debugmsg( "invalid \"repaired_with\" %s for %s.", _repaired_with.c_str(), id.c_str() );
129 }
130
131 if( _wind_resist && ( *_wind_resist > 100 || *_wind_resist < 0 ) ) {
132 debugmsg( "Wind resistance outside of range (100%% to 0%%, is %d%%) for %s.", *_wind_resist,
133 id.str() );
134 }
135 for( auto &ca : _compact_accepts ) {
136 if( !ca.is_valid() ) {
137 debugmsg( "invalid \"compact_accepts\" %s for %s.", ca.c_str(), id.c_str() );
138 }
139 }
140 for( auto &ci : _compacts_into ) {
141 if(
142 !ci.is_valid() ||
143 !item( ci, calendar::start_of_cataclysm ).only_made_of( std::set<material_id> { id } )
144 ) {
145 debugmsg( "invalid \"compacts_into\" %s for %s.", ci.c_str(), id.c_str() );
146 }
147 }
