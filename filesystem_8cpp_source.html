156//--------------------------------------------------------------------------------------------------
157// For non-empty path, call function for each file at path.
158//--------------------------------------------------------------------------------------------------
159template <typename Function>
160void for_each_dir_entry( const std::string &path, Function function )
161{
162 using dir_ptr = DIR*;
163
164 if( path.empty() ) {
165 return;
166 }
167
168 const dir_ptr root = opendir( path.c_str() );
169 if( !root ) {
170 const auto e_str = strerror( errno );
171 dbg( DL::Warn ) << "opendir [" << path << "] failed with \"" << e_str << "\".";
172 return;
173 }
174
175 while( const auto entry = readdir( root ) ) {
176 function( *entry );
177 }
178 closedir( root );
179}

271//--------------------------------------------------------------------------------------------------
272// If at_end is true, returns whether entry's name ends in match.
273// Otherwise, returns whether entry's name contains match.
274//--------------------------------------------------------------------------------------------------
275bool name_contains( const dirent &entry, const std::string &match, const bool at_end )
276{
277 const size_t len_fname = strlen( entry.d_name );
278 const size_t len_match = match.length();
279
280 if( len_match > len_fname ) {
281 return false;
282 }
283
284 const size_t offset = at_end ? ( len_fname - len_match ) : 0;
285 return strstr( entry.d_name + offset, match.c_str() ) != nullptr;
286}

288//--------------------------------------------------------------------------------------------------
289// Return every file at root_path matching predicate.
290//
291// If root_path is empty, search the current working directory.
292// If recursive_search is true, search breadth-first into the directory hierarchy.
293//
294// Results are ordered depth-first with directories searched in lexically order. Furthermore,
295// regular files at each level are also ordered lexically by file name.
296//
297// Files ending in ~ are excluded.
298//--------------------------------------------------------------------------------------------------
299template <typename Predicate>
300std::vector<std::string> find_file_if_bfs( const std::string &root_path,
301 const bool recursive_search,
302 Predicate predicate )
303{
304 std::deque<std::string> directories;
305 if( root_path.empty() ) {
306 directories.emplace_back( "./" );
307 } else if( string_ends_with( root_path, "/" ) ) {
308 directories.emplace_back( root_path );
309 } else {
310 directories.emplace_back( root_path + "/" );
311 }
312 std::vector<std::string> results;
313
314 while( !directories.empty() ) {
315 const auto path = std::move( directories.front() );
316 directories.pop_front();
317
318 const auto n_dirs = static_cast<std::ptrdiff_t>( directories.size() );
319 const auto n_results = static_cast<std::ptrdiff_t>( results.size() );
320
321 for_each_dir_entry( path, [&]( const dirent & entry ) {
322 // exclude special directories.
323 if( is_special_dir( entry ) ) {
324 return;
325 }
326
327 const auto full_path = path + entry.d_name;
328
329 // don't add files ending in '~'.
330 if( full_path.back() == '~' ) {
331 return;
332 }
333
334 // add sub directories to recursive_search if requested
335 const auto is_dir = is_directory( entry, full_path );
336 if( recursive_search && is_dir ) {
337 directories.emplace_back( full_path + "/" );
338 }
339
340 // check the file
341 if( !predicate( entry, is_dir ) ) {
342 return;
343 }
344
345 results.emplace_back( full_path );
346 } );
347
348 // Keep files and directories to recurse ordered consistently
349 // by sorting from the old end to the new end.
350 // NOLINTNEXTLINE(cata-use-localized-sorting)
351 std::sort( std::begin( directories ) + n_dirs, std::end( directories ) );
352 // NOLINTNEXTLINE(cata-use-localized-sorting)
353 std::sort( std::begin( results ) + n_results, std::end( results ) );
354 }
355
356 return results;
357}

370/**
371 * Find directories which containing pattern.
372 * @param pattern Search pattern.
373 * @param root_path Search root.
374 * @param recursive_search Be recurse or not.
375 * @return vector or directories without pattern filename at end.
376 */
377std::vector<std::string> get_directories_with( const std::string &pattern,
378 const std::string &root_path, const bool recursive_search )
379{
380 if( pattern.empty() ) {
381 return std::vector<std::string>();
382 }
383
384 auto files = find_file_if_bfs( root_path, recursive_search, [&]( const dirent & entry, bool ) {
385 return name_contains( entry, pattern, true );
386 } );
387
388 // Chop off the file names. Dir path MUST be splitted by '/'
389 for( auto &file : files ) {
390 file.erase( file.rfind( '/' ), std::string::npos );
391 }
392
393 files.erase( std::unique( std::begin( files ), std::end( files ) ), std::end( files ) );
394
395 return files;

398/**
399 * Find directories which containing pattern.
400 * @param patterns Search patterns.
401 * @param root_path Search root.
402 * @param recursive_search Be recurse or not.
403 * @return vector or directories without pattern filename at end.
404 */
405std::vector<std::string> get_directories_with( const std::vector<std::string> &patterns,
406 const std::string &root_path, const bool recursive_search )
407{
408 if( patterns.empty() ) {
409 return std::vector<std::string>();
