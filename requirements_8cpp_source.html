478template <typename T>
479void inline_requirements( std::vector<std::vector<T>> &list,
480 const std::function<const std::vector<std::vector<T>> & ( const requirement_data & )> &getter )
481{
482 // add a single component to the vector. If component already exists, chooses min count
483 const auto add_component = []( const T & comp, std::vector<T> &accum ) {
484 auto iter = std::find_if( accum.begin(), accum.end(), [&]( const T & req ) {
485 return !req.requirement && req.type == comp.type;
486 } );
487 if( iter == accum.end() ) {
488 accum.push_back( comp ); // component doesn't exist yet, adding it
489 } else {
490 // this is the list of alternatives (a OR b OR c), so if there are two identical components
491 // with the different counts as an alternatives, the one with the lowest count is selected
492 iter->count = std::min( comp.count, iter->count );
493 }
494 };
495
496 // if `comp` is a component, adds it to the `accum`
497 // if it's a requirement, recursively expands it into simple components and adds them to the vector
498 const auto inline_one = [&]( const T & comp, std::vector<T> &accum ) {
499 std::set<requirement_id> already_nested;
500 std::list<requirement_id> stack;
501
502 std::function<void( const T &comp )> rec;
503 rec = [&]( const T & comp ) {
504 // add simple component to the vector
505 if( !comp.requirement ) {
506 add_component( comp, accum );
507 return;
508 }
509 // otherwise expand component as requirement
510 const requirement_id r( comp.type.str() );
511 if( !r.is_valid() ) {
512 debugmsg( "Tried to inline unknown requirement %s", r.c_str() );
513 return;
514 }
515 // stack just holds the current path of inlining for debug purposes
516 stack.push_back( r );
517 if( already_nested.count( r ) ) {
518 // print debug msg and skip just this one requirement
519 debugmsg( "Tried to inline requirement %s which forms a cycle", r.c_str() );
520 stack.pop_back();
521 return;
522 }
523 already_nested.insert( r );
524
525 const auto &req = r.obj();
526 const requirement_data multiplied = req * comp.count;
527
528 const std::vector<std::vector<T>> &to_inline = getter( multiplied );
529 if( !to_inline.empty() ) {
530 const std::vector<T> &nested = to_inline.front();
531 for( const T &nested_comp : nested ) {
532 rec( nested_comp );
533 }
534 }
535 already_nested.erase( r );
536 stack.pop_back();
537 };
538
539 rec( comp );
540 };

542 for( std::vector<T> &vec : list ) {
543 const std::vector<T> vec_copy = vec;
544 vec.clear();
545 for( const T &comp : vec_copy ) {
546 inline_one( comp, vec );

863bool requirement_data::check_enough_materials( const item_comp &comp, const inventory &crafting_inv,
864 const std::function<bool( const item & )> &filter, int batch ) const
865{
866 if( comp.available != available_status::a_true ) {
867 return false;
868 }
869 const int cnt = std::abs( comp.count ) * batch;
870 const tool_comp *tq = find_by_type( tools, comp.type );
871 if( tq != nullptr && tq->available == available_status::a_true ) {
872 // The very same item type is also needed as tool!
873 // Use charges of it, or use it by count?
874 const int tc = tq->by_charges() ? 1 : std::abs( tq->count );
875 // Check for components + tool count. Check item amount (excludes
876 // pseudo items) and tool amount (includes pseudo items)
877 // Imagine: required = 1 welder (component) + 1 welder (tool),
878 // available = 1 welder (real item), 1 welding rig (creates
879 // a pseudo welder item). has_components(welder,2) returns false
880 // as there is only one real welder available, but has_tools(welder,2)
881 // returns true.
882 // Keep in mind that both requirements (tool+component) are checked
883 // before this. That assures that one real item is actually available,
884 // two welding rigs (and no real welder) would make this component
885 // non-available even before this function is called.
886 // Only ammo and (some) food is counted by charges, both are unlikely
887 // to appear as tool, but it's possible /-:
888 const item_comp i_tmp( comp.type, cnt + tc );
889 const tool_comp t_tmp( comp.type, -( cnt + tc ) ); // not by charges!
890 // batch factor is explicitly 1, because it's already included in the count.
891 if( !i_tmp.has( crafting_inv, filter, 1 ) && !t_tmp.has( crafting_inv, filter, 1 ) ) {
892 comp.available = available_status::a_insufficent;
893 }
894 }
895 const itype *it = &*comp.type;
896 for( const auto &ql : it->qualities ) {
897 const quality_requirement *qr = find_by_type( qualities, ql.first );
898 if( qr == nullptr || qr->level > ql.second ) {
899 continue;
900 }
901 // This item can be used for the quality requirement, same as above for specific
902 // tools applies.
903 if( !crafting_inv.has_quality( qr->type, qr->level, qr->count + std::abs( comp.count ) ) ) {
904 comp.available = available_status::a_insufficent;
905 }
906 }
907 return comp.available == available_status::a_true;
908}

911static bool apply_blacklist( std::vector<std::vector<T>> &vec, const itype_id &id )
912{
913 // remove all instances of @id type from each of the options
914 for( auto &opts : vec ) {
915 opts.erase( std::remove_if( opts.begin(), opts.end(), [&id]( const T & e ) {
916 return e.type == id;
917 } ), opts.end() );
918 }

920 // did we remove the last instance of an option group?
921 const bool blacklisted = std::any_of( vec.begin(), vec.end(), []( const std::vector<T> &e ) {
922 return e.empty();
923 } );
924

926 vec.erase( std::remove_if( vec.begin(), vec.end(), []( const std::vector<T> &e ) {
927 return e.empty();
928 } ), vec.end() );

930 return blacklisted;

939template <typename T>
940static void apply_replacement( std::vector<std::vector<T>> &vec, const itype_id &id,
941 const itype_id &replacement )
942{
943 // If the target and replacement are both present, remove the target.
944 // If only the target is present, replace it.
945 for( auto &opts : vec ) {
946 typename std::vector<T>::iterator target = opts.end();
947 typename std::vector<T>::iterator replacement_target = opts.end();
948 for( typename std::vector<T>::iterator iter = opts.begin(); iter != opts.end(); ++iter ) {
949 if( iter->type == id ) {
950 target = iter;
951 } else if( iter->type == replacement ) {
952 replacement_target = iter;
953 }
954 }
955 // No target to replace, do nothing.
956 if( target == opts.end() ) {
957 continue;
958 }
959 // Target but no replacement, replace.
960 if( replacement_target == opts.end() ) {
961 target->type = replacement;
962 continue;
963 }
964 // Both target and replacement, remove the target entry and leave the existing replacement.
965 opts.erase( target );
966 }
967}

1165template<typename T, typename Accum>
1166static std::vector<std::vector<T>> consolidate( std::vector<std::vector<T>> old_vec,
1167 const Accum &accum )
1168{
1169 const auto type_lt = []( const T & lhs, const T & rhs ) -> bool {
1170 //TODO change to use localized sorting
1171 return std::forward_as_tuple( lhs.type.str(), lhs.requirement )
1172 < std::forward_as_tuple( rhs.type.str(), rhs.requirement );
1173 };
1174 // in order to simplify blueprint requirements, we merge a longer requirement
1175 // list into a shorter requirement list whose types are a subsequence of the
1176 // longer list's types. However, this operation is not symmetric and depends
1177 // on the order of the requirement lists. Thus we sort the lists first, to
1178 // ensure consistent results when the order of construction requirements changes.
1179 for( std::vector<T> &old_inner : old_vec ) {
1180 std::sort( old_inner.begin(), old_inner.end(), type_lt );
1181 }
1182 std::sort( old_vec.begin(), old_vec.end(),
1183 [&type_lt]( const std::vector<T> &lhs, const std::vector<T> &rhs ) -> bool {
1184 return std::lexicographical_compare( lhs.begin(), lhs.end(), rhs.begin(), rhs.end(),
1185 type_lt );
1186 } );
1187
1188 std::vector<std::vector<T>> new_vec;
1189 for( std::vector<T> &old_inner : old_vec ) {
1190 bool match = false;
1191 for( std::vector<T> &new_inner : new_vec ) {
1192 // in order to simplify blueprint requirements, we merge a longer
1193 // requirement list into a shorter requirement list whose types are
1194 // a subsequence of the longer list's types.
1195 //
1196 // note this actually may make a requirement stricter.
1197 // for example, if the item requirement was
1198 // [ [ [ "a", 1 ], [ "b", 1 ], [ "c", 1 ] ],
1199 // [ [ "a", 1 ], [ "b", 1 ] ] ]
1200 // then you could satisfy it by having one "a" and one "b", one
1201 // "c" and one "a", two "a", or two "b", etc.
1202 //
1203 // but after consolidation, it becomes
1204 // [ [ [ "a", 2 ], [ "b", 2 ] ] ]
1205 // then you can only satisfy it by having either 2 "a" or 2 "b"
1206 if( std::includes( new_inner.begin(), new_inner.end(),
1207 old_inner.begin(), old_inner.end(),
1208 type_lt ) ) {
1209 // old_inner is a subsequence of new_inner
1210 match = true;
1211 std::swap( old_inner, new_inner );
1212 } else if( std::includes( old_inner.begin(), old_inner.end(),
1213 new_inner.begin(), new_inner.end(),
1214 type_lt ) ) {
1215 // new_inner is a subsequence of old_inner
1216 match = true;
1217 }
1218 if( match ) {
1219 for( auto it1 = new_inner.begin(), it2 = old_inner.begin();
1220 it1 < new_inner.end(); ++it2 ) {
1221 if( !type_lt( *it2, *it1 ) ) {
1222 // which means *it2 and *it1 have the same type, since
1223 // we know new_inner is a subsequence of old_inner
1224 *it1 = accum( *it1, *it2 );
1225 ++it1;
1226 }
1227 }
1228 break;
1229 }
1230 }
1231 if( !match ) {
1232 new_vec.emplace_back( old_inner );
1233 }
1234 }
1235 return new_vec;
1236}

1238void requirement_data::consolidate()
1239{
1240 qualities = ::consolidate( qualities,
1241 []( const quality_requirement & lhs, const quality_requirement & rhs ) {
1242 quality_requirement ret = lhs;
1243 ret.count = std::max( ret.count, rhs.count );
1244 ret.level = std::max( ret.level, rhs.level );
1245 return ret;
1246 } );

1248 tools = ::consolidate( tools,
1249 []( const tool_comp & lhs, const tool_comp & rhs ) {
1250 tool_comp ret = lhs;
1251 if( ret.count < 0 && rhs.count < 0 ) {
1252 ret.count = std::min( ret.count, rhs.count );
1253 } else if( ret.count > 0 && rhs.count > 0 ) {
1254 ret.count += rhs.count;
1255 } else {
1256 debugmsg( "required counts of the same tool have different signs" );
1257 }
1258 return ret;
1259 } );
1260
1261 components = ::consolidate( components,
1262 []( const item_comp & lhs, const item_comp & rhs ) {
1263 item_comp ret = lhs;
1264 ret.count += rhs.count;
1265 return ret;
1266 } );
1267}

1323///
1324/// The goal of this function is to consolidate a particular item_comp that
1325/// would otherwise be duplicated between two requirements.

1328/// searching for another item_comp to merge @p leftover with. For each
1329/// compatible item_comp found it performs that merger and writes out a
1330/// suitably updated form of the overall requirements to @p result.
1331///
1332/// If it chooses *not* to merge with any particular item_comp, then it deletes
1333/// that item_comp from the options, to avoid the duplication.
1334///
1335/// Lastly, it also outputs a version of the requirements where @p leftover
1336/// remains where it was, and all other compatible item_comp entries have been
1337/// deleted.
1338///
1339/// @param leftover The item_comp needing to be dealt with.
1340/// @param req_prefix The requirements considered so far; more will be appended
1341/// to this.
1342/// @param to_expand The original requirements we are working through to look
1343/// for a duplicate.
1344/// @param orig_index The index into the alter_item_comp_vector where @p
1345/// leftover was originally to be found. If it isn't merged with another item,
1346/// then it will be re-inserted at this position.
1347/// @param index The position within @p to_expand where we will next look for
1348/// duplicates of @p leftover to merge with.
1349/// @param result The finished requirements should be appended to this.
1350static void expand_item_in_reqs(
1351 const item_comp &leftover, requirement_data::alter_item_comp_vector req_prefix,
1352 const requirement_data::alter_item_comp_vector &to_expand, size_t orig_index, size_t index,
1353 std::vector<requirement_data::alter_item_comp_vector> &result )
1354{
1355 assert( req_prefix.size() >= orig_index );
1356 assert( orig_index < index );
1357
1358 if( index == to_expand.size() ) {
1359 // We reached the end without using the leftovers. So need to add them
1360 // as their own requirement, separate from everything else.
1361 req_prefix.insert( req_prefix.begin() + orig_index, { leftover } );
1362 result.push_back( req_prefix );
1363 return;
1364 }
1365
1366 std::vector<item_comp> this_requirement = to_expand[index];
1367 auto duplicate = std::find_if( this_requirement.begin(), this_requirement.end(),
1368 [&]( const item_comp & c ) {
1369 return c.type == leftover.type;
1370 } );
1371 if( duplicate == this_requirement.end() ) {
1372 // No match in this one; proceed to next
1373 req_prefix.push_back( this_requirement );
1374 expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1375 return;
1376 }
1377 // First option: amalgamate the leftovers into this requirement, which
1378 // forces us to pick that specific option:
1379 requirement_data::alter_item_comp_vector req = req_prefix;
1380 req.push_back( { item_comp( leftover.type, leftover.count + duplicate->count ) } );
1381 req.insert( req.end(), to_expand.begin() + index + 1, to_expand.end() );
1382 result.push_back( req );
1383
1384 // Second option: use a separate option for this requirement, which means
1385 // we need to recurse further to find something into which to amalgamate
1386 // the original requirement
1387 this_requirement.erase( duplicate );
1388 if( !this_requirement.empty() ) {
1389 req_prefix.push_back( this_requirement );
1390 expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1391 }
1392}

1394deduped_requirement_data::deduped_requirement_data( const requirement_data &in,
1395 const recipe_id &context )
1396{
1397 // This constructor works through a requirement_data, converting it into an
1398 // equivalent set of requirement_data alternatives, where each alternative
1399 // has the property that no item type appears more than once.
1400 //
1401 // We only deal with item requirements. Tool requirements could be handled
1402 // similarly, but no examples where they are a problem have yet been
1403 // raised.
1404 //
1405 // We maintain a queue of requirement_data component info to be split.
1406 // Each to_check struct has a vector of component requirements, and an
1407 // index. The index is the position within the vector to be checked next.
1408 struct to_check {
1409 alter_item_comp_vector components;
1410 size_t index;
1411 };
1412 std::stack<to_check, std::vector<to_check>> pending;
1413 pending.push( { in.get_components(), 0 } );
1414
1415 while( !pending.empty() ) {
1416 to_check next = pending.top();
1417 pending.pop();
1418
1419 if( next.index == next.components.size() ) {
1420 alternatives_.emplace_back( in.get_tools(), in.get_qualities(), next.components );
1421 continue;
1422 }
1423
1424 // Build a set of all the itypes used in later stages of this set of
1425 // requirements.
1426 std::unordered_set<itype_id> later_itypes;
1427 for( size_t i = next.index + 1; i != next.components.size(); ++i ) {
1428 std::transform( next.components[i].begin(), next.components[i].end(),
1429 std::inserter( later_itypes, later_itypes.end() ),
1430 []( const item_comp & c ) {
1431 return c.type;
1432 } );
1433 }
1434
1435 std::vector<item_comp> this_requirement = next.components[next.index];
1436
1437 auto first_duplicated = std::stable_partition(
1438 this_requirement.begin(), this_requirement.end(),
1439 [&]( const item_comp & c ) {
1440 return !later_itypes.count( c.type );
1441 }
1442 );
1443
1444 for( auto comp_it = first_duplicated; comp_it != this_requirement.end(); ++comp_it ) {
1445 // Factor this requirement out into its own separate case
1446
1447 alter_item_comp_vector req_prefix( next.components.begin(),
1448 next.components.begin() + next.index );
1449 std::vector<alter_item_comp_vector> result;
1450 expand_item_in_reqs( *comp_it, req_prefix, next.components, next.index, next.index + 1,
1451 result );
1452 for( const alter_item_comp_vector &v : result ) {
1453 // When v is smaller, that means the current requirement was
1454 // deleted, in which case we don't advance index.
1455 size_t index_inc = v.size() == next.components.size() ? 1 : 0;
1456 pending.push( { v, next.index + index_inc } );
1457 }
1458 }
1459
1460 // Deal with all the remaining, non-duplicated ones
1461 this_requirement.erase( first_duplicated, this_requirement.end() );
1462 if( !this_requirement.empty() ) {
1463 alter_item_comp_vector without_dupes = next.components;
1464 without_dupes[next.index] = this_requirement;
1465 pending.push( { without_dupes, next.index + 1 } );
1466 }
1467
1468 // Because this algorithm is super-exponential in the worst case, add a
1469 // sanity check to prevent things getting too far out of control.
1470 // The worst case in the core game currently is chainmail_suit_faraday
1471 // with 63 alternatives.
1472 static constexpr size_t max_alternatives = 100;
1473 if( alternatives_.size() + pending.size() > max_alternatives ) {
1474 debugmsg( "Construction of deduped_requirement_data generated too many alternatives. "
1475 "The recipe %s should be simplified. See the Recipe section in "
1476 "doc/JSON_INFO.md for more details.", context.str() );
1477 is_too_complex_ = true;
1478 alternatives_ = { in };
1479 return;
1480 }
1481 }
1482}
