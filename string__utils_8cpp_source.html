139bool wildcard_match( const std::string &text_in, const std::string &pattern_in )
140{
141 std::string text = text_in;
142
143 if( text.empty() ) {
144 return false;
145 } else if( text == "*" ) {
146 return true;
147 }
148
149 std::vector<std::string> pattern = string_split( wildcard_trim_rule( pattern_in ), '*' );
150
151 if( pattern.size() == 1 ) { // no * found
152 return ( text.length() == pattern[0].length() && ci_find_substr( text, pattern[0] ) != -1 );
153 }
154
155 for( auto it = pattern.begin(); it != pattern.end(); ++it ) {
156 if( it == pattern.begin() && !it->empty() ) {
157 if( text.length() < it->length() ||
158 ci_find_substr( text.substr( 0, it->length() ), *it ) == -1 ) {
159 return false;
160 }
161
162 text = text.substr( it->length(), text.length() - it->length() );
163 } else if( it == pattern.end() - 1 && !it->empty() ) {
164 if( text.length() < it->length() ||
165 ci_find_substr( text.substr( text.length() - it->length(),
166 it->length() ), *it ) == -1 ) {
167 return false;
168 }
169 } else {
170 if( !( *it ).empty() ) {
171 int pos = ci_find_substr( text, *it );
172 if( pos == -1 ) {
173 return false;
174 }
175
176 text = text.substr( pos + static_cast<int>( it->length() ),
177 static_cast<int>( text.length() ) - pos );
178 }
179 }
180 }
181
182 return true;
183}

185std::string wildcard_trim_rule( const std::string &pattern_in )
186{
187 std::string pattern = pattern_in;
188 size_t pos = pattern.find( "**" );
189
190 //Remove all double ** in pattern
191 while( pos != std::string::npos ) {
192 pattern = pattern.substr( 0, pos ) + pattern.substr( pos + 1, pattern.length() - pos - 1 );
193 pos = pattern.find( "**" );
