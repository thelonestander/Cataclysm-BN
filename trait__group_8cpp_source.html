35// NOTE: This function is largely based on item_group::get_unique_group_id()
36static Trait_group_tag get_unique_trait_group_id()
37{
38 // This is just a hint what id to use next. Overflow of it is defined and if the group
39 // name is already used, we simply go the next id.
40 static unsigned int next_id = 0;
41 // Prefixing with a character outside the ASCII range, so it is hopefully unique and
42 // (if actually printed somewhere) stands out. Theoretically those auto-generated group
43 // names should not be seen anywhere.
44 static const std::string unique_prefix = "\u01F7 ";
45 while( true ) {
46 const Trait_group_tag new_group( unique_prefix + std::to_string( next_id++ ) );
47 if( !new_group.is_valid() ) {
48 return new_group;
49 }
50 }
51}

53Trait_group_tag trait_group::load_trait_group( const JsonValue &value,
54 const std::string &default_subtype )
55{
56 if( value.test_string() ) {
57 return Trait_group_tag( value.get_string() );
58 } else if( value.test_object() ) {
59 const Trait_group_tag group = get_unique_trait_group_id();
60
61 JsonObject jo = value.get_object();
62 const std::string subtype = jo.get_string( "subtype", default_subtype );
63
64 mutation_branch::load_trait_group( jo, group, subtype );
65
66 return group;
67 } else if( value.test_array() ) {
68 const Trait_group_tag group = get_unique_trait_group_id();
69
70 if( default_subtype != "collection" && default_subtype != "distribution" ) {
71 value.throw_error( "invalid subtype for trait group" );
72 }
73
74 mutation_branch::load_trait_group( value.get_array(), group, default_subtype == "collection" );
75 return group;
76 } else {
77 value.throw_error( "invalid trait group, must be string (group id) or object/array (the group data)" );
78 return Trait_group_tag{};
79 }
80}

83void trait_group::debug_spawn()
84{
85 std::vector<Trait_group_tag> groups = mutation_branch::get_all_group_names();
86 uilist menu;
87 menu.text = _( "Test which group?" );
88 for( size_t i = 0; i < groups.size(); i++ ) {
89 menu.entries.emplace_back( static_cast<int>( i ), true, -2, groups[i].str() );
90 }
91 while( true ) {
92 menu.query();
93 const int index = menu.ret;
94 if( index >= static_cast<int>( groups.size() ) || index < 0 ) {
95 break;
96 }
97 // Spawn traits from the group 100 times
98 std::map<std::string, int> traitnames;
99 for( size_t a = 0; a < 100; a++ ) {
100 const auto traits = traits_from( groups[index] );
101 for( auto &tr : traits ) {
102 traitnames[mutation_branch::get_name( tr )]++;
103 }
104 }
105 // Invert the map to get sorting!
106 std::multimap<int, std::string> traitnames2;
107 for( const auto &e : traitnames ) {
108 traitnames2.insert( std::pair<int, std::string>( e.second, e.first ) );
109 }
110 uilist menu2;
111 menu2.text = _( "Result of 100 spawns:" );
112 for( const auto &e : traitnames2 ) {
113 menu2.entries.emplace_back( static_cast<int>( menu2.entries.size() ), true, -2,
114 string_format( _( "%d x %s" ), e.first, e.second ) );
115 }
116 menu2.query();

166Trait_list Trait_group_creator::create( RecursionList &rec ) const
167{
168
169 Trait_list result;
170 if( std::find( rec.begin(), rec.end(), id ) != rec.end() ) {
171 debugmsg( "recursion in trait creation list %s", id.c_str() );
172 return result;
173 }
174 rec.push_back( id );
175
176 if( !id.is_valid() ) {
177 debugmsg( "unknown trait creation list %s", id.c_str() );
178 return result;
179 }
180 const auto tcd = mutation_branch::get_group( id );
181
182 Trait_list tmplist = tcd->create( rec );
183 rec.pop_back();
184 result.insert( result.end(), tmplist.begin(), tmplist.end() );
185
186 return result;
187}

293Trait_list Trait_group_distribution::create( RecursionList &rec ) const
294{
295 Trait_list result;
296 int p = rng( 0, sum_prob - 1 );
297 for( const auto &creator : creators ) {
298 p -= ( creator )->probability;
299 if( p >= 0 ) {
300 continue;
301 }
302 Trait_list tmp = ( creator )->create( rec );
303 result.insert( result.end(), tmp.begin(), tmp.end() );
304 break;
