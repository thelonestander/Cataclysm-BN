102void node_t::load( const JsonObject &jo, const std::string & )
103{
104 // We don't initialize the node unless it has no children (opportunistic optimization).
105 // Instead we initialize a parallel struct that holds the labels until finalization.
106 if( jo.has_array( "children" ) ) {
107 temp_node_data.emplace_back();
108 node_data &new_data = temp_node_data.back();
109 new_data.id = id;
110 optional( jo, was_loaded, "children", new_data.children );
111 }
112 if( jo.has_string( "strategy" ) ) {
113 std::unordered_map<std::string, const strategy_t *>::iterator new_strategy =
114 behavior::strategy_map.find( jo.get_string( "strategy" ) );
115 if( new_strategy != strategy_map.end() ) {
116 strategy = new_strategy->second;
117 } else {
118 debugmsg( "While loading %s, failed to find strategy %s.",
119 id.str(), jo.get_string( "strategy" ) );
120 jo.throw_error( "Invalid strategy in behavior." );
121 }
122 }
123 if( jo.has_string( "predicate" ) ) {
124 auto new_predicate = predicate_map.find( jo.get_string( "predicate" ) );
125 if( new_predicate != predicate_map.end() ) {
126 predicate = new_predicate->second;
127 } else {
128 debugmsg( "While loading %s, failed to find predicate %s.",
129 id.str(), jo.get_string( "predicate" ) );
130 jo.throw_error( "Invalid strategy in behavior." );
131 }
132 }
133 optional( jo, was_loaded, "goal", _goal );
134}
