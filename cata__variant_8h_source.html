74// type_for<T>() returns the cata_variant_type enum value corresponding to the
75// given value type. e.g. type_for<mtype_id>() == cata_variant_type::mtype_id.
76
77template<typename T, size_t... I>
78constexpr cata_variant_type type_for_impl( std::index_sequence<I...> )
79{
80 constexpr size_t num_types = static_cast<size_t>( cata_variant_type::num_types );
81 constexpr std::array<bool, num_types> matches = {{
82 std::is_same<T, typename convert<static_cast<cata_variant_type>( I )>::type>::value...
83 }
84 };
85 for( size_t i = 0; i < num_types; ++i ) {
86 if( matches[i] ) {
87 return static_cast<cata_variant_type>( i );
88 }
89 }
90 // No match
91 return cata_variant_type::num_types;
92}
93
94template<typename T>
95constexpr cata_variant_type type_for()
96{
97 constexpr size_t num_types = static_cast<size_t>( cata_variant_type::num_types );
98 using SimpleT = std::remove_cv_t<std::remove_reference_t<T>>;
99 return type_for_impl<SimpleT>( std::make_index_sequence<num_types> {} );
100}

260 public:
261 // Default constructor for an 'empty' variant (you can't get a value
262 // from it).
263 cata_variant() : type_( cata_variant_type::void_ ) {}
264
265 // Constructor that attempts to infer the type from the type of the
266 // value passed.
267 template < typename Value,
268 typename = std::enable_if_t <(
269 cata_variant_detail::type_for<Value>() < cata_variant_type::num_types )>>
270 explicit cata_variant( Value && value ) {
271 constexpr cata_variant_type Type = cata_variant_detail::type_for<Value>();
272 type_ = Type;
273 value_ =
274 cata_variant_detail::convert<Type>::to_string( std::forward<Value>( value ) );
275 }
276
277 // Call this function to unambiguously construct a cata_variant instance,
278 // pass the type as a template parameter.
279 // In cases where the value type alone is unambiguous, the above
280 // simpler constructor call can be used.
281 template<cata_variant_type Type, typename Value>
282 static cata_variant make( Value &&value ) {
283 return cata_variant(
284 Type, cata_variant_detail::convert<Type>::to_string(
285 std::forward<Value>( value ) ) );
286 }
287
288 cata_variant_type type() const {
289 return type_;
290 }
291
292 template<cata_variant_type Type>
293 auto get() const -> typename cata_variant_detail::convert<Type>::type {
294 if( type_ != Type ) {
295 debugmsg( "Tried to extract type %s from cata_variant which contained %s",
296 io::enum_to_string( Type ),
297 io::enum_to_string( type_ ) );
298 return {};
299 }
300 return cata_variant_detail::convert<Type>::from_string( value_ );
301 }
