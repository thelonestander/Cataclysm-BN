14/**
15 * @brief Sort elements of a sequence by their rating (the smaller the better).
16 *
17 * @param begin An input/output iterator.
18 * @param end An input/output iterator.
19 * @param rating_func A functor which calculates ratings (the outcome
20 * must be comparable using '<' operator).
21 *
22 * The rating is calculated only once per each element.
23 */
24template<typename Iterator, typename RatingFunction>
25void sort_by_rating( Iterator begin, Iterator end, RatingFunction rating_func )
26{
27 using value_t = typename Iterator::value_type;
28 using pair_t = std::pair<value_t, decltype( rating_func( *begin ) )>;
29
30 std::vector<pair_t> rated_entries;
31 rated_entries.reserve( std::distance( begin, end ) );
32
33 std::transform( begin, end, std::back_inserter( rated_entries ),
34 [ &rating_func ]( const value_t &elem ) {
35 return std::make_pair( elem, rating_func( elem ) );
36 } );
37
38 std::sort( rated_entries.begin(), rated_entries.end(), []( const pair_t &lhs, const pair_t &rhs ) {
39 return lhs.second < rhs.second;
40 } );
41
42 std::transform( rated_entries.begin(), rated_entries.end(), begin, []( const pair_t &elem ) {
43 return elem.first;
44 } );
45}

47// Implementation detail of below find_cycles
48// This explores one branch of the given graph depth-first
49template<typename T>
50void find_cycles_impl(
51 const std::unordered_map<T, std::vector<T>> &edges,
52 const T &v,
53 std::unordered_set<T> &visited,
54 std::unordered_map<T, T> &on_current_branch,
55 std::vector<std::vector<T>> &result )
56{
57 bool new_vertex = visited.insert( v ).second;
58
59 if( !new_vertex ) {
60 return;
61 }
62 auto it = edges.find( v );
63 if( it == edges.end() ) {
64 return;
65 }
66
67 for( const T &next_v : it->second ) {
68 if( next_v == v ) {
69 // Trivial self-loop
70 result.push_back( { v } );
71 continue;
72 }
73 auto previous_match = on_current_branch.find( next_v );
74 if( previous_match != on_current_branch.end() ) {
75 // We have looped back to somewhere along the branch we took to
76 // reach this vertex, so reconstruct the loop and save it.
77 std::vector<T> loop;
78 T on_path = v;
79 while( true ) {
80 loop.push_back( on_path );
81 if( on_path == next_v ) {
82 break;
83 }
84 on_path = on_current_branch[on_path];
85 }
86 std::reverse( loop.begin(), loop.end() );
87 result.push_back( loop );
88 } else {
89 on_current_branch.emplace( next_v, v );
90 find_cycles_impl( edges, next_v, visited, on_current_branch, result );
91 on_current_branch.erase( next_v );
92 }
93 }
94}

96// Find and return a list of all cycles in a directed graph.
97// Each T defines a vertex.
98// For a vertex a, edges[a] is a list of all the vertices connected by edges
99// from a.
100// It is acceptable for some vertex keys to be missing from the edges map, if
101// those vertices have no out-edges.
102// Complexity should be O(V+E)
103// Based on https://www.geeksforgeeks.org/detect-cycle-in-a-graph/
104template<typename T>
105std::vector<std::vector<T>> find_cycles( const std::unordered_map<T, std::vector<T>> &edges )
106{
107 std::unordered_set<T> visited;
108 std::unordered_map<T, T> on_current_branch;
109 std::vector<std::vector<T>> result;
110
111 for( const auto &p : edges ) {
112 const T &root = p.first;
113
114 on_current_branch.emplace( root, root );
115 find_cycles_impl( edges, root, visited, on_current_branch, result );
116 on_current_branch.erase( root );
117 assert( on_current_branch.empty() );
118 }
119
120 return result;
121}
