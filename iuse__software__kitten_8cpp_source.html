418void robot_finds_kitten::process_input()
419{
420 input_context ctxt( "IUSE_SOFTWARE_KITTEN" );
421 ctxt.register_action( "QUIT" );
422 ctxt.register_action( "HELP_KEYBINDINGS" );
423 ctxt.register_action( "ANY_INPUT" );
424 switch( current_ui_state ) {
425 case ui_state::instructions: {
426 const std::string action = ctxt.handle_input();
427 if( action == "QUIT" ) {
428 current_ui_state = ui_state::exit;
429 } else if( action == "ANY_INPUT" && !ctxt.get_raw_input().sequence.empty() ) {
430 current_ui_state = ui_state::main;
431 }
432 break;
433 }
434 case ui_state::main:
435 case ui_state::invalid_input:
436 case ui_state::bogus_message: {
437 ctxt.register_cardinal();
438 const std::string action = ctxt.handle_input();
439 if( action == "QUIT" ) {
440 current_ui_state = ui_state::exit;
441 } else if( action == "ANY_INPUT" ) {
442 if( !ctxt.get_raw_input().sequence.empty() ) {
443 current_ui_state = ui_state::invalid_input;
444 }
445 } else if( action != "HELP_KEYBINDINGS" ) {
446 current_ui_state = ui_state::main;
447 point check = robot.pos;
448
449 if( action == "UP" ) {
450 check.y--;
451 } else if( action == "DOWN" ) {
452 check.y++;
453 } else if( action == "LEFT" ) {
454 check.x--;
455 } else if( action == "RIGHT" ) {
456 check.x++;
457 }
458
459 constexpr half_open_rectangle<point> bounds(
460 point( 0, 3 ), point( rfkCOLS, rfkLINES ) );
461 if( !bounds.contains( check ) ) {
462 /* Can't move past edge */
463 } else if( rfkscreen[check.x][check.y] != EMPTY ) {
464 switch( rfkscreen[check.x][check.y] ) {
465 case ROBOT:
466 /* We didn't move. */
467 break;
468 case KITTEN:
469 /* Found it! */
470 ret = true;
471 current_ui_state = ui_state::end_animation;
472 end_animation_frame = 0;
473 end_animation_last_input_left_or_up = action == "UP" || action == "LEFT";
474 break;
475 default:
476 current_ui_state = ui_state::bogus_message;
477 bogus_message_idx = bogus_messages[rfkscreen[check.x][check.y] - 2];
478 break;
479 }
480 } else {
481 /* Otherwise, move the robot. */
482 rfkscreen[robot.pos.x][robot.pos.y] = EMPTY;
483 robot.pos = check;
484 rfkscreen[robot.pos.x][robot.pos.y] = ROBOT;
485 }
486 }
487 break;
488 }
489 case ui_state::end_animation:
490 if( end_animation_frame + 1 == num_end_animation_frames ) {
491 const std::string action = ctxt.handle_input();
492 if( action != "HELP_KEYBINDINGS" ) {
493 current_ui_state = ui_state::exit;
494 }
495 } else {
496 refresh_display();
497 end_animation_frame++;
498 timespec ts;
499 ts.tv_sec = 1;
500 ts.tv_nsec = 0;
501 nanosleep( &ts, nullptr );
502 }
503 break;
504 case ui_state::exit:
505 break;
506 }
