93/**
94 * Value and metadata for one property of an item
95 *
96 * Contains the value of one property of an item, as well as various metadata items required to
97 * output that value. This is used primarily for user output of information about an item, for
98 * example in the various inventory menus. See @ref item::info() for the main example of how a
99 * class desiring to provide user output might obtain a class of this type.
100 *
101 * As an example, if the item being queried was a piece of clothing, then several properties might
102 * be returned. All would have sType "ARMOR". There would be one for the coverage stat with
103 * sName "Coverage: ", another for the warmth stat with sName "Warmth: ", etc.
104 */
105struct iteminfo {
106 public:
107 /** Category of item that owns this iteminfo. See @ref item_category. */
108 std::string sType;
109
110 /** Main text of this property's name */
111 std::string sName;
112
113 /** Formatting text to be placed between the name and value of this item. */
114 std::string sFmt;
115
116 /** Numerical value of this property. Set to -999 if no compare value is present */
117 std::string sValue;
118
119 /** Internal double floating point version of value, for numerical comparisons */
120 double dValue;
121
122 /** Flag indicating type of sValue. True if integer, false if single decimal */
123 bool is_int;
124
125 /** Flag indicating whether a newline should be printed after printing this item */
126 bool bNewLine;
127
128 /** Reverses behavior of red/green text coloring; smaller values are green if true */
129 bool bLowerIsBetter;
130
131 /** Whether to print sName. If false, use for comparisons but don't print for user. */
132 bool bDrawName;
133
134 /** Whether to print a sign on positive values */
135 bool bShowPlus;
136
137 /** Flag indicating decimal with three points of precision. */
138 bool three_decimal;
139
140 enum flags {
141 no_flags = 0,
142 is_decimal = 1 << 0, ///< Print as decimal rather than integer
143 is_three_decimal = 1 << 1, ///< Print as decimal with three points of precision
144 no_newline = 1 << 2, ///< Do not follow with a newline
145 lower_is_better = 1 << 3, ///< Lower values are better for this stat
146 no_name = 1 << 4, ///< Do not print the name
147 show_plus = 1 << 5, ///< Use a + sign for positive values
148 };

150 /**
151 * @param Type The item type of the item this iteminfo belongs to.
152 * @param Name The name of the property this iteminfo describes.
153 * @param Fmt Formatting text desired between item name and value
154 * @param Flags Additional flags to customize this entry
155 * @param Value Numerical value of this property, -999 for none.
156 */
157 iteminfo( const std::string &Type, const std::string &Name, const std::string &Fmt = "",
158 flags Flags = no_flags, double Value = -999 );
159 iteminfo( const std::string &Type, const std::string &Name, double Value );
160};
