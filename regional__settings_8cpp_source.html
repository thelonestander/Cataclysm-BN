585void load_region_overlay( const JsonObject &jo )
586{

612void apply_region_overlay( const JsonObject &jo, regional_settings &region )
613{
614 jo.read( "default_oter", region.default_oter );
615 jo.read( "river_scale", region.river_scale );
616 if( jo.has_array( "default_groundcover" ) ) {
617 region.default_groundcover_str.reset( new weighted_int_list<ter_str_id> );
618 for( JsonArray inner : jo.get_array( "default_groundcover" ) ) {
619 if( region.default_groundcover_str->add( ter_str_id( inner.get_string( 0 ) ),
620 inner.get_int( 1 ) ) == nullptr ) {
621 jo.throw_error( "'default_groundcover' must be a weighted list: an array of pairs [ \"id\", weight ]" );
622 }
623 }
624 }
625
626 JsonObject fieldjo = jo.get_object( "field_coverage" );
627 double tmpval = 0.0f;
628 if( fieldjo.read( "percent_coverage", tmpval ) ) {
629 region.field_coverage.mpercent_coverage = static_cast<int>( tmpval * 10000.0 );
630 }
631
632 fieldjo.read( "default_ter", region.field_coverage.default_ter_str );
633
634 for( const JsonMember member : fieldjo.get_object( "other" ) ) {
635 if( member.is_comment() ) {
636 continue;
637 }
638 region.field_coverage.percent_str[member.name()] = member.get_float();
639 }
640
641 if( fieldjo.read( "boost_chance", tmpval ) ) {
642 region.field_coverage.boost_chance = static_cast<int>( tmpval * 10000.0 );
643 }
644 if( fieldjo.read( "boosted_percent_coverage", tmpval ) ) {
645 if( region.field_coverage.boost_chance > 0.0f && tmpval == 0.0f ) {
646 fieldjo.throw_error( "boost_chance > 0 requires boosted_percent_coverage" );
647 }
648
649 region.field_coverage.boosted_mpercent_coverage = static_cast<int>( tmpval * 10000.0 );
650 }
651
652 if( fieldjo.read( "boosted_other_percent", tmpval ) ) {
653 if( region.field_coverage.boost_chance > 0.0f && tmpval == 0.0f ) {
654 fieldjo.throw_error( "boost_chance > 0 requires boosted_other_percent" );
655 }
656
657 region.field_coverage.boosted_other_mpercent = static_cast<int>( tmpval * 10000.0 );
658 }
659
660 for( const JsonMember member : fieldjo.get_object( "boosted_other" ) ) {
661 if( member.is_comment() ) {
662 continue;
663 }
664 region.field_coverage.boosted_percent_str[member.name()] = member.get_float();
665 }
666
667 if( region.field_coverage.boost_chance > 0.0f &&
668 region.field_coverage.boosted_percent_str.empty() ) {
669 fieldjo.throw_error( "boost_chance > 0 requires boosted_other { … }" );
670 }
671
672 load_forest_mapgen_settings( jo, region.forest_composition, false, true );
673
674 load_forest_trail_settings( jo, region.forest_trail, false, true );
675
676 for( const JsonMember zone : jo.get_object( "map_extras" ) ) {
677 if( zone.is_comment() ) {
678 continue;
679 }
680 JsonObject zonejo = zone.get_object();
681
682 int tmpval = 0;

684 region.region_extras[zone.name()].chance = tmpval;
685 }

687 for( const JsonMember member : zonejo.get_object( "extras" ) ) {
688 if( member.is_comment() ) {
689 continue;
690 }
691 region.region_extras[zone.name()].values.add_or_replace( member.name(), member.get_int() );
692 }
693 }
694
695 JsonObject cityjo = jo.get_object( "city" );
696
697 cityjo.read( "shop_radius", region.city_spec.shop_radius );
698 cityjo.read( "shop_sigma", region.city_spec.shop_sigma );
699 cityjo.read( "park_radius", region.city_spec.park_radius );
700 cityjo.read( "park_sigma", region.city_spec.park_sigma );
701
702 const auto load_building_types = [&cityjo]( const std::string & type, building_bin & dest ) {
703 for( const JsonMember member : cityjo.get_object( type ) ) {
704 if( member.is_comment() ) {
705 continue;
706 }
707 dest.add( overmap_special_id( member.name() ), member.get_int() );
708 }
709 };
710 load_building_types( "houses", region.city_spec.houses );
711 load_building_types( "shops", region.city_spec.shops );
712 load_building_types( "parks", region.city_spec.parks );

714 load_overmap_feature_flag_settings( jo, region.overmap_feature_flag, false, true );
715
716 load_overmap_forest_settings( jo, region.overmap_forest, false, true );

718 load_overmap_lake_settings( jo, region.overmap_lake, false, true );

720 load_region_terrain_and_furniture_settings( jo, region.region_terrain_and_furniture, false, true );

842ter_furn_id forest_biome::pick() const
843{
844 // Iterate through the biome components (which have already been put into sequence), roll for the
845 // one_in chance that component contributes a feature, and if so pick that feature and return it.
846 // If a given component does not roll as success, proceed to the next feature in sequence until
847 // a feature is picked or none are picked, in which case an empty feature is returned.
848 const ter_furn_id *result = nullptr;
849 for( auto &pr : biome_components ) {
850 if( one_in( pr.chance ) ) {
851 result = pr.types.pick();
852 break;
853 }
854 }
855
856 if( result == nullptr ) {
857 return ter_furn_id();
858 }
859
860 return *result;
861}
