29T *furn_at( const tripoint_abs_ms &p )
30{
31 tripoint_abs_sm p_abs_sm;
32 point_sm_ms p_within_sm;
33 std::tie( p_abs_sm, p_within_sm ) = project_remain<coords::sm>( p );
34

36 if( sm == nullptr ) {
37 return nullptr;

113void solar_tile::update_internal( time_point to, const tripoint_abs_ms &p, distribution_grid &grid )
114{
115 constexpr time_point zero = time_point::from_turn( 0 );
116 constexpr time_duration tick_length = 10_minutes;
117 constexpr int tick_turns = to_turns<int>( tick_length );
118 time_duration till_then = get_last_updated() - zero;
119 time_duration till_now = to - zero;
120 // This is just for rounding to nearest tick
121 time_duration ticks_then = till_then / tick_turns;
122 time_duration ticks_now = till_now / tick_turns;
123 // This is to cut down on sum_conditions
124 if( ticks_then == ticks_now ) {
125 return;
126 }
127 time_duration rounded_then = ticks_then * tick_turns;
128 time_duration rounded_now = ticks_now * tick_turns;
129
130 // TODO: Use something that doesn't calc a ton of worthless crap
131 float sunlight = sum_conditions( zero + rounded_then, zero + rounded_now,
132 p.raw() ).sunlight / default_daylight_level();
133 // int64 because we can have years in here
134 std::int64_t produced = power * static_cast<std::int64_t>( sunlight ) / 1000;
135 grid.mod_resource( static_cast<int>( std::min( static_cast<std::int64_t>( INT_MAX ), produced ) ) );
136}

246void charger_tile::update_internal( time_point to, const tripoint_abs_ms &p,
247 distribution_grid &grid )
248{
249 tripoint_abs_sm p_abs_sm;
250 point_sm_ms p_within_sm;
251 std::tie( p_abs_sm, p_within_sm ) = project_remain<coords::sm>( p );
252
253 submap *sm = MAPBUFFER.lookup_submap( p_abs_sm );
254 if( sm == nullptr ) {
255 return;
256 }
257 std::int64_t power = this->power * to_seconds<std::int64_t>( to - get_last_updated() );
258 // TODO: Make not a copy from map.cpp
259 for( item &outer : sm->get_items( p_within_sm.raw() ) ) {
260 outer.visit_items( [&power, &grid]( item * it ) {
261 item &n = *it;
262 if( !n.has_flag( flag_RECHARGE ) && !n.has_flag( flag_USE_UPS ) ) {
263 return VisitResponse::NEXT;
264 }
265 if( n.ammo_capacity() > n.ammo_remaining() ||
266 ( n.type->battery && n.type->battery->max_capacity > n.energy_remaining() ) ) {
267 while( power >= 1000 || x_in_y( power, 1000 ) ) {
268 const int missing = grid.mod_resource( -1 );
269 if( missing == 0 ) {
270 if( n.is_battery() ) {
271 n.mod_energy( 1_kJ );
272 } else {
273 n.ammo_set( itype_battery, n.ammo_remaining() + 1 );
274 }
275 }
276 power -= 1000;
277 }
278 return VisitResponse::ABORT;
279 }
280
281 return VisitResponse::SKIP;
282 } );
283 }
