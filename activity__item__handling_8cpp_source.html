405static std::list<pickup::act_item> convert_to_items( Character &p,
406 const drop_locations &drop,
407 std::function<bool( item_location loc )> filter )
408{
409 std::list<pickup::act_item> res;
410
411 for( const drop_location &rec : drop ) {
412 const item_location loc = rec.loc;
413 const int count = rec.count;
414
415 if( !filter( loc ) ) {
416 continue;
417 } else if( !p.is_worn( *loc ) && !p.is_wielding( *loc ) ) {
418 // Special case. After dropping the first few items, the remaining items are already separated.
419 // That means: `drop` already contains references to each of the items in
420 // `p.inv.const_stack`, and `count` will be 1 for each of them.
421 // If we continued without this check, we iterate over `p.inv.const_stack` multiple times,
422 // but each time stopping after visiting the first item.
423 // In the end, we would add references to the same item (the first one in the stack) multiple times.
424 if( count == 1 ) {
425 res.emplace_back( loc, 1, loc.obtain_cost( p, 1 ) );
426 continue;
427 }
428 int obtained = 0;
429 for( const item &it : p.inv.const_stack( p.get_item_position( &*loc ) ) ) {
430 if( obtained >= count ) {
431 break;
432 }
433 const int qty = it.count_by_charges() ? std::min<int>( it.charges, count - obtained ) : 1;
434 obtained += qty;
435 item_location loc( p, const_cast<item *>( &it ) );
436 res.emplace_back( loc, qty, loc.obtain_cost( p, qty ) );
437 }
438 } else {
439 res.emplace_back( loc, count, p.is_wielding( *loc ) ? 0 : loc.obtain_cost( p ) );
440 }
441 }
442
443 return res;
444}

449// Prepares items for dropping by reordering them so that the drop
450// cost is minimal and "dependent" items get taken off first.
451// Implements the "backpack" logic.
452std::list<act_item> reorder_for_dropping( Character &p, const drop_locations &drop )
453{
454 std::list<act_item> res = convert_to_items( p, drop,
455 [&p]( item_location loc ) {
456 return p.is_wielding( *loc );
457 } );
458 std::list<act_item> inv = convert_to_items( p, drop,
459 [&p]( item_location loc ) {
460 return !p.is_wielding( *loc ) && !p.is_worn( *loc );
461 } );
462 std::list<act_item> worn = convert_to_items( p, drop,
463 [&p]( item_location loc ) {
464 return p.is_worn( *loc );
465 } );
466
467 // Sort inventory items by volume in ascending order
468 inv.sort( []( const act_item & first, const act_item & second ) {
469 return first.loc->volume() < second.loc->volume();
470 } );
471 // Add missing dependent worn items (if any).
472 for( const auto &wait : worn ) {
473 for( item *dit : p.get_dependent_worn_items( *wait.loc ) ) {
474 const auto iter = std::find_if( worn.begin(), worn.end(),
475 [dit]( const act_item & ait ) {
476 return &*ait.loc == dit;
477 } );
478
479 if( iter == worn.end() ) {
480 // TODO: Use a calculated cost
481 const item_location loc( p, dit );
482 act_item act( loc, loc->count(), loc.obtain_cost( p, loc->count() ) );
483 worn.emplace_front( loc, loc->count(), loc.obtain_cost( p ) );
484 }
485 }
486 }
487 // Sort worn items by storage in descending order, but dependent items always go first.
488 worn.sort( []( const act_item & first, const act_item & second ) {
489 return first.loc->is_worn_only_with( *second.loc )
490 || ( first.loc->get_storage() > second.loc->get_storage()
491 && !second.loc->is_worn_only_with( *first.loc ) );
492 } );
493
494 // Avoid tumbling to the ground. Unload cleanly.
495 units::volume dropped_inv_contents = std::accumulate( inv.begin(), inv.end(), 0_ml,
496 []( units::volume acc, const act_item & ait ) {
497 return acc + ait.loc->volume();
498 } );
499 const units::volume dropped_worn_storage = std::accumulate( worn.begin(), worn.end(), 0_ml,
500 []( units::volume acc, const act_item & ait ) {
501 return acc + ait.loc->get_storage();
502 } );
503 std::set<int> inv_indices;
504 std::transform( inv.begin(), inv.end(), std::inserter( inv_indices, inv_indices.begin() ),
505 [&p]( const act_item & ait ) {
506 return p.get_item_position( &*ait.loc );
507 } );
508
509 units::volume excessive_volume = p.volume_carried() - dropped_inv_contents
510 - p.volume_capacity_reduced_by( dropped_worn_storage );
511 if( excessive_volume > 0_ml ) {
512 invslice old_inv = p.inv.slice();
513 for( size_t i = 0; i < old_inv.size() && excessive_volume > 0_ml; i++ ) {
514 // TODO: Reimplement random dropping?
515 if( inv_indices.count( i ) != 0 ) {
516 continue;
517 }
518 std::list<item> &inv_stack = *old_inv[i];
519 for( item &item : inv_stack ) {
520 // Note: zero cost, but won't be contained on drop
521 act_item to_drop = act_item( item_location( p, &item ), item.count(), 0 );
522 inv.push_back( to_drop );
523 excessive_volume -= to_drop.loc->volume();
524 if( excessive_volume <= 0_ml ) {
525 break;
526 }
527 }
528 }
529 // Need to re-sort
530 inv.sort( []( const act_item & first, const act_item & second ) {
531 return first.loc->volume() < second.loc->volume();
532 } );
533 }
534
535 // Cumulatively decreases
536 units::volume remaining_dropped_storage = dropped_worn_storage;
537
538 while( !worn.empty() && !inv.empty() ) {
539 units::volume front_storage = worn.front().loc->get_storage();
540 // Does not fit
541 // TODO: but maybe an item further down the line does
542 if( remaining_dropped_storage < inv.front().loc->volume() ) {
543 break;
544 }
545
546 res.push_back( worn.front() );
547 worn.pop_front();
548 remaining_dropped_storage -= front_storage;
549 while( !inv.empty() ) {
550 units::volume inventory_item_volume = inv.front().loc->volume();
551 if( front_storage < inventory_item_volume ) {
552 break;
553 }
554 front_storage -= inventory_item_volume;
555
556 res.push_back( inv.front() );
557 // Free of charge
558 res.back().consumed_moves = 0;
559
560 inv.pop_front();
561 }
562 }
563
564 // Now insert everything that remains
565 std::copy( inv.begin(), inv.end(), std::back_inserter( res ) );
566 std::copy( worn.begin(), worn.end(), std::back_inserter( res ) );
567
568 return res;
569}

605 p.takeoff( *ait.loc, &res );
606 } else if( ait.loc->count_by_charges() ) {
607 res.push_back( p.reduce_charges( const_cast<item *>( &*ait.loc ), ait.count ) );

610 }

612 // TODO: Get the item consistently instead of using back()
613 item &current_drop = res.back();
614
615 // Hack: if it consumes zero moves, it must have been contained
616 // TODO: Properly mark containment somehow
617 *current_drop.drop_token = token_provider.make_next( calendar::turn );;
618 if( ait.consumed_moves == 0 && last_storage_volume >= current_drop.volume() ) {

620 current_drop.drop_token->parent_number = last_token.parent_number;

622 last_token = *current_drop.drop_token;

814static double get_capacity_fraction( int capacity, int volume )
815{
816 // fraction of capacity the item would occupy
817 // fr = 1 is for capacity smaller than is size of item
818 // in such case, let's assume player does the trip for full cost with item in hands
819 double fr = 1;
820
821 if( capacity > volume ) {
822 fr = static_cast<double>( volume ) / capacity;
823 }
824
825 return fr;

828static int move_cost_inv( const item &it, const tripoint &src, const tripoint &dest )
829{
830 // to prevent potentially ridiculous number
831 const int MAX_COST = 500;
832
833 // it seems that pickup cost is flat 100
834 // in function pick_one_up, variable moves_taken has initial value of 100
835 // and never changes until it is finally used in function
836 // remove_from_map_or_vehicle
837 const int pickup_cost = 100;
838
839 // drop cost for non-tumbling items (from inventory overload) is also flat 100
840 // according to convert_to_items (it does contain todo to use calculated costs)
841 const int drop_cost = 100;
842
843 // typical flat ground move cost
844 const int mc_per_tile = 100;
845
846 // only free inventory capacity
847 const int inventory_capacity = units::to_milliliter( g->u.volume_capacity() -
848 g->u.volume_carried() );
849
850 const int item_volume = units::to_milliliter( it.volume() );
851
852 const double fr = get_capacity_fraction( inventory_capacity, item_volume );
853
854 // approximation of movement cost between source and destination
855 const int move_cost = mc_per_tile * rl_dist( src, dest ) * fr;
856
857 return std::min( pickup_cost + drop_cost + move_cost, MAX_COST );
858}

907// return false if it was not possible.
908static bool vehicle_activity( player &p, const tripoint &src_loc, int vpindex, char type )
909{
910 vehicle *veh = veh_pointer_or_null( g->m.veh_at( src_loc ) );
911 if( !veh ) {
912 return false;
913 }
914 int time_to_take = 0;
915 if( vpindex >= veh->part_count() ) {
916 // if parts got removed during our work, we can't just carry on removing, we want to repair parts!
917 // so just bail out, as we don't know if the next shifted part is suitable for repair.
918 if( type == 'r' ) {
919 return false;
920 } else if( type == 'o' ) {
921 vpindex = veh->get_next_shifted_index( vpindex, p );
922 if( vpindex == -1 ) {
923 return false;
924 }
925 }
926 }
927 const vpart_info &vp = veh->part_info( vpindex );
928 if( type == 'r' ) {
929 const vehicle_part part = veh->part( vpindex );
930 time_to_take = vp.repair_time( p ) * part.damage() / part.max_damage();
931 } else if( type == 'o' ) {
932 time_to_take = vp.removal_time( p );
933 }
934 p.assign_activity( ACT_VEHICLE, time_to_take, static_cast<int>( type ) );
935 // so , NPCs can remove the last part on a position, then there is no vehicle there anymore,
936 // for someone else who stored that position at the start of their activity.
937 // so we may need to go looking a bit further afield to find it , at activities end.
938 for( const auto pt : veh->get_points( true ) ) {
939 p.activity.coord_set.insert( g->m.getabs( pt ) );
940 }
941 // values[0]
942 p.activity.values.push_back( g->m.getabs( src_loc ).x );
943 // values[1]
944 p.activity.values.push_back( g->m.getabs( src_loc ).y );
945 // values[2]
946 p.activity.values.push_back( point_zero.x );
947 // values[3]
948 p.activity.values.push_back( point_zero.y );
949 // values[4]
950 p.activity.values.push_back( -point_zero.x );
951 // values[5]
952 p.activity.values.push_back( -point_zero.y );
953 // values[6]
954 p.activity.values.push_back( veh->index_of_part( &veh->part( vpindex ) ) );
955 p.activity.str_values.push_back( vp.get_id().str() );
956 // this would only be used for refilling tasks
957 item_location target;
958 p.activity.targets.emplace_back( std::move( target ) );
959 p.activity.placement = g->m.getabs( src_loc );
960 p.activity_vehicle_part_index = -1;
961 return true;

976 } else {
977 leftovers.charges = 0;

1033static activity_reason_info find_base_construction(
1034 const std::vector<construction> &list_constructions,
1035 player &p,
1036 const inventory &inv,
1037 const tripoint &loc,
1038 const cata::optional<construction_id> &part_con_idx,
1039 const construction_id idx,
1040 std::set<construction_id> &used )
1041{
1042 const construction &build = idx.obj();
1043 //already done?
1044 const furn_id furn = g->m.furn( loc );
1045 const ter_id ter = g->m.ter( loc );
1046 if( !build.post_terrain.empty() ) {
1047 if( build.post_is_furniture ) {
1048 if( furn_id( build.post_terrain ) == furn ) {
1049 return activity_reason_info::build( do_activity_reason::ALREADY_DONE, false, idx );
1050 }
1051 } else {
1052 if( ter_id( build.post_terrain ) == ter ) {
1053 return activity_reason_info::build( do_activity_reason::ALREADY_DONE, false, idx );
1054 }
1055 }
1056 }
1057 //if there's an appropriate partial construction on the tile, then we can work on it, no need to check inventories.
1058 const bool has_skill = p.meets_skill_requirements( build );
1059 if( part_con_idx && *part_con_idx == idx ) {
1060 if( !has_skill ) {
1061 return activity_reason_info::build( do_activity_reason::DONT_HAVE_SKILL, false, idx );
1062 }
1063 return activity_reason_info::build( do_activity_reason::CAN_DO_CONSTRUCTION, true, idx );
1064 }
1065 //can build?
1066 const bool cc = can_construct( build, loc );
1067 const bool pcb = player_can_build( p, inv, build );
1068 if( !has_skill ) {
1069 return activity_reason_info::build( do_activity_reason::DONT_HAVE_SKILL, false, idx );
1070 }
1071 if( cc ) {
1072 if( pcb ) {
1073 return activity_reason_info::build( do_activity_reason::CAN_DO_CONSTRUCTION, true, idx );
1074 }
1075 //can't build with current inventory, do not look for pre-req
1076 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, idx );
1077 }
1078
1079 // there are no pre-requisites.
1080 // so we need to potentially fetch components
1081 if( build.pre_terrain.empty() && build.pre_special( loc ) ) {
1082 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, idx );
1083 } else if( !build.pre_special( loc ) ) {
1084 return activity_reason_info::build( do_activity_reason::BLOCKING_TILE, false, idx );
1085 }
1086
1087 // can't build it
1088 // maybe we can build the pre-requisite instead
1089 // see if the reason is because of pre-terrain requirement
1090 if( !build.pre_terrain.empty() &&
1091 ( ( build.pre_is_furniture &&
1092 furn_id( build.pre_terrain ) == furn ) ||
1093 ( !build.pre_is_furniture &&
1094 ter_id( build.pre_terrain ) == ter ) ) ) {
1095 // the pre-req is already built, so the reason is due to lack of tools/components
1096 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, idx );
1097 }
1098
1099 //we can't immediately build it, looking for pre-req
1100 used.insert( idx );
1101 cata::optional<do_activity_reason> reason;
1102 construction_id pre_req_idx( -1 );
1103 //first step: try only constructions with the same description
1104 //second step: try all constructions
1105 for( int try_num = 0; try_num < 2; ++try_num ) {
1106 for( const construction &pre_build : list_constructions ) {
1107 //skip if already checked this one
1108 if( pre_build.id == idx || used.find( pre_build.id ) != used.end() ) {
1109 continue;
1110 }
1111 //skip unknown
1112 if( pre_build.post_terrain.empty() ) {
1113 continue;
1114 }
1115 //skip if it is not a pre-build required or gives the same result
1116 if( pre_build.post_terrain != build.pre_terrain &&
1117 pre_build.post_terrain != build.post_terrain ) {
1118 continue;
1119 }
1120 //at first step, try to get building with the same description
1121 if( try_num == 0 &&
1122 pre_build.description != build.description ) {
1123 continue;
1124 }
1125 activity_reason_info act_info_pre = find_base_construction( list_constructions,
1126 p, inv, loc, part_con_idx, pre_build.id, used );
1127 if( act_info_pre.can_do ) {
1128 return activity_reason_info::build( do_activity_reason::CAN_DO_PREREQ, true,
1129 *act_info_pre.con_idx );
1130 }
1131 //find first pre-req failed reason
1132 if( !reason ) {
1133 reason = act_info_pre.reason;
1134 pre_req_idx = *act_info_pre.con_idx;
1135 }
1136 if( act_info_pre.reason == do_activity_reason::ALREADY_DONE ) {
1137 //pre-req is already here, but we still can't build over it
1138 reason.reset();
1139 break;
1140 }
1141 }
1142 }
1143 //have a partial construction which is not leading to the required construction
1144 if( part_con_idx ) {
1145 return activity_reason_info::build( do_activity_reason::BLOCKING_TILE, false, idx );
1146 }
1147 //pre-req failed?
1148 if( reason ) {
1149 if( *reason == do_activity_reason::NO_COMPONENTS ) {
1150 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS_PREREQ, false, pre_req_idx );
1151 }
1152 return activity_reason_info::build( *reason, false, pre_req_idx );
1153 }
1154 if( !pcb ) {
1155 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, idx );
1156 }
1157 //only cc failed, no pre-req
1158 return activity_reason_info::build( do_activity_reason::BLOCKING_TILE, false, idx );
1159}

1204 // if, however they are already near the work spot, then the crafting / inventory functions will have their own method to use or discount them.
1205 if( in_loot_zones ) {
1206 // skip tiles in IGNORE zone and tiles on fire
1207 // (to prevent taking out wood off the lit brazier)
1208 // and inaccessible furniture, like filled charcoal kiln
1209 if( mgr.has( zone_type_LOOT_IGNORE, g->m.getabs( elem ) ) ||
1210 g->m.dangerous_field_at( elem ) ||
1211 !g->m.can_put_items_ter_furn( elem ) ) {
1212 continue;
1213 }
1214 }
1215 for( const auto &elem2 : g->m.i_at( elem ) ) {
1216 if( in_loot_zones && elem2.made_of( LIQUID ) ) {
1217 continue;
1218 }
1219 if( check_weight ) {
1220 // this fetch task will need to pick up an item. so check for its weight/volume before setting off.
1221 if( in_loot_zones && ( elem2.volume() > volume_allowed ||
1222 elem2.weight() > weight_allowed ) ) {
1223 continue;
1224 }
1225 }
1226 temp_inv += elem2;
1227 }
1228 if( !in_loot_zones ) {
1229 if( const cata::optional<vpart_reference> vp = g->m.veh_at( elem ).part_with_feature( "CARGO",
1230 false ) ) {
1231 vehicle &src_veh = vp->vehicle();
1232 int src_part = vp->part_index();
1233 for( auto &it : src_veh.get_items( src_part ) ) {
1234 temp_inv += it;
1235 }
1236 }
1237 }
1238 }
1239 // use nearby welding rig without needing to drag it or position yourself on the right side of the vehicle.
1240 if( !found_welder ) {
1241 for( const tripoint &elem : g->m.points_in_radius( src_loc, PICKUP_RANGE - 1 ) ) {
1242 const optional_vpart_position vp = g->m.veh_at( elem );
1243 if( vp ) {
1244 vehicle &veh = vp->vehicle();
1245 const cata::optional<vpart_reference> weldpart = vp.part_with_feature( "WELDRIG", true );
1246 if( weldpart ) {
1247 item welder( itype_welder, calendar::start_of_cataclysm );
1248 welder.charges = veh.fuel_left( itype_battery, true );
1249 welder.set_flag( "PSEUDO" );
1250 temp_inv.add_item( welder );
1251 item soldering_iron( itype_soldering_iron, calendar::start_of_cataclysm );
1252 soldering_iron.charges = veh.fuel_left( itype_battery, true );
1253 soldering_iron.set_flag( "PSEUDO" );
1254 temp_inv.add_item( soldering_iron );
1255 }
1256 }
1257 }
1258 }
1259 return needed_things.obj().can_make_with_inventory( temp_inv, is_crafting_component );
1260}

1281 act == ACT_VEHICLE_REPAIR ) {

1285 return activity_reason_info::fail( do_activity_reason::NO_ZONE );

1298 guy_work_spot = g->m.getlocal( guy.activity.placement );
1299 }
1300 // If their position or intended position or player position/intended position
1301 // then discount, don't need to move each other out of the way.
1302 if( g->m.getlocal( g->u.activity.placement ) == src_loc ||
1303 guy_work_spot == src_loc || guy.pos() == src_loc || ( p.is_npc() && g->u.pos() == src_loc ) ) {

1308 // working on same vehicle - store the index to check later.
1309 if( other_veh && other_veh == veh && guy.activity_vehicle_part_index != -1 ) {
1310 already_working_indexes.push_back( guy.activity_vehicle_part_index );
1311 }
1312 }
1313 if( g->u.activity_vehicle_part_index != -1 ) {
1314 already_working_indexes.push_back( g->u.activity_vehicle_part_index );
1315 }
1316 }
1317 if( act == ACT_VEHICLE_DECONSTRUCTION ) {
1318 // find out if there is a vehicle part here we can remove.

1320 for( vehicle_part *part_elem : parts ) {
1321 const vpart_info &vpinfo = part_elem->info();

1323 // if part is not on this vehicle, or if its attached to another part that needs to be removed first.
1324 if( vpindex == -1 || !veh->can_unmount( vpindex ) ) {
1325 continue;
1326 }
1327 // this is the same part that somebody else wants to work on, or already is.
1328 if( std::find( already_working_indexes.begin(), already_working_indexes.end(),
1329 vpindex ) != already_working_indexes.end() ) {
1330 continue;
1331 }

1334 continue;
1335 }

1338 // no wheel removal yet
1339 continue;

1343 TOOL_LIFT_FACTOR );

1345 const bool use_str = p.can_lift( base.lift_strength() );
1346 if( !( use_aid || use_str ) ) {
1347 continue;
1348 }
1349 const auto &reqs = vpinfo.removal_requirements();
1350 const inventory &inv = p.crafting_inventory( false );
1351
1352 const bool can_make = reqs.can_make_with_inventory( inv, is_crafting_component );
1353 p.set_value( "veh_index_type", vpinfo.name() );
1354 // temporarily store the intended index, we do this so two NPCs don't try and work on the same part at same time.
1355 p.activity_vehicle_part_index = vpindex;
1356 if( !can_make ) {
1357 return activity_reason_info::fail( do_activity_reason::NEEDS_VEH_DECONST );
1358 } else {
1359 return activity_reason_info::ok( do_activity_reason::NEEDS_VEH_DECONST );
1360 }
1361 }
1362 } else if( act == ACT_VEHICLE_REPAIR ) {
1363 // find out if there is a vehicle part here we can repair.
1364 std::vector<vehicle_part *> parts = veh->get_parts_at( src_loc, "", part_status_flag::any );
1365 for( vehicle_part *part_elem : parts ) {
1366 const vpart_info &vpinfo = part_elem->info();
1367 int vpindex = veh->index_of_part( part_elem, true );
1368 // if part is undamaged or beyond repair - can skip it.
1369 if( part_elem->is_broken() || part_elem->damage() == 0 ||
1370 part_elem->info().repair_requirements().is_empty() ) {
1371 continue;
1372 }
1373 if( std::find( already_working_indexes.begin(), already_working_indexes.end(),
1374 vpindex ) != already_working_indexes.end() ) {
1375 continue;
1376 }
1377 // don't have skill to repair it
1378 if( !has_skill_for_vehicle_work( vpinfo.repair_skills, p ) ) {
1379 continue;
1380 }
1381 const auto &reqs = vpinfo.repair_requirements();
1382 const inventory &inv = p.crafting_inventory( src_loc, PICKUP_RANGE - 1, false );
1383 const bool can_make = reqs.can_make_with_inventory( inv, is_crafting_component );
1384 p.set_value( "veh_index_type", vpinfo.name() );
1385 // temporarily store the intended index, we do this so two NPCs don't try and work on the same part at same time.
1386 p.activity_vehicle_part_index = vpindex;
1387 if( !can_make ) {
1388 return activity_reason_info::fail( do_activity_reason::NEEDS_VEH_REPAIR );
1389 } else {
1390 return activity_reason_info::ok( do_activity_reason::NEEDS_VEH_REPAIR );
1391 }
1392 }
1393 }
1394 p.activity_vehicle_part_index = -1;
1395 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1396 }
1397 if( act == ACT_MULTIPLE_MINE ) {
1398 if( !g->m.has_flag( "MINEABLE", src_loc ) ) {
1399 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1400 }
1401 std::vector<item *> mining_inv = p.items_with( []( const item & itm ) {
1402 return ( itm.has_flag( flag_DIG_TOOL ) && !itm.type->can_use( "JACKHAMMER" ) ) ||
1403 ( itm.type->can_use( "JACKHAMMER" ) && itm.ammo_sufficient() );
1404 } );
1405 if( mining_inv.empty() ) {
1406 return activity_reason_info::fail( do_activity_reason::NEEDS_MINING );
1407 } else {
1408 return activity_reason_info::ok( do_activity_reason::NEEDS_MINING );
1409 }
1410 }
1411 if( act == ACT_MULTIPLE_FISH ) {
1412 if( !g->m.has_flag( flag_FISHABLE, src_loc ) ) {
1413 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1414 }
1415 std::vector<item *> rod_inv = p.items_with( []( const item & itm ) {
1416 return itm.has_flag( flag_FISH_POOR ) || itm.has_flag( flag_FISH_GOOD );
1417 } );
1418 if( rod_inv.empty() ) {
1419 return activity_reason_info::fail( do_activity_reason::NEEDS_FISHING );
1420 } else {
1421 return activity_reason_info::ok( do_activity_reason::NEEDS_FISHING );
1422 }
1423 }
1424 if( act == ACT_MULTIPLE_CHOP_TREES ) {
1425 if( g->m.has_flag( flag_TREE, src_loc ) || g->m.ter( src_loc ) == t_trunk ||
1426 g->m.ter( src_loc ) == t_stump ) {
1427 if( p.has_quality( qual_AXE ) ) {
1428 return activity_reason_info::ok( do_activity_reason::NEEDS_TREE_CHOPPING );
1429 } else {
1430 return activity_reason_info::fail( do_activity_reason::NEEDS_TREE_CHOPPING );
1431 }
1432 } else {
1433 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1434 }
1435 }
1436 if( act == ACT_MULTIPLE_BUTCHER ) {
1437 std::vector<item> corpses;
1438 int big_count = 0;
1439 int small_count = 0;
1440 for( const auto &i : g->m.i_at( src_loc ) ) {
1441 // make sure nobody else is working on that corpse right now
1442 if( i.is_corpse() && !i.has_var( "activity_var" ) ) {
1443 const mtype corpse = *i.get_mtype();
1444 if( corpse.size >= MS_MEDIUM ) {
1445 big_count += 1;
1446 } else {
1447 small_count += 1;
1448 }
1449 corpses.push_back( i );
1450 }
1451 }
1452 bool b_rack_present = false;
1453 for( const tripoint &pt : g->m.points_in_radius( src_loc, 2 ) ) {
1454 const inventory &inv = p.crafting_inventory();
1455 if( g->m.has_flag_furn( flag_BUTCHER_EQ, pt ) || inv.has_item_with( []( const item & it ) {
1456 return it.has_flag( "BUTCHER_RACK" );
1457 } ) ) {
1458 b_rack_present = true;
1459 }
1460 }
1461 if( !corpses.empty() ) {
1462 if( big_count > 0 && small_count == 0 ) {
1463 const inventory &inv = p.crafting_inventory();
1464 if( !b_rack_present || !( g->m.has_nearby_table( src_loc, PICKUP_RANGE ) ||
1465 inv.has_item_with( []( const item & it ) {
1466 return it.has_flag( "FLAT_SURFACE" );
1467 } ) ) ) {
1468 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1469 }
1470 if( p.has_quality( quality_id( qual_BUTCHER ), 1 ) ) {
1471 return activity_reason_info::ok( do_activity_reason::NEEDS_BIG_BUTCHERING );
1472 } else {
1473 return activity_reason_info::fail( do_activity_reason::NEEDS_BIG_BUTCHERING );
1474 }
1475 }
1476 if( ( big_count > 0 && small_count > 0 ) || ( big_count == 0 ) ) {
1477 // there are small corpses here, so we can ignore any big corpses here for the moment.
1478 if( p.has_quality( qual_BUTCHER, 1 ) ) {
1479 return activity_reason_info::ok( do_activity_reason::NEEDS_BUTCHERING );
1480 } else {
1481 return activity_reason_info::fail( do_activity_reason::NEEDS_BUTCHERING );
1482 }
1483 }
1484 }
1485 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1486 }
1487 if( act == ACT_MULTIPLE_CHOP_PLANKS ) {
1488 //are there even any logs there?
1489 for( auto &i : g->m.i_at( src_loc ) ) {
1490 if( i.typeId() == itype_log ) {
1491 // do we have an axe?
1492 if( p.has_quality( qual_AXE, 1 ) ) {
1493 return activity_reason_info::ok( do_activity_reason::NEEDS_CHOPPING );

1500 }

1503 mgr.has_near( z_camp_storage, g->m.getabs( src_loc ), distance ) ) {

1511 g->m.getabs( src_loc ) );

1513 cata::optional<construction_id> part_con_idx;
1514 if( part_con ) {
1515 part_con_idx = part_con->id;
1516 }
1517 const map_stack stuff_there = g->m.i_at( src_loc );
1518
1519 // PICKUP_RANGE -1 because we will be adjacent to the spot when arriving.
1520 const inventory pre_inv = p.crafting_inventory( src_loc, PICKUP_RANGE - 1 );
1521 for( const zone_data &zone : zones ) {
1522 const blueprint_options &options = dynamic_cast<const blueprint_options &>( zone.get_options() );
1523 const construction_id index = options.get_index();
1524 if( !stuff_there.empty() ) {
1525 return activity_reason_info::build( do_activity_reason::BLOCKING_TILE, false, index );
1526 }
1527 std::set<construction_id> used_idx;
1528 const activity_reason_info act_info = find_base_construction( list_constructions, p, pre_inv,
1529 src_loc, part_con_idx, index, used_idx );
1530 return act_info;
1531 }
1532 } else if( act == ACT_MULTIPLE_FARM ) {
1533 zones = mgr.get_zones( zone_type_FARM_PLOT,
1534 g->m.getabs( src_loc ) );
1535 for( const zone_data &zone : zones ) {
1536 if( g->m.has_flag_furn( flag_GROWTH_HARVEST, src_loc ) ) {
1537 // simple work, pulling up plants, nothing else required.
1538 return activity_reason_info::ok( do_activity_reason::NEEDS_HARVESTING );
1539 } else if( g->m.has_flag( flag_PLOWABLE, src_loc ) && !g->m.has_furn( src_loc ) ) {
1540 if( p.has_quality( qual_DIG, 1 ) ) {
1541 // we have a shovel/hoe already, great
1542 return activity_reason_info::ok( do_activity_reason::NEEDS_TILLING );
1543 } else {
1544 // we need a shovel/hoe
1545 return activity_reason_info::fail( do_activity_reason::NEEDS_TILLING );
1546 }
1547 } else if( g->m.has_flag_ter_or_furn( flag_PLANTABLE, src_loc ) &&
1548 warm_enough_to_plant( src_loc ) ) {
1549 if( g->m.has_items( src_loc ) ) {
1550 return activity_reason_info::fail( do_activity_reason::BLOCKING_TILE );
1551 } else {
1552 // do we have the required seed on our person?
1553 const plot_options &options = dynamic_cast<const plot_options &>( zone.get_options() );
1554 const itype_id seed = options.get_seed();
1555 // If its a farm zone with no specified seed, and we've checked for tilling and harvesting.
1556 // then it means no further work can be done here
1557 if( seed.is_empty() ) {
1558 return activity_reason_info::fail( do_activity_reason::ALREADY_DONE );
1559 }
1560 std::vector<item *> seed_inv = p.items_with( []( const item & itm ) {
1561 return itm.is_seed();
1562 } );
1563 for( const auto elem : seed_inv ) {
1564 if( elem->typeId() == itype_id( seed ) ) {
1565 return activity_reason_info::ok( do_activity_reason::NEEDS_PLANTING );
1566 }
1567 }
1568 // didn't find the seed, but maybe there are overlapping farm zones
1569 // and another of the zones is for a seed that we have
1570 // so loop again, and return false once all zones done.
1571 }
1572
1573 } else {
1574 // can't plant, till or harvest
1575 return activity_reason_info::fail( do_activity_reason::ALREADY_DONE );
1576 }
1577
1578 }
1579 // looped through all zones, and only got here if its plantable, but have no seeds.
1580 return activity_reason_info::fail( do_activity_reason::NEEDS_PLANTING );
1581 } else if( act == ACT_FETCH_REQUIRED ) {
1582 // we check if its possible to get all the requirements for fetching at two other places.
1583 // 1. before we even assign the fetch activity and;
1584 // 2. when we form the src_set to loop through at the beginning of the fetch activity.
1585 return activity_reason_info::ok( do_activity_reason::CAN_DO_FETCH );
1586 }
1587 // Shouldn't get here because the zones were checked previously. if it does, set enum reason as "no zone"
1588 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1589}

1607static std::vector<std::tuple<tripoint, itype_id, int>> requirements_map( player &p,
1608 const int distance = ACTIVITY_SEARCH_DISTANCE )
1609{
1610 std::vector<std::tuple<tripoint, itype_id, int>> requirement_map;
1611 if( p.backlog.empty() || p.backlog.front().str_values.empty() ) {
1612 return requirement_map;
1613 }
1614 const requirement_data things_to_fetch = requirement_id( p.backlog.front().str_values[0] ).obj();
1615 const activity_id activity_to_restore = p.backlog.front().id();
1616 // NOLINTNEXTLINE(performance-unnecessary-copy-initialization)
1617 requirement_id things_to_fetch_id = things_to_fetch.id();
1618 std::vector<std::vector<item_comp>> req_comps = things_to_fetch.get_components();
1619 std::vector<std::vector<tool_comp>> tool_comps = things_to_fetch.get_tools();
1620 std::vector<std::vector<quality_requirement>> quality_comps = things_to_fetch.get_qualities();
1621 zone_manager &mgr = zone_manager::get_manager();
1622 const bool pickup_task = p.backlog.front().id() == ACT_MULTIPLE_FARM ||
1623 p.backlog.front().id() == ACT_MULTIPLE_CHOP_PLANKS ||
1624 p.backlog.front().id() == ACT_MULTIPLE_BUTCHER ||
1625 p.backlog.front().id() == ACT_MULTIPLE_CHOP_TREES ||
1626 p.backlog.front().id() == ACT_VEHICLE_DECONSTRUCTION ||
1627 p.backlog.front().id() == ACT_VEHICLE_REPAIR ||
1628 p.backlog.front().id() == ACT_MULTIPLE_FISH ||
1629 p.backlog.front().id() == ACT_MULTIPLE_MINE;
1630 // where it is, what it is, how much of it, and how much in total is required of that item.
1631 std::vector<std::tuple<tripoint, itype_id, int>> final_map;
1632 std::vector<tripoint> loot_spots;
1633 std::vector<tripoint> already_there_spots;
1634 std::vector<tripoint> combined_spots;
1635 std::map<itype_id, int> total_map;
1636 tripoint src_loc = g->m.getlocal( p.backlog.front().placement );
1637 for( const tripoint &elem : g->m.points_in_radius( src_loc,
1638 PICKUP_RANGE - 1 ) ) {
1639 already_there_spots.push_back( elem );
1640 combined_spots.push_back( elem );
1641 }
1642 for( const tripoint &elem : mgr.get_point_set_loot( g->m.getabs( p.pos() ), distance,
1643 p.is_npc() ) ) {
1644 // if there is a loot zone that's already near the work spot, we don't want it to be added twice.
1645 if( std::find( already_there_spots.begin(), already_there_spots.end(),
1646 elem ) != already_there_spots.end() ) {
1647 // construction tasks don't need the loot spot *and* the already_there/combined spots both added.
1648 // but a farming task will need to go and fetch the tool no matter if its near the work spot.
1649 // whereas the construction will automatically use what's nearby anyway.
1650 if( pickup_task ) {
1651 loot_spots.push_back( elem );
1652 } else {
1653 continue;
1654 }
1655 } else {
1656 loot_spots.push_back( elem );
1657 combined_spots.push_back( elem );
1658 }
1659 }
1660 add_basecamp_storage_to_loot_zone_list( mgr, src_loc, p, loot_spots, combined_spots );
1661 // if the requirements aren't available, then stop.
1662 if( !are_requirements_nearby( pickup_task ? loot_spots : combined_spots, things_to_fetch_id, p,
1663 activity_to_restore, pickup_task, src_loc ) ) {
1664 return requirement_map;
1665 }
1666 // if the requirements are already near the work spot and its a construction/crafting task, then no need to fetch anything more.
1667 if( !pickup_task &&
1668 are_requirements_nearby( already_there_spots, things_to_fetch_id, p, activity_to_restore,
1669 false, src_loc ) ) {
1670 return requirement_map;

1698 comp_elem.count -= stack_elem.count();

1702 // we only want the one welder that has the required charge.
1703 if( stack_elem.ammo_remaining() >= comp_elem.count ) {
1704 temp_map[stack_elem.typeId()] += stack_elem.ammo_remaining();
1705 }

1707 temp_map[stack_elem.typeId()] += stack_elem.count();
1708 }
1709 }
1710 }
1711 }

1716 const int qual_level = comp_elem.level;
1717 if( stack_elem.has_quality( tool_qual, qual_level ) ) {
1718 if( !pickup_task &&
1719 std::find( already_there_spots.begin(), already_there_spots.end(),
1720 point_elem ) != already_there_spots.end() ) {
1721 comp_elem.count -= stack_elem.count();
1722 }
1723 temp_map[stack_elem.typeId()] += stack_elem.count();
1724 }
1725 }
1726 }
1727 }

1729 total_map[map_elem.first] += map_elem.second;
1730 // if its a construction/crafting task, we can discount any items already near the work spot.

1743 // we need to consolidate them, and winnow it down to the minimum required counts, instead of all matching.

1764 quantity_required ) ) );
1765 if( quantity_here >= quantity_required ) {
1766 line_found = true;
1767 break;
1768 } else {
1769 remainder = quantity_required - quantity_here;
1770 }
1771 break;
1772 }
1773 it++;
1774 }
1775 if( line_found ) {
1776 while( true ) {
1777 // go back over things
1778 if( it == requirement_map.begin() ) {
1779 break;
1780 }
1781 if( remainder <= 0 ) {
1782 line_found = true;
1783 break;
1784 }

1786 itype_id item_here2 = std::get<1>( *it );

1809 int item_quantity = 0;
1810 auto it = requirement_map.begin();
1811 int remainder = 0;
1812 while( it != requirement_map.end() ) {
1813 tripoint pos_here = std::get<0>( *it );
1814 itype_id item_here = std::get<1>( *it );

2064 p.set_moves( 0 );
2065 p.activity = player_activity();
2066 p.backlog.clear();
2067 return false;
