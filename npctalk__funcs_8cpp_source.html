900void talk_function::start_training( npc &p )
901{
902 int cost;
903 time_duration time = 0_turns;
904 std::string name;
905 const skill_id &skill = p.chatbin.skill;
906 const matype_id &style = p.chatbin.style;
907 const spell_id &sp_id = p.chatbin.dialogue_spell;
908 int expert_multiplier = 1;
909 if( skill.is_valid() && g->u.get_skill_level( skill ) < p.get_skill_level( skill ) ) {
910 cost = calc_skill_training_cost( p, skill );
911 time = calc_skill_training_time( p, skill );
912 name = skill.str();
913 } else if( p.chatbin.style.is_valid() && !g->u.martial_arts_data->has_martialart( style ) ) {
914 cost = calc_ma_style_training_cost( p, style );
915 time = calc_ma_style_training_time( p, style );
916 name = p.chatbin.style.str();
917 // already checked if can learn this spell in npctalk.cpp
918 } else if( p.chatbin.dialogue_spell.is_valid() ) {
919 const spell &temp_spell = p.magic->get_spell( sp_id );
920 const bool knows = g->u.magic->knows_spell( sp_id );
921 cost = p.calc_spell_training_cost( knows, temp_spell.get_difficulty(), temp_spell.get_level() );
922 name = temp_spell.id().str();
923 expert_multiplier = knows ? temp_spell.get_level() - g->u.magic->get_spell( sp_id ).get_level() : 1;
924 // quicker to learn with instruction as opposed to books.
925 // if this is a known spell, then there is a set time to gain some exp.
926 // if player doesn't know this spell, then the NPC will teach all of it
927 // which takes max 6 hours, min 3 hours.
928 // TODO: a system for NPCs to train new stuff in bits and pieces
929 // and remember the progress.
930 if( knows ) {
931 time = 1_hours;
932 } else {
933 int seconds = g->u.magic->time_to_learn_spell( g->u, sp_id ) / 50;
934 time = time_duration::from_seconds( clamp( seconds, 7200, 21600 ) );
935 }
936 } else {
937 debugmsg( "start_training with no valid skill or style set" );
938 return;
939 }
940
941 mission *miss = p.chatbin.mission_selected;
942 if( miss != nullptr && miss->get_assigned_player_id() == g->u.getID() &&
943 miss->is_complete( g->u.getID() ) ) {
944 clear_mission( p );
945 } else if( !npc_trading::pay_npc( p, cost ) ) {
946 return;
947 }
948 player_activity act = player_activity( ACT_TRAIN, to_moves<int>( time ),
949 p.getID().get_value(), 0, name );
950 act.values.push_back( expert_multiplier );
951 g->u.assign_activity( act );
952
953 p.add_effect( effect_asked_to_train, 6_hours );
954}
