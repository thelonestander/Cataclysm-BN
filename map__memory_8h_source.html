56 inline void set_tile( const point &p, const memorized_terrain_tile &value ) {
57 if( tiles.empty() ) {
58 // call 'reserve' first to force allocation of exact size
59 tiles.reserve( SEEX * SEEY );
60 tiles.resize( SEEX * SEEY, default_tile );
61 }
62 tiles[p.y * SEEX + p.x] = value;
63 }
