51enum class type : int {
52 invalid = -1,

60/** For the purposes of iteration. */
61const std::array<type, 4> all = {{ type::north, type::east, type::south, type::west }};
62const size_t size = all.size();
63
64const std::array<std::string, 4> all_suffixes = {{ "_north", "_east", "_south", "_west" }};
65const std::string invalid_dir_suffix;
66const std::array<int, 4> all_cw_rotations = {{ 0, 1, 2, 3 }};
67const int invalid_dir_rotations = 0;
68
69/** Returns directional suffix associated with the value, e.g. _north or _west. */
70constexpr const std::string &get_suffix( type dir )
71{
72 if( dir == type::invalid ) {
73 return invalid_dir_suffix;
74 } else {
75 return all_suffixes[static_cast<size_t>( dir )];
76 }
77}
78
79/** Returns number of clockwise rotations needed to reach this direction from 'north'. */
80constexpr int get_num_cw_rotations( type dir )
81{
82 if( dir == type::invalid ) {
83 return invalid_dir_rotations;
84 } else {
85 return all_cw_rotations[static_cast<size_t>( dir )];
86 }
87}
