195itype_id mtype::get_meat_itype() const
196{

204 return itype_veggy_tainted;

206 return itype_bone_tainted;

208 } else {
209 if( made_of( material_id( "flesh" ) ) || made_of( material_id( "hflesh" ) ) ) {
210 if( has_flag( MF_HUMAN ) ) {
211 return itype_human_flesh;

213 return itype_fish;

218 //In the future, insects could drop insect flesh rather than plain ol' meat.
219 return itype_meat;

221 return itype_veggy;
