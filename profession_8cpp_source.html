389std::list<item> profession::items( bool male, const std::vector<trait_id> &traits ) const
390{
391 std::list<item> result;
392 auto add_legacy_items = [&result]( const itypedecvec & vec ) {
393 for( const itypedec &elem : vec ) {
394 item it( elem.type_id, advanced_spawn_time(), item::default_charges_tag {} );
395 if( !elem.snip_id.is_null() ) {
396 it.set_snippet( elem.snip_id );
397 }
398 it = it.in_its_container();
399 result.push_back( it );
400 }
401 };
402
403 add_legacy_items( legacy_starting_items );
404 add_legacy_items( male ? legacy_starting_items_male : legacy_starting_items_female );
405
406 const std::vector<item> group_both = item_group::items_from( _starting_items,
407 advanced_spawn_time() );
408 const std::vector<item> group_gender = item_group::items_from( male ? _starting_items_male :
409 _starting_items_female, advanced_spawn_time() );
410 result.insert( result.begin(), group_both.begin(), group_both.end() );
411 result.insert( result.begin(), group_gender.begin(), group_gender.end() );
412
413 std::vector<itype_id> bonus = item_substitutions.get_bonus_items( traits );
414 for( const itype_id &elem : bonus ) {
415 if( elem != no_bonus ) {
416 result.push_back( item( elem, advanced_spawn_time(), item::default_charges_tag {} ) );
417 }
418 }
419 for( auto iter = result.begin(); iter != result.end(); ) {
420 const auto sub = item_substitutions.get_substitution( *iter, traits );
421 if( !sub.empty() ) {
422 result.insert( result.begin(), sub.begin(), sub.end() );
423 iter = result.erase( iter );
424 } else {
425 ++iter;
426 }
427 }
428 for( item &it : result ) {
429 clear_faults( it );
430 if( it.is_holster() && it.contents.num_item_stacks() == 1 ) {
431 clear_faults( it.contents.front() );
432 }
433 if( it.has_flag( "VARSIZE" ) ) {
434 it.set_flag( "FIT" );
435 }
436 }
437
438 if( result.empty() ) {
439 // No need to do the below stuff. Plus it would cause said below stuff to crash
440 return result;
441 }
442
443 // Merge charges for items that stack with each other
444 for( auto outer = result.begin(); outer != result.end(); ++outer ) {
445 if( !outer->count_by_charges() ) {
446 continue;
447 }
448 for( auto inner = std::next( outer ); inner != result.end(); ) {
449 if( outer->stacks_with( *inner ) ) {
450 outer->merge_charges( *inner );
451 inner = result.erase( inner );
452 } else {
453 ++inner;
454 }
455 }
456 }
457
458 result.sort( []( const item & first, const item & second ) {
459 return first.get_layer() < second.get_layer();
460 } );
461 return result;
462}

573void json_item_substitution::load( const JsonObject &jo )
574{
575 const bool item_mode = jo.has_string( "item" );
576 const std::string title = jo.get_string( item_mode ? "item" : "trait" );
577
578 auto check_duplicate_item = [&]( const itype_id & it ) {
579 return substitutions.find( it ) != substitutions.end() ||
580 std::find_if( bonuses.begin(), bonuses.end(),
581 [&it]( const std::pair<itype_id, trait_requirements> &p ) {
582 return p.first == it;
583 } ) != bonuses.end();
584 };
585 if( item_mode && check_duplicate_item( itype_id( title ) ) ) {
586 jo.throw_error( "Duplicate definition of item" );
587 }
588
589 if( item_mode ) {
590 if( jo.has_member( "bonus" ) ) {
591 bonuses.emplace_back( itype_id( title ),
592 trait_requirements( jo.get_object( "bonus" ) ) );
593 }
594
595 for( const JsonValue sub : jo.get_array( "sub" ) ) {
596 substitution s;
597 JsonObject obj = sub.get_object();
598 s.trait_reqs = trait_requirements( obj );
599 for( const JsonValue info : obj.get_array( "new" ) ) {
600 s.infos.emplace_back( info );
601 }
602 substitutions[itype_id( title )].push_back( s );
603 }
604 } else {
605 for( const JsonObject sub : jo.get_array( "sub" ) ) {
606 substitution s;
607 itype_id old_it;
608 sub.read( "item", old_it, true );
609 if( check_duplicate_item( old_it ) ) {
610 sub.throw_error( "Duplicate definition of item" );
611 }
612 s.trait_reqs.present.push_back( trait_id( title ) );
613 for( const JsonValue info : sub.get_array( "new" ) ) {
614 s.infos.emplace_back( info );
615 }
616 substitutions[old_it].push_back( s );
617 }
618 }

621void json_item_substitution::check_consistency()
622{
623 auto check_if_trait = []( const trait_id & t ) {
624 if( !t.is_valid() ) {
625 debugmsg( "%s is not a trait", t.c_str() );
626 }
627 };
628 auto check_if_itype = []( const itype_id & i ) {
629 if( !i.is_valid() ) {
630 debugmsg( "%s is not an itype_id", i.c_str() );
631 }
632 };
633 auto check_trait_reqs = [&check_if_trait]( const trait_requirements & tr ) {
634 for( const trait_id &str : tr.present ) {
635 check_if_trait( str );
636 }
637 for( const trait_id &str : tr.absent ) {
638 check_if_trait( str );
639 }
640 };
641
642 for( const auto &pair : substitutions ) {
643 check_if_itype( pair.first );
644 for( const substitution &s : pair.second ) {
645 check_trait_reqs( s.trait_reqs );
646 for( const substitution::info &inf : s.infos ) {
647 check_if_itype( inf.new_item );
648 }
649 }
650 }
651 for( const auto &pair : bonuses ) {
652 check_if_itype( pair.first );
653 check_trait_reqs( pair.second );
654 }

657bool json_item_substitution::trait_requirements::meets_condition( const std::vector<trait_id>
658 &traits ) const
659{
660 const auto pred = [&traits]( const trait_id & s ) {
661 return std::find( traits.begin(), traits.end(), s ) != traits.end();
662 };
663 return std::all_of( present.begin(), present.end(), pred ) &&
664 std::none_of( absent.begin(), absent.end(), pred );
665}

667std::vector<item> json_item_substitution::get_substitution( const item &it,
668 const std::vector<trait_id> &traits ) const
669{
670 auto iter = substitutions.find( it.typeId() );
671 std::vector<item> ret;
672 if( iter == substitutions.end() ) {
673 for( const item *con : it.contents.all_items_top() ) {
674 const auto sub = get_substitution( *con, traits );
675 ret.insert( ret.end(), sub.begin(), sub.end() );
676 }
677 return ret;
678 }
679
680 const auto sub = std::find_if( iter->second.begin(),
681 iter->second.end(), [&traits]( const substitution & s ) {
682 return s.trait_reqs.meets_condition( traits );
683 } );
684 if( sub == iter->second.end() ) {
685 return ret;
686 }
687
688 const int old_amt = it.count();
689 for( const substitution::info &inf : sub->infos ) {
