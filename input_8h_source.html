87/**
88 * An instance of an input, like a keypress etc.
89 *
90 * Gamepad, mouse and keyboard keypresses will be represented as `int`.
91 * Whether a gamepad, mouse or keyboard was used can be checked using the
92 * `type` member.
93 *
94 */
95struct input_event {
96 input_event_t type;
97
98 std::vector<int> modifiers; // Keys that need to be held down for
99 // this event to be activated.
100
101 std::vector<int> sequence; // The sequence of key or mouse events that
102 // triggers this event. For single-key
103 // events, simply make this of size 1.
104
105 point mouse_pos; // Mouse click co-ordinates, if applicable
106
107 // Actually entered text (if any), UTF-8 encoded, might be empty if
108 // the input is not UTF-8 or not even text.
109 std::string text;
110 std::string edit;
111 bool edit_refresh;
112
113#if defined(__ANDROID__)
114 // Used exclusively by the quick shortcuts to determine how stale a shortcut is
115 int shortcut_last_used_action_counter;
116#endif
117
118 input_event() : edit_refresh( false ) {
119 type = CATA_INPUT_ERROR;
120#if defined(__ANDROID__)
121 shortcut_last_used_action_counter = 0;
122#endif
123 }
124 input_event( int s, input_event_t t )
125 : type( t ), edit_refresh( false ) {
126 sequence.push_back( s );
127#if defined(__ANDROID__)
128 shortcut_last_used_action_counter = 0;
129#endif
130 }
131
132 int get_first_input() const;
133
134 void add_input( const int input ) {
135 sequence.push_back( input );
136 }
137
138#if defined(__ANDROID__)
139 input_event &operator=( const input_event &other ) {
140 type = other.type;
141 modifiers = other.modifiers;
142 sequence = other.sequence;
143 mouse_pos = other.mouse_pos;
144 text = other.text;
145 shortcut_last_used_action_counter = other.shortcut_last_used_action_counter;
146 return *this;
147 }
148#endif
149
150 bool operator==( const input_event &other ) const {
151 if( type != other.type ) {
152 return false;
153 }
154
155 if( sequence.size() != other.sequence.size() ) {
156 return false;
157 }
158 for( size_t i = 0; i < sequence.size(); ++i ) {
159 if( sequence[i] != other.sequence[i] ) {
160 return false;
161 }
162 }
163
164 if( modifiers.size() != other.modifiers.size() ) {
165 return false;
166 }
167 for( size_t i = 0; i < modifiers.size(); ++i ) {
168 if( modifiers[i] != other.modifiers[i] ) {
169 return false;
170 }
171 }
172
173 return true;
174 }
