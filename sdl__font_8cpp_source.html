11static int test_face_size( const std::string &f, int size, int faceIndex )
12{
13 const TTF_Font_Ptr fnt( TTF_OpenFontIndex( f.c_str(), size, faceIndex ) );
14 if( fnt ) {
15 char *style = TTF_FontFaceStyleName( fnt.get() );
16 if( style != nullptr ) {
17 int faces = TTF_FontFaces( fnt.get() );
18 for( int i = faces - 1; i >= 0; i-- ) {
19 const TTF_Font_Ptr tf( TTF_OpenFontIndex( f.c_str(), size, i ) );
20 char *ts = nullptr;
21 if( tf ) {
22 if( nullptr != ( ts = TTF_FontFaceStyleName( tf.get() ) ) ) {
23 if( lcequal( ts, style ) && TTF_FontHeight( tf.get() ) <= size ) {
24 return i;
25 }
26 }
27 }
28 }
29 }
30 }

36 const std::string &typeface, int fontsize, int width,

40{

272SDL_Texture_Ptr CachedTTFFont::create_glyph( const SDL_Renderer_Ptr &renderer,
273 const std::string &ch,
274 const int color )
275{
276 const auto function = fontblending ? TTF_RenderUTF8_Blended : TTF_RenderUTF8_Solid;
277 SDL_Surface_Ptr sglyph( function( font.get(), ch.c_str(), windowsPalette[color] ) );
278 if( !sglyph ) {
279 dbg( DL::Error ) << "Failed to create glyph for " << ch << ": " << TTF_GetError();
280 return nullptr;
281 }
282 /* SDL interprets each pixel as a 32-bit number, so our masks must depend
283 on the endianness (byte order) of the machine */
284#if SDL_BYTEORDER == SDL_BIG_ENDIAN
285 static const Uint32 rmask = 0xff000000;
286 static const Uint32 gmask = 0x00ff0000;
287 static const Uint32 bmask = 0x0000ff00;
288 static const Uint32 amask = 0x000000ff;
289#else
290 static const Uint32 rmask = 0x000000ff;
291 static const Uint32 gmask = 0x0000ff00;
292 static const Uint32 bmask = 0x00ff0000;
293 static const Uint32 amask = 0xff000000;
294#endif
295 const int wf = utf8_wrapper( ch ).display_width();
296 // Note: bits per pixel must be 8 to be synchronized with the surface
297 // that TTF_RenderGlyph above returns. This is important for SDL_BlitScaled
298 SDL_Surface_Ptr surface = CreateRGBSurface( 0, width * wf, height, 32, rmask, gmask, bmask,
299 amask );
300 SDL_Rect src_rect = { 0, 0, sglyph->w, sglyph->h };
301 SDL_Rect dst_rect = { 0, 0, width * wf, height };
302 if( src_rect.w < dst_rect.w ) {
303 dst_rect.x = ( dst_rect.w - src_rect.w ) / 2;
304 dst_rect.w = src_rect.w;
305 } else if( src_rect.w > dst_rect.w ) {
306 src_rect.x = ( src_rect.w - dst_rect.w ) / 2;
307 src_rect.w = dst_rect.w;
308 }
309 if( src_rect.h < dst_rect.h ) {
310 dst_rect.y = ( dst_rect.h - src_rect.h ) / 2;
311 dst_rect.h = src_rect.h;
312 } else if( src_rect.h > dst_rect.h ) {
313 src_rect.y = ( src_rect.h - dst_rect.h ) / 2;
314 src_rect.h = dst_rect.h;
315 }
316
317 if( !printErrorIf( SDL_BlitSurface( sglyph.get(), &src_rect, surface.get(), &dst_rect ) != 0,
318 "SDL_BlitSurface failed" ) ) {
319 sglyph = std::move( surface );
320 }
321
322 return CreateTextureFromSurface( renderer, sglyph );

361BitmapFont::BitmapFont(
362 SDL_Renderer_Ptr &renderer, SDL_PixelFormat_Ptr &format,
363 const int w, const int h,
364 const palette_array &palette,
365 const std::string &typeface_path )
366 : Font( w, h, palette )
367{
368 dbg( DL::Info ) << "Loading bitmap font [" + typeface_path + "].";
369 SDL_Surface_Ptr asciiload = load_image( typeface_path.c_str() );
370 assert( asciiload );
371 if( asciiload->w * asciiload->h < ( width * height * 256 ) ) {
372 throw std::runtime_error( "bitmap for font is to small" );
373 }
374 Uint32 key = SDL_MapRGB( asciiload->format, 0xFF, 0, 0xFF );
375 SDL_SetColorKey( asciiload.get(), SDL_TRUE, key );
376 SDL_Surface_Ptr ascii_surf[std::tuple_size<decltype( ascii )>::value];
377 ascii_surf[0].reset( SDL_ConvertSurface( asciiload.get(), format.get(), 0 ) );
378 SDL_SetSurfaceRLE( ascii_surf[0].get(), 1 );
379 asciiload.reset();
380
381 for( size_t a = 1; a < std::tuple_size<decltype( ascii )>::value; ++a ) {
382 ascii_surf[a].reset( SDL_ConvertSurface( ascii_surf[0].get(), format.get(), 0 ) );
383 SDL_SetSurfaceRLE( ascii_surf[a].get(), 1 );
384 }
385
386 for( size_t a = 0; a < std::tuple_size<decltype( ascii )>::value - 1; ++a ) {
387 SDL_LockSurface( ascii_surf[a].get() );
388 int size = ascii_surf[a]->h * ascii_surf[a]->w;
389 Uint32 *pixels = static_cast<Uint32 *>( ascii_surf[a]->pixels );
390 Uint32 color = ( windowsPalette[a].r << 16 ) | ( windowsPalette[a].g << 8 ) | windowsPalette[a].b;
391 for( int i = 0; i < size; i++ ) {
392 if( pixels[i] == 0xFFFFFF ) {
393 pixels[i] = color;
394 }
395 }
396 SDL_UnlockSurface( ascii_surf[a].get() );
397 }
398 tilewidth = ascii_surf[0]->w / width;
399
400 //convert ascii_surf to SDL_Texture
401 for( size_t a = 0; a < std::tuple_size<decltype( ascii )>::value; ++a ) {
402 ascii[a] = CreateTextureFromSurface( renderer, ascii_surf[a] );
403 }
404}
