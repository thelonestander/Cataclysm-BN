33int calc_xp_gain( const vpart_info &vp, const skill_id &sk, const Character &who )
34{
35 const auto iter = vp.install_skills.find( sk );
36 if( iter == vp.install_skills.end() ) {
37 return 0;
38 }
39
40 // how many levels are we above the requirement?
41 const int lvl = std::max( who.get_skill_level( sk ) - iter->second, 1 );
42
43 // scale xp gain per hour according to relative level
44 // 0-1: 60 xp /h
45 // 2: 15 xp /h
46 // 3: 6 xp /h
47 // 4: 4 xp /h
48 // 5: 3 xp /h
49 // 6: 2 xp /h
50 // 7+: 1 xp /h
51 return std::ceil( static_cast<double>( vp.install_moves ) /
52 to_moves<int>( 1_minutes * std::pow( lvl, 2 ) ) );
53}

55vehicle_part &most_repairable_part( vehicle &veh, Character &who, bool only_repairable )
56{
57 const auto &inv = who.crafting_inventory();
58
59 enum class repairable_status {
60 not_repairable = 0,
61 need_replacement,
62 repairable
63 };
64 std::map<const vehicle_part *, repairable_status> repairable_cache;
65 for( const vpart_reference &vpr : veh.get_all_parts() ) {
66 const auto &info = vpr.info();
67 repairable_cache[ &vpr.part() ] = repairable_status::not_repairable;
68 if( vpr.part().removed || vpr.part().damage() <= 0 ) {
69 continue;
70 }
71
72 if( vpr.part().is_broken() ) {
73 if( info.install_requirements().can_make_with_inventory( inv, is_crafting_component ) ) {
74 repairable_cache[ &vpr.part()] = repairable_status::need_replacement;
75 }
76
77 continue;
78 }
79
80 if( info.is_repairable() &&
81 ( info.repair_requirements() * vpr.part().damage_level( 4 ) ).can_make_with_inventory( inv,
82 is_crafting_component ) ) {
83 repairable_cache[ &vpr.part()] = repairable_status::repairable;
84 }
85 }
86
87 const auto part_damage_comparison = [&repairable_cache]( const vpart_reference & a,
88 const vpart_reference & b ) {
89 return ( repairable_cache[ &b.part() ] > repairable_cache[ &a.part()] ) ||
90 ( repairable_cache[ &b.part()] == repairable_cache[ &a.part()] &&
91 b.part().damage() > a.part().damage() );
92 };
93 const vehicle_part_range vp_range = veh.get_all_parts();
94 const auto high_damage_iterator = std::max_element( vp_range.begin(),
95 vp_range.end(),
96 part_damage_comparison );
97 if( high_damage_iterator == vp_range.end() ||
98 high_damage_iterator->part().removed ||
99 !high_damage_iterator->info().is_repairable() ||
100 ( only_repairable &&
101 repairable_cache[ &( *high_damage_iterator ).part() ] != repairable_status::not_repairable ) ) {
102 static vehicle_part nullpart;
103 return nullpart;
104 }
105
106 return high_damage_iterator->part();
107}

124 // as they have the handicap of not being able to use the veh interaction menu

139 base = obj;
140 }
141 }
142 }
143
144 for( const auto &e : reqs.get_tools() ) {

146 }

148 who.invalidate_crafting_inventory();
149

151 who.practice( sk.first, calc_xp_gain( vp, sk.first, who ) );
152 }
153
154 // If part is broken, it will be destroyed and references invalidated
155 std::string partname = pt.name( false );
156 const std::string startdurability = colorize( pt.get_base().damage_symbol(),
157 pt.get_base().damage_color() );
158 bool wasbroken = pt.is_broken();
159 if( wasbroken ) {
160 const units::angle dir = pt.direction;
161 point loc = pt.mount;
162 auto replacement_id = pt.info().get_id();
163 get_map().spawn_items( who.pos(), pt.pieces_for_broken_part() );
164 veh.remove_part( part_index );
165 const int partnum = veh.install_part( loc, replacement_id, std::move( base ) );
166 veh.part( partnum ).direction = dir;
167 veh.part_removal_cleanup();
168 } else {
169 veh.set_hp( pt, pt.info().durability );
170 }
171
172 // TODO: NPC doing that
173 who.add_msg_if_player( m_good,
174 wasbroken ? _( "You replace the %1$s's %2$s. (was %3$s)" ) :
175 _( "You repair the %1$s's %2$s. (was %3$s)" ), veh.name,
176 partname, startdurability );
177 return true;
178}
