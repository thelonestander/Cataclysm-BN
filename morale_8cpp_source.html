73struct morale_mult {
74 morale_mult(): good( 1.0 ), bad( 1.0 ) {}
75 morale_mult( double good, double bad ): good( good ), bad( bad ) {}
76 morale_mult( double both ): good( both ), bad( both ) {}
77
78 double good; // For good morale
79 double bad; // For bad morale
80
81 morale_mult operator * ( const morale_mult &rhs ) const {
82 return morale_mult( *this ) *= rhs;
83 }
84
85 morale_mult &operator *= ( const morale_mult &rhs ) {
86 good *= rhs.good;
87 bad *= rhs.bad;
88 return *this;
89 }
90};

93{

177void player_morale::morale_point::add( const int new_bonus, const int new_max_bonus,
178 time_duration new_duration, time_duration new_decay_start, bool new_cap )
179{
180 new_duration = std::max( 0_turns, new_duration );
181 new_decay_start = std::max( 0_turns, new_decay_start );
182
183 const bool same_sign = ( bonus > 0 ) == ( new_max_bonus > 0 );
184
185 if( new_cap || new_duration == 0_turns ) {
186 duration = new_duration;
187 decay_start = new_decay_start;
188 } else {
189 duration = pick_time( duration, new_duration, same_sign );
190 decay_start = pick_time( decay_start, new_decay_start, same_sign );
191 }
192
193 int sqrt_of_sum_of_squares;
194 if( new_cap || !same_sign ) {
195 // If the morale bonus is capped apply the full bonus
196 // This is because some morale types build up slowly to a cap over time (e.g. MORALE_WET)
197 // If the new bonus is opposing apply the full bonus
198 sqrt_of_sum_of_squares = get_net_bonus() + new_bonus;
199 } else {
200 // Otherwise use the sqrt of sum of squares to nerf stacking
201 sqrt_of_sum_of_squares = std::pow( get_net_bonus(), 2 ) + std::pow( new_bonus, 2 );
202 sqrt_of_sum_of_squares = std::sqrt( sqrt_of_sum_of_squares );
203 sqrt_of_sum_of_squares *= sgn( bonus );
204 }
205
206 bonus = normalize_bonus( sqrt_of_sum_of_squares, new_max_bonus, new_cap );
207 age = 0_turns; // Brand new. The assignment should stay below get_net_bonus() and pick_time().
208}

303 const time_duration &duration, const time_duration &decay_start,
304 bool capped )
305{
306 if( ( duration == 0_turns ) && !type->is_permanent() ) {
307 std::string full_desc = subtype.has_description()
308 ? type.obj().describe( subtype.describe() )
309 : type.obj().describe();
310 debugmsg( "Tried to set a non-permanent morale \"%s\" as permanent.", full_desc );
311 return;
312 }
313
314 for( morale_point &m : points ) {
315 if( m.matches( type, subtype ) ) {
316 const int prev_bonus = m.get_net_bonus();

318 m.add( bonus, max_bonus, duration, decay_start, capped );
319
320 if( m.is_expired() ) {
321 remove_expired();
322 } else if( m.get_net_bonus() != prev_bonus ) {
323 invalidate();
324 }
325
326 return;
327 }
328 }

330 morale_point new_morale( type, subtype, bonus, max_bonus, duration, decay_start, capped );
331
332 if( !new_morale.is_expired() ) {
333 points.push_back( new_morale );
334 invalidate();
335 }

436void player_morale::calculate_percentage()
437{
438 const morale_mult mult = get_temper_mult();
439
440 int sum_of_positive_squares = 0;
441 int sum_of_negative_squares = 0;
442
443 for( const morale_point &m : points ) {
444 const int bonus = m.get_net_bonus( mult );
445 if( bonus > 0 ) {
446 sum_of_positive_squares += bonus * bonus;
447 } else {
448 sum_of_negative_squares += bonus * bonus;
449 }
450 }
451
452 for( morale_point &m : points ) {
453 const int bonus = m.get_net_bonus( mult );
454 if( bonus > 0 ) {
455 m.set_percent_contribution( bonus * bonus * 100.0 / sum_of_positive_squares );
456 } else {
457 m.set_percent_contribution( bonus * bonus * 100.0 / sum_of_negative_squares );
458 }
459 }

841bool player_morale::consistent_with( const player_morale &morale ) const
842{
843 const auto test_points = []( const player_morale & lhs, const player_morale & rhs ) {
844 for( const auto &lhp : lhs.points ) {
845 if( !lhp.is_permanent() ) {
846 continue;
847 }
848
849 const auto iter = std::find_if( rhs.points.begin(), rhs.points.end(),
850 [ &lhp ]( const morale_point & rhp ) {
851 return lhp.matches( rhp );
852 } );
853
854 if( iter == rhs.points.end() || lhp.get_net_bonus() != iter->get_net_bonus() ) {
855 debugmsg( "Morale \"%s\" is inconsistent: lhs %d != rhs %s",
856 lhp.get_name(), lhp.get_net_bonus(),
857 ( iter == rhs.points.end() ? "N/A" : std::to_string( iter->get_net_bonus() ).c_str() ) );
858 return false;
859 }
860 }
861
862 return true;
863 };
864
865 if( took_prozac != morale.took_prozac ) {
866 debugmsg( "player_morale::took_prozac is inconsistent." );
867 return false;
868 } else if( took_prozac_bad != morale.took_prozac_bad ) {
869 debugmsg( "player_morale::took_prozac (bad) is inconsistent." );
870 return false;
871 } else if( stylish != morale.stylish ) {
872 debugmsg( "player_morale::stylish is inconsistent." );
873 return false;
874 } else if( perceived_pain != morale.perceived_pain ) {
875 debugmsg( "player_morale::perceived_pain is inconsistent." );
876 return false;
877 }
878
879 return test_points( *this, morale ) && test_points( morale, *this );
880}

1000void player_morale::set_worn( const item &it, bool worn )
1001{
1002 const bool fancy = it.has_flag( "FANCY" );
1003 const bool super_fancy = it.has_flag( "SUPER_FANCY" );
1004 const bool filthy_gear = it.has_flag( "FILTHY" );
1005 const int sign = ( worn ) ? 1 : -1;
1006
1007 const auto update_body_part = [&]( body_part_data & bp_data ) {
1008 if( fancy || super_fancy ) {
1009 bp_data.fancy += sign;
1010 }
1011 if( filthy_gear ) {
1012 bp_data.filthy += sign;
1013 }
1014 bp_data.covered += sign;
1015 };
1016
1017 const body_part_set covered( it.get_covered_body_parts() );
1018
1019 if( covered.any() ) {
1020 for( const bodypart_id &bp_id : get_player_character().get_all_body_parts() ) {
1021 auto bp = bp_id->token;
1022 if( covered.test( bp ) ) {
1023 const bodypart_id bp_id = convert_bp( bp ).id();
1024 update_body_part( body_parts[bp_id] );
1025 }
1026 }
1027 } else {
1028 update_body_part( no_body_part );
1029 }
1030
1031 if( super_fancy ) {
1032 const auto id = it.typeId();
1033 const auto iter = super_fancy_items.find( id );
1034
1035 if( iter != super_fancy_items.end() ) {
1036 iter->second += sign;
1037 if( iter->second == 0 ) {
1038 super_fancy_items.erase( iter );
1039 }
1040 } else if( worn ) {
1041 super_fancy_items[id] = 1;
1042 } else {
1043 debugmsg( "Tried to take off \"%s\" which isn't worn.", id.c_str() );
1044 }
1045 }
1046 if( fancy || super_fancy ) {
1047 update_stylish_bonus();
1048 }
1049 if( filthy_gear ) {
1050 update_squeamish_penalty();
1051 }
1052 update_constrained_penalty();
1053}
