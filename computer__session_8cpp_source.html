199bool computer_session::hack_attempt( player &p, int Security )
200{
201 if( Security == -1 ) {
202 Security = comp.security; // Set to main system security if no value passed
203 }
204 const int hack_skill = p.get_skill_level( skill_computer );
205
206 // Every time you dig for lab notes, (or, in future, do other suspicious stuff?)
207 // +2 dice to the system's hack-resistance
208 // So practical max files from a given terminal = 5, at 10 Computer
209 if( comp.alerts > 0 ) {
210 Security += ( comp.alerts * 2 );
211 }
212
213 p.moves -= 10 * ( 5 + Security * 2 ) / std::max( 1, hack_skill + 1 );
214 int player_roll = hack_skill;
215 ///\EFFECT_INT <8 randomly penalizes hack attempts, 50% of the time
216 if( p.int_cur < 8 && one_in( 2 ) ) {
217 player_roll -= rng( 0, 8 - p.int_cur );
218 ///\EFFECT_INT >8 randomly benefits hack attempts, 33% of the time
219 } else if( p.int_cur > 8 && one_in( 3 ) ) {
220 player_roll += rng( 0, p.int_cur - 8 );
221 }
222
223 ///\EFFECT_COMPUTER increases chance of successful hack attempt, vs Security level
224 bool successful_attempt = ( dice( player_roll, 6 ) >= dice( Security, 6 ) );
225 p.practice( skill_computer, successful_attempt ? ( 15 + Security * 3 ) : 7 );
226 return successful_attempt;
227}
