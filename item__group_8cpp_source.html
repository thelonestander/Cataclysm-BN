50item Single_item_creator::create_single( const time_point &birthday, RecursionList &rec ) const
51{
52 item tmp;
53 if( type == S_ITEM ) {
54 if( id == "corpse" ) {
55 tmp = item::make_corpse( mtype_id::NULL_ID(), birthday );
56 } else {
57 tmp = item( id, birthday );
58 }
59 } else if( type == S_ITEM_GROUP ) {
60 if( std::find( rec.begin(), rec.end(), id ) != rec.end() ) {
61 debugmsg( "recursion in item spawn list %s", id.c_str() );

87Item_spawn_data::ItemList Single_item_creator::create( const time_point &birthday,
88 RecursionList &rec ) const
89{
90 ItemList result;
91 int cnt = 1;
92 if( modifier ) {
93 auto modifier_count = modifier->count;
94 cnt = ( modifier_count.first == modifier_count.second ) ? modifier_count.first : rng(
95 modifier_count.first, modifier_count.second );
96 }
97 for( ; cnt > 0; cnt-- ) {
98 if( type == S_ITEM ) {
99 const auto itm = create_single( birthday, rec );
100 if( !itm.is_null() ) {
101 result.push_back( itm );
102 }
103 } else {
104 if( std::find( rec.begin(), rec.end(), id ) != rec.end() ) {
105 debugmsg( "recursion in item spawn list %s", id.c_str() );
106 return result;
107 }
108 rec.push_back( id );
109 Item_spawn_data *isd = item_controller->get_group( item_group_id( id ) );
110 if( isd == nullptr ) {
111 debugmsg( "unknown item spawn list %s", id.c_str() );
112 return result;
113 }
114 ItemList tmplist = isd->create( birthday, rec );
115 rec.erase( rec.end() - 1 );
116 if( modifier ) {
117 for( auto &elem : tmplist ) {
118 modifier->modify( elem );
119 }
120 }
121 result.insert( result.end(), tmplist.begin(), tmplist.end() );
122 }
123 }
124 return result;

239void Item_modifier::modify( item &new_item ) const
240{
241 if( new_item.is_null() ) {
242 return;
243 }
244
245 new_item.set_damage( rng( damage.first, damage.second ) );
246 // no need for dirt if it's a bow
247 if( new_item.is_gun() && !new_item.has_flag( flag_PRIMITIVE_RANGED_WEAPON ) &&
248 !new_item.has_flag( flag_NON_FOULING ) ) {
249 int random_dirt = rng( dirt.first, dirt.second );
250 // if gun RNG is dirty, must add dirt fault to allow cleaning
251 if( random_dirt > 0 ) {
252 new_item.set_var( "dirt", random_dirt );
253 new_item.faults.emplace( "fault_gun_dirt" );
254 // chance to be unlubed, but only if it's not a laser or something
255 } else if( one_in( 10 ) && !new_item.has_flag( flag_NEEDS_NO_LUBE ) ) {
256 new_item.faults.emplace( "fault_gun_unlubricated" );
257 }
258 }
259
260 // create container here from modifier or from default to get max charges later
261 item cont;
262 if( container != nullptr ) {
263 cont = container->create_single( new_item.birthday() );
264 }
265 if( cont.is_null() && new_item.type->default_container.has_value() ) {
266 const itype_id &cont_value = new_item.type->default_container.value_or( "null" );
267 if( !cont_value.is_null() ) {
268 cont = item( cont_value, new_item.birthday() );
269 }
270 }
271
272 int max_capacity = -1;
273 if( charges.first != -1 && charges.second == -1 ) {
274 const int max_ammo = new_item.ammo_capacity();
275 if( max_ammo > 0 ) {
276 max_capacity = max_ammo;
277 }
278 }
279
280 if( max_capacity == -1 && !cont.is_null() && ( new_item.made_of( LIQUID ) ||
281 ( !new_item.is_tool() && !new_item.is_gun() && !new_item.is_magazine() ) ) ) {
282 max_capacity = new_item.charges_per_volume( cont.get_container_capacity() );
283 }
284
285 const bool charges_not_set = charges.first == -1 && charges.second == -1;
286 int ch = -1;
287 if( !charges_not_set ) {
288 int charges_min = charges.first == -1 ? 0 : charges.first;
289 int charges_max = charges.second == -1 ? max_capacity : charges.second;
290
291 if( charges_min == -1 && charges_max != -1 ) {
292 charges_min = 0;
293 }
294
295 if( max_capacity != -1 && ( charges_max > max_capacity || ( charges_min != 1 &&
296 charges_max == -1 ) ) ) {
297 charges_max = max_capacity;
298 }
299
300 if( charges_min > charges_max ) {
301 charges_min = charges_max;
302 }
303
304 ch = charges_min == charges_max ? charges_min : rng( charges_min,
305 charges_max );
306 } else if( !cont.is_null() && new_item.made_of( LIQUID ) ) {
307 new_item.charges = std::max( 1, max_capacity );
308 }
309
310 if( ch != -1 ) {
311 if( new_item.count_by_charges() || new_item.made_of( LIQUID ) ) {
312 // food, ammo
313 // count_by_charges requires that charges is at least 1. It makes no sense to
314 // spawn a "water (0)" item.
315 new_item.charges = std::max( 1, ch );
316 } else if( new_item.is_tool() ) {
317 const int qty = std::min( ch, new_item.ammo_capacity() );
318 new_item.charges = qty;
319 if( !new_item.ammo_types().empty() && qty > 0 ) {
320 new_item.ammo_set( new_item.ammo_default(), qty );
321 }
322 } else if( new_item.type->can_have_charges() ) {
323 new_item.charges = ch;
324 }
325 }
326
327 if( ch > 0 && ( new_item.is_gun() || new_item.is_magazine() ) ) {
328 if( ammo == nullptr ) {
329 // In case there is no explicit ammo item defined, use the default ammo
330 if( !new_item.ammo_types().empty() ) {
331 new_item.ammo_set( new_item.ammo_default(), ch );
332 }
333 } else {
334 const item am = ammo->create_single( new_item.birthday() );
335 new_item.ammo_set( am.typeId(), ch );
336 }
337 // Make sure the item is in valid state
338 if( new_item.ammo_data() && new_item.magazine_integral() ) {
339 new_item.charges = std::min( new_item.charges, new_item.ammo_capacity() );
340 } else {
341 new_item.charges = 0;
342 }
343 }
344
345 if( new_item.is_tool() || new_item.is_gun() || new_item.is_magazine() ) {
346 bool spawn_ammo = rng( 0, 99 ) < with_ammo && new_item.ammo_remaining() == 0 && ch == -1 &&
347 ( !new_item.is_tool() || new_item.type->tool->rand_charges.empty() );
348 bool spawn_mag = rng( 0, 99 ) < with_magazine && !new_item.magazine_integral() &&
349 !new_item.magazine_current();
350
351 if( spawn_mag ) {
352 new_item.put_in( item( new_item.magazine_default(), new_item.birthday() ) );
353 }
354
355 if( spawn_ammo ) {
356 if( ammo ) {
357 const item am = ammo->create_single( new_item.birthday() );
358 new_item.ammo_set( am.typeId() );
359 } else {
360 new_item.ammo_set( new_item.ammo_default() );
361 }
362 }
363 }
364
365 if( !cont.is_null() ) {
366 cont.put_in( new_item );
367 new_item = cont;
368 }
369
370 if( contents != nullptr ) {
371 Item_spawn_data::ItemList contentitems = contents->create( new_item.birthday() );
372 for( const item &it : contentitems ) {
373 new_item.put_in( it );
374 }
375 }
376
377 for( auto &flag : custom_flags ) {
378 new_item.set_flag( flag );
379 }

477Item_spawn_data::ItemList Item_group::create( const time_point &birthday, RecursionList &rec ) const
478{
479 ItemList result;
480 if( type == G_COLLECTION ) {
481 for( const auto &elem : items ) {
482 if( rng( 0, 99 ) >= ( elem )->probability ) {
483 continue;
484 }
485 ItemList tmp = ( elem )->create( birthday, rec );
486 result.insert( result.end(), tmp.begin(), tmp.end() );
487 }
488 } else if( type == G_DISTRIBUTION ) {
489 int p = rng( 0, sum_prob - 1 );
490 for( const auto &elem : items ) {
491 p -= ( elem )->probability;
492 if( p >= 0 ) {
493 continue;
494 }
495 ItemList tmp = ( elem )->create( birthday, rec );
496 result.insert( result.end(), tmp.begin(), tmp.end() );
497 break;
498 }
499 }
500
501 return result;

632static item_group_id get_unique_group_id()
633{
634 // This is just a hint what id to use next. Overflow of it is defined and if the group
635 // name is already used, we simply go the next id.
636 static unsigned int next_id = 0;
637 // Prefixing with a character outside the ASCII range, so it is hopefully unique and
638 // (if actually printed somewhere) stands out. Theoretically those auto-generated group
639 // names should not be seen anywhere.
640 static const std::string unique_prefix = "\u01F7 ";
641 while( true ) {
642 const item_group_id new_group = item_group_id( unique_prefix + std::to_string( next_id++ ) );
643 if( !item_group::group_is_defined( new_group ) ) {
644 return new_group;
645 }
646 }

649item_group_id item_group::load_item_group( const JsonValue &value,
650 const std::string &default_subtype )
651{
652 if( value.test_string() ) {
653 return item_group_id( value.get_string() );
654 } else if( value.test_object() ) {
655 const item_group_id group = get_unique_group_id();
656
657 JsonObject jo = value.get_object();
658 const std::string subtype = jo.get_string( "subtype", default_subtype );
659 item_controller->load_item_group( jo, group, subtype );
660
661 return group;
662 } else if( value.test_array() ) {
663 const item_group_id group = get_unique_group_id();
664
665 JsonArray jarr = value.get_array();
666 // load_item_group needs a bool, invalid subtypes are unexpected and most likely errors
667 // from the caller of this function.
668 if( default_subtype != "collection" && default_subtype != "distribution" ) {
669 debugmsg( "invalid subtype for item group: %s", default_subtype.c_str() );
670 }
671 item_controller->load_item_group( jarr, group, default_subtype == "collection", 0, 0 );
672
673 return group;
674 } else {
675 value.throw_error( "invalid item group, must be string (group id) or object/array (the group data)" );
676 // stream.error always throws, this is here to prevent a warning
677 return item_group_id{};
