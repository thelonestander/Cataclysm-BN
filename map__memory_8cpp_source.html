120bool map_memory::prepare_region( const tripoint &p1, const tripoint &p2 )
121{
122 assert( p1.z == p2.z );
123 assert( p1.x <= p2.x && p1.y <= p2.y );
124
125 tripoint sm_p1 = coord_pair( p1 ).sm - point_south_east;
126 tripoint sm_p2 = coord_pair( p2 ).sm + point_south_east;
127
128 tripoint sm_pos = sm_p1;
129 point sm_size = sm_p2.xy() - sm_p1.xy();
130
131 if( sm_pos.z == cache_pos.z ) {
132 inclusive_rectangle<point> rect( cache_pos.xy(), cache_pos.xy() + cache_size );
133 if( rect.contains( sm_p1.xy() ) && rect.contains( sm_p2.xy() ) ) {
134 return false;
135 }
136 }
137
138 dbg( DL::Info ) << "Preparing memory map for area: pos: " << sm_pos << " size: " << sm_size;
139
140 cache_pos = sm_pos;
141 cache_size = sm_size;
142 cached.clear();
143 cached.reserve( cache_size.x * cache_size.y );
144 for( int dy = 0; dy < cache_size.y; dy++ ) {
145 for( int dx = 0; dx < cache_size.x; dx++ ) {
146 cached.push_back( fetch_submap( cache_pos + point( dx, dy ) ) );
147 }
148 }
149 return true;
150}

165shared_ptr_fast<mm_submap> map_memory::allocate_submap( const tripoint &sm_pos )
166{
167 // Since all save/load operations are done on regions of submaps,
168 // we need to allocate the whole region at once.
169 shared_ptr_fast<mm_submap> ret;
170 tripoint reg = reg_coord_pair( sm_pos ).reg;
171
172 dbg( DL::Info ) << "Allocated mm_region " << reg << " [" << mmr_to_sm_copy( reg ) << "]";
173
174 for( size_t y = 0; y < MM_REG_SIZE; y++ ) {
175 for( size_t x = 0; x < MM_REG_SIZE; x++ ) {
176 tripoint pos = mmr_to_sm_copy( reg ) + tripoint( x, y, 0 );
177 shared_ptr_fast<mm_submap> sm = make_shared_fast<mm_submap>();
178 if( pos == sm_pos ) {
179 ret = sm;
180 }
181 submaps.insert( std::make_pair( pos, sm ) );
182 }
183 }
184
185 return ret;

199{
200 if( test_mode ) {
201 return nullptr;
202 }
203
204 const std::string dirname = find_mm_dir();
205 reg_coord_pair p( sm_pos );
206 const std::string path = find_region_path( dirname, p.reg );
207
208 if( !dir_exist( dirname ) ) {
209 // Old saves don't have [plname].mm1 folder

277void map_memory::load( const tripoint &pos )
278{
279 const std::string dirname = find_mm_dir();
280
281 clear_cache();
282
283 if( !dir_exist( dirname ) ) {
284 // Old saves have [plname].mm file and no [plname].mm1 folder
285 const std::string legacy_file = find_legacy_mm_file();
286 if( file_exist( legacy_file ) ) {

292 debugmsg( "Failed to load legacy memory map file: %s", err.what() );
293 }
294 }

296 }
297
298 coord_pair p( pos );
299 tripoint start = p.sm - tripoint( MM_SIZE / 2, MM_SIZE / 2, 0 );
300 dbg( DL::Info ) << "[LOAD] Loading memory map around " << p.sm << ". Loading submaps within "
301 << start << "->" << start + tripoint( MM_SIZE, MM_SIZE, 0 );
302 for( int dy = 0; dy < MM_SIZE; dy++ ) {
303 for( int dx = 0; dx < MM_SIZE; dx++ ) {
304 fetch_submap( start + tripoint( dx, dy, 0 ) );
305 }
306 }
307 dbg( DL::Info ) << "[LOAD] Done.";

310bool map_memory::save( const tripoint &pos )
311{
312 tripoint sm_center = coord_pair( pos ).sm;
313 const std::string dirname = find_mm_dir();
314 assure_dir_exist( dirname );
315
316 clear_cache();
317
318 dbg( DL::Info ) << "N submaps before save: " << submaps.size();
319
320 // Since mm_submaps are always allocated in regions,
321 // we are certain that each region will be filled.
322 std::map<tripoint, mm_region> regions;
323 for( auto &it : submaps ) {
324 const reg_coord_pair p( it.first );
325 regions[p.reg].submaps[p.sm_loc.x][p.sm_loc.y] = it.second;
326 }
327 submaps.clear();
328
329 constexpr point MM_HSIZE_P = point( MM_SIZE / 2, MM_SIZE / 2 );
330 half_open_rectangle<point> rect_keep( sm_center.xy() - MM_HSIZE_P, sm_center.xy() + MM_HSIZE_P );
331
332 dbg( DL::Info ) << "[SAVE] Saving memory map around " << sm_center << ". Keeping submaps within "
333 << rect_keep.p_min << "->" << rect_keep.p_max;
334
335 bool result = true;
336
337 for( auto &it : regions ) {
338 const tripoint &regp = it.first;
339 mm_region &reg = it.second;
340 if( !reg.is_empty() ) {
341 const std::string path = find_region_path( dirname, regp );
342 const std::string descr = string_format(
343 _( "memory map region for (%d,%d,%d)" ),
344 regp.x, regp.y, regp.z
345 );
346
347 const auto writer = [&]( std::ostream & fout ) -> void {
348 fout << serialize_wrapper( [&]( JsonOut & jsout )
349 {
350 reg.serialize( jsout );
351 } );
352 };
353
354 const bool res = write_to_file( path, writer, descr.c_str() );
355 result = result & res;
356 }
357 tripoint regp_sm = mmr_to_sm_copy( regp );
358 half_open_rectangle<point> rect_reg(
359 regp_sm.xy(),
360 regp_sm.xy() + point( MM_REG_SIZE, MM_REG_SIZE )
361 );
362 if( rect_reg.overlaps( rect_keep ) ) {
363 dbg( DL::Info ) << "Keeping mm_region " << regp << " [" << regp_sm << "]";
364 // Put submaps back
365 for( size_t y = 0; y < MM_REG_SIZE; y++ ) {
366 for( size_t x = 0; x < MM_REG_SIZE; x++ ) {
367 tripoint p = regp_sm + tripoint( x, y, 0 );
368 shared_ptr_fast<mm_submap> &sm = reg.submaps[x][y];
369 submaps.insert( std::make_pair( p, sm ) );
370 }
371 }
372 } else {
373 dbg( DL::Info ) << "Dropping mm_region " << regp << " [" << regp_sm << "]";
374 }
375 }
376
377 dbg( DL::Info ) << "[SAVE] Done.";
378 dbg( DL::Info ) << "N submaps after save: " << submaps.size();
379
380 return result;
381}
