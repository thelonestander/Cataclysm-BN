110void computer::load_legacy_data( const std::string &data )
111{
112 options.clear();
113 failures.clear();
114
115 std::istringstream dump( data );
116 dump.imbue( std::locale::classic() );
117
118 dump >> name >> security >> mission_id;
119
120 name = replace_all( name, "_", " " );
121
122 // Pull in options
123 int optsize;
124 dump >> optsize;
125 for( int n = 0; n < optsize; n++ ) {
126 std::string tmpname;
127
128 int tmpaction;
129 int tmpsec;
130
131 dump >> tmpname >> tmpaction >> tmpsec;
132 // Legacy missle launch option that got removed before `computer_action` was
133 // refactored to be saved and loaded as string ids. Do not change this number:
134 // `computer_action` now has different underlying values from back then!
135 if( tmpaction == 15 ) {
136 continue;
137 }
138 add_option( replace_all( tmpname, "_", " " ), computer_action_from_legacy_enum( tmpaction ),
139 tmpsec );
140 }
141
142 // Pull in failures
143 int failsize;
144 dump >> failsize;
145 for( int n = 0; n < failsize; n++ ) {
146 int tmpfail;
147 dump >> tmpfail;
148 add_failure( computer_failure_type_from_legacy_enum( tmpfail ) );
149 }
150
151 std::string tmp_access_denied;
152 dump >> tmp_access_denied;
153
154 // For backwards compatibility, only set the access denied message if it
155 // isn't empty. This is to avoid the message becoming blank when people
156 // load old saves.
157 if( !tmp_access_denied.empty() ) {
158 access_denied = replace_all( tmp_access_denied, "_", " " );
159 }
160}
