994{
995 if( !warm_enough_to_plant( g->u.pos() ) ) {
996 popup( _( "It is too cold to plant anything now." ) );

999 std::vector<item *> seed_inv = g->u.items_with( []( const item & itm ) {
1000 return itm.is_seed() && itm.typeId() != itype_marloss_seed &&
1001 itm.typeId() != itype_fungal_seeds;
1002 } );
1003 if( seed_inv.empty() ) {
1004 popup( _( "You have no seeds to plant!" ) );
1005 return;
1006 }
1007
1008 int empty_plots = 0;
1009 int free_seeds = 0;
1010
1011 std::vector<itype_id> seed_types;
1012 std::vector<std::string> seed_names;
1013 for( auto &seed : seed_inv ) {
1014 if( std::find( seed_types.begin(), seed_types.end(), seed->typeId() ) == seed_types.end() ) {
1015 seed_types.push_back( seed->typeId() );
1016 seed_names.push_back( seed->tname() );
1017 }
1018 }
1019 // Choose seed if applicable
1020 const int seed_index = uilist( _( "Which seeds do you wish to have planted?" ), seed_names );
1021 // Did we cancel?
1022 if( seed_index < 0 || static_cast<size_t>( seed_index ) >= seed_types.size() ) {
1023 popup( _( "You saved your seeds for later." ) );
1024 return;
1025 }
1026
1027 const auto &seed_id = seed_types[seed_index];
1028 if( item::count_by_charges( seed_id ) ) {
1029 free_seeds = g->u.charges_of( seed_id );
1030 } else {
1031 free_seeds = g->u.amount_of( seed_id );
1032 }
1033
1034 //Now we need to find how many free plots we have to plant in...
1035 const tripoint_abs_omt site = overmap_buffer.find_closest(
1036 g->u.global_omt_location(), place, 20, false );
1037 tinymap bay;
1038 bay.load( project_to<coords::sm>( site ), false );
1039 for( const tripoint &plot : bay.points_on_zlevel() ) {
1040 if( bay.ter( plot ) == t_dirtmound ) {
1041 empty_plots++;
1042 }
1043 }
1044
1045 if( empty_plots == 0 ) {
1046 popup( _( "You have no room to plant seeds…" ) );
1047 return;
1048 }
1049
1050 int limiting_number = free_seeds;
1051 if( free_seeds > empty_plots ) {
1052 limiting_number = empty_plots;
1053 }
1054
1055 signed int a = limiting_number * 300;
1056 if( a > g->u.cash ) {
1057 popup( _( "I'm sorry, you don't have enough money to plant those seeds…" ) );
1058 return;
1059 }
1060 if( !query_yn( _( "Do you wish to have %d %s planted here for $%d?" ), limiting_number,
1061 seed_names[seed_index], limiting_number * 3 ) ) {
1062 return;
1063 }
1064
1065 //Plant the actual seeds
1066 for( const tripoint &plot : bay.points_on_zlevel() ) {
1067 if( bay.ter( plot ) == t_dirtmound && limiting_number > 0 ) {
1068 std::list<item> used_seed;
1069 if( item::count_by_charges( seed_id ) ) {
1070 used_seed = g->u.use_charges( seed_id, 1 );
1071 } else {
1072 used_seed = g->u.use_amount( seed_id, 1 );
1073 }
1074 used_seed.front().set_age( 0_turns );
1075 bay.add_item_or_charges( plot, used_seed.front() );
1076 bay.set( plot, t_dirt, f_plant_seed );
1077 limiting_number--;
1078 }
1079 }

1081 bay.save();
1082 popup( _( "After counting your money and collecting your seeds, %s calls forth a labor party "
1083 "to plant your field." ), p.name );
