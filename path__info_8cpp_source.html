119std::string find_translated_file( const std::string &base_path, const std::string &extension,
120 const std::string &fallback )
121{
122 std::vector<std::string> opts = get_lang_path_substring( get_language().id );
123
124 for( const std::string &s : opts ) {
125 const std::string local_path = base_path + s + extension;
126 if( file_exist( local_path ) ) {
127 return local_path;
128 }
129 }
130
131 return fallback;
132}
