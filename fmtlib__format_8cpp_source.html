14template <typename T>
15int format_float( char *buf, std::size_t size, const char *format, int precision,
16 T value )
17{
18#ifdef FMT_FUZZ
19 if( precision > 100000 )
20 throw std::runtime_error(
21 "fuzz mode - avoid large allocation inside snprintf" );
22#endif
23 // Suppress the warning about nonliteral format string.
24 int ( *snprintf_ptr )( char *, size_t, const char *, ... ) = FMT_SNPRINTF;
25 return precision < 0 ? snprintf_ptr( buf, size, format, value )
26 : snprintf_ptr( buf, size, format, precision, value );
27}
