36constexpr int map_squares_per( scale s )
37{
38 static_assert( SEEX == SEEY, "we assume submaps are square" );
39 static_assert( OMAPX == OMAPY, "we assume overmaps are square" );
40
41 switch( s ) {
42 case scale::map_square:
43 return 1;
44 case scale::submap:
45 return SEEX;
46 case scale::overmap_terrain:
47 return SEEX * 2;
48 case scale::segment:
49 return SEG_SIZE * map_squares_per( scale::overmap_terrain );
50 case scale::overmap:
51 return OMAPX * map_squares_per( scale::overmap_terrain );
52 default:
53 constexpr_fatal( 0, "Requested scale of %d", s );
54 }
55}

306template<scale ResultScale, typename Point, origin Origin, scale SourceScale>
307inline coord_point<Point, Origin, ResultScale> project_to(
308 const coord_point<Point, Origin, SourceScale> &src )
309{
310 constexpr int scale_down = map_squares_per( ResultScale ) / map_squares_per( SourceScale );
311 constexpr int scale_up = map_squares_per( SourceScale ) / map_squares_per( ResultScale );
312 return project_to_impl<scale_up, scale_down, ResultScale>()( src );
313}

316struct quotient_remainder_helper {
317 constexpr static origin RemainderOrigin = origin_from_scale( CoarseScale );
318 using quotient_type = coord_point<point, Origin, CoarseScale>;
319 using quotient_type_tripoint = coord_point<tripoint, Origin, CoarseScale>;

321 using remainder_type_tripoint = coord_point<tripoint, RemainderOrigin, FineScale>;

324template<origin Origin, scale CoarseScale, scale FineScale>
325struct quotient_remainder_point {
326 using helper = quotient_remainder_helper<Origin, CoarseScale, FineScale>;
327 using quotient_type = typename helper::quotient_type;
328 using remainder_type = typename helper::remainder_type;
329
330 quotient_type quotient;
331 remainder_type remainder;
332
333 // For assigning to std::tie( q, r );
334 operator std::tuple<quotient_type &, remainder_type &>() {
335 return std::tie( quotient, remainder );
336 }
337};
338
339template<origin Origin, scale CoarseScale, scale FineScale>
340struct quotient_remainder_tripoint {
341 using helper = quotient_remainder_helper<Origin, CoarseScale, FineScale>;
342 using quotient_type = typename helper::quotient_type;
343 using quotient_type_tripoint = typename helper::quotient_type_tripoint;
344 using remainder_type = typename helper::remainder_type;
345 using remainder_type_tripoint = typename helper::remainder_type_tripoint;
346
347 // Annoyingly, for the conversion operators below to work correctly, we
348 // need to have point and tripoint version of both quotient and remainder
349 // ready here, so that we can take references to any of them.
350 // Luckily the entire lifetime of this struct should be pretty short, so
351 // the compiler can do its magic to remove the overhead of initializing the
352 // ones that don't actually get used.
353 quotient_type quotient;
354 remainder_type remainder;
355 quotient_type_tripoint quotient_tripoint;
356 remainder_type_tripoint remainder_tripoint;
357
358 // For assigning to std::tie( q, r );
359 // Exactly one of the two resulting types should be a tripoint, so that the
360 // z-coordinate doesn't get duplicated.
361 operator std::tuple<quotient_type_tripoint &, remainder_type &>() {
362 return std::tie( quotient_tripoint, remainder );
363 }
364 operator std::tuple<quotient_type &, remainder_type_tripoint &>() {
365 return std::tie( quotient, remainder_tripoint );
366 }
367};

369// project_remain returns a helper struct, intended to be used with std::tie
370// to pull out the two components of the result.
371// For exmaple, when splitting a point:
372// point_abs_sm val;
373// point_abs_om quotient;
374// point_om_sm remainder;
375// std::tie( quotient, remainder ) = project_remain<coords::om>( val );
376// If passing a tripoint to project_remain, you must choose exactly one of the
377// quotient or remainder to get the z coordinate. Both of these should work:
378// tripoint_abs_sm val;
379// tripoint_abs_om quotient;
380// point_om_sm remainder;
381// std::tie( quotient, remainder ) = project_remain<coords::om>( val );
382//
383// point_abs_om quotient;
384// tripoint_om_sm remainder;
385// std::tie( quotient, remainder ) = project_remain<coords::om>( val );
386template<scale ResultScale, origin Origin, scale SourceScale>
387inline quotient_remainder_point<Origin, ResultScale, SourceScale> project_remain(
388 const coord_point<point, Origin, SourceScale> &src )
389{
390 constexpr int ScaleDown = map_squares_per( ResultScale ) / map_squares_per( SourceScale );
391 static_assert( ScaleDown > 0, "You can only project to coarser coordinate systems" );
392 constexpr static origin RemainderOrigin = origin_from_scale( ResultScale );
393 coord_point<point, Origin, ResultScale> quotient(
394 divide_xy_round_to_minus_infinity( src.raw(), ScaleDown ) );
395 coord_point<point, RemainderOrigin, SourceScale> remainder(
396 src.raw() - quotient.raw() * ScaleDown );
397
398 return { quotient, remainder };
399}
400
401template<scale ResultScale, origin Origin, scale SourceScale>
402inline quotient_remainder_tripoint<Origin, ResultScale, SourceScale> project_remain(
403 const coord_point<tripoint, Origin, SourceScale> &src )
404{
405 quotient_remainder_point<Origin, ResultScale, SourceScale> point_result =
406 project_remain<ResultScale>( src.xy() );
407 return { point_result.quotient, point_result.remainder,
408 { point_result.quotient, src.z() }, { point_result.remainder, src.z() } };
409}

411template<typename PointL, typename PointR, origin CoarseOrigin, scale CoarseScale,
412 origin FineOrigin, scale FineScale>
413inline auto project_combine(
414 const coord_point<PointL, CoarseOrigin, CoarseScale> &coarse,
415 const coord_point<PointR, FineOrigin, FineScale> &fine )
416{
417 static_assert( origin_from_scale( CoarseScale ) == FineOrigin,
418 "given point types are not compatible for combination" );
419 static_assert( PointL::dimension != 3 || PointR::dimension != 3,
420 "two tripoints should not be combined; it's unclear how to handle z" );
421 using PointResult = decltype( PointL() + PointR() );
422 const coord_point<PointL, CoarseOrigin, FineScale> refined_coarse =
423 project_to<FineScale>( coarse );
424 return coord_point<PointResult, CoarseOrigin, FineScale>( refined_coarse.raw() + fine.raw() );
425}

616/* find appropriate subdivided coordinates for absolute tile coordinate.
617 * This is less obvious than one might think, for negative coordinates, so this
618 * was created to give a definitive answer.
619 *
620 * 'absolute' is defined as the -actual- submap x,y * SEEX + position in submap, and
621 * can be obtained from map.getabs(x, y);
622 * usage:
623 * real_coords rc( g->m.getabs(g->u.posx(), g->u.posy() ) );
624 */
625struct real_coords {
626 static const int tiles_in_sub = SEEX;
627 static const int tiles_in_sub_n = tiles_in_sub - 1;

646 void fromabs( const point &abs ) {
647 const point norm( std::abs( abs.x ), std::abs( abs.y ) );
648 abs_pos = abs;
649
650 if( abs.x < 0 ) {
651 abs_sub.x = ( abs.x - SEEX + 1 ) / SEEX;
652 sub_pos.x = SEEX - 1 - ( ( norm.x - 1 ) % SEEX );
653 abs_om.x = ( abs_sub.x - subs_in_om_n ) / subs_in_om;
654 om_sub.x = subs_in_om_n - ( ( ( norm.x - 1 ) / SEEX ) % subs_in_om );
655 } else {
656 abs_sub.x = norm.x / SEEX;
657 sub_pos.x = abs.x % SEEX;
658 abs_om.x = abs_sub.x / subs_in_om;
659 om_sub.x = abs_sub.x % subs_in_om;
660 }
661 om_pos.x = om_sub.x / 2;
662
663 if( abs.y < 0 ) {
664 abs_sub.y = ( abs.y - SEEY + 1 ) / SEEY;
665 sub_pos.y = SEEY - 1 - ( ( norm.y - 1 ) % SEEY );
666 abs_om.y = ( abs_sub.y - subs_in_om_n ) / subs_in_om;
667 om_sub.y = subs_in_om_n - ( ( ( norm.y - 1 ) / SEEY ) % subs_in_om );
668 } else {
669 abs_sub.y = norm.y / SEEY;
670 sub_pos.y = abs.y % SEEY;
671 abs_om.y = abs_sub.y / subs_in_om;
672 om_sub.y = abs_sub.y % subs_in_om;
673 }
674 om_pos.y = om_sub.y / 2;
675 }

678 void fromomap( const point &rel_om, const point &rel_om_pos ) {
679 const point a = om_to_omt_copy( rel_om ) + rel_om_pos;
680 fromabs( omt_to_ms_copy( a ) );
681 }

683 point_abs_omt abs_omt() const {
684 return project_to<coords::omt>( point_abs_sm( abs_sub ) );
685 }

689 point begin_sub() {
690 return point( abs_sub.x * tiles_in_sub, abs_sub.y * tiles_in_sub );
691 }
692 point begin_om_pos() {
693 return point( ( abs_om.x * subs_in_om * tiles_in_sub ) + ( om_pos.x * 2 * tiles_in_sub ),
694 ( abs_om.y * subs_in_om * tiles_in_sub ) + ( om_pos.y * 2 * tiles_in_sub ) );
695 }
696 point begin_om() {
697 return point( abs_om.x * subs_in_om * tiles_in_sub, abs_om.y * subs_in_om * tiles_in_sub );
698 }
