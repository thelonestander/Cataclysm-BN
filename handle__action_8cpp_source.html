147 int moves_elapsed() {
148 const float turn_duration = get_option<float>( "TURN_DURATION" );
149 // Magic number 0.005 chosen due to option menu's 2 digit precision and
150 // the option menu UI rounding <= 0.005 down to "0.00" in the display.
151 // This conditional will catch values (e.g. 0.003) that the options menu
152 // would round down to "0.00" in the options menu display. This prevents
153 // the user from being surprised by floating point rounding near zero.
154 if( turn_duration <= 0.005 ) {
155 return 0;
156 }
157 auto now = std::chrono::steady_clock::now();
158 std::chrono::milliseconds elapsed_ms =
159 std::chrono::duration_cast<std::chrono::milliseconds>( now - user_turn_start );
160 return elapsed_ms.count() / ( 10.0 * turn_duration );
161 }

439static void pldrive( const tripoint &p )
440{
441 if( !g->check_safe_mode_allowed() ) {
442 return;
443 }
444 player &u = g->u;
445 vehicle *veh = g->remoteveh();
446 bool remote = true;
447 int part = -1;
448 if( !veh ) {
449 if( const optional_vpart_position vp = g->m.veh_at( u.pos() ) ) {
450 veh = &vp->vehicle();
451 part = vp->part_index();
452 }
453 remote = false;
454 }
455 if( !veh ) {
456 debugmsg( "game::pldrive error: can't find vehicle! Drive mode is now off." );
457 u.in_vehicle = false;
458 return;
459 }
460 if( veh->is_on_ramp && p.x != 0 ) {
461 add_msg( m_bad, _( "You can't turn the vehicle while on a ramp." ) );
462 return;
463 }
464 if( !remote ) {
465 static const itype_id fuel_type_animal( "animal" );
466 const bool has_animal_controls = veh->part_with_feature( part, "CONTROL_ANIMAL", true ) >= 0;
467 const bool has_controls = veh->part_with_feature( part, "CONTROLS", true ) >= 0;
468 const bool has_animal = veh->has_engine_type( fuel_type_animal, false ) &&
469 veh->has_harnessed_animal();
470 if( !has_controls && !has_animal_controls ) {
471 add_msg( m_info, _( "You can't drive the vehicle from here. You need controls!" ) );
472 u.controlling_vehicle = false;
473 return;
474 } else if( !has_controls && has_animal_controls && !has_animal ) {
475 add_msg( m_info, _( "You can't drive this vehicle without an animal to pull it." ) );
476 u.controlling_vehicle = false;
477 return;
478 }
479 } else {
480 if( empty( veh->get_avail_parts( "REMOTE_CONTROLS" ) ) ) {
481 add_msg( m_info, _( "Can't drive this vehicle remotely. It has no working controls." ) );
482 return;
483 }
484 }
485 if( p.z != 0 && !g->m.has_zlevels() ) {
486 u.add_msg_if_player( m_info, _( "This vehicle doesn't look very airworthy." ) );
487 return;
488 }
489 if( p.z == -1 ) {
490 if( veh->check_heli_descend( u ) ) {
491 u.add_msg_if_player( m_info, _( "You steer the vehicle into a descent." ) );
492 } else {
493 return;
494 }
495 } else if( p.z == 1 ) {
496 if( veh->check_heli_ascend( u ) ) {
497 u.add_msg_if_player( m_info, _( "You steer the vehicle into an ascent." ) );
498 } else {
499 return;
500 }
501 }
502 veh->pldrive( get_avatar(), p.xy(), p.z );
503}

511{

516 pgettext( "no door, gate, curtain, etc.", "There is nothing that can be opened nearby." ),
517 ACTION_OPEN, false );
518

523
524 u.moves -= 100;
525
526 if( const optional_vpart_position vp = m.veh_at( openp ) ) {
527 vehicle *const veh = &vp->vehicle();
528 int openable = veh->next_part_to_open( vp->part_index() );
529 if( openable >= 0 ) {
530 const vehicle *player_veh = veh_pointer_or_null( m.veh_at( u.pos() ) );
531 bool outside = !player_veh || player_veh != veh;
532 if( !outside ) {
533 if( !veh->handle_potential_theft( dynamic_cast<player &>( g->u ) ) ) {
534 u.moves += 100;
535 return;
536 } else {
537 veh->open( openable );
538 }
539 } else {
540 // Outside means we check if there's anything in that tile outside-openable.
541 // If there is, we open everything on tile. This means opening a closed,
542 // curtained door from outside is possible, but it will magically open the
543 // curtains as well.
544 int outside_openable = veh->next_part_to_open( vp->part_index(), true );
545 if( outside_openable == -1 ) {
546 const std::string name = veh->part_info( openable ).name();
547 add_msg( m_info, _( "That %s can only opened from the inside." ), name );
548 u.moves += 100;
549 } else {
550 if( !veh->handle_potential_theft( dynamic_cast<player &>( g->u ) ) ) {
551 u.moves += 100;
552 return;
553 } else {
554 veh->open_all_at( openable );
555 }
556 }
557 }
558 } else {
559 // If there are any OPENABLE parts here, they must be already open
560 if( const cata::optional<vpart_reference> already_open = vp.part_with_feature( "OPENABLE",
561 true ) ) {
562 const std::string name = already_open->info().name();
563 add_msg( m_info, _( "That %s is already open." ), name );
564 }
565 u.moves += 100;
566 }
567 return;
568 }
569
570 bool didit = m.open_door( openp, !m.is_outside( u.pos() ) );
571
572 if( !didit ) {
573 const ter_str_id tid = m.ter( openp ).id();
574
575 if( m.has_flag( flag_LOCKED, openp ) ) {
576 add_msg( m_info, _( "The door is locked!" ) );

578 } else if( tid.obj().close ) {
579 // if the following message appears unexpectedly, the prior check was for t_door_o
580 add_msg( m_info, _( "That door is already open." ) );
581 u.moves += 100;
582 return;
583 }
584 add_msg( m_info, _( "No door there." ) );
585 u.moves += 100;
586 }

1409static void cast_spell()
1410{
1411 player &u = g->u;
1412
1413 std::vector<spell_id> spells = u.magic->spells();
1414
1415 if( spells.empty() ) {
1416 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1417 _( "You don't know any spells to cast." ) );
1418 return;
1419 }
1420
1421 bool can_cast_spells = false;
1422 for( spell_id sp : spells ) {
1423 spell temp_spell = u.magic->get_spell( sp );
1424 if( temp_spell.can_cast( u ) ) {
1425 can_cast_spells = true;
1426 }
1427 }
1428
1429 if( !can_cast_spells ) {
1430 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1431 _( "You can't cast any of the spells you know!" ) );
1432 }
1433
1434 const int spell_index = u.magic->select_spell( u );
1435 if( spell_index < 0 ) {
1436 return;
1437 }
1438
1439 spell &sp = *u.magic->get_spells()[spell_index];
1440
1441 if( u.is_armed() && !sp.has_flag( spell_flag::NO_HANDS ) &&
1442 !u.weapon.has_flag( flag_MAGIC_FOCUS ) ) {
1443 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1444 _( "You need your hands free to cast this spell!" ) );
1445 return;
1446 }
1447
1448 if( !u.magic->has_enough_energy( u, sp ) ) {
1449 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1450 _( "You don't have enough %s to cast the spell." ),
1451 sp.energy_string() );
1452 return;
1453 }
1454
1455 if( sp.energy_source() == hp_energy && !u.has_quality( qual_CUT ) ) {
1456 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1457 _( "You cannot cast Blood Magic without a cutting implement." ) );
1458 return;
1459 }
1460
1461 player_activity cast_spell( ACT_SPELLCASTING, sp.casting_time( u ) );
1462 // [0] this is used as a spell level override for items casting spells
1463 cast_spell.values.emplace_back( -1 );
1464 // [1] if this value is 1, the spell never fails
1465 cast_spell.values.emplace_back( 0 );
1466 // [2] this value overrides the mana cost if set to 0
1467 cast_spell.values.emplace_back( 1 );
1468 cast_spell.name = sp.id().c_str();
1469 if( u.magic->casting_ignore ) {
1470 const std::vector<distraction_type> ignored_distractions = {
1471 distraction_type::noise,
1472 distraction_type::pain,
1473 distraction_type::attacked,
1474 distraction_type::hostile_spotted_near,
1475 distraction_type::hostile_spotted_far,
1476 distraction_type::talked_to,
1477 distraction_type::asthma,
1478 distraction_type::motion_alarm,
1479 distraction_type::weather_change
1480 };
1481 for( const distraction_type ignored : ignored_distractions ) {
1482 cast_spell.ignore_distraction( ignored );
1483 }
1484 }
1485 u.assign_activity( cast_spell, false );
1486}

1514bool game::handle_action()
1515{
1516 std::string action;
1517 input_context ctxt;
1518 action_id act = ACTION_NULL;
1519 user_turn current_turn;
1520 // Check if we have an auto-move destination
1521 if( u.has_destination() ) {
1522 act = u.get_next_auto_move_direction();
1523 if( act == ACTION_NULL ) {
1524 add_msg( m_info, _( "Auto-move canceled" ) );
1525 u.clear_destination();
1526 return false;
1527 }
1528 } else if( u.has_destination_activity() ) {
1529 // starts destination activity after the player successfully reached his destination
1530 u.start_destination_activity();
1531 return false;
1532 } else {
1533 // No auto-move, ask player for input
1534 ctxt = get_player_input( action );
1535 }
1536
1537 const optional_vpart_position vp = m.veh_at( u.pos() );
1538 bool veh_ctrl = !u.is_dead_state() &&
1539 ( ( vp && vp->vehicle().player_in_control( u ) ) || remoteveh() != nullptr );
1540
1541 // If performing an action with right mouse button, co-ordinates
1542 // of location clicked.
1543 cata::optional<tripoint> mouse_target;
1544
1545 if( uquit == QUIT_WATCH && action == "QUIT" ) {
1546 uquit = QUIT_DIED;
1547 return false;
1548 }
1549
1550 if( act == ACTION_NULL ) {
1551 act = look_up_action( action );
1552
1553 if( act == ACTION_KEYBINDINGS ) {
1554 // already handled by input context
1555 return false;
1556 }
1557
1558 if( act == ACTION_MAIN_MENU ) {
1559 if( uquit == QUIT_WATCH ) {
1560 return false;
1561 }
1562 // No auto-move actions have or can be set at this point.
1563 u.clear_destination();
1564 destination_preview.clear();
1565 act = handle_main_menu();
1566 if( act == ACTION_NULL ) {
1567 return false;
1568 }
1569 }
1570
1571 if( act == ACTION_ACTIONMENU ) {
1572 if( uquit == QUIT_WATCH ) {
1573 return false;
1574 }
1575 // No auto-move actions have or can be set at this point.
1576 u.clear_destination();
1577 destination_preview.clear();
1578 act = handle_action_menu();
1579 if( act == ACTION_NULL ) {
1580 return false;
1581 }
1582#if defined(__ANDROID__)
1583 if( get_option<bool>( "ANDROID_ACTIONMENU_AUTOADD" ) && ctxt.get_category() == "DEFAULTMODE" ) {
1584 add_best_key_for_action_to_quick_shortcuts( act, ctxt.get_category(), false );
1585 }
1586#endif
1587 }
1588
1589 if( act == ACTION_KEYBINDINGS ) {
1590 u.clear_destination();
1591 destination_preview.clear();
1592 act = ctxt.display_menu( true );
1593 if( act == ACTION_NULL ) {
1594 return false;
1595 }
1596 }
1597
1598 if( can_action_change_worldstate( act ) ) {
1599 user_action_counter += 1;
1600 }
1601
1602 if( act == ACTION_SELECT || act == ACTION_SEC_SELECT ) {
1603 // Mouse button click
1604 if( veh_ctrl ) {
1605 // No mouse use in vehicle
1606 return false;
1607 }
1608
1609 if( u.is_dead_state() ) {
1610 // do not allow mouse actions while dead
1611 return false;
1612 }
1613
1614 const cata::optional<tripoint> mouse_pos = ctxt.get_coordinates( w_terrain );
1615 if( !mouse_pos ) {
1616 return false;
1617 } else if( !u.sees( *mouse_pos ) ) {
1618 // Not clicked in visible terrain
1619 return false;
1620 }
1621 mouse_target = mouse_pos;
1622
1623 if( act == ACTION_SELECT ) {
1624 // Note: The following has the potential side effect of
1625 // setting auto-move destination state in addition to setting
1626 // act.
1627 if( !try_get_left_click_action( act, *mouse_target ) ) {
1628 return false;
1629 }
1630 } else if( act == ACTION_SEC_SELECT ) {
1631 if( !try_get_right_click_action( act, *mouse_target ) ) {
1632 return false;
1633 }
1634 }
1635 } else if( act != ACTION_TIMEOUT ) {
1636 // act has not been set for an auto-move, so clearing possible
1637 // auto-move destinations. Since initializing an auto-move with
1638 // the mouse may span across multiple actions, we do not clear the
1639 // auto-move destination if the action is only a timeout, as this
1640 // would require the user to double click quicker than the
1641 // timeout delay.
1642 u.clear_destination();
1643 destination_preview.clear();
1644 }
1645 }
1646
1647 if( act == ACTION_NULL ) {
1648 const input_event &&evt = ctxt.get_raw_input();
1649 if( !evt.sequence.empty() ) {
1650 const int ch = evt.get_first_input();
1651 const std::string &&name = inp_mngr.get_keyname( ch, evt.type, true );
1652 if( !get_option<bool>( "NO_UNKNOWN_COMMAND_MSG" ) ) {
1653 add_msg( m_info, _( "Unknown command: \"%s\" (%ld)" ), name, ch );
1654 if( const cata::optional<std::string> hint =
1655 press_x_if_bound( ACTION_KEYBINDINGS ) ) {
1656 add_msg( m_info, _( "%s at any time to see and edit keybindings relevant to "
1657 "the current context." ),
1658 *hint );
1659 }
1660 }
1661 }
1662 return false;
1663 }
1664
1665 // This has no action unless we're in a special game mode.
1666 gamemode->pre_action( act );
1667
1668 int soffset = get_option<int>( "MOVE_VIEW_OFFSET" );
1669
1670 int before_action_moves = u.moves;
1671
1672 // These actions are allowed while deathcam is active. Registered in game::get_player_input
1673 if( uquit == QUIT_WATCH || !u.is_dead_state() ) {
1674 switch( act ) {
1675 case ACTION_TOGGLE_MAP_MEMORY:
1676 u.toggle_map_memory();
1677 break;
1678
1679 case ACTION_CENTER:
1680 u.view_offset.x = driving_view_offset.x;
1681 u.view_offset.y = driving_view_offset.y;
1682 break;
1683
1684 case ACTION_SHIFT_N:
1685 case ACTION_SHIFT_NE:
1686 case ACTION_SHIFT_E:
1687 case ACTION_SHIFT_SE:
1688 case ACTION_SHIFT_S:
1689 case ACTION_SHIFT_SW:
1690 case ACTION_SHIFT_W:
1691 case ACTION_SHIFT_NW: {
1692 static const std::map<action_id, std::pair<point, point>> shift_delta = {
1693 { ACTION_SHIFT_N, { point_north, point_north_east } },
1694 { ACTION_SHIFT_NE, { point_north_east, point_east } },
1695 { ACTION_SHIFT_E, { point_east, point_south_east } },
1696 { ACTION_SHIFT_SE, { point_south_east, point_south } },
1697 { ACTION_SHIFT_S, { point_south, point_south_west } },
1698 { ACTION_SHIFT_SW, { point_south_west, point_west } },
1699 { ACTION_SHIFT_W, { point_west, point_north_west } },
1700 { ACTION_SHIFT_NW, { point_north_west, point_north } },
1701 };

1703 shift_delta.at( act ).second * soffset : shift_delta.at( act ).first * soffset;
1704 }
1705 break;
1706

1708 look_around();
1709 break;
1710
1711 case ACTION_KEYBINDINGS:
1712 // already handled by input context
1713 break;
1714
1715 default:
1716 break;
1717 }
1718 }
1719
1720 // actions allowed only while alive
1721 if( !u.is_dead_state() ) {
1722 switch( act ) {
1723 case ACTION_NULL:
1724 case NUM_ACTIONS:
1725 break; // dummy entries
1726 case ACTION_ACTIONMENU:
1727 case ACTION_MAIN_MENU:
1728 case ACTION_KEYBINDINGS:
1729 break; // handled above
1730
1731 case ACTION_TIMEOUT:
1732 if( check_safe_mode_allowed( false ) ) {
1733 u.pause();
1734 }
1735 break;
1736
1737 case ACTION_PAUSE:
1738 if( check_safe_mode_allowed() ) {
1739 u.pause();
1740 }
1741 break;

1745 break;
1746

1749 break;

1753 break;
1754
1755 case ACTION_TOGGLE_CROUCH:
1756 u.toggle_crouch_mode();
1757 break;
1758
1759 case ACTION_OPEN_MOVEMENT:
1760 open_movement_mode_menu();
1761 break;
1762
1763 case ACTION_MOVE_FORTH:
1764 case ACTION_MOVE_FORTH_RIGHT:
1765 case ACTION_MOVE_RIGHT:
1766 case ACTION_MOVE_BACK_RIGHT:
1767 case ACTION_MOVE_BACK:
1768 case ACTION_MOVE_BACK_LEFT:

1784 u.posy() + dest_delta.y * ( SEEX - i ),

1820 pldrive( tripoint_below );

1835 pldrive( tripoint_above );
