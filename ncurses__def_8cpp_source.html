251input_event input_manager::get_input_event()
252{
253 int key = ERR;
254 input_event rval;
255 do {
256 previously_pressed_key = 0;
257 // flush any output
258 catacurses::doupdate();
259 key = getch();
260 if( key != ERR ) {
261 int newch;
262 // Clear the buffer of characters that match the one we're going to act on.
263 const int prev_timeout = input_timeout;
264 set_timeout( 0 );
265 do {
266 newch = getch();
267 } while( newch != ERR && newch == key );
268 set_timeout( prev_timeout );
269 // If we read a different character than the one we're going to act on, re-queue it.
270 if( newch != ERR && newch != key ) {
271 ungetch( newch );
272 }
273 }
274 rval = input_event();
275 if( key == ERR ) {
276 if( input_timeout > 0 ) {
277 rval.type = CATA_INPUT_TIMEOUT;
278 } else {
279 rval.type = CATA_INPUT_ERROR;
280 }
281 // ncurses mouse handling
282 } else if( key == KEY_RESIZE ) {
283 catacurses::resizeterm();
284 } else if( key == KEY_MOUSE ) {
285 MEVENT event;
286 if( getmouse( &event ) == OK ) {
287 rval.type = CATA_INPUT_MOUSE;
288 rval.mouse_pos = point( event.x, event.y );
289 if( event.bstate & BUTTON1_CLICKED ) {
290 rval.add_input( MOUSE_BUTTON_LEFT );
291 } else if( event.bstate & BUTTON3_CLICKED ) {
292 rval.add_input( MOUSE_BUTTON_RIGHT );
293 } else if( event.bstate & REPORT_MOUSE_POSITION ) {
294 rval.add_input( MOUSE_MOVE );
295 if( input_timeout > 0 ) {
296 // Mouse movement seems to clear ncurses timeout
297 set_timeout( input_timeout );
298 }
299 } else {
300 rval.type = CATA_INPUT_ERROR;
301 }
302 } else {
303 rval.type = CATA_INPUT_ERROR;
304 }
305 } else {
306 if( key == 127 ) { // == Unicode DELETE
307 previously_pressed_key = KEY_BACKSPACE;
308 return input_event( KEY_BACKSPACE, CATA_INPUT_KEYBOARD );
309 }
310 rval.type = CATA_INPUT_KEYBOARD;
311 rval.text.append( 1, static_cast<char>( key ) );
312 // Read the UTF-8 sequence (if any)
313 if( key < 127 ) {
314 // Single byte sequence
315 } else if( 194 <= key && key <= 223 ) {
316 rval.text.append( 1, static_cast<char>( getch() ) );
317 } else if( 224 <= key && key <= 239 ) {
318 rval.text.append( 1, static_cast<char>( getch() ) );
319 rval.text.append( 1, static_cast<char>( getch() ) );
320 } else if( 240 <= key && key <= 244 ) {
321 rval.text.append( 1, static_cast<char>( getch() ) );
322 rval.text.append( 1, static_cast<char>( getch() ) );
323 rval.text.append( 1, static_cast<char>( getch() ) );
324 } else {
325 // Other control character, etc. - no text at all, return an event
326 // without the text property
327 previously_pressed_key = key;
328 return input_event( key, CATA_INPUT_KEYBOARD );
329 }
330 // Now we have loaded an UTF-8 sequence (possibly several bytes)
331 // but we should only return *one* key, so return the code point of it.
332 const uint32_t cp = UTF8_getch( rval.text );
333 if( cp == UNKNOWN_UNICODE ) {
334 // Invalid UTF-8 sequence, this should never happen, what now?
335 // Maybe return any error instead?
336 previously_pressed_key = key;
337 return input_event( key, CATA_INPUT_KEYBOARD );
338 }
339 previously_pressed_key = cp;
340 // for compatibility only add the first byte, not the code point
341 // as it would conflict with the special keys defined by ncurses
342 rval.add_input( key );
343 }
344 } while( key == KEY_RESIZE );
345
346 return rval;
347}

389void ensure_term_size()
390{
391 // do not use ui_adaptor here to avoid re-entry
392 const int minHeight = FULL_SCREEN_HEIGHT;
393 const int minWidth = FULL_SCREEN_WIDTH;
394 int maxy = getmaxy( catacurses::stdscr );
395 int maxx = getmaxx( catacurses::stdscr );
396
397 while( maxy < minHeight || maxx < minWidth ) {
398 catacurses::erase();
399 if( maxy < minHeight && maxx < minWidth ) {
400 fold_and_print( catacurses::stdscr, point_zero, maxx, c_white,
401 _( "Whoa! Your terminal is tiny! This game requires a minimum terminal size of "
402 "%dx%d to work properly. %dx%d just won't do. Maybe a smaller font would help?" ),
403 minWidth, minHeight, maxx, maxy );
404 } else if( maxx < minWidth ) {
405 fold_and_print( catacurses::stdscr, point_zero, maxx, c_white,
406 _( "Oh! Hey, look at that. Your terminal is just a little too narrow. This game "
407 "requires a minimum terminal size of %dx%d to function. It just won't work "
408 "with only %dx%d. Can you stretch it out sideways a bit?" ),
409 minWidth, minHeight, maxx, maxy );
410 } else {
411 fold_and_print( catacurses::stdscr, point_zero, maxx, c_white,
412 _( "Woah, woah, we're just a little short on space here. The game requires a "
413 "minimum terminal size of %dx%d to run. %dx%d isn't quite enough! Can you "
414 "make the terminal just a smidgen taller?" ),
415 minWidth, minHeight, maxx, maxy );
416 }
417 catacurses::refresh();
418 // do not use input_manager or input_context here to avoid re-entry
419 getch();
420 maxy = getmaxy( catacurses::stdscr );
421 maxx = getmaxx( catacurses::stdscr );
422 }
423}
