67item vehicle_part::properties_to_item() const
68{
69 item tmp = base;
70 tmp.unset_flag( "VEHICLE" );
71
72 // Cables get special handling: their target coordinates need to remain
73 // stored, and if a cable actually drops, it should be half-connected.
74 // Except grid-connected ones, for now.
75 if( tmp.has_flag( "CABLE_SPOOL" ) && !tmp.has_flag( "TOW_CABLE" ) ) {
76 if( has_flag( targets_grid ) ) {
77 // Ideally, we'd drop the cable on the charger instead
78 tmp.erase_var( "source_x" );
79 tmp.erase_var( "source_y" );
80 tmp.erase_var( "source_z" );
81 tmp.erase_var( "state" );
82 tmp.active = false;
83 tmp.charges = tmp.type->maximum_charges();
84 } else {
85 const tripoint local_pos = g->m.getlocal( target.first );
86 if( !g->m.veh_at( local_pos ) ) {
87 // That vehicle ain't there no more.
88 tmp.set_flag( "NO_DROP" );
89 }
90
91 tmp.set_var( "source_x", target.first.x );
92 tmp.set_var( "source_y", target.first.y );
93 tmp.set_var( "source_z", target.first.z );
94 tmp.set_var( "state", "pay_out_cable" );
95 tmp.active = true;
96 }
97 }
98
99 // force rationalization of damage values to the middle value of each damage level so
100 // that parts will stack nicely
101 tmp.set_damage( tmp.damage_level( 4 ) * itype::damage_scale );

259 if( is_tank() && liquid->phase >= LIQUID && qty != 0 ) {

302double vehicle_part::consume_energy( const itype_id &ftype, double energy_j )
303{
304 if( base.contents.empty() || !is_fuel_store() ) {
305 return 0.0f;
306 }
307
308 item &fuel = base.contents.back();
309 if( fuel.typeId() == ftype ) {
310 assert( fuel.is_fuel() );
311 // convert energy density in MJ/L to J/ml
312 const double energy_p_mL = fuel.fuel_energy() * 1000;
313 const int ml_to_use = static_cast<int>( std::floor( energy_j / energy_p_mL ) );
314 int charges_to_use = fuel.charges_per_volume( ml_to_use * 1_ml );
315
316 if( !charges_to_use ) {
317 return 0.0;
318 }
319 if( charges_to_use >= fuel.charges ) {
320 charges_to_use = fuel.charges;
321 base.contents.clear_items();
322 } else {
323 fuel.charges -= charges_to_use;
324 }
325 item fuel_consumed( ftype, calendar::turn, charges_to_use );
326 return energy_p_mL * units::to_milliliter<int>( fuel_consumed.volume( true ) );
327 }
328 return 0.0;

331bool vehicle_part::can_reload( const item &obj ) const
332{
333 // first check part is not destroyed and can contain ammo
334 if( !is_fuel_store() ) {
335 return false;
336 }
337
338 if( !obj.is_null() ) {
339 const itype_id obj_type = obj.typeId();
340 if( is_reactor() ) {
341 return base.is_reloadable_with( obj_type );
342 }
343
344 // forbid filling tanks with solids or non-material things
345 if( is_tank() && ( obj.made_of( SOLID ) || obj.made_of( PNULL ) ) ) {
346 return false;
347 }
348 // forbid putting liquids, gasses, and plasma in things that aren't tanks
349 else if( !obj.made_of( SOLID ) && !is_tank() ) {
350 return false;
351 }
352 // prevent mixing of different ammo
353 if( !ammo_current().is_null() && ammo_current() != obj_type ) {
354 return false;
355 }
356 // For storage with set type, prevent filling with different types
357 if( info().fuel_type != fuel_type_none && info().fuel_type != obj_type ) {
358 return false;
359 }
360 // don't fill magazines with inappropriate fuel
361 if( !is_tank() && !base.is_reloadable_with( obj_type ) ) {
362 return false;
363 }
364 }
365
366 return ammo_remaining() < ammo_capacity();
367}
