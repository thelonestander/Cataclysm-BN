106bool tutorial_game::init()
107{
108 // TODO: clean up old tutorial
109 Character &player_character = get_player_character();
110
111 // Start at noon
112 calendar::turn = calendar::turn_zero + 12_hours;
113 tutorials_seen.clear();
114 g->scent.reset();
115 get_weather().temperature = 65;
116 // We use a Z-factor of 10 so that we don't plop down tutorial rooms in the
117 // middle of the "real" game world
118 g->u.normalize();
119 g->u.str_cur = g->u.str_max;
120 g->u.per_cur = g->u.per_max;
121 g->u.int_cur = g->u.int_max;
122 g->u.dex_cur = g->u.dex_max;
123
124 g->u.set_all_parts_hp_to_max();
125
126 const oter_id rock( "rock" );
127 //~ default name for the tutorial
128 g->u.name = _( "John Smith" );
129 g->u.prof = profession::generic();
130 // overmap terrain coordinates
131 const tripoint_om_omt lp( 50, 50, 0 );
132 // Assume overmap zero
133 const tripoint_abs_omt lp_abs = project_combine( point_abs_om(), lp );
134 auto &starting_om = overmap_buffer.get( point_abs_om() );
135 for( int i = 0; i < OMAPX; i++ ) {
136 for( int j = 0; j < OMAPY; j++ ) {
137 tripoint_om_omt p( i, j, 0 );
138 starting_om.ter_set( p + tripoint_below, rock );
139 // Start with the overmap revealed
140 starting_om.seen( p ) = true;
141 }
142 }
143 starting_om.ter_set( lp, oter_id( "tutorial" ) );
144 starting_om.ter_set( lp + tripoint_below, oter_id( "tutorial" ) );
145 starting_om.clear_mon_groups();
146
147 g->u.toggle_trait( trait_QUICK );
148 item lighter( "lighter", calendar::start_of_cataclysm );
149 lighter.invlet = 'e';
150 player_character.inv.add_item( lighter, true, false );
151 player_character.set_skill_level( skill_gun, 5 );
152 player_character.set_skill_level( skill_melee, 5 );
153 g->load_map( project_to<coords::sm>( lp_abs ) );
154 player_character.setx( 2 );
155 player_character.sety( 4 );
156
157 // This shifts the view to center the players pos
158 g->update_map( g->u );
159 return true;
160}

162void tutorial_game::per_turn()
163{
164 // note that add_message does nothing if the message was already shown
165 add_message( tut_lesson::LESSON_INTRO );
166 add_message( tut_lesson::LESSON_MOVE );
167 add_message( tut_lesson::LESSON_LOOK );
168
169 if( g->light_level( g->u.posz() ) == 1 ) {
170 if( g->u.has_amount( itype_flashlight, 1 ) ) {
171 add_message( tut_lesson::LESSON_DARK );
172 } else {
173 add_message( tut_lesson::LESSON_DARK_NO_FLASH );
174 }
175 }
176
177 if( g->u.get_pain() > 0 ) {
178 add_message( tut_lesson::LESSON_PAIN );
179 }
180
181 if( g->u.recoil >= MAX_RECOIL ) {
182 add_message( tut_lesson::LESSON_RECOIL );
183 }
184
185 if( !tutorials_seen[tut_lesson::LESSON_BUTCHER] ) {
186 for( const item &it : g->m.i_at( point( g->u.posx(), g->u.posy() ) ) ) {
187 if( it.is_corpse() ) {
188 add_message( tut_lesson::LESSON_BUTCHER );
189 break;
190 }
191 }
192 }
193
194 for( const tripoint &p : g->m.points_in_radius( g->u.pos(), 1 ) ) {
195 if( g->m.ter( p ) == t_door_o ) {
196 add_message( tut_lesson::LESSON_OPEN );
197 break;
198 } else if( g->m.ter( p ) == t_door_c ) {
199 add_message( tut_lesson::LESSON_CLOSE );
200 break;
201 } else if( g->m.ter( p ) == t_window ) {
202 add_message( tut_lesson::LESSON_SMASH );
203 break;
204 } else if( g->m.furn( p ) == f_rack && !g->m.i_at( p ).empty() ) {
205 add_message( tut_lesson::LESSON_EXAMINE );
206 break;
207 } else if( g->m.ter( p ) == t_stairs_down ) {
208 add_message( tut_lesson::LESSON_STAIRS );
209 break;
210 } else if( g->m.ter( p ) == t_water_sh ) {
211 add_message( tut_lesson::LESSON_PICKUP_WATER );
212 break;
213 }
214 }
215
216 if( !g->m.i_at( point( g->u.posx(), g->u.posy() ) ).empty() ) {
217 add_message( tut_lesson::LESSON_PICKUP );
218 }
219}

235void tutorial_game::post_action( action_id act )
236{
237 switch( act ) {
238 case ACTION_RELOAD_WEAPON:
239 if( g->u.weapon.is_gun() && !tutorials_seen[tut_lesson::LESSON_GUN_FIRE] ) {
240 g->place_critter_at( mon_zombie, tripoint( g->u.posx(), g->u.posy() - 6, g->u.posz() ) );
241 g->place_critter_at( mon_zombie, tripoint( g->u.posx() + 2, g->u.posy() - 5, g->u.posz() ) );
242 g->place_critter_at( mon_zombie, tripoint( g->u.posx() - 2, g->u.posy() - 5, g->u.posz() ) );
243 add_message( tut_lesson::LESSON_GUN_FIRE );
244 }
245 break;
246
247 case ACTION_OPEN:
248 add_message( tut_lesson::LESSON_CLOSE );
249 break;
250
251 case ACTION_CLOSE:
252 add_message( tut_lesson::LESSON_SMASH );
253 break;
254
255 case ACTION_USE:
256 if( g->u.has_amount( itype_grenade_act, 1 ) ) {
257 add_message( tut_lesson::LESSON_ACT_GRENADE );
258 }
259 for( const tripoint &dest : g->m.points_in_radius( g->u.pos(), 1 ) ) {
260 if( g->m.tr_at( dest ).id == trap_str_id( "tr_bubblewrap" ) ) {
261 add_message( tut_lesson::LESSON_ACT_BUBBLEWRAP );
262 }
263 }
264 break;
265
266 case ACTION_EAT:
267 if( g->u.last_item == itype_codeine ) {
268 add_message( tut_lesson::LESSON_TOOK_PAINKILLER );
269 } else if( g->u.last_item == itype_cig ) {
270 add_message( tut_lesson::LESSON_TOOK_CIG );
271 } else if( g->u.last_item == itype_water ) {
272 add_message( tut_lesson::LESSON_DRANK_WATER );
273 }
274 break;
275
276 case ACTION_WEAR: {
277 item it( g->u.last_item, calendar::start_of_cataclysm );
278 if( it.is_armor() ) {
279 if( it.get_coverage() >= 2 || it.get_thickness() >= 2 ) {
280 add_message( tut_lesson::LESSON_WORE_ARMOR );
281 }
282 if( it.get_storage() >= units::from_liter( 5 ) ) {
283 add_message( tut_lesson::LESSON_WORE_STORAGE );
284 }
285 if( it.get_env_resist() >= 2 ) {
286 add_message( tut_lesson::LESSON_WORE_MASK );
287 }
288 }
289 }
290 break;
291
292 case ACTION_WIELD:
293 if( g->u.weapon.is_gun() ) {
294 add_message( tut_lesson::LESSON_GUN_LOAD );
295 }
296 break;
297
298 case ACTION_EXAMINE:
299 add_message( tut_lesson::LESSON_INTERACT );
300 /* fallthrough */
301 case ACTION_PICKUP: {
302 item it( g->u.last_item, calendar::start_of_cataclysm );
303 if( it.is_armor() ) {
304 add_message( tut_lesson::LESSON_GOT_ARMOR );
305 } else if( it.is_gun() ) {
306 add_message( tut_lesson::LESSON_GOT_GUN );
307 } else if( it.is_ammo() ) {
308 add_message( tut_lesson::LESSON_GOT_AMMO );
309 } else if( it.is_tool() ) {
310 add_message( tut_lesson::LESSON_GOT_TOOL );
311 } else if( it.is_food() ) {
312 add_message( tut_lesson::LESSON_GOT_FOOD );
313 } else if( it.is_melee() ) {
314 add_message( tut_lesson::LESSON_GOT_WEAPON );
315 }
316
317 }
318 break;
319
320 default:
321 // TODO: add more actions here
322 break;
323
324 }
