187
188 // populate the precalc_array for O(1) lookups
189 void precalc() {
190 precalc_array.clear();
191 // to avoid additional reallocations
192 precalc_array.reserve( this->total_weight );
193 // weights [3,1,5] will produce vector of indices [0,0,0,1,2,2,2,2,2]
194 for( size_t i = 0; i < this->objects.size(); i++ ) {
195 precalc_array.resize( precalc_array.size() + this->objects[i].weight, i );
196 }
197 }
198
199 protected:
200
201 size_t pick_ent( unsigned int randi ) const override {
202 if( this->objects.size() == 1 ) {
203 return 0;
204 }
205 size_t i;
206 const int picked = ( randi % ( this->total_weight ) ) + 1;
207 if( !precalc_array.empty() ) {
208 // if the precalc_array is populated, use it for O(1) lookup
209 i = precalc_array[picked - 1];
210 } else {
211 // otherwise do O(N) search through items
212 int accumulated_weight = 0;
213 for( i = 0; i < this->objects.size(); i++ ) {
214 accumulated_weight += this->objects[i].weight;
215 if( accumulated_weight >= picked ) {
216 break;
217 }
218 }
219 }
220 return i;
221 }
222
223 void invalidate_precalc() override {
224 precalc_array.clear();
225 }
