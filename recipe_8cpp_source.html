48int recipe::batch_time( int batch, float multiplier, size_t assistants ) const
49{
50 // 1.0f is full speed
51 // 0.33f is 1/3 speed
52 if( multiplier == 0.0f ) {
53 // If an item isn't craftable in the dark, show the time to complete as if you could craft it
54 multiplier = 1.0f;
55 }
56
57 const float local_time = static_cast<float>( time ) / multiplier;
58
59 // if recipe does not benefit from batching and we have no assistants, don't do unnecessary additional calculations
60 if( batch_rscale == 0.0 && assistants == 0 ) {
61 return static_cast<int>( local_time ) * batch;
62 }
63
64 float total_time = 0.0;
65 // if recipe does not benefit from batching but we do have assistants, skip calculating the batching scale factor
66 if( batch_rscale == 0.0 ) {
67 total_time = local_time * batch;
68 } else {
69 // recipe benefits from batching, so batching scale factor needs to be calculated
70 // At batch_rsize, incremental time increase is 99.5% of batch_rscale
71 const double scale = batch_rsize / 6.0;
72 for( int x = 0; x < batch; x++ ) {
73 // scaled logistic function output
74 const double logf = ( 2.0 / ( 1.0 + std::exp( -( x / scale ) ) ) ) - 1.0;
75 total_time += local_time * ( 1.0 - ( batch_rscale * logf ) );
76 }
77 }
78
79 //Assistants can decrease the time for production but never less than that of one unit
80 if( assistants == 1 ) {
81 total_time = total_time * .75;
82 } else if( assistants >= 2 ) {
83 total_time = total_time * .60;
84 }
85 if( total_time < local_time ) {
86 total_time = local_time;
87 }
88
89 return static_cast<int>( total_time );
90}

365std::string recipe::get_consistency_error() const
366{
367 if( category == "CC_BUILDING" ) {
368 if( is_blueprint() || oter_str_id( result_.c_str() ).is_valid() ) {
369 return std::string();
370 }
371 return "defines invalid result";
372 }
373
374 if( !result_.is_valid() ) {
375 return "defines invalid result";
376 }
377
378 if( charges && !item::count_by_charges( result_ ) ) {
379 return "specifies charges but result is not counted by charges";
380 }
381
382 const auto is_invalid_bp = []( const std::pair<itype_id, int> &elem ) {
383 return !elem.first.is_valid();
384 };
385
386 if( std::any_of( byproducts.begin(), byproducts.end(), is_invalid_bp ) ) {
387 return "defines invalid byproducts";
388 }
389
390 if( !contained && !container.is_null() ) {
391 return "defines container but not contained";
392 }
393
394 if( !container.is_valid() ) {
395 return "specifies unknown container";
396 }
397
398 const auto is_invalid_skill = []( const std::pair<skill_id, int> &elem ) {
399 return !elem.first.is_valid();
400 };
401
402 if( ( skill_used && !skill_used.is_valid() ) ||
403 std::any_of( required_skills.begin(), required_skills.end(), is_invalid_skill ) ) {
404 return "uses invalid skill";
405 }
406
407 const auto is_invalid_book = []( const std::pair<itype_id, int> &elem ) {
408 return !elem.first->book;
409 };
410
411 if( std::any_of( booksets.begin(), booksets.end(), is_invalid_book ) ) {
412 return "defines invalid book";
413 }
414
415 // TODO: IDs instead of raw string
416 if( !blueprint.empty() && !mapgen::has_update_id( blueprint ) ) {
417 return "has invalid mapgen id";
418 }
419
420 return std::string();

423item recipe::create_result() const
424{
425 item newit( result_, calendar::turn, item::default_charges_tag{} );
426 if( charges ) {
427 newit.charges = *charges;
428 }
429
430 if( !newit.craft_has_charges() ) {
431 newit.charges = 0;
432 } else if( result_mult != 1 ) {
433 // TODO: Make it work for charge-less items (update makes amount)
434 newit.charges *= result_mult;
435 }
436
437 // Show crafted items as fitting
438 // They might end up not fitting, but it's rare
439 if( newit.has_flag( flag_VARSIZE ) ) {
440 newit.item_tags.insert( flag_FIT );
441 }
442
443 if( contained ) {
444 newit = newit.in_container( container );
445 }
446
447 return newit;
448}

450std::vector<item> recipe::create_results( int batch ) const
451{
452 std::vector<item> items;
453
454 const bool by_charges = item::count_by_charges( result_ );
455 if( contained || !by_charges ) {
456 // by_charges items get their charges multiplied in create_result
457 const int num_results = by_charges ? batch : batch * result_mult;
458 for( int i = 0; i < num_results; i++ ) {
459 item newit = create_result();
460 items.push_back( newit );
461 }
462 } else {
463 item newit = create_result();
464 newit.charges *= batch;
465 items.push_back( newit );
466 }
467
468 return items;

501// Format a std::pair<skill_id, int> for the crafting menu.
502// skill colored green (or yellow if beyond characters skill)
503// optionally with the skill level (player / difficulty)
504template<typename Iter>
505std::string required_skills_as_string( Iter first, Iter last, const Character *c,
506 const bool print_skill_level )
507{
508 if( first == last ) {
509 return _( "<color_cyan>none</color>" );
510 }
511
512 return enumerate_as_string( first, last,
513 [&]( const std::pair<skill_id, int> &skill ) {
514 const int player_skill = c ? c->get_skill_level( skill.first ) : 0;
515 std::string difficulty_color = skill.second > player_skill ? "yellow" : "green";
516 std::string skill_level_string = print_skill_level ? "" : ( std::to_string( player_skill ) + "/" );
517 skill_level_string += std::to_string( skill.second );
518 return string_format( "<color_cyan>%s</color> <color_%s>(%s)</color>",
519 skill.first.obj().name(), difficulty_color, skill_level_string );
520 } );
521}

606std::function<bool( const item & )> recipe::get_component_filter(
607 const recipe_filter_flags flags ) const
608{
609 const item result = create_result();
610
611 // Disallow crafting of non-perishables with rotten components
612 // Make an exception for items with the ALLOW_ROTTEN flag such as seeds
613 const bool recipe_forbids_rotten =
614 result.is_food() && !result.goes_bad() && !has_flag( "ALLOW_ROTTEN" );
615 const bool flags_forbid_rotten =
616 static_cast<bool>( flags & recipe_filter_flags::no_rotten ) && result.goes_bad_after_opening();
617 std::function<bool( const item & )> rotten_filter = return_true<item>;
618 if( recipe_forbids_rotten || flags_forbid_rotten ) {
619 rotten_filter = []( const item & component ) {
620 return !component.rotten();
621 };
622 }
623
624 // Disallow usage of non-full magazines as components
625 // This is primarily used to require a fully charged battery, but works for any magazine.
626 std::function<bool( const item & )> magazine_filter = return_true<item>;
627 if( has_flag( "FULL_MAGAZINE" ) ) {
628 magazine_filter = []( const item & component ) {
629 return !component.is_magazine() || ( component.ammo_remaining() >= component.ammo_capacity() );
630 };
631 }
632
633 // Filter out filthy components here instead of with is_crafting_component
634 // Make an exception for recipes with the ALLOW_FILTHY flag
635 std::function<bool( const item & )> filthy_filter = return_true<item>;
636 if( !has_flag( flag_ALLOW_FILTHY ) ) {
637 filthy_filter = []( const item & component ) {
638 return !component.is_filthy();
639 };
640 }
641
642 return [ rotten_filter, magazine_filter, filthy_filter ]( const item & component ) {
643 return is_crafting_component_allow_filthy( component ) &&
644 rotten_filter( component ) &&
645 magazine_filter( component ) &&
646 filthy_filter( component );
647 };
648}

685static std::string dump_requirements(
686 const requirement_data &reqs,
687 int move_cost,
688 const std::map<skill_id, int> &skills )
689{
690 std::ostringstream os;
691 JsonOut jsout( os, /*pretty_print=*/true );
692
693 jsout.start_object();
694
695 jsout.member( "time" );
696 if( move_cost % 100 == 0 ) {
697 dump_to_json_string( time_duration::from_turns( move_cost / 100 ),
698 jsout, time_duration::units );
699 } else {
700 // cannot precisely represent the value using time_duration format,
701 // write integer instead.
702 jsout.write( move_cost );
703 }
704
705 jsout.member( "skills" );
706 jsout.start_array( /*wrap=*/!skills.empty() );
707 for( const std::pair<const skill_id, int> &p : skills ) {
708 jsout.start_array();
709 jsout.write( p.first );
710 jsout.write( p.second );
711 jsout.end_array();
712 }
713 jsout.end_array();
714
715 jsout.member( "inline" );
716 reqs.dump( jsout );
717
718 jsout.end_object();
719 return os.str();
720}

722void recipe::check_blueprint_requirements()
723{
724 if( !blueprint.empty() && !mapgen::has_update_id( blueprint ) ) {
725 debugmsg( "Blueprint %1$s has invalid mapgen id %2$s", ident_.str(), blueprint );
726 return;
727 }
728 build_reqs total_reqs;
729 const std::pair<std::map<ter_id, int>, std::map<furn_id, int>> &changed_ids
730 = get_changed_ids_from_update( blueprint );
731 get_build_reqs_for_furn_ter_ids( changed_ids, total_reqs );
732 requirement_data req_data_blueprint = std::accumulate(
733 reqs_blueprint.begin(), reqs_blueprint.end(), requirement_data(),
734 []( const requirement_data & lhs, const std::pair<requirement_id, int> &rhs ) {
735 return lhs + ( *rhs.first * rhs.second );
736 } );
737 requirement_data req_data_calc = std::accumulate(
738 total_reqs.reqs.begin(), total_reqs.reqs.end(), requirement_data(),
739 []( const requirement_data & lhs, const std::pair<requirement_id, int> &rhs ) {
740 return lhs + ( *rhs.first * rhs.second );
741 } );
742 // do not consolidate req_data_blueprint: it actually changes the meaning of the requirement.
743 // instead we enforce specifying the exact consolidated requirement.
744 req_data_calc.consolidate();
745 if( time_blueprint != total_reqs.time || skills_blueprint != total_reqs.skills
746 || !req_data_blueprint.has_same_requirements_as( req_data_calc ) ) {
747 std::string calc_req_str = dump_requirements( req_data_calc, total_reqs.time, total_reqs.skills );
748 std::string got_req_str = dump_requirements( req_data_blueprint, total_reqs.time,
749 total_reqs.skills );
750
751 std::stringstream ss;
752 for( auto &id_count : changed_ids.first ) {
753 ss << string_format( "%s: %d\n", id_count.first.id(), id_count.second );
754 }
755 for( auto &id_count : changed_ids.second ) {
756 ss << string_format( "%s: %d\n", id_count.first.id(), id_count.second );
757 }
758
759 debugmsg( "Specified blueprint requirements of %1$s does not match calculated requirements. "
760 "Specify \"check_blueprint_needs\": false to disable the check or "
761 "Update \"blueprint_needs\" to the following value (you can use tools/update_blueprint_needs.py):\n"
762 // mark it for the auto-update python script
763 "~~~ auto-update-blueprint: %1$s\n"
764 "%2$s\n"
765 "~~~ end-auto-update\n"
766 "If the stated requirements match the above, but the error still appears, it is a bug.\n"
767 "Stated requirements are expanded to:\n"
768 "%3$s\n"
769 "---Begin expected tile changes---\n%4$s---End expected tile changes--",
770 ident_.str(), calc_req_str, got_req_str, ss.str() );
771 }

774bool recipe::hot_result() const
775{
776 // Check if the recipe tools make this food item hot upon making it.
777 // We don't actually know which specific tool the player used/will use here, but
778 // we're checking for a class of tools; because of the way requirements
779 // processing works, the "surface_heat" id gets nuked into an actual
780 // list of tools, see data/json/recipes/cooking_tools.json.
781 //
782 // Currently it's only checking for a hotplate because that's a
783 // suitable item in both the "surface_heat" and "water_boiling_heat"
784 // tools, and it's usually the first item in a list of tools so if this
785 // does get heated we'll find it right away.
786 //
787 // TODO: Make this less of a hack
788 if( create_result().is_food() ) {
789 const requirement_data::alter_tool_comp_vector &tool_lists = simple_requirements().get_tools();
790 for( const std::vector<tool_comp> &tools : tool_lists ) {
791 for( const tool_comp &t : tools ) {
792 if( t.type == itype_hotplate ) {
793 return true;
794 }
795 }
796 }
797 }
798 return false;
799}
