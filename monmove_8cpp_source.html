81bool monster::is_immune_field( const field_type_id &fid ) const
82{
83 if( fid == fd_fungal_haze ) {
84 return has_flag( MF_NO_BREATHE ) || type->in_species( FUNGUS );
85 }
86 if( fid == fd_fungicidal_gas ) {
87 return !type->in_species( FUNGUS );
88 }
89 if( fid == fd_insecticidal_gas ) {
90 return !type->in_species( INSECT ) && !type->in_species( SPIDER );
91 }
92 const field_type &ft = fid.obj();
93 if( ft.has_fume ) {
94 return has_flag( MF_NO_BREATHE );
95 }
96 if( ft.has_acid ) {
97 return has_flag( MF_ACIDPROOF ) || flies();
98 }
99 if( ft.has_fire ) {
100 return has_flag( MF_FIREPROOF );
101 }
102 if( ft.has_elec ) {
103 return has_flag( MF_ELECTRIC );
104 }
105 if( ft.immune_mtypes.count( type->id ) > 0 ) {
106 return true;
107 }
108 // No specific immunity was found, so fall upwards
109 return Creature::is_immune_field( fid );
110}

117bool monster::will_move_to( const tripoint &p ) const
118{
119 if( g->m.impassable( p ) ) {
120 if( digging() ) {
121 if( !g->m.has_flag( "BURROWABLE", p ) ) {
122 return false;
123 }
124 } else if( !( can_climb() && g->m.has_flag( "CLIMBABLE", p ) ) ) {
125 return false;
126 }
127 }
128
129 if( ( !can_submerge() && !flies() ) && g->m.has_flag( TFLAG_DEEP_WATER, p ) ) {
130 return false;
131 }
132
133 if( digs() && !g->m.has_flag( "DIGGABLE", p ) && !g->m.has_flag( "BURROWABLE", p ) ) {
134 return false;
135 }
136
137 if( has_flag( MF_AQUATIC ) && !g->m.has_flag( "SWIMMABLE", p ) ) {
138 return false;
139 }
140
141 if( has_flag( MF_SUNDEATH ) && g->is_in_sunlight( p ) ) {
142 return false;
143 }
144
145 if( get_size() > MS_MEDIUM && g->m.has_flag_ter( TFLAG_SMALL_PASSAGE, p ) ) {
146 return false; // if a large critter, can't move through tight passages
147 }
148
149 // Various avoiding behaviors.
150
151 bool avoid_fire = has_flag( MF_AVOID_FIRE );
152 bool avoid_fall = has_flag( MF_AVOID_FALL );
153 bool avoid_simple = has_flag( MF_AVOID_DANGER_1 );
154 bool avoid_complex = has_flag( MF_AVOID_DANGER_2 );
155 /*
156 * Because some avoidance behaviors are supersets of others,
157 * we can cascade through the implications. Complex implies simple,
158 * and simple implies fire and fall.
159 * unfortunately, fall does not necessarily imply fire, nor the converse.
160 */
161 if( avoid_complex ) {
162 avoid_simple = true;
163 }
164 if( avoid_simple ) {
165 avoid_fire = true;
166 avoid_fall = true;
167 }
168
169 // technically this will shortcut in evaluation from fire or fall
170 // before hitting simple or complex but this is more explicit
171 if( avoid_fire || avoid_fall || avoid_simple || avoid_complex ) {
172 const ter_id target = g->m.ter( p );
173
174 // Don't enter lava if we have any concept of heat being bad
175 if( avoid_fire && target == t_lava ) {
176 return false;
177 }
178
179 if( avoid_fall ) {
180 // Don't throw ourselves off cliffs if we have a concept of falling
181 if( !g->m.has_floor( p ) && !flies() ) {
182 return false;
183 }
184
185 // Don't enter open pits ever unless tiny, can fly or climb well
186 if( !( type->size == MS_TINY || can_climb() ) &&
187 ( target == t_pit || target == t_pit_spiked || target == t_pit_glass ) ) {
188 return false;
189 }
190 }
191
192 // Some things are only avoided if we're not attacking
193 if( attitude( &g->u ) != MATT_ATTACK ) {
194 // Sharp terrain is ignored while attacking
195 if( avoid_simple && g->m.has_flag( "SHARP", p ) &&
196 !( type->size == MS_TINY || flies() ) ) {
197 return false;
198 }
199 }
200
201 const field &target_field = g->m.field_at( p );
202
203 // Higher awareness is needed for identifying these as threats.
204 if( avoid_complex ) {
205 const trap &target_trap = g->m.tr_at( p );
206 // Don't enter any dangerous fields
207 if( is_dangerous_fields( target_field ) ) {
208 return false;
209 }
210 // Don't step on any traps (if we can see)
211 if( has_flag( MF_SEES ) && !target_trap.is_benign() && g->m.has_floor( p ) ) {
212 return false;
213 }
214 }
215
216 // Without avoid_complex, only fire and electricity are checked for field avoidance.
217 if( avoid_fire && target_field.find_field( fd_fire ) ) {
218 return false;
219 }
220 if( avoid_simple && target_field.find_field( fd_electricity ) ) {
221 return false;
222 }
223 }
224
225 return true;
226}

228bool monster::can_reach_to( const tripoint &p ) const
229{
230 map &here = get_map();
231 if( p.z > pos().z && z_is_valid( pos().z ) ) {
232 if( here.has_flag( TFLAG_RAMP_UP, tripoint( p.xy(), p.z - 1 ) ) ) {
233 return true;
234 }
235 if( !here.has_flag( TFLAG_GOES_UP, pos() ) && !here.has_flag( TFLAG_NO_FLOOR, p ) ) {
236 // can't go through the roof
237 return false;
238 }
239 } else if( p.z < pos().z && z_is_valid( pos().z ) ) {
240 if( !here.has_flag( TFLAG_GOES_DOWN, pos() ) ) {
241 // can't go through the floor
242 // you would fall anyway if there was no floor, so no need to check for that here
243 return false;
244 }
245 }
246 return true;
247}

273float monster::rate_target( Creature &c, float best, bool smart ) const
274{
275 const auto d = rl_dist_fast( pos(), c.pos() );
276 if( d <= 0 ) {
277 return FLT_MAX;
278 }
279
280 // Check a very common and cheap case first
281 if( !smart && d >= best ) {
282 return FLT_MAX;
283 }
284
285 if( !sees( c ) ) {
286 return FLT_MAX;
287 }
288
289 if( !smart ) {
290 return int( d );
291 }
292
293 float power = c.power_rating();
294 monster *mon = dynamic_cast< monster * >( &c );
295 // Their attitude to us and not ours to them, so that bobcats won't get gunned down
296 if( mon != nullptr && mon->attitude_to( *this ) == Attitude::A_HOSTILE ) {
297 power += 2;
298 }
299
300 if( power > 0 ) {
301 return int( d ) / power;
302 }
303
304 return FLT_MAX;
305}

307void monster::plan()
308{
309 const auto &factions = g->critter_tracker->factions();
310
311 // Bots are more intelligent than most living stuff
312 bool smart_planning = has_flag( MF_PRIORITIZE_TARGETS );
313 Creature *target = nullptr;
314 int max_sight_range = std::max( type->vision_day, type->vision_night );
315 // 8.6f is rating for tank drone 60 tiles away, moose 16 or boomer 33
316 float dist = !smart_planning ? max_sight_range : 8.6f;
317 bool fleeing = false;
318 bool docile = friendly != 0 && has_effect( effect_docile );
319 bool waiting = has_effect( effect_ai_waiting );
320
321 const bool angers_hostile_weak = type->has_anger_trigger( mon_trigger::HOSTILE_WEAK );
322 const int angers_hostile_near = type->has_anger_trigger( mon_trigger::HOSTILE_CLOSE ) ? 5 : 0;
323 const int angers_mating_season = type->has_anger_trigger( mon_trigger::MATING_SEASON ) ? 3 : 0;
324 const int angers_cub_threatened = type->has_anger_trigger( mon_trigger::PLAYER_NEAR_BABY ) ? 8 : 0;
325 const int fears_hostile_near = type->has_fear_trigger( mon_trigger::HOSTILE_CLOSE ) ? 5 : 0;
326
327 bool group_morale = has_flag( MF_GROUP_MORALE ) && morale < type->morale;
328 bool swarms = has_flag( MF_SWARMS );
329 auto mood = attitude();
330
331 // If we can see the player, move toward them or flee, simpleminded animals are too dumb to follow the player.
332 if( friendly == 0 && sees( g->u ) && !has_flag( MF_PET_WONT_FOLLOW ) && !waiting ) {
333 dist = rate_target( g->u, dist, smart_planning );
334 fleeing = fleeing || is_fleeing( g->u );
335 target = &g->u;
336 if( dist <= 5 ) {
337 anger += angers_hostile_near;
338 morale -= fears_hostile_near;
339 if( angers_mating_season > 0 ) {
340 bool mating_angry = false;
341 season_type season = season_of_year( calendar::turn );
342 for( auto &elem : type->baby_flags ) {
343 if( ( season == SUMMER && elem == "SUMMER" ) ||
344 ( season == WINTER && elem == "WINTER" ) ||
345 ( season == SPRING && elem == "SPRING" ) ||
346 ( season == AUTUMN && elem == "AUTUMN" ) ) {
347 mating_angry = true;
348 break;
349 }
350 }
351 if( mating_angry ) {
352 anger += angers_mating_season;
353 }
354 }
355 }
356 if( angers_cub_threatened > 0 ) {
357 for( monster &tmp : g->all_monsters() ) {
358 if( type->baby_monster == tmp.type->id ) {
359 // baby nearby; is the player too close?
360 dist = tmp.rate_target( g->u, dist, smart_planning );
361 if( dist <= 3 ) {
362 //proximity to baby; monster gets furious and less likely to flee
363 anger += angers_cub_threatened;
364 morale += angers_cub_threatened / 2;
365 }
366 }
367 }
368 }
369 } else if( friendly != 0 && !docile && !waiting ) {
370 for( monster &tmp : g->all_monsters() ) {
371 if( tmp.friendly == 0 ) {
372 float rating = rate_target( tmp, dist, smart_planning );
373 if( rating < dist ) {
374 target = &tmp;
375 dist = rating;
376 }
377 }
378 }
379 }
380
381 if( waiting ) {
382 set_dest( pos() );
383 return;
384 }
385
386 if( docile ) {
387 if( friendly != 0 && target != nullptr ) {
388 set_dest( target->pos() );
389 }
390
391 return;
392 }
393
394 int valid_targets = ( target == nullptr ) ? 1 : 0;
395 for( npc &who : g->all_npcs() ) {
396 auto faction_att = faction.obj().attitude( who.get_monster_faction() );
397 if( faction_att == MFA_NEUTRAL || faction_att == MFA_FRIENDLY ) {
398 continue;
399 }
400
401 float rating = rate_target( who, dist, smart_planning );
402 bool fleeing_from = is_fleeing( who );
403 if( rating == dist && ( fleeing || attitude( &who ) == MATT_ATTACK ) ) {
404 ++valid_targets;
405 if( one_in( valid_targets ) ) {
406 target = &who;
407 }
408 }
409 // Switch targets if closer and hostile or scarier than current target
410 if( ( rating < dist && fleeing ) ||
411 ( faction_att == MFA_HATE ) ||
412 ( rating < dist && attitude( &who ) == MATT_ATTACK ) ||
413 ( !fleeing && fleeing_from ) ) {
414 target = &who;
415 dist = rating;
416 valid_targets = 1;
417 }
418 fleeing = fleeing || fleeing_from;
419 if( rating <= 5 ) {
420 anger += angers_hostile_near;
421 morale -= fears_hostile_near;
422 if( angers_mating_season > 0 ) {
423 bool mating_angry = false;
424 season_type season = season_of_year( calendar::turn );
425 for( auto &elem : type->baby_flags ) {
426 if( ( season == SUMMER && elem == "SUMMER" ) ||
427 ( season == WINTER && elem == "WINTER" ) ||
428 ( season == SPRING && elem == "SPRING" ) ||
429 ( season == AUTUMN && elem == "AUTUMN" ) ) {
430 mating_angry = true;
431 break;
432 }
433 }
434 if( mating_angry ) {
435 anger += angers_mating_season;
436 }
437 }
438 }
439 }
440
441 fleeing = fleeing || ( mood == MATT_FLEE );
442 if( friendly == 0 ) {
443 for( const auto &fac : factions ) {
444 auto faction_att = faction.obj().attitude( fac.first );
445 if( faction_att == MFA_NEUTRAL || faction_att == MFA_FRIENDLY ) {
446 continue;
447 }
448
449 for( const weak_ptr_fast<monster> &weak : fac.second ) {
450 const shared_ptr_fast<monster> shared = weak.lock();
451 if( !shared ) {
452 continue;
453 }
454 monster &mon = *shared;
455 float rating = rate_target( mon, dist, smart_planning );
456 if( rating == dist ) {
457 ++valid_targets;
458 if( one_in( valid_targets ) ) {
459 target = &mon;
460 }
461 }
462 if( rating < dist ) {
463 target = &mon;
464 dist = rating;
465 valid_targets = 1;
466 }
467 if( rating <= 5 ) {
468 anger += angers_hostile_near;
469 morale -= fears_hostile_near;
470 }
471 }
472 }
473 }
474
475 // Friendly monsters here
476 // Avoid for hordes of same-faction stuff or it could get expensive
477 const auto actual_faction = friendly == 0 ? faction : mfaction_str_id( "player" );
478 const auto &myfaction_iter = factions.find( actual_faction );
479 if( myfaction_iter == factions.end() ) {
480 DebugLog( DL::Error, DC::Game ) << disp_name() << " tried to find faction "
481 << actual_faction.id().str()
482 << " which wasn't loaded in game::monmove";
483 swarms = false;
484 group_morale = false;
485 }
486 swarms = swarms && target == nullptr; // Only swarm if we have no target
487 if( group_morale || swarms ) {
488 for( const weak_ptr_fast<monster> &weak : myfaction_iter->second ) {
489 const shared_ptr_fast<monster> shared = weak.lock();
490 if( !shared ) {
491 continue;
492 }
493 monster &mon = *shared;
494 float rating = rate_target( mon, dist, smart_planning );
495 if( group_morale && rating <= 10 ) {
496 morale += 10 - rating;
497 }
498 if( swarms ) {
499 if( rating < 5 ) { // Too crowded here
500 wander_pos.x = posx() * rng( 1, 3 ) - mon.posx();
501 wander_pos.y = posy() * rng( 1, 3 ) - mon.posy();
502 wandf = 2;
503 target = nullptr;
504 // Swarm to the furthest ally you can see
505 } else if( rating < FLT_MAX && rating > dist && wandf <= 0 ) {
506 target = &mon;
507 dist = rating;
508 }
509 }
510 }
511 }
512
513 // Operating monster keep you safe while they operate, how nice....
514 if( type->has_special_attack( "OPERATE" ) ) {
515 int prev_friendlyness = friendly;
516 if( has_effect( effect_operating ) ) {
517 friendly = 100;
518 for( auto critter : g->m.get_creatures_in_radius( pos(), 6 ) ) {
519 monster *mon = dynamic_cast<monster *>( critter );
520 if( mon != nullptr && mon->type->in_species( ZOMBIE ) ) {
521 anger = 100;
522 } else {
523 anger = 0;
524 }
525 }
526 } else {
527 friendly = prev_friendlyness;
528 }
529 }
530
531 if( has_effect( effect_dragging ) ) {
532
533 if( type->has_special_attack( "OPERATE" ) ) {
534
535 bool found_path_to_couch = false;
536 tripoint tmp( pos() + point( 12, 12 ) );
537 tripoint couch_loc;
538 for( const auto &couch_pos : g->m.find_furnitures_with_flag_in_radius( pos(), 10,
539 flag_AUTODOC_COUCH ) ) {
540 if( g->m.clear_path( pos(), couch_pos, 10, 0, 100 ) ) {
541 if( rl_dist( pos(), couch_pos ) < rl_dist( pos(), tmp ) ) {
542 tmp = couch_pos;
543 found_path_to_couch = true;
544 couch_loc = couch_pos;
545 }
546 }
547 }
548
549 if( !found_path_to_couch ) {
550 anger = 0;
551 remove_effect( effect_dragging );
552 } else {
553 set_dest( couch_loc );
554 }
555 }
556
557 } else if( target != nullptr ) {
558
559 tripoint dest = target->pos();
560 auto att_to_target = attitude_to( *target );
561 if( att_to_target == Attitude::A_HOSTILE && !fleeing ) {
562 set_dest( dest );
563 } else if( fleeing ) {
564 set_dest( tripoint( posx() * 2 - dest.x, posy() * 2 - dest.y, posz() ) );
565 }
566 if( angers_hostile_weak && att_to_target != Attitude::A_FRIENDLY ) {
567 int hp_per = target->hp_percentage();
568 if( hp_per <= 70 ) {
569 anger += 10 - static_cast<int>( hp_per / 10 );
570 }
571 }
572 } else if( friendly > 0 && one_in( 3 ) ) {
573 // Grow restless with no targets
574 friendly--;
575 } else if( friendly < 0 && sees( g->u ) ) {
576 if( rl_dist( pos(), g->u.pos() ) > 2 ) {
577 set_dest( g->u.pos() );
578 } else {
579 unset_dest();
580 }
581 }
582}

584/**
585 * Method to make monster movement speed consistent in the face of staggering behavior and
586 * differing distance metrics.
587 * It works by scaling the cost to take a step by
588 * how much that step reduces the distance to your goal.
589 * Since it incorporates the current distance metric,
590 * it also scales for diagonal vs orthogonal movement.
591 **/
592static float get_stagger_adjust( const tripoint &source, const tripoint &destination,
593 const tripoint &next_step )
594{
595 // TODO: push this down into rl_dist
596 const float initial_dist =
597 trigdist ? trig_dist( source, destination ) : rl_dist( source, destination );
598 const float new_dist =
599 trigdist ? trig_dist( next_step, destination ) : rl_dist( next_step, destination );
600 // If we return 0, it wil cancel the action.
601 return std::max( 0.01f, initial_dist - new_dist );
602}

626// General movement.
627// Currently, priority goes:
628// 1) Special Attack
629// 2) Sight-based tracking
630// 3) Scent-based tracking
631// 4) Sound-based tracking
632void monster::move()
633{
634 // We decrement wandf no matter what. We'll save our wander_to plans until
635 // after we finish out set_dest plans, UNLESS they time out first.
636 if( wandf > 0 ) {
637 wandf--;
638 }
639
640 //Hallucinations have a chance of disappearing each turn
641 if( is_hallucination() && one_in( 25 ) ) {
642 die( nullptr );
643 return;
644 }
645 map &here = get_map();
646
647 behavior::monster_oracle_t oracle( this );
648 behavior::tree goals;
649 goals.add( type->get_goals() );
650 std::string action = goals.tick( &oracle );
651 //The monster can consume objects it stands on. Check if there are any.
652 //If there are. Consume them.
653 // TODO: Stick this in a map and dispatch to it via the action string.
654 if( action == "consume_items" ) {
655 if( g->u.sees( *this ) ) {
656 add_msg( _( "The %s flows around the objects on the floor and they are quickly dissolved!" ),
657 name() );
658 }
659 static const auto volume_per_hp = 250_ml;
660 for( auto &elem : g->m.i_at( pos() ) ) {
661 hp += elem.volume() / volume_per_hp; // Yeah this means it can get more HP than normal.
662 if( has_flag( MF_ABSORBS_SPLITS ) ) {
663 while( hp / 2 > type->hp ) {
664 monster *const spawn = g->place_critter_around( type->id, pos(), 1 );
665 if( !spawn ) {
666 break;
667 }
668 hp -= type->hp;
669 //this is a new copy of the monster. Ideally we should copy the stats/effects that affect the parent
670 spawn->make_ally( *this );
671 if( g->u.sees( *this ) ) {
672 add_msg( _( "The %s splits in two!" ), name() );
673 }
674 }
675 }
676 }
677 g->m.i_clear( pos() );
678 }
679 // record position before moving to put the player there if we're dragging
680 tripoint drag_to = g->m.getabs( pos() );
681
682 const bool pacified = has_effect( effect_pacified );
683
684 // First, use the special attack, if we can!
685 // The attack may change `monster::special_attacks` (e.g. by transforming
686 // this into another monster type). Therefore we can not iterate over it
687 // directly and instead iterate over the map from the monster type
688 // (properties of monster types should never change).
689 for( const auto &sp_type : type->special_attacks ) {
690 const std::string &special_name = sp_type.first;
691 const auto local_iter = special_attacks.find( special_name );
692 if( local_iter == special_attacks.end() ) {
693 continue;
694 }
695 mon_special_attack &local_attack_data = local_iter->second;
696 if( !local_attack_data.enabled ) {
697 continue;
698 }
699
700 // Cooldowns are decremented in monster::process_turn
701
702 if( local_attack_data.cooldown == 0 && !pacified && !is_hallucination() ) {
703 if( !sp_type.second->call( *this ) ) {
704 continue;
705 }
706
707 // `special_attacks` might have changed at this point. Sadly `reset_special`
708 // doesn't check the attack name, so we need to do it here.
709 if( special_attacks.count( special_name ) == 0 ) {
710 continue;
711 }
712 reset_special( special_name );
713 }
714 }
715
716 // Check if they're dragging a foe and find their hapless victim
717 player *dragged_foe = find_dragged_foe();
718
719 // Give nursebots a chance to do surgery.
720 nursebot_operate( dragged_foe );
721
722 // The monster can sometimes hang in air due to last fall being blocked
723 if( !flies() && g->m.has_flag( TFLAG_NO_FLOOR, pos() ) ) {
724 g->m.creature_on_trap( *this, false );
725 if( is_dead() ) {
726 return;
727 }
728 }
729
730 // if the monster is in a deep water tile, it has a chance to drown
731 if( die_if_drowning( pos(), 10 ) ) {
732 return;
733 }
734
735 if( moves < 0 ) {
736 return;
737 }
738
739 // TODO: Move this to attack_at/move_to/etc. functions
740 bool attacking = false;
741 if( !move_effects( attacking ) ) {
742 moves = 0;
743 return;
744 }
745 if( has_flag( MF_IMMOBILE ) || has_flag( MF_RIDEABLE_MECH ) ) {
746 moves = 0;
747 return;
748 }
749 if( has_effect( effect_stunned ) ) {
750 stumble();
751 moves = 0;
752 return;
753 }
754 if( friendly > 0 ) {
755 --friendly;
756 }
757 if( has_effect( effect_ai_waiting ) ) {
758 moves = 0;
759 return;
760 }
761
762 // don't move if a passenger in a moving vehicle
763 auto vp = g->m.veh_at( pos() );
764 bool harness_part = static_cast<bool>( g->m.veh_at( pos() ).part_with_feature( "ANIMAL_CTRL",
765 true ) );
766 if( vp && vp->vehicle().is_moving() && vp->vehicle().get_pet( vp->part_index() ) ) {
767 moves = 0;
768 return;
769 // Don't move if harnessed, even if vehicle is stationary
770 } else if( vp && has_effect( effect_harnessed ) ) {
771 moves = 0;
772 return;
773 // If harnessed monster finds itself moved from the harness point, the harness probably broke!
774 } else if( !harness_part && has_effect( effect_harnessed ) ) {
775 remove_effect( effect_harnessed );
776 }
777 // Set attitude to attitude to our current target
778 monster_attitude current_attitude = attitude( nullptr );
779 if( !wander() ) {
780 if( goal == g->u.pos() ) {
781 current_attitude = attitude( &g->u );
782 } else {
783 for( const npc &guy : g->all_npcs() ) {
784 if( goal == guy.pos() ) {
785 current_attitude = attitude( &guy );
786 }
787 }
788 }
789 }
790
791 if( current_attitude == MATT_IGNORE ||
792 ( current_attitude == MATT_FOLLOW && rl_dist( pos(), goal ) <= MONSTER_FOLLOW_DIST ) ) {
793 moves -= 100;
794 stumble();
795 return;
796 }
797
798 bool moved = false;
799 tripoint destination;
800
801 bool try_to_move = false;
802 for( const tripoint &dest : g->m.points_in_radius( pos(), 1 ) ) {
803 if( dest != pos() ) {
804 if( can_move_to( dest ) &&
805 g->critter_at( dest, true ) == nullptr ) {
806 try_to_move = true;
807 break;
808 }
809 }
810 }
811 // If true, don't try to greedily avoid locally bad paths
812 bool pathed = false;
813 if( try_to_move ) {
814 if( !wander() ) {
815 while( !path.empty() && path.front() == pos() ) {
816 path.erase( path.begin() );
817 }
818
819 const auto &pf_settings = get_pathfinding_settings();
820 if( pf_settings.max_dist >= rl_dist( pos(), goal ) &&
821 ( path.empty() || rl_dist( pos(), path.front() ) >= 2 || path.back() != goal ) ) {
822 // We need a new path
823 path = g->m.route( pos(), goal, pf_settings, get_path_avoid() );
824 }
825
826 // Try to respect old paths, even if we can't pathfind at the moment
827 if( !path.empty() && path.back() == goal ) {
828 destination = path.front();
829 moved = true;
830 pathed = true;
831 } else {
832 // Straight line forward, probably because we can't pathfind (well enough)
833 destination = goal;
834 moved = true;
835 }
836 }
837 }
838 if( !moved && has_flag( MF_SMELLS ) ) {
839 // No sight... or our plans are invalid (e.g. moving through a transparent, but
840 // solid, square of terrain). Fall back to smell if we have it.
841 unset_dest();
842 tripoint tmp = scent_move();
843 if( tmp.x != -1 ) {
844 destination = tmp;
845 moved = true;
846 }
847 }
848 if( wandf > 0 && !moved && friendly == 0 ) { // No LOS, no scent, so as a fall-back follow sound
849 unset_dest();
850 if( wander_pos != pos() ) {
851 destination = wander_pos;
852 moved = true;
853 }
854 }
855
856 if( !g->m.has_zlevels() ) {
857 // Otherwise weird things happen
858 destination.z = posz();
859 }
860
861 point new_d( destination.xy() - pos().xy() );
862
863 // toggle facing direction for sdl flip
864 if( !tile_iso ) {
865 if( new_d.x < 0 ) {
866 facing = FD_LEFT;
867 } else if( new_d.x > 0 ) {
868 facing = FD_RIGHT;
869 }
870 } else {
871 if( new_d.y <= 0 && new_d.x <= 0 ) {
872 facing = FD_LEFT;
873 }
874 if( new_d.x >= 0 && new_d.y >= 0 ) {
875 facing = FD_RIGHT;
876 }
877 }
878
879 tripoint next_step;
880 const bool can_open_doors = has_flag( MF_CAN_OPEN_DOORS );
881 const bool staggers = has_flag( MF_STUMBLES );
882 if( moved ) {
883 // Implement both avoiding obstacles and staggering.
884 moved = false;
885 float switch_chance = 0.0;
886 const bool can_bash = bash_skill() > 0;
887 // This is a float and using trig_dist() because that Does the Right Thing(tm)
888 // in both circular and roguelike distance modes.
889 const float distance_to_target = trig_dist( pos(), destination );
890 for( tripoint &candidate : squares_closer_to( pos(), destination ) ) {
891 bool via_ramp = false;
892 if( here.has_flag( TFLAG_RAMP_UP, candidate ) ) {
893 via_ramp = true;
894 candidate.z += 1;
895 } else if( here.has_flag( TFLAG_RAMP_DOWN, candidate ) ) {
896 via_ramp = true;
897 candidate.z -= 1;
898 }
899 tripoint candidate_abs = g->m.getabs( candidate );
900
901 if( candidate.z != posz() ) {
902 bool can_z_move = true;
903 if( !here.valid_move( pos(), candidate, false, true, via_ramp ) ) {
904 // Can't phase through floor
905 can_z_move = false;
906 }
907
908 // If we're trying to go up but can't fly, check if we can climb. If we can't, then don't
909 // This prevents non-climb/fly enemies running up walls
910 if( candidate.z > posz() && !( via_ramp || flies() ) ) {
911 if( !can_climb() || !here.has_floor_or_support( candidate ) ) {
912 // Can't "jump" up a whole z-level
913 can_z_move = false;
914 }
915 }
916
917 // Last chance - we can still do the z-level stair teleport bullshit that isn't removed yet
918 // TODO: Remove z-level stair bullshit teleport after aligning all stairs
919 if( !can_z_move &&
920 posx() / ( SEEX * 2 ) == candidate.x / ( SEEX * 2 ) &&
921 posy() / ( SEEY * 2 ) == candidate.y / ( SEEY * 2 ) ) {
922 const tripoint &upper = candidate.z > posz() ? candidate : pos();
923 const tripoint &lower = candidate.z > posz() ? pos() : candidate;
924 if( g->m.has_flag( TFLAG_GOES_DOWN, upper ) && g->m.has_flag( TFLAG_GOES_UP, lower ) ) {
925 can_z_move = true;
926 }
927 }
928
929 if( !can_z_move ) {
930 continue;
931 }
932 }
933
934 // A flag to allow non-stumbling critters to stumble when the most direct choice is bad.
935 bool bad_choice = false;
936
937 const Creature *target = g->critter_at( candidate, is_hallucination() );
938 if( target != nullptr ) {
939 const Creature::Attitude att = attitude_to( *target );
940 if( att == A_HOSTILE ) {
941 // When attacking an adjacent enemy, we're direct.
942 moved = true;
943 next_step = candidate_abs;
944 break;
945 } else if( att == A_FRIENDLY && ( target->is_player() || target->is_npc() ) ) {
946 continue; // Friendly firing the player or an NPC is illegal for gameplay reasons
947 } else if( !has_flag( MF_ATTACKMON ) && !has_flag( MF_PUSH_MON ) ) {
948 // Bail out if there's a non-hostile monster in the way and we're not pushy.
949 continue;
950 }
951 // Friendly fire and pushing are always bad choices - they take a lot of time
952 bad_choice = true;
953 }
954
955 map &here = g->m;
956 // is there an openable door?
957 if( can_open_doors &&
958 here.open_door( candidate, !here.is_outside( pos() ), true ) ) {
959 moved = true;
960 next_step = candidate_abs;
961 continue;
962 }
963
964 // Try to shove vehicle out of the way
965 shove_vehicle( destination, candidate );
966 // Bail out if we can't move there and we can't bash.
967 if( !pathed && !can_move_to( candidate ) ) {
968 if( !can_bash ) {
969 continue;
970 }
971 // Don't bash if we're just tracking a noise.
972 if( wander() && destination == wander_pos ) {
973 continue;
974 }
975 const int estimate = here.bash_rating( bash_estimate(), candidate );
976 if( estimate <= 0 ) {
977 continue;
978 }
979
980 if( estimate < 5 ) {
981 bad_choice = true;
982 }
983 }
984
985 const float progress = distance_to_target - trig_dist( candidate, destination );
986 // The x2 makes the first (and most direct) path twice as likely,
987 // since the chance of switching is 1/1, 1/4, 1/6, 1/8
988 switch_chance += progress * 2;
989 // Randomly pick one of the viable squares to move to weighted by distance.
990 if( progress > 0 && ( !moved || x_in_y( progress, switch_chance ) ) ) {
991 moved = true;
992 next_step = candidate_abs;
993 // If we stumble, pick a random square, otherwise take the first one,
994 // which is the most direct path.
995 // Except if the direct path is bad, then check others
996 // Or if the path is given by pathfinder
997 if( !staggers && ( !bad_choice || pathed ) ) {
998 break;
999 }
1000 }
1001 }
1002 }
1003 // Finished logic section. By this point, we should have chosen a square to
1004 // move to (moved = true).
1005 const tripoint local_next_step = g->m.getlocal( next_step );
1006 if( moved ) { // Actual effects of moving to the square we've chosen
1007 const bool did_something =
1008 ( !pacified && attack_at( local_next_step ) ) ||
1009 ( !pacified && can_open_doors && g->m.open_door( local_next_step, !g->m.is_outside( pos() ) ) ) ||
1010 ( !pacified && bash_at( local_next_step ) ) ||
1011 ( !pacified && push_to( local_next_step, 0, 0 ) ) ||
1012 move_to( local_next_step, false, false, get_stagger_adjust( pos(), destination, local_next_step ) );
1013
1014 if( !did_something ) {
1015 moves -= 100; // If we don't do this, we'll get infinite loops.
1016 }
1017 if( has_effect( effect_dragging ) && dragged_foe != nullptr ) {
1018
1019 if( !dragged_foe->has_effect( effect_grabbed ) ) {
1020 dragged_foe = nullptr;
1021 remove_effect( effect_dragging );
1022 } else if( g->m.getlocal( drag_to ) != pos() &&
1023 g->critter_at( g->m.getlocal( drag_to ) ) == nullptr ) {
1024 dragged_foe->setpos( g->m.getlocal( drag_to ) );
1025 }
1026 }
1027 } else {
1028 moves -= 100;
1029 stumble();
1030 path.clear();
1031 }
1032}

1035{

1038 dragged_foe_id = character_id();
1039 return nullptr;
1040 }

1057void monster::nursebot_operate( player *dragged_foe )
1058{
1059 // No dragged foe, nothing to do.
1060 if( dragged_foe == nullptr ) {
1061 return;
1062 }
1063
1064 // Nothing to do if they can't operate, or they don't think they're dragging.
1065 if( !( type->has_special_attack( "OPERATE" ) && has_effect( effect_dragging ) ) ) {
1066 return;
1067 }
1068
1069 if( rl_dist( pos(), goal ) == 1 && !g->m.has_flag_furn( flag_AUTODOC_COUCH, goal ) &&
1070 !has_effect( effect_operating ) ) {
1071 if( dragged_foe->has_effect( effect_grabbed ) && !has_effect( effect_countdown ) &&
1072 ( g->critter_at( goal ) == nullptr || g->critter_at( goal ) == dragged_foe ) ) {
1073 add_msg( m_bad, _( "The %1$s slowly but firmly puts %2$s down onto the autodoc couch." ), name(),
1074 dragged_foe->disp_name() );
1075
1076 dragged_foe->setpos( goal );
1077
1078 // There's still time to get away
1079 add_effect( effect_countdown, 2_turns );
1080 add_msg( m_bad, _( "The %s produces a syringe full of some translucent liquid." ), name() );
1081 } else if( g->critter_at( goal ) != nullptr && has_effect( effect_dragging ) ) {
1082 sounds::sound( pos(), 8, sounds::sound_t::electronic_speech,
1083 string_format(
1084 _( "a soft robotic voice say, \"Please step away from the autodoc, this patient needs immediate care.\"" ) ) );
1085 // TODO: Make it able to push NPC/player
1086 push_to( goal, 4, 0 );
1087 }
1088 }
1089 if( get_effect_dur( effect_countdown ) == 1_turns && !has_effect( effect_operating ) ) {
1090 if( dragged_foe->has_effect( effect_grabbed ) ) {
1091
1092 const bionic_collection &collec = *dragged_foe->my_bionics;
1093 const int index = rng( 0, collec.size() - 1 );
1094 const bionic &target_cbm = collec[index];
1095
1096 //8 intelligence*4 + 8 first aid*4 + 3 computer *3 + 4 electronic*1 = 77
1097 const float adjusted_skill = static_cast<float>( 77 ) - std::min( static_cast<float>( 40 ),
1098 static_cast<float>( 77 ) - static_cast<float>( 77 ) / static_cast<float>( 10.0 ) );
1099
1100 g->u.uninstall_bionic( target_cbm, *this, *dragged_foe, adjusted_skill );
1101
1102 dragged_foe->remove_effect( effect_grabbed );
1103 remove_effect( effect_dragging );
1104 dragged_foe_id = character_id();
1105
1106 }
1107 }

1154tripoint monster::scent_move()
1155{
1156 // TODO: Remove when scentmap is 3D
1157 if( std::abs( posz() - g->get_levz() ) > SCENT_MAP_Z_REACH ) {
1158 return { -1, -1, INT_MIN };
1159 }
1160
1161 const std::set<scenttype_id> &tracked_scents = type->scents_tracked;
1162 const std::set<scenttype_id> &ignored_scents = type->scents_ignored;
1163
1164 std::vector<tripoint> smoves;
1165
1166 int bestsmell = 10; // Squares with smell 0 are not eligible targets.
1167 int smell_threshold = 200; // Squares at or above this level are ineligible.
1168 if( has_flag( MF_KEENNOSE ) ) {
1169 bestsmell = 1;
1170 smell_threshold = 400;
1171 }
1172
1173 const bool fleeing = is_fleeing( g->u );
1174 if( fleeing ) {
1175 bestsmell = g->scent.get( pos() );
1176 }
1177
1178 tripoint next( -1, -1, posz() );
1179 if( ( !fleeing && g->scent.get( pos() ) > smell_threshold ) ||
1180 ( fleeing && bestsmell == 0 ) ) {
1181 return next;
1182 }
1183 const bool can_bash = bash_skill() > 0;
1184 for( const auto &dest : g->m.points_in_radius( pos(), 1, SCENT_MAP_Z_REACH ) ) {
1185 int smell = g->scent.get( dest );
1186 const scenttype_id &type_scent = g->scent.get_type( dest );
1187
1188 bool right_scent = false;
1189 // is the monster tracking this scent
1190 if( !tracked_scents.empty() ) {
1191 right_scent = tracked_scents.find( type_scent ) != tracked_scents.end();
1192 }
1193 //is this scent recognised by the monster species
1194 if( !type_scent.is_empty() ) {
1195 const std::set<species_id> &receptive_species = type_scent->receptive_species;
1196 const std::set<species_id> &monster_species = type->species;
1197 std::vector<species_id> v_intersection;
1198 std::set_intersection( receptive_species.begin(), receptive_species.end(), monster_species.begin(),
1199 monster_species.end(), std::back_inserter( v_intersection ) );
1200 if( !v_intersection.empty() ) {
1201 right_scent = true;
1202 }
1203 }
1204 // is the monster actually ignoring this scent
1205 if( !ignored_scents.empty() && ( ignored_scents.find( type_scent ) != ignored_scents.end() ) ) {
1206 right_scent = false;
1207 }
1208
1209 if( ( !fleeing && smell < bestsmell ) || ( fleeing && smell > bestsmell ) || !right_scent ) {
1210 continue;
1211 }
1212 if( g->m.valid_move( pos(), dest, can_bash, true ) &&
1213 ( can_move_to( dest ) || ( dest == g->u.pos() ) ||
1214 ( can_bash && g->m.bash_rating( bash_estimate(), dest ) > 0 ) ) ) {
1215 if( ( !fleeing && smell > bestsmell ) || ( fleeing && smell < bestsmell ) ) {
1216 smoves.clear();
1217 smoves.push_back( dest );
1218 bestsmell = smell;
1219 } else if( ( !fleeing && smell == bestsmell ) || ( fleeing && smell == bestsmell ) ) {
1220 smoves.push_back( dest );
1221 }
1222 }
1223 }
1224
1225 return random_entry( smoves, next );
1226}

1228int monster::calc_movecost( const tripoint &f, const tripoint &t ) const
1229{
1230 int movecost = 0;
1231
1232 const int source_cost = g->m.move_cost( f );
1233 const int dest_cost = g->m.move_cost( t );
1234 // Digging and flying monsters ignore terrain cost
1235 if( flies() || ( digging() && g->m.has_flag( "DIGGABLE", t ) ) ) {
1236 movecost = 100;
1237 // Swimming monsters move super fast in water
1238 } else if( swims() ) {
1239 if( g->m.has_flag( "SWIMMABLE", f ) ) {
1240 movecost += 25;
1241 } else {
1242 movecost += 50 * g->m.move_cost( f );
1243 }
1244 if( g->m.has_flag( "SWIMMABLE", t ) ) {
1245 movecost += 25;
1246 } else {
1247 movecost += 50 * g->m.move_cost( t );
1248 }
1249 } else if( can_submerge() ) {
1250 // No-breathe monsters have to walk underwater slowly
1251 if( g->m.has_flag( "SWIMMABLE", f ) ) {
1252 movecost += 250;
1253 } else {
1254 movecost += 50 * g->m.move_cost( f );
1255 }
1256 if( g->m.has_flag( "SWIMMABLE", t ) ) {
1257 movecost += 250;
1258 } else {
1259 movecost += 50 * g->m.move_cost( t );
1260 }
1261 movecost /= 2;
1262 } else if( climbs() ) {
1263 if( g->m.has_flag( "CLIMBABLE", f ) ) {
1264 movecost += 150;
1265 } else {
1266 movecost += 50 * g->m.move_cost( f );
1267 }
1268 if( g->m.has_flag( "CLIMBABLE", t ) ) {
1269 movecost += 150;
1270 } else {
1271 movecost += 50 * g->m.move_cost( t );
1272 }
1273 movecost /= 2;
1274 } else {
1275 movecost = ( ( 50 * source_cost ) + ( 50 * dest_cost ) ) / 2.0;
1276 }
1277
1278 return movecost;
1279}

1297/*
1298 * Return points of an area extending 1 tile to either side and
1299 * (maxdepth) tiles behind basher.
1300 */
1301static std::vector<tripoint> get_bashing_zone( const tripoint &bashee, const tripoint &basher,
1302 int maxdepth )
1303{
1304 std::vector<tripoint> direction;
1305 direction.push_back( bashee );
1306 direction.push_back( basher );
1307 // Draw a line from the target through the attacker.
1308 std::vector<tripoint> path = continue_line( direction, maxdepth );
1309 // Remove the target.
1310 path.insert( path.begin(), basher );
1311 std::vector<tripoint> zone;
1312 // Go ahead and reserve enough room for all the points since
1313 // we know how many it will be.
1314 zone.reserve( 3 * maxdepth );
1315 tripoint previous = bashee;
1316 for( const tripoint &p : path ) {
1317 std::vector<point> swath = squares_in_direction( previous.xy(), p.xy() );
1318 for( const point &q : swath ) {
1319 zone.push_back( tripoint( q, bashee.z ) );
1320 }
1321
1322 previous = p;
1323 }
1324 return zone;
1325}

1327bool monster::bash_at( const tripoint &p )
1328{

1386int monster::group_bash_skill( const tripoint &target )
1387{
1388 if( !has_flag( MF_GROUP_BASH ) ) {
1389 return bash_skill();
1390 }
1391 int bashskill = 0;
1392
1393 // pileup = more bash skill, but only help bashing mob directly in front of target
1394 const int max_helper_depth = 5;
1395 const std::vector<tripoint> bzone = get_bashing_zone( target, pos(), max_helper_depth );
1396
1397 for( const tripoint &candidate : bzone ) {
1398 // Drawing this line backwards excludes the target and includes the candidate.
1399 std::vector<tripoint> path_to_target = line_to( target, candidate, 0, 0 );
1400 bool connected = true;
1401 monster *mon = nullptr;
1402 for( const tripoint &in_path : path_to_target ) {
1403 // If any point in the line from zombie to target is not a cooperating zombie,
1404 // it can't contribute.
1405 mon = g->critter_at<monster>( in_path );
1406 if( !mon ) {
1407 connected = false;
1408 break;
1409 }
1410 monster &helpermon = *mon;
1411 if( !helpermon.has_flag( MF_GROUP_BASH ) || helpermon.is_hallucination() ) {
1412 connected = false;
1413 break;
1414 }
1415 }
1416 if( !connected || !mon ) {
1417 continue;
1418 }
1419 // If we made it here, the last monster checked was the candidate.
1420 monster &helpermon = *mon;
1421 // Contribution falls off rapidly with distance from target.
1422 bashskill += helpermon.bash_skill() / rl_dist( candidate, target );
1423 }
1424
1425 return bashskill;
1426}

1493bool monster::move_to( const tripoint &p, bool force, bool step_on_critter,
1494 const float stagger_adjustment )
1495{
1496 const bool on_ground = !digging() && !flies();
1497
1498 const bool z_move = p.z != pos().z;
1499 const bool going_up = p.z > pos().z;
1500
1501 tripoint destination = p;
1502
1503 // This is stair teleportation hackery.
1504 // TODO: Remove this in favor of stair alignment
1505 if( going_up ) {
1506 if( g->m.has_flag( TFLAG_GOES_UP, pos() ) ) {
1507 destination = find_closest_stair( p, TFLAG_GOES_DOWN );
1508 }
1509 } else if( z_move ) {
1510 if( g->m.has_flag( TFLAG_GOES_DOWN, pos() ) ) {
1511 destination = find_closest_stair( p, TFLAG_GOES_UP );
1512 }
1513 }
1514
1515 // Allows climbing monsters to move on terrain with movecost <= 0
1516 Creature *critter = g->critter_at( destination, is_hallucination() );
1517 if( g->m.has_flag( "CLIMBABLE", destination ) ) {
1518 if( g->m.impassable( destination ) && critter == nullptr ) {
1519 if( flies() ) {
1520 moves -= 100;
1521 force = true;
1522 if( g->u.sees( *this ) ) {
1523 add_msg( _( "The %1$s flies over the %2$s." ), name(),
1524 g->m.has_flag_furn( "CLIMBABLE", p ) ? g->m.furnname( p ) :
1525 g->m.tername( p ) );
1526 }
1527 } else if( climbs() ) {
1528 moves -= 150;
1529 force = true;
1530 if( g->u.sees( *this ) ) {
1531 add_msg( _( "The %1$s climbs over the %2$s." ), name(),
1532 g->m.has_flag_furn( "CLIMBABLE", p ) ? g->m.furnname( p ) :
1533 g->m.tername( p ) );
1534 }
1535 }
1536 }
1537 }
1538
1539 if( critter != nullptr && !step_on_critter ) {
1540 return false;
1541 }
1542
1543 // Make sure that we can move there, unless force is true.
1544 if( !force && !can_move_to( destination ) ) {
1545 return false;
1546 }
1547
1548 if( !force ) {
1549 // This adjustment is to make it so that monster movement speed relative to the player
1550 // is consistent even if the monster stumbles,
1551 // and the same regardless of the distance measurement mode.
1552 // Note: Keep this as float here or else it will cancel valid moves
1553 const float cost = stagger_adjustment *
1554 static_cast<float>( climbs() &&
1555 g->m.has_flag( TFLAG_NO_FLOOR, p ) ? calc_climb_cost( pos(), destination ) : calc_movecost( pos(),
1556 destination ) );
1557 if( cost > 0.0f ) {
1558 moves -= static_cast<int>( std::ceil( cost ) );
1559 } else {
1560 return false;
1561 }
1562 }
1563
1564 //Check for moving into/out of water
1565 bool was_water = g->m.is_divable( pos() );
1566 bool will_be_water = on_ground && can_submerge() && g->m.is_divable( destination );
1567
1568 // Attitude check is kinda slow, better gate it
1569 if( was_water != will_be_water && !flies() ) {
1570 if( attitude( &g->u ) != MATT_ATTACK ) {
1571 // Nothing, no need to spam
1572 } else if( was_water && !will_be_water && g->u.sees( p ) ) {
1573 // Use more dramatic messages for swimming monsters
1574 //~ Message when a monster emerges from water
1575 //~ %1$s: monster name, %2$s: leaps/emerges, %3$s: terrain name
1576 add_msg( m_warning, pgettext( "monster movement", "A %1$s %2$s from the %3$s!" ), name(),
1577 swims() || has_flag( MF_AQUATIC ) ? _( "leaps" ) : _( "emerges" ),
1578 g->m.tername( pos() ) );
1579 } else if( !was_water && will_be_water && g->u.sees( destination ) ) {
1580 //~ Message when a monster enters water
1581 //~ %1$s: monster name, %2$s: dives/sinks, %3$s: terrain name
1582 add_msg( m_warning, pgettext( "monster movement", "A %1$s %2$s into the %3$s!" ), name(),
1583 swims() || has_flag( MF_AQUATIC ) ? _( "dives" ) : _( "sinks" ),
1584 g->m.tername( destination ) );
1585 }
1586 }
1587
1588 setpos( destination );
1589 footsteps( destination );
1590 underwater = will_be_water;
1591 if( is_hallucination() ) {
1592 //Hallucinations don't do any of the stuff after this point
1593 return true;
1594 }
1595
1596 if( type->size != MS_TINY && on_ground ) {
1597 const int sharp_damage = rng( 1, 10 );
1598 const int rough_damage = rng( 1, 2 );
1599 if( g->m.has_flag( "SHARP", pos() ) && !one_in( 4 ) &&
1600 get_armor_cut( bodypart_id( "torso" ) ) < sharp_damage ) {
1601 apply_damage( nullptr, bodypart_id( "torso" ), sharp_damage );
1602 }
1603 if( g->m.has_flag( "ROUGH", pos() ) && one_in( 6 ) &&
1604 get_armor_cut( bodypart_id( "torso" ) ) < rough_damage ) {
1605 apply_damage( nullptr, bodypart_id( "torso" ), rough_damage );
1606 }
1607 }
1608
1609 if( g->m.has_flag( "UNSTABLE", destination ) && on_ground ) {
1610 add_effect( effect_bouldering, 1_turns, num_bp );
1611 } else if( has_effect( effect_bouldering ) ) {
1612 remove_effect( effect_bouldering );
1613 }
1614
1615 if( g->m.has_flag_ter_or_furn( TFLAG_NO_SIGHT, destination ) && on_ground ) {
1616 add_effect( effect_no_sight, 1_turns, num_bp );
1617 } else if( has_effect( effect_no_sight ) ) {
1618 remove_effect( effect_no_sight );
1619 }
1620
1621 g->m.creature_on_trap( *this );
1622 if( is_dead() ) {
1623 return true;

1626 underwater = g->m.has_flag( "DIGGABLE", pos() );

1629 if( digging() && g->m.has_flag( "DIGGABLE", pos() ) ) {

1634 break;
1635 case MS_SMALL:
1636 factor = 30;
1637 break;
1638 case MS_MEDIUM:
1639 factor = 6;
1640 break;

1642 factor = 3;

1645 factor = 1;
1646 break;

1648 factor = 6;
1649 break;

1652 if( one_in( factor ) ) {
1653 g->m.ter_set( pos(), t_dirtmound );
1654 }
1655 }

1657 if( has_flag( MF_ACIDTRAIL ) ) {

1692 g->m.add_item_or_charges( pos(), item( "gasoline" ) );
1693 }
1694 }
1695 return true;

1698bool monster::push_to( const tripoint &p, const int boost, const size_t depth )
1699{
1700 if( is_hallucination() ) {
1701 // Don't let hallucinations push, not even other hallucinations
1702 return false;
1703 }
1704
1705 if( !has_flag( MF_PUSH_MON ) || depth > 2 || has_effect( effect_pushed ) ) {
1706 return false;
1707 }
1708
1709 // TODO: Generalize this to Creature
1710 monster *const critter = g->critter_at<monster>( p );
1711 if( critter == nullptr || critter == this ||
1712 p == pos() || critter->movement_impaired() ) {
1713 return false;
1714 }
1715
1716 if( !can_move_to( p ) ) {
1717 return false;
1718 }
1719
1720 if( critter->is_hallucination() ) {
1721 // Kill the hallu, but return false so that the regular move_to is uses instead
1722 critter->die( nullptr );
1723 return false;
1724 }
1725
1726 // Stability roll of the pushed critter
1727 const int defend = critter->stability_roll();
1728 // Stability roll of the pushing zed
1729 const int attack = stability_roll() + boost;
1730 if( defend > attack ) {
1731 return false;
1732 }
1733
1734 const int movecost_from = 50 * g->m.move_cost( p );
1735 const int movecost_attacker = std::max( movecost_from, 200 - 10 * ( attack - defend ) );
1736 const tripoint dir = p - pos();
1737
1738 // Mark self as pushed to simplify recursive pushing
1739 add_effect( effect_pushed, 1_turns );
1740
1741 for( size_t i = 0; i < 6; i++ ) {
1742 const point d{ rng( -1, 1 ), rng( -1, 1 ) };
1743 if( d.x == 0 && d.y == 0 ) {
1744 continue;
1745 }
1746
1747 // Pushing forward is easier than pushing aside
1748 const int direction_penalty = std::abs( d.x - dir.x ) + std::abs( d.y - dir.y );
1749 if( direction_penalty > 2 ) {
1750 continue;
1751 }
1752
1753 tripoint dest( p + d );
1754 const int dest_movecost_from = 50 * g->m.move_cost( dest );
1755
1756 // Pushing into cars/windows etc. is harder
1757 const int movecost_penalty = g->m.move_cost( dest ) - 2;
1758 if( movecost_penalty <= -2 ) {
1759 // Can't push into unpassable terrain
1760 continue;
1761 }
1762
1763 int roll = attack - ( defend + direction_penalty + movecost_penalty );
1764 if( roll < 0 ) {
1765 continue;
1766 }
1767
1768 Creature *critter_recur = g->critter_at( dest );
1769 if( !( critter_recur == nullptr || critter_recur->is_hallucination() ) ) {
1770 // Try to push recursively
1771 monster *mon_recur = dynamic_cast< monster * >( critter_recur );
1772 if( mon_recur == nullptr ) {
1773 continue;
1774 }
1775
1776 if( critter->push_to( dest, roll, depth + 1 ) ) {
1777 // The tile isn't necessarily free, need to check
1778 if( !g->critter_at( p ) ) {
1779 move_to( p );
1780 }
1781
1782 moves -= movecost_attacker;
1783
1784 // Don't knock down a creature that successfully
1785 // pushed another creature, just reduce moves
1786 critter->moves -= dest_movecost_from;
1787 return true;
1788 } else {
1789 return false;
1790 }
1791 }
1792
1793 critter_recur = g->critter_at( dest );
1794 if( critter_recur != nullptr ) {
1795 if( critter_recur->is_hallucination() ) {
1796 critter_recur->die( nullptr );
1797 }
1798 } else if( !critter->has_flag( MF_IMMOBILE ) ) {
1799 critter->setpos( dest );
1800 move_to( p );
1801 moves -= movecost_attacker;
1802 critter->add_effect( effect_downed, time_duration::from_turns( movecost_from / 100 + 1 ) );
1803 }
1804 return true;
1805 }
1806
1807 // Try to trample over a much weaker zed (or one with worse rolls)
1808 // Don't allow trampling with boost
1809 if( boost > 0 || attack < 2 * defend ) {
1810 return false;
1811 }
1812
1813 g->swap_critters( *critter, *this );
1814 critter->add_effect( effect_stunned, rng( 0_turns, 2_turns ) );
1815 // Only print the message when near player or it can get spammy
1816 if( rl_dist( g->u.pos(), pos() ) < 4 && g->u.sees( *critter ) ) {
1817 add_msg( m_warning, _( "The %1$s tramples %2$s" ),
1818 name(), critter->disp_name() );
1819 }
1820
1821 moves -= movecost_attacker;
1822 if( movecost_from > 100 ) {
1823 critter->add_effect( effect_downed, time_duration::from_turns( movecost_from / 100 + 1 ) );
1824 } else {
1825 critter->moves -= movecost_from;
1826 }
1827
1828 return true;
1829}

1832 * Stumble in a random direction, but with some caveats.
1833 */
1834void monster::stumble()
1835{
1836 // Only move every 10 turns.
1837 if( !one_in( 10 ) ) {
1838 return;
1839 }
1840
1841 map &here = get_map();
1842
1843 std::vector<tripoint> valid_stumbles;
1844 valid_stumbles.reserve( 11 );
1845 const bool avoid_water = has_flag( MF_NO_BREATHE ) && !swims() && !has_flag( MF_AQUATIC );
1846 for( const tripoint &dest : here.points_in_radius( pos(), 1 ) ) {
1847 if( dest != pos() ) {
1848 if( here.has_flag( TFLAG_RAMP_DOWN, dest ) ) {
1849 valid_stumbles.push_back( tripoint( dest.xy(), dest.z - 1 ) );
1850 } else if( here.has_flag( TFLAG_RAMP_UP, dest ) ) {
1851 valid_stumbles.push_back( tripoint( dest.xy(), dest.z + 1 ) );
1852 } else {
1853 valid_stumbles.push_back( dest );
1854 }
1855 }
1856 }
1857
1858 if( here.has_zlevels() ) {
1859 tripoint below( posx(), posy(), posz() - 1 );
1860 if( here.valid_move( pos(), below, false, true ) ) {
1861 valid_stumbles.push_back( below );
1862 }
1863 }
1864 while( !valid_stumbles.empty() && !is_dead() ) {
1865 const tripoint dest = random_entry_removed( valid_stumbles );
1866 if( can_move_to( dest ) &&
1867 //Stop zombies and other non-breathing monsters wandering INTO water
1868 //(Unless they can swim/are aquatic)
1869 //But let them wander OUT of water if they are there.
1870 !( avoid_water &&
1871 here.has_flag( TFLAG_SWIMMABLE, dest ) &&
1872 !here.has_flag( TFLAG_SWIMMABLE, pos() ) ) &&
1873 ( g->critter_at( dest, is_hallucination() ) == nullptr ) ) {
1874 if( move_to( dest, true, false ) ) {
1875 break;
1876 }
1877 }
1878 }

1881void monster::knock_back_to( const tripoint &to )
1882{
1883 if( to == pos() ) {
1884 return; // No effect
1885 }
1886
1887 if( is_hallucination() ) {
1888 die( nullptr );
1889 return;
1890 }
1891
1892 bool u_see = g->u.sees( to );
1893
1894 // First, see if we hit another monster
1895 if( monster *const z = g->critter_at<monster>( to ) ) {
1896 apply_damage( z, bodypart_id( "torso" ), z->type->size );
1897 add_effect( effect_stunned, 1_turns );
1898 if( type->size > 1 + z->type->size ) {
1899 z->knock_back_from( pos() ); // Chain reaction!
1900 z->apply_damage( this, bodypart_id( "torso" ), type->size );
1901 z->add_effect( effect_stunned, 1_turns );
1902 } else if( type->size > z->type->size ) {
1903 z->apply_damage( this, bodypart_id( "torso" ), type->size );
1904 z->add_effect( effect_stunned, 1_turns );
1905 }
1906 z->check_dead_state();
1907
1908 if( u_see ) {
1909 add_msg( _( "The %1$s bounces off a %2$s!" ), name(), z->name() );
1910 }
1911
1912 return;
1913 }
1914
1915 if( npc *const p = g->critter_at<npc>( to ) ) {
1916 apply_damage( p, bodypart_id( "torso" ), 3 );
1917 add_effect( effect_stunned, 1_turns );
1918 p->deal_damage( this, bodypart_id( "torso" ), damage_instance( DT_BASH, type->size ) );
1919 if( u_see ) {
1920 add_msg( _( "The %1$s bounces off %2$s!" ), name(), p->name );
1921 }
1922
1923 p->check_dead_state();
1924 return;
1925 }
1926
1927 // If we're still in the function at this point, we're actually moving a tile!
1928 // die_if_drowning will kill the monster if necessary, but if the deep water
1929 // tile is on a vehicle, we should check for swimmers out of water
1930 if( !die_if_drowning( to ) && has_flag( MF_AQUATIC ) ) {
1931 die( nullptr );
1932 if( u_see ) {
1933 add_msg( _( "The %s flops around and dies!" ), name() );
1934 }
1935 }
1936
1937 if( g->m.impassable( to ) ) {
1938
1939 // It's some kind of wall.
1940 apply_damage( nullptr, bodypart_id( "torso" ), type->size );
1941 add_effect( effect_stunned, 2_turns );
1942 if( u_see ) {
1943 add_msg( _( "The %1$s bounces off a %2$s." ), name(),
1944 g->m.obstacle_name( to ) );
1945 }
1946
1947 } else { // It's no wall
1948 setpos( to );
1949 }
1950 check_dead_state();
1951}

1953/* will_reach() is used for determining whether we'll get to stairs (and
1954 * potentially other locations of interest). It is generally permissive.
1955 * TODO: Pathfinding;

1957 Injure monsters if they're gonna be walking through pits or whatever

1959bool monster::will_reach( const point &p )
