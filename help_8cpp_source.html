43void help::deserialize( JsonIn &jsin )
44{
45 hotkeys.clear();
46
47 std::string note_colors = get_note_colors();
48 std::string dir_grid = get_dir_grid();
49
50 jsin.start_array();
51 while( !jsin.end_array() ) {
52 JsonObject jo = jsin.get_object();
53
54 std::vector<std::string> messages;
55 jo.read( "messages", messages );
56
57 for( auto &line : messages ) {
58 if( line == "<DRAW_NOTE_COLORS>" ) {
59 line = replace_all( line, "<DRAW_NOTE_COLORS>", note_colors );
60 continue;
61 } else if( line == "<HELP_DRAW_DIRECTIONS>" ) {
62 line = replace_all( line, "<HELP_DRAW_DIRECTIONS>", dir_grid );
63 continue;
64 }
65 }
66
67 jo.get_string( "type" ); // Mark as visited
68 std::string name = jo.get_string( "name" );
69 help_texts[jo.get_int( "order" )] = std::make_pair( name, messages );
70 hotkeys.push_back( get_hotkeys( name ) );
71 }
72}

74std::string help::get_dir_grid()
75{
76 static const std::array<action_id, 9> movearray = {{
77 ACTION_MOVE_FORTH_LEFT, ACTION_MOVE_FORTH, ACTION_MOVE_FORTH_RIGHT,
78 ACTION_MOVE_LEFT, ACTION_PAUSE, ACTION_MOVE_RIGHT,
79 ACTION_MOVE_BACK_LEFT, ACTION_MOVE_BACK, ACTION_MOVE_BACK_RIGHT
80 }
81 };
82
83 std::string movement = "<LEFTUP_0> <UP_0> <RIGHTUP_0> <LEFTUP_1> <UP_1> <RIGHTUP_1>\n"
84 " \\ | / \\ | /\n"
85 " \\|/ \\|/\n"
86 "<LEFT_0>--<pause_0>--<RIGHT_0> <LEFT_1>--<pause_1>--<RIGHT_1>\n"
87 " /|\\ /|\\\n"
88 " / | \\ / | \\\n"
89 "<LEFTDOWN_0> <DOWN_0> <RIGHTDOWN_0> <LEFTDOWN_1> <DOWN_1> <RIGHTDOWN_1>";
90
91 for( auto dir : movearray ) {
92 std::vector<char> keys = keys_bound_to( dir );
93 for( size_t i = 0; i < 2; i++ ) {
94 std::string what = "<" + action_ident( dir ) + string_format( "_%d>", i );
95 std::string with = i < keys.size()
96 ? string_format( "<color_light_blue>%s</color>", keys[i] )
97 : "<color_red>?</color>";
98 movement = replace_all( movement, what, with );
99 }
100 }
101
102 return movement;
103}
