100void damage_instance::add( const damage_unit &new_du )
101{
102 auto iter = std::find_if( damage_units.begin(), damage_units.end(),
103 [&new_du]( const damage_unit & du ) {
104 return du.type == new_du.type;
105 } );
106 if( iter == damage_units.end() ) {
107 damage_units.emplace_back( new_du );
108 } else {
109 damage_unit &du = *iter;
110 // Actually combining two instances of damage is complex and ambiguous
111 // So let's just add/multiply the values
112 du.amount += new_du.amount;
113 du.res_pen += new_du.res_pen;
114
115 du.damage_multiplier *= new_du.damage_multiplier;
116 du.res_mult *= new_du.res_mult;
117 }
118}

294static damage_unit load_damage_unit( const JsonObject &curr )
295{
296 damage_type dt = dt_by_name( curr.get_string( "damage_type" ) );
297 if( dt == DT_NULL ) {
298 curr.throw_error( "Invalid damage type" );
299 }
300
301 float amount = curr.get_float( "amount", 0 );
302 float arpen = curr.get_float( "armor_penetration", 0 );
303 float armor_mul = curr.get_float( "armor_multiplier", 1.0f );
304 float damage_mul = curr.get_float( "damage_multiplier", 1.0f );
305
306 // Legacy
307 float unc_armor_mul = curr.get_float( "constant_armor_multiplier", 1.0f );
308 float unc_damage_mul = curr.get_float( "constant_damage_multiplier", 1.0f );
309
310 return damage_unit( dt, amount, arpen, armor_mul * unc_armor_mul, damage_mul * unc_damage_mul );
311}

313static damage_unit load_damage_unit_inherit( const JsonObject &curr, const damage_instance &parent )
314{
315 damage_unit ret = load_damage_unit( curr );
316
317 const std::vector<damage_unit> &parent_damage = parent.damage_units;
318 auto du_iter = std::find_if( parent_damage.begin(),
319 parent_damage.end(), [&ret]( const damage_unit & dmg ) {
320 return dmg.type == ret.type;
321 } );
322
323 if( du_iter == parent_damage.end() ) {
324 return ret;
325 }
326
327 const damage_unit &parent_du = *du_iter;
328
329 if( !curr.has_float( "amount" ) ) {
330 ret.amount = parent_du.amount;
331 }
332 if( !curr.has_float( "armor_penetration" ) ) {
333 ret.res_pen = parent_du.res_pen;
334 }
335 if( !curr.has_float( "armor_multiplier" ) ) {
336 ret.res_mult = parent_du.res_mult;
337 }
338 if( !curr.has_float( "damage_multiplier" ) ) {
339 ret.damage_multiplier = parent_du.damage_multiplier;
340 }
341
342 return ret;
343}
