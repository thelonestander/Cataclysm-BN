84static std::vector<page> split_to_pages( const std::vector<talk_data> &responses, int page_w,
85 int page_h )
86{
87 page this_page;
88 std::vector<page> ret;
89 int fold_width = page_w - 3;
90 int this_h = 0;
91 for( const talk_data &resp : responses ) {
92 // Assemble single entry for printing
93 const std::vector<std::string> folded = foldstring( resp.text, fold_width );
94 page_entry this_entry;
95 this_entry.col = resp.col;
96 if( !folded.empty() ) {
97 this_entry.lines.push_back( string_format( "%c: %s", resp.letter, folded[0] ) );
98 for( size_t i = 1; i < folded.size(); i++ ) {
99 this_entry.lines.push_back( string_format( " %s", folded[i] ) );
100 }
101 }
102
103 // Add entry to current / new page
104 if( this_h + static_cast<int>( folded.size() ) > page_h ) {
105 ret.push_back( this_page );
106 this_page = page();
107 this_h = 0;
108 }
109 this_h += this_entry.lines.size();
110 this_page.entries.push_back( this_entry );
111 }
112 if( !this_page.entries.empty() ) {
113 ret.push_back( this_page );
114 }
115 return ret;
116}
