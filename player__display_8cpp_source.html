59static bool should_combine_bps( const player &p, body_part l, body_part r,
60 const item *selected_clothing )
61{
62 const auto enc_data = p.get_encumbrance();
63 return l != r && // are different parts
64 l == other_part( r ) && r == other_part( l ) && // are complementary parts
65 // same encumberance & temperature
66 enc_data[l] == enc_data[r] &&
67 temperature_print_rescaling( p.temp_conv[l] ) == temperature_print_rescaling( p.temp_conv[r] ) &&
68 // selected_clothing covers both or neither parts
69 ( !selected_clothing || ( selected_clothing->covers( l ) == selected_clothing->covers( r ) ) );
70}

72static std::vector<std::pair<body_part, bool>> list_and_combine_bps( const player &p,
73 const item *selected_clothing )
74{
75 // bool represents whether the part has been combined with its other half
76 std::vector<std::pair<body_part, bool>> bps;
77 for( auto bp : all_body_parts ) {
78 // assuming that a body part has at most one other half
79 if( other_part( other_part( bp ) ) != bp ) {
80 debugmsg( "Bodypart %d has more than one other half!", bp );
81 }
82 if( should_combine_bps( p, bp, other_part( bp ), selected_clothing ) ) {
83 if( bp < other_part( bp ) ) {
84 // only add the earlier one
85 bps.emplace_back( bp, true );
86 }
87 } else {
88 bps.emplace_back( bp, false );
89 }
90 }
91 return bps;

94void player::print_encumbrance( const catacurses::window &win, const int line,
95 const item *const selected_clothing ) const
96{
97 // bool represents whether the part has been combined with its other half
98 const std::vector<std::pair<body_part, bool>> bps = list_and_combine_bps( *this,
99 selected_clothing );
100
101 // width/height excluding title & scrollbar
102 const int height = getmaxy( win ) - 1;
103 bool draw_scrollbar = height < static_cast<int>( bps.size() );
104 const int width = getmaxx( win ) - ( draw_scrollbar ? 1 : 0 );
105 // index of the first printed bodypart from `bps`
106 const int firstline = clamp( line - height / 2, 0, std::max( 0,
107 static_cast<int>( bps.size() ) - height ) );
108
109 /*** I chose to instead only display X+Y instead of X+Y=Z. More room was needed ***
110 *** for displaying triple digit encumbrance, due to new encumbrance system. ***
111 *** If the player wants to see the total without having to do them maths, the ***
112 *** armor layers ui shows everything they want :-) -Davek ***/
113 const auto enc_data = get_encumbrance();
114 for( int i = 0; i < height; ++i ) {
115 int thisline = firstline + i;
116 if( thisline < 0 ) {
117 continue;
118 }
119 if( static_cast<size_t>( thisline ) >= bps.size() ) {
120 break;
121 }
122 const body_part bp = bps[thisline].first;
123 const bool combine = bps[thisline].second;
124 const encumbrance_data &e = enc_data[bp];
125 const bool highlighted = selected_clothing ? selected_clothing->covers( bp ) : false;
126 std::string out = body_part_name_as_heading( bp, combine ? 2 : 1 );
127 if( utf8_width( out ) > 7 ) {
128 out = utf8_truncate( out, 7 );
129 }
130
131 // Two different highlighting schemes, highlight if the line is selected as per line being set.
132 // Make the text green if this part is covered by the passed in item.
133 nc_color limb_color = thisline == line ?
134 ( highlighted ? h_green : h_light_gray ) :
135 ( highlighted ? c_green : c_light_gray );
136 mvwprintz( win, point( 1, 1 + i ), limb_color, "%s", out );
137 // accumulated encumbrance from clothing, plus extra encumbrance from layering
138 mvwprintz( win, point( 8, 1 + i ), encumb_color( e.encumbrance ), "%3d",
139 e.encumbrance - e.layer_penalty );
140 // separator in low toned color
141 mvwprintz( win, point( 11, 1 + i ), c_light_gray, "+" );
142 // take into account the new encumbrance system for layers
143 mvwprintz( win, point( 12, 1 + i ), encumb_color( e.encumbrance ), "%-3d", e.layer_penalty );
144 // print warmth, tethered to right hand side of the window
145 mvwprintz( win, point( width - 6, 1 + i ), bodytemp_color( bp ), "(% 3d)",
146 temperature_print_rescaling( temp_conv[bp] ) );
147 }
148
149 if( draw_scrollbar ) {
150 scrollbar().
151 offset_x( width ).
152 offset_y( 1 ).
153 content_size( bps.size() ).
154 viewport_pos( firstline ).
155 viewport_size( height ).
156 scroll_to_last( false ).
157 apply( win );
158 }
159}

766 // NOLINTNEXTLINE(cata-use-named-point-constants)

770 int pen = 0;
771 unsigned int line = 3;
772 if( you.weight_carried() > you.weight_capacity() ) {

774 mvwprintz( w_speed, point( 1, line ), c_red,
775 pgettext( "speed penalty", "Overburdened -%2d%%" ), pen );

777 }
778 pen = you.get_pain_penalty().speed;
779 if( pen >= 1 ) {
780 mvwprintz( w_speed, point( 1, line ), c_red,
781 pgettext( "speed penalty", "Pain -%2d%%" ), pen );
782 line++;
783 }
784 if( you.get_thirst() > thirst_levels::very_thirsty ) {
785 pen = std::abs( player::thirst_speed_penalty( you.get_thirst() ) );
786 mvwprintz( w_speed, point( 1, line ), c_red,
787 pgettext( "speed penalty", "Thirst -%2d%%" ), pen );
788 line++;
789 }
790 if( you.kcal_speed_penalty() < 0 ) {
791 pen = std::abs( you.kcal_speed_penalty() );
792 //~ %s: Starving/Underfed (already left-justified), %2d: speed penalty
793 mvwprintz( w_speed, point( 1, line ), c_red, pgettext( "speed penalty", "%s-%2d%%" ),
794 left_justify( _( "Starving" ), 20 ), pen );
795 line++;
796 }
797 if( you.has_trait( trait_id( "SUNLIGHT_DEPENDENT" ) ) && !g->is_in_sunlight( you.pos() ) ) {
798 pen = ( g->light_level( you.posz() ) >= 12 ? 5 : 10 );
799 mvwprintz( w_speed, point( 1, line ), c_red,
800 pgettext( "speed penalty", "Out of Sunlight -%2d%%" ), pen );
801 line++;
802 }
803
804 const float temperature_speed_modifier = you.mutation_value( "temperature_speed_modifier" );
805 if( temperature_speed_modifier != 0 ) {
806 nc_color pen_color;
807 std::string pen_sign;

810 pen_color = c_green;
811 pen_sign = "+";
812 } else if( player_local_temp < 65 ) {
813 pen_color = c_red;
814 pen_sign = "-";
815 }

817 pen = ( player_local_temp - 65 ) * temperature_speed_modifier;
818 mvwprintz( w_speed, point( 1, line ), pen_color,
819 //~ %s: sign of bonus/penalty, %2d: speed bonus/penalty
820 pgettext( "speed modifier", "Cold-Blooded %s%2d%%" ), pen_sign, std::abs( pen ) );
821 line++;
822 }
823 }
824
825 int quick_bonus = static_cast<int>( newmoves - ( newmoves / 1.1 ) );
826 int bio_speed_bonus = quick_bonus;
827 if( you.has_trait( trait_id( "QUICK" ) ) && you.has_bionic( bionic_id( "bio_speed" ) ) ) {
828 bio_speed_bonus = static_cast<int>( newmoves / 1.1 - ( newmoves / 1.1 / 1.1 ) );
829 std::swap( quick_bonus, bio_speed_bonus );
830 }
831 if( you.has_trait( trait_id( "QUICK" ) ) ) {
832 mvwprintz( w_speed, point( 1, line ), c_green,
833 pgettext( "speed bonus", "Quick +%2d%%" ), quick_bonus );
834 line++;
835 }
836 if( you.has_bionic( bionic_id( "bio_speed" ) ) ) {
837 mvwprintz( w_speed, point( 1, line ), c_green,
838 pgettext( "speed bonus", "Bionic Speed +%2d%%" ), bio_speed_bonus );
839 line++;
840 }
841
842 for( const std::pair<const std::string, int> &speed_effect : speed_effects ) {
843 nc_color col = ( speed_effect.second > 0 ? c_green : c_red );
844 mvwprintz( w_speed, point( 1, line ), col, "%s", speed_effect.first );
845 mvwprintz( w_speed, point( 21, line ), col, ( speed_effect.second > 0 ? "+" : "-" ) );
846 mvwprintz( w_speed, point( std::abs( speed_effect.second ) >= 10 ? 22 : 23, line ), col, "%d%%",
847 std::abs( speed_effect.second ) );
848 line++;
849 }
850
851 int runcost = you.run_cost( 100 );
852 nc_color col = ( runcost <= 100 ? c_green : c_red );
853 mvwprintz( w_speed, point( 21 + ( runcost >= 100 ? 0 : ( runcost < 10 ? 2 : 1 ) ), 1 ), col,
854 "%d", runcost );
855 col = ( newmoves >= 100 ? c_green : c_red );
856 mvwprintz( w_speed, point( 21 + ( newmoves >= 100 ? 0 : ( newmoves < 10 ? 2 : 1 ) ), 2 ), col,
857 "%d", newmoves );
858 wnoutrefresh( w_speed );

924static bool handle_player_display_action( player &you, unsigned int &line,
925 player_display_tab &curtab, input_context &ctxt,
926 const ui_adaptor &ui_tip, const ui_adaptor &ui_info,
927 const ui_adaptor &ui_stats, const ui_adaptor &ui_encumb,
928 const ui_adaptor &ui_traits, const ui_adaptor &ui_bionics,
929 const ui_adaptor &ui_effects, const ui_adaptor &ui_skills,
930 const std::vector<trait_id> &traitslist,
931 const std::vector<bionic> &bionicslist,
932 const std::vector<std::pair<std::string, std::string>> &effect_name_and_text,
933 const std::vector<HeaderSkill> &skillslist )
934{
935 const auto invalidate_tab = [&]( const player_display_tab tab ) {
936 switch( tab ) {
937 case player_display_tab::stats:
938 ui_stats.invalidate_ui();
939 break;
940 case player_display_tab::encumbrance:
941 ui_encumb.invalidate_ui();
942 break;
943 case player_display_tab::traits:
944 ui_traits.invalidate_ui();
945 break;
946 case player_display_tab::bionics:
947 ui_bionics.invalidate_ui();
948 break;
949 case player_display_tab::effects:
950 ui_effects.invalidate_ui();
951 break;
952 case player_display_tab::skills:
953 ui_skills.invalidate_ui();
954 break;
955 case player_display_tab::num_tabs:
956 abort();
957 }
958 };
959
960 unsigned int line_beg = 0;
961 unsigned int line_end = 0;
962 switch( curtab ) {
963 case player_display_tab::stats:
964 line_end = 6;
965 break;
966 case player_display_tab::encumbrance: {
967 const std::vector<std::pair<body_part, bool>> bps = list_and_combine_bps( you, nullptr );
968 line_end = bps.size();
969 break;
970 }
971 case player_display_tab::traits:
972 line_end = traitslist.size();
973 break;
974 case player_display_tab::bionics:
975 line_end = bionicslist.size();
976 break;
977 case player_display_tab::effects:
978 line_end = effect_name_and_text.size();
979 break;
980 case player_display_tab::skills:
981 line_beg = 1; // skip first header
982 line_end = skillslist.size();
983 break;
984 case player_display_tab::num_tabs:
985 abort();
986 }
987 if( line_beg >= line_end || line < line_beg ) {
988 line = line_beg;
989 } else if( line > line_end - 1 ) {
990 line = line_end - 1;
991 }
992
993 bool done = false;
994 std::string action = ctxt.handle_input();
995
996 if( action == "UP" ) {
997 if( line > line_beg ) {
998 --line;
999 } else {
1000 line = line_end - 1;
1001 }
1002 if( curtab == player_display_tab::skills && skillslist[line].is_header ) {
1003 --line;
1004 }
1005 invalidate_tab( curtab );
1006 ui_info.invalidate_ui();
1007 } else if( action == "DOWN" ) {
1008 if( line + 1 < line_end ) {
1009 ++line;
1010 } else {
1011 line = line_beg;
1012 }
1013 if( curtab == player_display_tab::skills && skillslist[line].is_header ) {
1014 ++line;
1015 }
1016 invalidate_tab( curtab );
1017 ui_info.invalidate_ui();
1018 } else if( action == "NEXT_TAB" || action == "PREV_TAB" ) {
1019 line = 0;
1020 invalidate_tab( curtab );
1021 curtab = action == "NEXT_TAB" ? next_tab( curtab ) : prev_tab( curtab );
1022 invalidate_tab( curtab );
1023 ui_info.invalidate_ui();
1024 } else if( action == "QUIT" ) {
1025 done = true;
1026 } else if( action == "CONFIRM" ) {
1027 switch( curtab ) {
1028 default:
1029 break;
1030 case player_display_tab::stats:
1031 if( line < 4 && get_option<bool>( "STATS_THROUGH_KILLS" ) && you.is_avatar() ) {
1032 you.as_avatar()->upgrade_stat_prompt( static_cast<character_stat>( line ) );
1033 }
1034 invalidate_tab( curtab );
1035 break;
1036 case player_display_tab::skills: {
1037 const Skill *selectedSkill = nullptr;
1038 if( line < skillslist.size() && !skillslist[line].is_header ) {
1039 selectedSkill = skillslist[line].skill;
1040 }
1041 if( selectedSkill ) {
1042 you.get_skill_level_object( selectedSkill->ident() ).toggleTraining();
1043 }
1044 invalidate_tab( curtab );
1045 break;
1046 }
1047 }
1048 } else if( action == "CHANGE_PROFESSION_NAME" ) {
1049 string_input_popup popup;
1050 popup.title( _( "Profession Name: " ) )
1051 .width( 25 )
1052 .text( "" )
1053 .max_length( 25 )
1054 .query();
1055
1056 you.custom_profession = popup.text();
1057 ui_tip.invalidate_ui();
1058 }
1059 return done;
1060}
