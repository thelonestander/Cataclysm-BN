77/**
78 * Represents queued active tile / furniture transformations.
79 *
80 * Some active tiles can turn into other active tiles, or even inactive tiles, as a result
81 * of an update. If such transformation is applied immediately, it could trigger recalculation of
82 * the grid that's being updated, which would require additional code to handle.
83 *
84 * As a simpler alternative, we queue active tile transformations and apply them only after
85 * all grids have been updated. The transformations are applied according to FIFO method,
86 * so if some tile has multiple competing transforms queued, the last one will win out.
87 */
88class grid_furn_transform_queue
89{
90 private:
91 std::vector<transform_queue_entry> queue;
92
93 public:
94 void add( const tripoint_abs_ms &p, const furn_str_id &id, const std::string &msg ) {
95 queue.emplace_back( transform_queue_entry{ p, id, msg } );
96 }
97
98 void apply( mapbuffer &mb, distribution_grid_tracker &grid_tracker, Character &u, map &m );
99
100 void clear() {
101 queue.clear();
102 }
103
104 bool operator==( const grid_furn_transform_queue &l ) const {
105 return queue == l.queue;
106 }
107
108 std::string to_string() const;
109};

191/** Traverses the graph of connected vehicles and grids.
