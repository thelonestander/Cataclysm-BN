210void SkillLevel::train( int amount, bool skip_scaling )
211{
212 // Working off rust to regain levels goes twice as fast as reaching levels in the first place
213 if( _level < _highestLevel ) {
214 amount *= 2;
215 }
216
217 if( skip_scaling ) {
218 _exercise += amount;
219 } else {
220 const double scaling = get_option<float>( "SKILL_TRAINING_SPEED" );
221 if( scaling > 0.0 ) {
222 _exercise += roll_remainder( amount * scaling );
223 }
224 }
225
226 if( _exercise >= 100 * ( _level + 1 ) * ( _level + 1 ) ) {
227 _exercise = 0;
228 ++_level;
229 if( _level > _highestLevel ) {
230 _highestLevel = _level;
231 }
232 }
233}

256bool SkillLevel::rust( bool charged_bio_mem, int character_rate )
257{
258 const time_duration delta = calendar::turn - _lastPracticed;
259 const float char_rate = character_rate / 100.0;
260 const time_duration skill_rate = rustRate( _level );
261 if( to_turns<int>( skill_rate ) * char_rate <= 0 || delta <= 0_turns ||
262 delta % ( skill_rate * char_rate ) != 0_turns ) {
263 return false;
264 }
265
266 if( charged_bio_mem ) {
267 return one_in( 5 );
268 }
269
270 _exercise -= _level;
271 const std::string &rust_type = get_option<std::string>( "SKILL_RUST" );
272 if( _exercise < 0 ) {
273 if( rust_type == "vanilla" || rust_type == "int" ) {
274 _exercise = ( 100 * _level * _level ) - 1;
275 --_level;
276 } else {
277 _exercise = 0;
278 }
279 }
280
281 return false;
282}
