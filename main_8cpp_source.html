66static int pfd[2];
67static pthread_t thr;
68static const char *tag = "cdda";
69
70static void *thread_func( void * )
71{
72 ssize_t rdsz;
73 char buf[128];
74 for( ;; ) {
75 if( ( ( rdsz = read( pfd[0], buf, sizeof buf - 1 ) ) > 0 ) ) {
76 if( buf[rdsz - 1] == '\n' ) {
77 --rdsz;
78 }
79 buf[rdsz] = 0; /* add null-terminator */
80 __android_log_write( ANDROID_LOG_DEBUG, tag, buf );
81 }
82 }
83 return 0;
84}

86int start_logger( const char *app_name )
87{
88 tag = app_name;
89
90 /* make stdout line-buffered and stderr unbuffered */
91 setvbuf( stdout, 0, _IOLBF, 0 );
92 setvbuf( stderr, 0, _IONBF, 0 );
93
94 /* create the pipe and redirect stdout and stderr */
95 pipe( pfd );
96 dup2( pfd[1], 1 );
97 dup2( pfd[1], 2 );
98
99 /* spawn the logging thread */
100 if( pthread_create( &thr, 0, thread_func, 0 ) == -1 ) {
101 return -1;
102 }
103 pthread_detach( thr );
104 return 0;
105}

740void printHelpMessage( const arg_handler *first_pass_arguments,
741 size_t num_first_pass_arguments,
742 const arg_handler *second_pass_arguments,
743 size_t num_second_pass_arguments )
744{
745
746 // Group all arguments by help_group.
747 std::multimap<std::string, const arg_handler *> help_map;
748 for( size_t i = 0; i < num_first_pass_arguments; ++i ) {
749 std::string help_group;
750 if( first_pass_arguments[i].help_group ) {
751 help_group = first_pass_arguments[i].help_group;
752 }
753 help_map.insert( std::make_pair( help_group, &first_pass_arguments[i] ) );
754 }
755 for( size_t i = 0; i < num_second_pass_arguments; ++i ) {
756 std::string help_group;
757 if( second_pass_arguments[i].help_group ) {
758 help_group = second_pass_arguments[i].help_group;
759 }
760 help_map.insert( std::make_pair( help_group, &second_pass_arguments[i] ) );
761 }
762
763 cata_printf( "Command line parameters:\n" );
764 std::string current_help_group;
765 auto it = help_map.begin();
766 auto it_end = help_map.end();
767 for( ; it != it_end; ++it ) {
768 if( it->first != current_help_group ) {
769 current_help_group = it->first;
770 cata_printf( "\n%s\n", current_help_group.c_str() );
771 }
772
773 const arg_handler *handler = it->second;
774 cata_printf( "%s", handler->flag );
775 if( handler->param_documentation ) {
776 cata_printf( " %s", handler->param_documentation );
777 }
778 cata_printf( "\n" );
779 if( handler->documentation ) {
780 cata_printf( " %s\n", handler->documentation );
781 }
782 }
