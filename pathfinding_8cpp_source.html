163template<class Set1, class Set2>
164bool is_disjoint( const Set1 &set1, const Set2 &set2 )
165{
166 if( set1.empty() || set2.empty() ) {
167 return true;
168 }
169
170 typename Set1::const_iterator it1 = set1.begin();
171 typename Set1::const_iterator it1_end = set1.end();
172
173 typename Set2::const_iterator it2 = set2.begin();
174 typename Set2::const_iterator it2_end = set2.end();
175
176 if( *set2.rbegin() < *it1 || *set1.rbegin() < *it2 ) {
177 return true;
178 }
179
180 while( it1 != it1_end && it2 != it2_end ) {
181 if( *it1 == *it2 ) {
182 return false;
183 }
184 if( *it1 < *it2 ) {
185 it1++;
186 } else {
187 it2++;
188 }
189 }
190
191 return true;
192}
