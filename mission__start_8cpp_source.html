129/*
130 * Find a location to place a computer. In order, prefer:
131 * 1) Broken consoles.
132 * 2) Corners or coords adjacent to a bed/dresser? (this logic may be flawed, dates from Whales in 2011)
133 * 3) A spot near the center of the tile that is not a console
134 * 4) A random spot near the center of the tile.
135 */
136static tripoint find_potential_computer_point( const tinymap &compmap )
137{
138 constexpr int rng_x_min = 10;
139 constexpr int rng_x_max = SEEX * 2 - 11;
140 constexpr int rng_y_min = 10;
141 constexpr int rng_y_max = SEEY * 2 - 11;
142 static_assert( rng_x_min <= rng_x_max && rng_y_min <= rng_y_max, "invalid randomization range" );
143 std::vector<tripoint> broken;
144 std::vector<tripoint> potential;
145 std::vector<tripoint> last_resort;
146 for( const tripoint &p : compmap.points_on_zlevel() ) {
147 if( compmap.ter( p ) == t_console_broken ) {
148 broken.emplace_back( p );
149 } else if( broken.empty() && compmap.ter( p ) == t_floor && compmap.furn( p ) == f_null ) {
150 for( const tripoint &p2 : compmap.points_in_radius( p, 1 ) ) {
151 if( compmap.furn( p2 ) == f_bed || compmap.furn( p2 ) == f_dresser ) {
152 potential.emplace_back( p );
153 break;
154 }
155 }
156 int wall = 0;
157 for( const tripoint &p2 : compmap.points_in_radius( p, 1 ) ) {
158 if( compmap.has_flag_ter( "WALL", p2 ) ) {
159 wall++;
160 }
161 }
162 if( wall == 5 ) {
163 if( compmap.is_last_ter_wall( true, p.xy(), point( SEEX * 2, SEEY * 2 ), direction::NORTH ) &&
164 compmap.is_last_ter_wall( true, p.xy(), point( SEEX * 2, SEEY * 2 ), direction::SOUTH ) &&
165 compmap.is_last_ter_wall( true, p.xy(), point( SEEX * 2, SEEY * 2 ), direction::WEST ) &&
166 compmap.is_last_ter_wall( true, p.xy(), point( SEEX * 2, SEEY * 2 ), direction::EAST ) ) {
167 potential.emplace_back( p );
168 }
169 }
170 } else if( broken.empty() && potential.empty() && p.x >= rng_x_min && p.x <= rng_x_max
171 && p.y >= rng_y_min && p.y <= rng_y_max && compmap.ter( p ) != t_console ) {
172 last_resort.emplace_back( p );
173 }
174 }
175 std::vector<tripoint> *used = &broken;
176 if( used->empty() ) {
177 used = &potential;
178 }
179 if( used->empty() ) {
180 used = &last_resort;
181 }
182 // if there's no possible location, then we have to overwrite an existing console...
183 const tripoint fallback( rng( rng_x_min, rng_x_max ), rng( rng_y_min, rng_y_max ),
184 compmap.get_abs_sub().z );
185 return random_entry( *used, fallback );
186}

197
198 std::string type = "house";
199
200 if( dev->myclass == NC_HACKER ) {
201 miss->item_id = itype_software_hacking;
202 } else if( dev->myclass == NC_DOCTOR ) {
203 miss->item_id = itype_software_medical;
204 type = "s_pharm";
205 miss->follow_up = mission_type_id( "MISSION_GET_ZOMBIE_BLOOD_ANAL" );
206 } else if( dev->myclass == NC_SCIENTIST ) {
207 miss->item_id = itype_software_math;
208 } else {
209 miss->item_id = itype_software_useless;
210 }
211
212 tripoint_abs_omt place;
213 if( type == "house" ) {
214 place = mission_util::random_house_in_closest_city();
215 } else {
216 place = overmap_buffer.find_closest( dev->global_omt_location(), type, 0, false );
217 }
218 miss->target = place;
219 overmap_buffer.reveal( place, 6 );
220
221 tinymap compmap;
222 compmap.load( project_to<coords::sm>( place ), false );
223 tripoint comppoint;
224
225 oter_id oter = overmap_buffer.ter( place );
226 if( is_ot_match( "house", oter, ot_match_type::prefix ) ||
227 is_ot_match( "s_pharm", oter, ot_match_type::type ) || oter == "" ) {
228 comppoint = find_potential_computer_point( compmap );
229 }
230
231 compmap.ter_set( comppoint, t_console );
232 computer *tmpcomp = compmap.add_computer( comppoint, string_format( _( "%s's Terminal" ),
233 dev->name ), 0 );
234 tmpcomp->set_mission( miss->get_id() );
235 tmpcomp->add_option( _( "Download Software" ), COMPACT_DOWNLOAD_SOFTWARE, 0 );
236 compmap.save();

302void mission_start::find_safety( mission *miss )
303{
304 const tripoint_abs_omt place = get_player_character().global_omt_location();
305 for( int radius = 0; radius <= 20; radius++ ) {
306 for( int dist = 0 - radius; dist <= radius; dist++ ) {
307 int offset = rng( 0, 3 ); // Randomizes the direction we check first
308 for( int i = 0; i <= 3; i++ ) { // Which direction?
309 tripoint_abs_omt check = place;
310 switch( ( offset + i ) % 4 ) {
311 case 0:
312 check.x() += dist;
313 check.y() -= radius;
314 break;
315 case 1:
316 check.x() += dist;
317 check.y() += radius;
318 break;
319 case 2:
320 check.y() += dist;
321 check.x() -= radius;
322 break;
323 case 3:
324 check.y() += dist;
325 check.x() += radius;
326 break;
327 }
328 if( overmap_buffer.is_safe( check ) ) {
329 miss->target = check;
330 return;
331 }
332 }
333 }
334 }
335 // Couldn't find safety; so just set the target to far away
336 switch( rng( 0, 3 ) ) {
337 case 0:
338 miss->target = place + point( -20, -20 );
339 break;
340 case 1:
341 miss->target = place + point( -20, 20 );
342 break;
343 case 2:
344 miss->target = place + point( 20, -20 );
345 break;
346 case 3:
347 miss->target = place + point( 20, 20 );
348 break;
349 }
350}
