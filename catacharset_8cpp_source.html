18uint32_t UTF8_getch( const char **src, int *srclen )
19{
20 const unsigned char *p = *reinterpret_cast<const unsigned char **>( src );
21 int left = 0;
22 bool overlong = false;
23 bool underflow = false;
24 uint32_t ch = UNKNOWN_UNICODE;
25
26 if( *srclen == 0 ) {
27 return UNKNOWN_UNICODE;
28 }
29 if( p[0] >= 0xFC ) {
30 if( ( p[0] & 0xFE ) == 0xFC ) {
31 if( p[0] == 0xFC && ( p[1] & 0xFC ) == 0x80 ) {
32 overlong = true;
33 }
34 ch = static_cast<uint32_t>( p[0] & 0x01 );
35 left = 5;
36 }
37 } else if( p[0] >= 0xF8 ) {
38 if( ( p[0] & 0xFC ) == 0xF8 ) {
39 if( p[0] == 0xF8 && ( p[1] & 0xF8 ) == 0x80 ) {
40 overlong = true;
41 }
42 ch = static_cast<uint32_t>( p[0] & 0x03 );
43 left = 4;
44 }
45 } else if( p[0] >= 0xF0 ) {
46 if( ( p[0] & 0xF8 ) == 0xF0 ) {
47 if( p[0] == 0xF0 && ( p[1] & 0xF0 ) == 0x80 ) {
48 overlong = true;
49 }
50 ch = static_cast<uint32_t>( p[0] & 0x07 );
51 left = 3;
52 }
53 } else if( p[0] >= 0xE0 ) {
54 if( ( p[0] & 0xF0 ) == 0xE0 ) {
55 if( p[0] == 0xE0 && ( p[1] & 0xE0 ) == 0x80 ) {
56 overlong = true;
57 }
58 ch = static_cast<uint32_t>( p[0] & 0x0F );
59 left = 2;
60 }
61 } else if( p[0] >= 0xC0 ) {
62 if( ( p[0] & 0xE0 ) == 0xC0 ) {
63 if( ( p[0] & 0xDE ) == 0xC0 ) {
64 overlong = true;
65 }
66 ch = static_cast<uint32_t>( p[0] & 0x1F );
67 left = 1;
68 }
69 } else {
70 if( ( p[0] & 0x80 ) == 0x00 ) {
71 ch = static_cast<uint32_t>( p[0] );
72 }
73 }
74 ++*src;
75 --*srclen;
76 while( left > 0 && *srclen > 0 ) {
77 ++p;
78 if( ( p[0] & 0xC0 ) != 0x80 ) {
79 ch = UNKNOWN_UNICODE;
80 break;
81 }
82 ch <<= 6;
83 ch |= ( p[0] & 0x3F );
84 ++*src;
85 --*srclen;
86 --left;
87 }
88 if( left > 0 ) {
89 underflow = true;
90 }
91 if( overlong || underflow ||
92 ( ch >= 0xD800 && ch <= 0xDFFF ) ||
93 ( ch == 0xFFFE || ch == 0xFFFF ) || ch > 0x10FFFF ) {
94 ch = UNKNOWN_UNICODE;
95 }
96 return ch;
97}

99std::string utf32_to_utf8( uint32_t ch )
100{
101 char out[5];
102 char *buf = out;
103 static const unsigned char utf8FirstByte[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
104 int utf8Bytes;
105 if( ch < 0x80 ) {
106 utf8Bytes = 1;
107 } else if( ch < 0x800 ) {
108 utf8Bytes = 2;
109 } else if( ch < 0x10000 ) {
110 utf8Bytes = 3;
111 } else if( ch <= 0x10FFFF ) {
112 utf8Bytes = 4;
113 } else {
114 utf8Bytes = 3;
115 ch = UNKNOWN_UNICODE;
116 }
117
118 buf += utf8Bytes;
119 switch( utf8Bytes ) {
120 case 4:
121 *--buf = ( ch | 0x80 ) & 0xBF;
122 ch >>= 6;
123 /* fallthrough */
124 case 3:
125 *--buf = ( ch | 0x80 ) & 0xBF;
126 ch >>= 6;
127 /* fallthrough */
128 case 2:
129 *--buf = ( ch | 0x80 ) & 0xBF;
130 ch >>= 6;
131 /* fallthrough */
132 case 1:
133 *--buf = ch | utf8FirstByte[utf8Bytes];
134 }
135 out[utf8Bytes] = '\0';
136 return out;
137}

199//Convert cursor position to byte offset
200//returns the first character position in bytes behind the cursor position.
201//If the cursor is not on the first half of the character,
202//prevpos (which points to the first byte of the cursor located char)
203// should be a different value.
204int cursorx_to_position( const char *line, int cursorx, int *prevpos, int maxlen )
205{
206 int dummy;
207 int i = 0;
208 int c = 0;
209 int *p = prevpos ? prevpos : &dummy;
210 *p = 0;
211 while( c < cursorx ) {
212 const char *utf8str = line + i;
213 int len = ANY_LENGTH;
214 if( utf8str[0] == 0 ) {
215 break;
216 }
217 uint32_t ch = UTF8_getch( &utf8str, &len );
218 int cw = mk_wcwidth( ch );
219 len = ANY_LENGTH - len;
220 if( len <= 0 ) {
221 len = 1;
222 }
223 if( maxlen >= 0 && maxlen < ( i + len ) ) {
224 break;
225 }
226 i += len;
227 if( cw <= 0 ) {
228 cw = 0;
229 }
230 c += cw;
231 if( c <= cursorx ) {
232 *p = i;
233 }
234 }
235 return i;
236}

277{
278 //assume it is already encoded
279 if( !str.empty() && str[0] == '#' ) {
280 return str;
281 }
282
283 int input_length = str.length();
284 int output_length = 4 * ( ( input_length + 2 ) / 3 );
285
286 std::string encoded_data( output_length, '\0' );
287 const unsigned char *data = reinterpret_cast<const unsigned char *>( str.c_str() );
288
289 for( int i = 0, j = 0; i < input_length; ) {
290
291 unsigned octet_a = i < input_length ? data[i++] : 0;
292 unsigned octet_b = i < input_length ? data[i++] : 0;
293 unsigned octet_c = i < input_length ? data[i++] : 0;
294
295 unsigned triple = ( octet_a << 0x10 ) + ( octet_b << 0x08 ) + octet_c;
296
297 encoded_data[j++] = base64_encoding_table[( triple >> 3 * 6 ) & 0x3F];
298 encoded_data[j++] = base64_encoding_table[( triple >> 2 * 6 ) & 0x3F];
299 encoded_data[j++] = base64_encoding_table[( triple >> 1 * 6 ) & 0x3F];
300 encoded_data[j++] = base64_encoding_table[( triple >> 0 * 6 ) & 0x3F];
301 }
302

304 encoded_data[output_length - 1 - i] = '=';
305 }

536size_t utf8_wrapper::byte_start_display( size_t bstart, size_t start ) const
537{

540 }
541 if( start == 0 ) {
542 return 0;

545 int len = _data.length() - bstart;
