193{

197 load_mod_info( mod_file, out );
198 }
199
200 std::set<mod_id> idents;
201 std::set<mod_id> has_dupes;
202
203 for( const MOD_INFORMATION &it : out ) {

205 has_dupes.emplace( it.ident );
206 } else {
207 idents.emplace( it.ident );
208 }
209 }
210
211 for( const mod_id &ident : has_dupes ) {

215
216 for( auto it = out.begin(); it != out.end(); ) {
217 if( it->ident == ident ) {
218 msg += " - ";
219 msg += it->path_full;
220 msg += "\n";
221 it = out.erase( it );
222 } else {
223 it++;
224 }
225 }
226
227 msg += _( "None of them will be loaded to avoid data corruption." );
228
229 debugmsg( msg );
230 }

235cata::optional<MOD_INFORMATION> load_modfile( const JsonObject &jo, const std::string &path )
236{
237 if( !jo.has_string( "type" ) || jo.get_string( "type" ) != "MOD_INFO" ) {
238 // Ignore anything that is not a mod-info
239 jo.allow_omitted_members();
240 return cata::nullopt;
241 }
242
243 // TEMPORARY until 0.G: Remove "ident" support
244 const mod_id m_ident( jo.has_string( "ident" ) ? jo.get_string( "ident" ) : jo.get_string( "id" ) );
245
246 std::string m_cat = jo.get_string( "category", "" );
247 std::pair<int, std::string> p_cat = { -1, "" };
248 bool bCatFound = false;
249
250 do {
251 for( size_t i = 0; i < get_mod_list_categories().size(); ++i ) {
252 if( get_mod_list_categories()[i].first == m_cat ) {
253 p_cat = { static_cast<int>( i ), get_mod_list_categories()[i].second };
254 bCatFound = true;
255 break;
256 }
257 }
258
259 if( !bCatFound && !m_cat.empty() ) {
260 m_cat.clear();
261 } else {
262 break;
263 }
264 } while( !bCatFound );
265
266 MOD_INFORMATION modfile;
267 modfile.ident = m_ident;
268 modfile.category = p_cat;
269
270 if( assign( jo, "path", modfile.path ) ) {
271 modfile.path = path + "/" + modfile.path;
272 } else {
273 modfile.path = path;
274 }
275
276 const std::string m_name = jo.get_string( "name", "" );
277 const std::string m_descr = jo.get_string( "description", "" );
278 modfile.set_translatable_info( translatable_mod_info( m_name, m_descr, modfile.path ) );
279
280 assign( jo, "authors", modfile.authors );
281 assign( jo, "maintainers", modfile.maintainers );
282 assign( jo, "version", modfile.version );
283 assign( jo, "dependencies", modfile.dependencies );
284 assign( jo, "conflicts", modfile.conflicts );
285 assign( jo, "core", modfile.core );
286 assign( jo, "obsolete", modfile.obsolete );
287
288 if( std::find( modfile.dependencies.begin(), modfile.dependencies.end(),
289 modfile.ident ) != modfile.dependencies.end() ) {
290 jo.throw_error( "mod specifies self as a dependency", "dependencies" );
291 }
292 for( const auto &conf : modfile.conflicts ) {
293 if( conf == modfile.ident ) {
294 jo.throw_error( "mod specifies self as a conflict", "conflicts" );
295 }
296 if( std::find( modfile.dependencies.begin(), modfile.dependencies.end(),
297 conf ) != modfile.dependencies.end() ) {
298 jo.throw_error( string_format( "mod specifies \"%s\" as both a dependency and a conflict", conf ),
299 "conflicts" );
300 }
301 }
302
303 return { std::move( modfile ) };
304}

306void load_mod_info( const std::string &info_file_path, std::vector<MOD_INFORMATION> &out )
307{
308 const std::string main_path = info_file_path.substr( 0, info_file_path.find_last_of( "/\\" ) );
309 read_from_file_optional_json( info_file_path, [&]( JsonIn & jsin ) {
310 if( jsin.test_object() ) {
311 // find type and dispatch single object
312 JsonObject jo = jsin.get_object();
313 cata::optional<MOD_INFORMATION> mf = load_modfile( jo, main_path );
314 if( mf ) {
315 mf->path_full = info_file_path;
316 out.push_back( std::move( *mf ) );
317 }
318 jo.finish();
319 } else if( jsin.test_array() ) {
320 jsin.start_array();
321 // find type and dispatch each object until array close
322 while( !jsin.end_array() ) {
323 JsonObject jo = jsin.get_object();
324 cata::optional<MOD_INFORMATION> mf = load_modfile( jo, main_path );
325 if( mf ) {
326 mf->path_full = info_file_path;
327 out.push_back( std::move( *mf ) );
328 }
329 jo.finish();
330 }
331 } else {
332 // not an object or an array?
333 jsin.error( "expected array or object" );
334 }
335 } );
336}

363void mod_manager::add_mods( std::vector<MOD_INFORMATION> &&list )
364{
365 struct replacement {
366 mod_id id;
367 std::string path_old;
368 std::string path_new;
369 };
370
371 std::vector<replacement> replaced;
372
373 for( MOD_INFORMATION &info : list ) {
374 auto it = mod_map.find( info.ident );
375 if( it == mod_map.end() ) {
376 mod_map.emplace( info.ident, std::move( info ) );
377 } else {
378 replacement r;
379 r.id = info.ident;
380 r.path_old = it->second.path_full;
381 r.path_new = info.path_full;
382 replaced.push_back( std::move( r ) );
383 it->second = std::move( info );
384 }
385 }
386
387 if( !replaced.empty() ) {
388 DebugLog( DL::Info, DC::Main ) << "[Mod manager] Replaced " << replaced.size() <<
389 " mod(s) with user overrides.";
390 std::string msg;
391 for( const replacement &it : replaced ) {
392 msg += string_format( "[%s] from \"%s\" with \"%s\"\n", it.id, it.path_old, it.path_new );
393 }
394 DebugLog( DL::Debug, DC::Main ) << msg;
395 }

427void mod_manager::load_mods_list( WORLDPTR world ) const
428{
429 if( world == nullptr ) {
430 return;
431 }
432 std::vector<mod_id> &amo = world->active_mod_order;
433 amo.clear();
434 bool obsolete_mod_found = false;
435 read_from_file_optional_json( get_mods_list_file( world ), [&]( JsonIn & jsin ) {
436 for( const std::string line : jsin.get_array() ) {
437 const mod_id mod( line );
438 if( std::find( amo.begin(), amo.end(), mod ) != amo.end() ) {
439 continue;
440 }
441 const auto iter = mod_replacements.find( mod );
442 if( iter != mod_replacements.end() ) {
443 if( !iter->second.is_empty() ) {
444 amo.push_back( iter->second );
445 }
446 obsolete_mod_found = true;
447 continue;
448 }
449 amo.push_back( mod );
450 }
451 } );
452 if( obsolete_mod_found ) {
453 // If we found an obsolete mod, overwrite the mod list without the obsolete one.
454 save_mods_list( world );
455 }
