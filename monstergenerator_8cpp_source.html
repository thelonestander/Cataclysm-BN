248static int calc_bash_skill( const mtype &t )
249{
250 // IOW, the critter's max bashing damage
251 int ret = t.melee_dice * t.melee_sides;
252 // This is for stuff that goes through solid rock: minerbots, dark wyrms, etc
253 if( t.has_flag( MF_BORES ) ) {
254 ret *= 15;
255 } else if( t.has_flag( MF_DESTROYS ) ) {
256 ret *= 2.5;
257 } else if( !t.has_flag( MF_BASHES ) ) {
258 ret = 0;
259 }
260
261 return ret;
262}

288void monster_adjustment::apply( mtype &mon )
289{
290 if( !mon.in_species( species ) ) {
291 return;
292 }
293 if( !stat.empty() ) {
294 if( stat == "speed" ) {
295 mon.speed *= stat_adjust;
296 } else if( stat == "hp" ) {
297 mon.hp *= stat_adjust;
298 }
299 }
300 if( !flag.empty() ) {
301 mon.set_flag( io::string_to_enum<m_flag>( flag ), flag_val );
302 }
303 if( !special.empty() ) {
304 if( special == "nightvision" ) {
305 mon.vision_night = mon.vision_day;
306 }
307 }
308}

317void load_monster_adjustment( const JsonObject &jsobj )
318{
319 monster_adjustment adj;
320 adj.species = species_id( jsobj.get_string( "species" ) );
321 if( jsobj.has_member( "stat" ) ) {
322 JsonObject stat = jsobj.get_object( "stat" );
323 stat.read( "name", adj.stat );
324 stat.read( "modifier", adj.stat_adjust );
325 }
326 if( jsobj.has_member( "flag" ) ) {
327 JsonObject flag = jsobj.get_object( "flag" );
328 flag.read( "name", adj.flag );
329 flag.read( "value", adj.flag_val );
330 }
331 if( jsobj.has_member( "special" ) ) {
332 jsobj.read( "special", adj.special );
333 }
334 adjustments.push_back( adj );
335}

345void MonsterGenerator::finalize_mtypes()
346{
347 mon_templates->finalize();
348 for( const auto &elem : mon_templates->get_all() ) {
349 mtype &mon = const_cast<mtype &>( elem );
350 apply_species_attributes( mon );
351 set_species_ids( mon );
352 mon.size = volume_to_size( mon.volume );
353
354 // adjust for worldgen difficulty parameters
355 mon.speed *= get_option<int>( "MONSTER_SPEED" ) / 100.0;
356 mon.hp *= get_option<int>( "MONSTER_RESILIENCE" ) / 100.0;
357
358 for( monster_adjustment adj : adjustments ) {
359 adj.apply( mon );
360 }
361
362 if( mon.bash_skill < 0 ) {
363 mon.bash_skill = calc_bash_skill( mon );
364 }
365
366 if( mon.armor_bash < 0 ) {
367 mon.armor_bash = 0;
368 }
369 if( mon.armor_cut < 0 ) {
370 mon.armor_cut = 0;
371 }
372 if( mon.armor_stab < 0 ) {
373 mon.armor_stab = mon.armor_cut * 0.8;
374 }
375 if( mon.armor_bullet < 0 ) {
376 mon.armor_bullet = 0;
377 }
378 if( mon.armor_acid < 0 ) {
379 mon.armor_acid = mon.armor_cut * 0.5;
380 }
381 if( mon.armor_fire < 0 ) {
382 mon.armor_fire = 0;
383 }
384
385 // Lower bound for hp scaling
386 mon.hp = std::max( mon.hp, 1 );
387
388 build_behavior_tree( mon );
389 finalize_pathfinding_settings( mon );
390 }
391
392 for( const auto &mon : mon_templates->get_all() ) {
393 if( !mon.has_flag( MF_NOT_HALLU ) ) {
394 hallucination_monsters.push_back( mon.id );
395 }
396 }
397}

683void mtype::load( const JsonObject &jo, const std::string &src )
684{
685 bool strict = src == "dda";
686
687 MonsterGenerator &gen = MonsterGenerator::generator();
688
689 name.make_plural();
690 mandatory( jo, was_loaded, "name", name );
691
692 optional( jo, was_loaded, "description", description );
693
694 optional( jo, was_loaded, "material", mat, auto_flags_reader<material_id> {} );
695 optional( jo, was_loaded, "species", species, auto_flags_reader<species_id> {} );
696 optional( jo, was_loaded, "categories", categories, auto_flags_reader<> {} );
697
698 // See monfaction.cpp
699 if( !was_loaded || jo.has_member( "default_faction" ) ) {
700 const auto faction = mfaction_str_id( jo.get_string( "default_faction" ) );
701 default_faction = monfactions::get_or_add_faction( faction );
702 }
703
704 if( !was_loaded || jo.has_member( "symbol" ) ) {
705 sym = jo.get_string( "symbol" );
706 if( utf8_wrapper( sym ).display_width() != 1 ) {
707 jo.throw_error( "monster symbol should be exactly one console cell width", "symbol" );
708 }
709 }
710 if( was_loaded && jo.has_member( "copy-from" ) && looks_like.empty() ) {
711 looks_like = jo.get_string( "copy-from" );
712 }
713 jo.read( "looks_like", looks_like );
714
715 assign( jo, "bodytype", bodytype );
716 assign( jo, "color", color );
717 assign( jo, "volume", volume, strict, 0_ml );
718 assign( jo, "weight", weight, strict, 0_gram );
719
720 optional( jo, was_loaded, "phase", phase, make_flag_reader( gen.phase_map, "phase id" ), SOLID );
721
722 assign( jo, "diff", difficulty_base, strict, 0 );
723 assign( jo, "hp", hp, strict, 1 );
724 assign( jo, "speed", speed, strict, 0 );
725 assign( jo, "aggression", agro, strict, -100, 100 );
726 assign( jo, "morale", morale, strict );
727
728 assign( jo, "mountable_weight_ratio", mountable_weight_ratio, strict );
729
730 assign( jo, "attack_cost", attack_cost, strict, 0 );
731 assign( jo, "melee_skill", melee_skill, strict, 0 );
732 assign( jo, "melee_dice", melee_dice, strict, 0 );
733 assign( jo, "melee_dice_sides", melee_sides, strict, 0 );
734
735 assign( jo, "grab_strength", grab_strength, strict, 0 );
736
737 assign( jo, "dodge", sk_dodge, strict, 0 );
738 assign( jo, "armor_bash", armor_bash, strict, 0 );
739 assign( jo, "armor_cut", armor_cut, strict, 0 );
740 assign( jo, "armor_bullet", armor_bullet, strict, 0 );
741 assign( jo, "armor_stab", armor_stab, strict, 0 );
742 assign( jo, "armor_acid", armor_acid, strict, 0 );
743 assign( jo, "armor_fire", armor_fire, strict, 0 );
744
745 assign( jo, "vision_day", vision_day, strict, 0 );
746 assign( jo, "vision_night", vision_night, strict, 0 );
747
748 optional( jo, was_loaded, "regenerates", regenerates, 0 );
749 optional( jo, was_loaded, "regenerates_in_dark", regenerates_in_dark, false );
750 optional( jo, was_loaded, "regen_morale", regen_morale, false );
751
752 if( !was_loaded || jo.has_member( "regeneration_modifiers" ) ) {
753 regeneration_modifiers.clear();
754 add_regeneration_modifiers( jo, "regeneration_modifiers", src );
755 } else {
756 // Note: regeneration_modifiers left as is, new modifiers are added to it!
757 // Note: member name prefixes are compatible with those used by generic_typed_reader
758 if( jo.has_object( "extend" ) ) {
759 JsonObject tmp = jo.get_object( "extend" );
760 tmp.allow_omitted_members();
761 add_regeneration_modifiers( tmp, "regeneration_modifiers", src );
762 }
763 if( jo.has_object( "delete" ) ) {
764 JsonObject tmp = jo.get_object( "delete" );
765 tmp.allow_omitted_members();
766 remove_regeneration_modifiers( tmp, "regeneration_modifiers", src );
767 }
768 }
769
770 optional( jo, was_loaded, "starting_ammo", starting_ammo );
771 optional( jo, was_loaded, "luminance", luminance, 0 );
772 optional( jo, was_loaded, "revert_to_itype", revert_to_itype, itype_id() );
773 optional( jo, was_loaded, "attack_effs", atk_effs, mon_attack_effect_reader{} );
774 optional( jo, was_loaded, "mech_weapon", mech_weapon, itype_id() );
775 optional( jo, was_loaded, "mech_str_bonus", mech_str_bonus, 0 );
776 optional( jo, was_loaded, "mech_battery", mech_battery, itype_id() );
777
778 // TODO: make this work with `was_loaded`
779 if( jo.has_array( "melee_damage" ) ) {
780 melee_damage = load_damage_instance( jo.get_array( "melee_damage" ) );
781 } else if( jo.has_object( "melee_damage" ) ) {
782 melee_damage = load_damage_instance( jo );
783 }
784
785 if( jo.has_array( "scents_tracked" ) ) {
786 for( const std::string line : jo.get_array( "scents_tracked" ) ) {
787 scents_tracked.emplace( line );
788 }
789 }
790
791 if( jo.has_array( "scents_ignored" ) ) {
792 for( const std::string line : jo.get_array( "scents_ignored" ) ) {
793 scents_ignored.emplace( line );
794 }
795 }
796
797 int bonus_cut = 0;
798 if( jo.has_int( "melee_cut" ) ) {
799 bonus_cut = jo.get_int( "melee_cut" );
800 melee_damage.add_damage( DT_CUT, bonus_cut );
801 }
802
803 if( jo.has_member( "death_drops" ) ) {
804 death_drops = item_group::load_item_group( jo.get_member( "death_drops" ),
805 "distribution" );
806 }
807
808 assign( jo, "harvest", harvest );
809
810 const auto death_reader = make_flag_reader( gen.death_map, "monster death function" );
811 optional( jo, was_loaded, "death_function", dies, death_reader );
812 if( dies.empty() ) {
813 // TODO: really needed? Is an empty `dies` container not allowed?
814 dies.push_back( mdeath::normal );
815 }
816
817 if( jo.has_array( "emit_fields" ) ) {
818 JsonArray jar = jo.get_array( "emit_fields" );
819 if( jar.has_string( 0 ) ) { // TEMPORARY until 0.F
820 for( const std::string id : jar ) {
821 emit_fields.emplace( emit_id( id ), 1_seconds );
822 }
823 } else {
824 while( jar.has_more() ) {
825 JsonObject obj = jar.next_object();
826 emit_fields.emplace( emit_id( obj.get_string( "emit_id" ) ),
827 read_from_json_string<time_duration>( *obj.get_raw( "delay" ), time_duration::units ) );
828 }
829 }
830 }
831
832 if( jo.has_member( "special_when_hit" ) ) {
833 JsonArray jsarr = jo.get_array( "special_when_hit" );
834 const auto iter = gen.defense_map.find( jsarr.get_string( 0 ) );
835 if( iter == gen.defense_map.end() ) {
836 jsarr.throw_error( "Invalid monster defense function" );
837 }
838 sp_defense = iter->second;
839 def_chance = jsarr.get_int( 1 );
840 } else if( !was_loaded ) {
841 sp_defense = &mdefense::none;
842 def_chance = 0;
843 }
844
845 if( !was_loaded || jo.has_member( "special_attacks" ) ) {
846 special_attacks.clear();
847 special_attacks_names.clear();
848 add_special_attacks( jo, "special_attacks", src );
849 } else {
850 // Note: special_attacks left as is, new attacks are added to it!
851 // Note: member name prefixes are compatible with those used by generic_typed_reader
852 if( jo.has_object( "extend" ) ) {
853 JsonObject tmp = jo.get_object( "extend" );
854 tmp.allow_omitted_members();
855 add_special_attacks( tmp, "special_attacks", src );
856 }
857 if( jo.has_object( "delete" ) ) {
858 JsonObject tmp = jo.get_object( "delete" );
859 tmp.allow_omitted_members();
860 remove_special_attacks( tmp, "special_attacks", src );
861 }
862 }
863
864 // Disable upgrading when JSON contains `"upgrades": false`, but fallback to the
865 // normal behavior (including error checking) if "upgrades" is not boolean or not `false`.
866 if( jo.has_bool( "upgrades" ) && !jo.get_bool( "upgrades" ) ) {
867 upgrade_group = mongroup_id::NULL_ID();
868 upgrade_into = mtype_id::NULL_ID();
869 upgrades = false;
870 } else if( jo.has_member( "upgrades" ) ) {
871 JsonObject up = jo.get_object( "upgrades" );
872 optional( up, was_loaded, "half_life", half_life, -1 );
873 optional( up, was_loaded, "age_grow", age_grow, -1 );
874 optional( up, was_loaded, "into_group", upgrade_group, auto_flags_reader<mongroup_id> {},
875 mongroup_id::NULL_ID() );
876 optional( up, was_loaded, "into", upgrade_into, auto_flags_reader<mtype_id> {},
877 mtype_id::NULL_ID() );
878 upgrades = true;
879 }
880
881 //Reproduction
882 if( jo.has_member( "reproduction" ) ) {
883 JsonObject repro = jo.get_object( "reproduction" );
884 optional( repro, was_loaded, "baby_count", baby_count, -1 );
885 if( repro.has_int( "baby_timer" ) ) {
886 baby_timer = time_duration::from_days( repro.get_int( "baby_timer" ) );
887 } else if( repro.has_string( "baby_timer" ) ) {
888 baby_timer = read_from_json_string<time_duration>( *repro.get_raw( "baby_timer" ),
889 time_duration::units );
890 }
891 optional( repro, was_loaded, "baby_monster", baby_monster, auto_flags_reader<mtype_id> {},
892 mtype_id::NULL_ID() );
893 optional( repro, was_loaded, "baby_egg", baby_egg, auto_flags_reader<itype_id> {},
894 itype_id::NULL_ID() );
895 reproduces = true;
896 }
897
898 if( jo.has_member( "baby_flags" ) ) {
899 // Because this determines mating season and some monsters have a mating season but not in-game offspring, declare this separately
900 baby_flags.clear();
901 for( const std::string line : jo.get_array( "baby_flags" ) ) {
902 baby_flags.push_back( line );
903 }
904 }
905
906 optional( jo, was_loaded, "burn_into", burn_into, auto_flags_reader<mtype_id> {},
907 mtype_id::NULL_ID() );
908 assign( jo, "zombify_into", zombify_into );
909 assign( jo, "fungalize_into", fungalize_into );
910
911 const auto flag_reader = enum_flags_reader<m_flag> { "monster flag" };
912 optional( jo, was_loaded, "flags", flags, flag_reader );
913 // Can't calculate yet - we want all flags first
914 optional( jo, was_loaded, "bash_skill", bash_skill, -1 );
915
916 const auto trigger_reader = enum_flags_reader<mon_trigger> { "monster trigger" };
917 optional( jo, was_loaded, "anger_triggers", anger, trigger_reader );
918 optional( jo, was_loaded, "placate_triggers", placate, trigger_reader );
919 optional( jo, was_loaded, "fear_triggers", fear, trigger_reader );
920
921 if( jo.has_member( "path_settings" ) ) {
922 auto jop = jo.get_object( "path_settings" );
923 // Here rather than in pathfinding.cpp because we want monster-specific defaults and was_loaded
924 optional( jop, was_loaded, "max_dist", path_settings.max_dist, 0 );
925 optional( jop, was_loaded, "max_length", path_settings.max_length, -1 );
926 optional( jop, was_loaded, "bash_strength", path_settings.bash_strength, -1 );
927 optional( jop, was_loaded, "allow_open_doors", path_settings.allow_open_doors, false );
928 optional( jop, was_loaded, "avoid_traps", path_settings.avoid_traps, false );
929 optional( jop, was_loaded, "allow_climb_stairs", path_settings.allow_climb_stairs, true );
930 optional( jop, was_loaded, "avoid_sharp", path_settings.avoid_sharp, false );
931 }
932 difficulty = ( melee_skill + 1 ) * melee_dice * ( bonus_cut + melee_sides ) * 0.04 +
933 ( sk_dodge + 1 ) * ( 3 + armor_bash + armor_cut ) * 0.04 +
934 ( difficulty_base + special_attacks.size() + 8 * emit_fields.size() );
935 difficulty *= ( hp + speed - attack_cost + ( morale + agro ) * 0.1 ) * 0.01 +
936 ( vision_day + 2 * vision_night ) * 0.01;
937}

1015{

1017 const std::string type = obj.get_string( "type", "monster_attack" );
1018 const std::string attack_type = obj.get_string( "attack_type", type );
1019

1023 "type" );
1024 }
1025

1028 const std::string id = obj.get_string( "id" );
1029 const auto &iter = attack_map.find( id );
1030 if( iter == attack_map.end() ) {
1031 obj.throw_error( "Monster attacks must specify type and/or id", "type" );
1032 }
1033
1034 new_attack = iter->second->clone();
1035 } else if( attack_type == "leap" ) {
1036 new_attack = std::make_unique<leap_actor>();
1037 } else if( attack_type == "melee" ) {
1038 new_attack = std::make_unique<melee_actor>();
1039 } else if( attack_type == "bite" ) {
1040 new_attack = std::make_unique<bite_actor>();
1041 } else if( attack_type == "gun" ) {
1042 new_attack = std::make_unique<gun_actor>();
1043 } else if( attack_type == "spell" ) {
1044 new_attack = std::make_unique<mon_spellcasting_actor>();
1045 } else {
1046 obj.throw_error( "unknown monster attack", "attack_type" );
1047 }
1048
1049 new_attack->load( obj, src );
1050 return mtype_special_attack( std::move( new_attack ) );
