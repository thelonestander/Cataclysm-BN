112/** Sums the two terms, being careful to not trigger overflow.
113 * Doesn't handle underflow.
114 *
115 * @param a The first addend.
116 * @param b The second addend.
117 * @return the sum of the addends, but truncated to std::numeric_limits<int>::max().
118 */
119template <typename T>
120static T sum_no_wrap( T a, T b )
121{
122 if( a > std::numeric_limits<T>::max() - b ||
123 b > std::numeric_limits<T>::max() - a ) {
124 return std::numeric_limits<T>::max();
125 }
126 return a + b;
127}

131{

381static VisitResponse visit_internal( const std::function<VisitResponse( item *, item * )> &func,
382 item *node, item *parent = nullptr )
383{
384 switch( func( node, parent ) ) {
385 case VisitResponse::ABORT:
386 return VisitResponse::ABORT;
387
388 case VisitResponse::NEXT:
389 if( node->is_gun() || node->is_magazine() ) {
390 // Content of guns and magazines are accessible only via their specific accessors
391 return VisitResponse::NEXT;
392 }
393
394 if( node->contents.visit_contents( func, node ) == VisitResponse::ABORT ) {
395 return VisitResponse::ABORT;
396 }
397 /* intentional fallthrough */
398
399 case VisitResponse::SKIP:
400 return VisitResponse::NEXT;
401 }
402
403 /* never reached but suppresses GCC warning */
404 return VisitResponse::ABORT;
405}

612/** @relates visitable */
613template <>
614std::list<item> visitable<inventory>::remove_items_with( const
615 std::function<bool( const item &e )> &filter, int count )
616{
617 auto inv = static_cast<inventory *>( this );
618 std::list<item> res;
619
620 if( count <= 0 ) {
621 // nothing to do
622 return res;
623 }
624
625 for( auto stack = inv->items.begin(); stack != inv->items.end() && count > 0; ) {
626 std::list<item> &istack = *stack;
627 const auto original_invlet = istack.front().invlet;
628
629 for( auto istack_iter = istack.begin(); istack_iter != istack.end() && count > 0; ) {
630 if( filter( *istack_iter ) ) {
631 count--;
632 res.splice( res.end(), istack, istack_iter++ );
633 // The non-first items of a stack may have different invlets, the code
634 // in inventory only ever checks the invlet of the first item. This
635 // ensures that the first item of a stack always has the same invlet, even
636 // after the original first item was removed.
637 if( istack_iter == istack.begin() && istack_iter != istack.end() ) {
638 istack_iter->invlet = original_invlet;
639 }
640
641 } else {
642 istack_iter->contents.remove_internal( filter, count, res );
643 ++istack_iter;
644 }
645 }
646
647 if( istack.empty() ) {
648 stack = inv->items.erase( stack );
649 } else {
650 ++stack;
651 }
652 }
653
654 // Invalidate binning cache
655 inv->binned = false;
656 inv->items_type_cached = false;
657
658 return res;
659}

662template <>
663std::list<item> visitable<Character>::remove_items_with( const
664 std::function<bool( const item &e )> &filter, int count )
665{
666 auto ch = static_cast<Character *>( this );
667 std::list<item> res;
668
669 if( count <= 0 ) {
670 // nothing to do
671 return res;
672 }
673
674 // first try and remove items from the inventory
675 res = ch->inv.remove_items_with( filter, count );
676 count -= res.size();
677 if( count == 0 ) {
678 return res;
679 }
680
681 // then try any worn items
682 for( auto iter = ch->worn.begin(); iter != ch->worn.end(); ) {
683 if( filter( *iter ) ) {
684 iter->on_takeoff( *ch );
685 res.splice( res.end(), ch->worn, iter++ );
686 if( --count == 0 ) {
687 return res;
688 }
689 } else {
690 iter->contents.remove_internal( filter, count, res );
691 if( count == 0 ) {
692 return res;
693 }
694 ++iter;
695 }
696 }
697
698 // finally try the currently wielded item (if any)
699 if( filter( ch->weapon ) ) {
700 res.push_back( ch->remove_weapon() );
701 count--;
702 } else {
703 ch->weapon.contents.remove_internal( filter, count, res );
704 }
705
706 return res;

716

777std::list<item> visitable<vehicle_cursor>::remove_items_with( const

779{

782
783 if( count <= 0 ) {

785 return res;
786 }
787
788 int idx = cur->veh.part_with_feature( cur->part, "CARGO", false );
789 if( idx < 0 ) {
790 return res;
791 }
792
793 vehicle_part &part = cur->veh.part( idx );

795 if( filter( *iter ) ) {
796 // remove from the active items cache (if it isn't there does nothing)
797 cur->veh.active_items.remove( &*iter );
798
799 res.push_back( *iter );
800 iter = part.items.erase( iter );
801
802 if( --count == 0 ) {
803 return res;
804 }
805 } else {
806 iter->contents.remove_internal( filter, count, res );
807 if( count == 0 ) {
808 return res;
809 }
810 ++iter;
811 }
812 }
813
814 if( !res.empty() ) {
815 // if we removed any items then invalidate the cached mass
816 cur->veh.invalidate_mass();
817 }
818
819 return res;
820}

899template <typename T, typename M>
900static int charges_of_internal( const T &self, const M &main, const itype_id &id, int limit,
901 const std::function<bool( const item & )> &filter,
902 std::function<void( int )> visitor )
903{
904 int qty = 0;
905
906 bool found_tool_with_UPS = false;
907 self.visit_items( [&]( const item * e ) {
908 if( filter( *e ) ) {
909 if( e->is_tool() ) {
910 if( e->typeId() == id ) {
911 // includes charges from any included magazine.
912 qty = sum_no_wrap( qty, e->ammo_remaining() );
913 if( e->has_flag( "USE_UPS" ) ) {
914 found_tool_with_UPS = true;
915 }
916 }
917 return qty < limit ? VisitResponse::SKIP : VisitResponse::ABORT;
918
919 } else if( e->count_by_charges() ) {
920 if( e->typeId() == id ) {
921 qty = sum_no_wrap( qty, e->charges );
922 }
923 // items counted by charges are not themselves expected to be containers
924 return qty < limit ? VisitResponse::SKIP : VisitResponse::ABORT;
925 }
926 }
927 // recurse through any nested containers
928 return qty < limit ? VisitResponse::NEXT : VisitResponse::ABORT;
929 } );
930
931 if( qty < limit && found_tool_with_UPS ) {
932 qty += main.charges_of( itype_UPS, limit - qty );
933 if( visitor ) {
934 visitor( qty );
935 }
936 }
937
938 return std::min( qty, limit );
939}

980int visitable<Character>::charges_of( const itype_id &what, int limit,
981 const std::function<bool( const item & )> &filter,

985 auto p = dynamic_cast<const player *>( self );
986
987 if( what == itype_toolset ) {
988 if( p && p->has_active_bionic( bio_tools ) ) {

991 return 0;
992 }

997 int power_charges = 0;

999 for( const bionic &bio : *self->my_bionics ) {

1007 power_charges = units::to_kilojoule( self->as_player()->get_power_level() ) * efficiency;

1009 return std::min( power_charges, limit );
1010 }
1011
1012 if( what == itype_UPS ) {
1013 int qty = 0;
1014 qty = sum_no_wrap( qty, charges_of( itype_UPS_off ) );
1015 qty = sum_no_wrap( qty, static_cast<int>( charges_of( itype_adv_UPS_off ) / 0.6 ) );

1017 qty = sum_no_wrap( qty, units::to_kilojoule( p->get_power_level() ) );

1020 auto mons = p->mounted_creature.get();

1022 qty = sum_no_wrap( qty, mons->battery_item->ammo_remaining() );

1026 }

1028 return charges_of_internal( *this, *this, what, limit, filter, visitor );
