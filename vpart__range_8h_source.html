17/**
18 * Exposes (multiple) parts of one vehicle as @ref vpart_reference.
19 * Uses an arbitrary @ref range_type to supply
20 * - the vehicle reference,
21 * - information which parts should be iterated over (and which
22 * should be skipped).
23 */
24template<typename range_type>
25class vehicle_part_iterator
26{
27 private:
28 std::reference_wrapper<const range_type> range_;
29 cata::optional<vpart_reference> vp_;
30
31 const range_type &range() const {
32 return range_.get();
33 }
34 void skip_to_next_valid( size_t i ) {
35 while( i < range().part_count() &&
36 !range().matches( i ) ) {
37 ++i;
38 }
39 if( i < range().part_count() ) {
40 vp_.emplace( range().vehicle(), i );
41 } else {
42 vp_.reset();
43 }
44 }
45
46 public:
47 vehicle_part_iterator( const range_type &r, size_t i ) : range_( r ) {
48 assert( i <= range().part_count() );
49 skip_to_next_valid( i );
50 }
51 vehicle_part_iterator( const vehicle_part_iterator & ) = default;
52 vehicle_part_iterator &operator=( const vehicle_part_iterator & ) = default;
53
54 const vpart_reference &operator*() const {
55 assert( vp_ );
56 return *vp_;
57 }
58 const vpart_reference *operator->() const {
59 assert( vp_ );
60 return &*vp_;
61 }
62
63 vehicle_part_iterator &operator++() {
64 assert( vp_ );
65 skip_to_next_valid( vp_->part_index() + 1 );
66 return *this;
67 }
68
69 bool operator==( const vehicle_part_iterator &rhs ) const {
70 if( vp_.has_value() != rhs.vp_.has_value() ) {
71 return false;
72 }
73 if( !vp_.has_value() ) {
74 return true;
75 }
76 return &vp_->vehicle() == &rhs.vp_->vehicle() && vp_->part_index() == rhs.vp_->part_index();
77 }
78 bool operator!=( const vehicle_part_iterator &rhs ) const {
79 return !operator==( rhs );
80 }
81};
