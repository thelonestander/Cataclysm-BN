186void stats_tracker::unwatch( base_watcher *watcher )
187{
188 // Use a slow O(n) approach for now; if it proves problematic we can build
189 // an index, but that seems over-complex.
190 auto erase_from = [watcher]( auto & map_of_watcher_sets ) {
191 for( auto &p : map_of_watcher_sets ) {
192 auto &set = p.second;
193 if( set.erase( watcher ) ) {
194 return true;
195 }
196 }
197 return false;
198 };
199
200
201 if( erase_from( event_type_watchers ) ||
202 erase_from( event_transformation_watchers ) ||
203 erase_from( stat_watchers ) ) {
204 return;
205 }
206 debugmsg( "unwatch for a watcher not found" );
207}
