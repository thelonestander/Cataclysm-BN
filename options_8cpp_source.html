503 const options_manager::Group &group,

505{
506 if( !adding_to_group_.empty() ) {

527
528 entries( page_id );

530 adding_to_group_.clear();

550{

558 const std::string &existingOptionType = existingOption.getType();

565 }

1013/** Fill a mapping with values.
1014 * Scans all directories in @p dirname directory for
1015 * a file named @p filename.
1016 * All found values added to resource_option as name, resource_dir.
1017 * Furthermore, it builds possible values list for cOpt class.
1018 */
1019static std::vector<options_manager::id_and_option> build_resource_list(
1020 std::map<std::string, std::string> &resource_option, const std::string &operation_name,
1021 const std::string &dirname, const std::string &filename )
1022{
1023 std::vector<options_manager::id_and_option> resource_names;
1024
1025 resource_option.clear();
1026 const auto resource_dirs = get_directories_with( filename, dirname, true );
1027
1028 for( auto &resource_dir : resource_dirs ) {
1029 read_from_file( resource_dir + "/" + filename, [&]( std::istream & fin ) {
1030 std::string resource_name;
1031 std::string view_name;
1032 // should only have 2 values inside it, otherwise is going to only load the last 2 values
1033 while( !fin.eof() ) {
1034 std::string sOption;
1035 fin >> sOption;
1036
1037 if( sOption.empty() ) {
1038 getline( fin, sOption ); // Empty line, chomp it
1039 } else if( sOption[0] == '#' ) { // # indicates a comment
1040 getline( fin, sOption );
1041 } else {
1042 if( sOption.find( "NAME" ) != std::string::npos ) {
1043 resource_name.clear();
1044 getline( fin, resource_name );
1045 resource_name = trim( resource_name );
1046 } else if( sOption.find( "VIEW" ) != std::string::npos ) {
1047 view_name.clear();
1048 getline( fin, view_name );
1049 view_name = trim( view_name );
1050 break;
1051 }
1052 }
1053 }
1054 resource_names.emplace_back( resource_name,
1055 view_name.empty() ? no_translation( resource_name ) : to_translation( view_name ) );
1056 if( resource_option.count( resource_name ) != 0 ) {
1057 debugmsg( "Found \"%s\" duplicate with name \"%s\" (new definition will be ignored)",
1058 operation_name, resource_name );
1059 } else {
1060 resource_option.insert( std::pair<std::string, std::string>( resource_name, resource_dir ) );
1061 }
1062 } );
1063 }
1064
1065 return resource_names;
1066}
