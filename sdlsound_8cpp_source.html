94bool init_sound()
95{
96 int audio_rate = 44100;
97 Uint16 audio_format = AUDIO_S16;
98 int audio_channels = 2;
99 int chunksize = 2048;
100 // Autodetect device
101 const char *audio_device = nullptr;
102 // We only care about sample format and number of channels.
103 // SDL will be allowed to choose a different frequency if it needs to.
104 int flags = SDL_AUDIO_ALLOW_FREQUENCY_CHANGE;
105
106 // We should only need to init once
107 if( !sound_init_success ) {
108 const char *driver = SDL_GetCurrentAudioDriver();
109 DebugLog( DL::Info, DC::Main ) << "Active audio driver: " << driver;
110
111 int open_code = Mix_OpenAudioDevice( audio_rate, audio_format, audio_channels,
112 chunksize, audio_device, flags );
113 if( open_code == 0 ) {
114 int dev_rate = 0;
115 Uint16 dev_format = 0;
116 int dev_channels = 0;
117 if( Mix_QuerySpec( &dev_rate, &dev_format, &dev_channels ) ) {
118 DebugLog( DL::Info, DC::Main )
119 << string_format( "Opened mixer with specs: %d Hz, %d channel(s), format %#x",
120 dev_rate, dev_channels, dev_format );
121 } else {
122 DebugLog( DL::Warn, DC::SDL ) << "Mix_QuerySpec failed: " << Mix_GetError();
123 }
124
125 Mix_AllocateChannels( 128 );
126 Mix_ReserveChannels( static_cast<int>( sfx::channel::MAX_CHANNEL ) );
127
128 // For the sound effects system.
129 Mix_GroupChannels( static_cast<int>( sfx::channel::daytime_outdoors_env ),
130 static_cast<int>( sfx::channel::nighttime_outdoors_env ),
131 static_cast<int>( sfx::group::time_of_day ) );
132 Mix_GroupChannels( static_cast<int>( sfx::channel::underground_env ),
133 static_cast<int>( sfx::channel::outdoor_blizzard ),
134 static_cast<int>( sfx::group::weather ) );
135 Mix_GroupChannels( static_cast<int>( sfx::channel::danger_extreme_theme ),
136 static_cast<int>( sfx::channel::danger_low_theme ),
137 static_cast<int>( sfx::group::context_themes ) );
138 Mix_GroupChannels( static_cast<int>( sfx::channel::stamina_75 ),
139 static_cast<int>( sfx::channel::stamina_35 ),
140 static_cast<int>( sfx::group::fatigue ) );
141
142 sound_init_success = true;
143 } else {
144 dbg( DL::Error ) << "Failed to open audio mixer, sound won't work: " << Mix_GetError();

189/** Callback called when we finish playing music. */
190void musicFinished()
191{
192 if( test_mode ) {
193 return;
194 }
195
196 Mix_HaltMusic();
197 Mix_FreeMusic( current_music );
198 current_music = nullptr;
199
200 const auto iter = playlists.find( current_playlist );
201 if( iter == playlists.end() ) {
202 return;
203 }
204 const music_playlist &list = iter->second;
205 if( list.entries.empty() ) {
206 return;
207 }
208
209 // Load the next file to play.
210 absolute_playlist_at++;
211
212 // Wrap around if we reached the end of the playlist.
213 if( absolute_playlist_at >= list.entries.size() ) {
214 absolute_playlist_at = 0;
215 }
216
217 current_playlist_at = playlist_indexes.at( absolute_playlist_at );
218
219 const music_playlist::entry &next = list.entries[current_playlist_at];
220 play_music_file( next.file, next.volume );
221}

223void play_music( const std::string &playlist )
224{
225 const auto iter = playlists.find( playlist );
226 if( iter == playlists.end() ) {
227 return;
228 }
229 const music_playlist &list = iter->second;
230 if( list.entries.empty() ) {
231 return;
232 }
233
234 // Don't interrupt playlist that's already playing.
235 if( playlist == current_playlist ) {
236 return;
237 }
238
239 for( size_t i = 0; i < list.entries.size(); i++ ) {
240 playlist_indexes.push_back( i );
241 }
242 if( list.shuffle ) {
243 // Son't need to worry about the determinism check here because it only
244 // affects audio, not game logic.
245 // NOLINTNEXTLINE(cata-determinism)
246 static auto eng = cata_default_random_engine(
247 std::chrono::system_clock::now().time_since_epoch().count() );
248 std::shuffle( playlist_indexes.begin(), playlist_indexes.end(), eng );
249 }
250
251 current_playlist = playlist;
252 current_playlist_at = playlist_indexes.at( absolute_playlist_at );
253
254 const music_playlist::entry &next = list.entries[current_playlist_at];
255 play_music_file( next.file, next.volume );
256}

439static Mix_Chunk *do_pitch_shift( Mix_Chunk *s, float pitch )
440{
441 Uint32 s_in = s->alen / 4;
442 Uint32 s_out = static_cast<Uint32>( static_cast<float>( s_in ) * pitch );
443 float pitch_real = static_cast<float>( s_out ) / static_cast<float>( s_in );
444 Mix_Chunk *result = static_cast<Mix_Chunk *>( malloc( sizeof( Mix_Chunk ) ) );
445 result->allocated = 1;
446 result->alen = s_out * 4;
447 result->abuf = static_cast<Uint8 *>( malloc( result->alen * sizeof( Uint8 ) ) );
448 result->volume = s->volume;
449 for( Uint32 i = 0; i < s_out; i++ ) {
450 Sint16 lt = 0;
451 Sint16 rt = 0;
452 Sint16 lt_out = 0;
453 Sint16 rt_out = 0;
454 Sint64 lt_avg = 0;
455 Sint64 rt_avg = 0;
456 Uint32 begin = static_cast<Uint32>( static_cast<float>( i ) / pitch_real );
457 Uint32 end = static_cast<Uint32>( static_cast<float>( i + 1 ) / pitch_real );
458
459 // check for boundary case
460 if( end > 0 && ( end >= ( s->alen / 4 ) ) ) {
461 end = begin;
462 }
463
464 for( Uint32 j = begin; j <= end; j++ ) {
465 lt = ( s->abuf[( 4 * j ) + 1] << 8 ) | ( s->abuf[( 4 * j ) + 0] );
466 rt = ( s->abuf[( 4 * j ) + 3] << 8 ) | ( s->abuf[( 4 * j ) + 2] );
467 lt_avg += lt;
468 rt_avg += rt;
469 }
470 lt_out = static_cast<Sint16>( static_cast<float>( lt_avg ) / static_cast<float>
471 ( end - begin + 1 ) );
472 rt_out = static_cast<Sint16>( static_cast<float>( rt_avg ) / static_cast<float>
473 ( end - begin + 1 ) );
474 result->abuf[( 4 * i ) + 1] = static_cast<Uint8>( ( lt_out >> 8 ) & 0xFF );
475 result->abuf[( 4 * i ) + 0] = static_cast<Uint8>( lt_out & 0xFF );
476 result->abuf[( 4 * i ) + 3] = static_cast<Uint8>( ( rt_out >> 8 ) & 0xFF );
477 result->abuf[( 4 * i ) + 2] = static_cast<Uint8>( rt_out & 0xFF );
478 }
479 return result;
480}

531 if( is_pitched ) {
532 double pitch_random = rng_float( pitch_min, pitch_max );
533 effect_to_play = do_pitch_shift( effect_to_play, static_cast<float>( pitch_random ) );
534 }
535 Mix_VolumeChunk( effect_to_play,

538 bool failed = ( channel == -1 );
539 if( !failed && is_pitched ) {
540 if( Mix_RegisterEffect( channel, empty_effect, cleanup_when_channel_finished,
541 effect_to_play ) == 0 ) {
542 // We'll be unable to de-allocate the chunk, stop the playback right now.
543 failed = true;
544 dbg( DL::Warn ) << "Mix_RegisterEffect failed: " << Mix_GetError();
545 Mix_HaltChannel( channel );
546 }
547 }
548 if( !failed ) {
549 if( Mix_SetPosition( channel, static_cast<Sint16>( to_degrees( angle ) ), 1 ) == 0 ) {
550 // Not critical
551 dbg( DL::Info ) << "Mix_SetPosition failed: " << Mix_GetError();
552 }
553 }
554 if( failed ) {
555 dbg( DL::Error ) << "Failed to play sound effect: " << Mix_GetError();
556 if( is_pitched ) {
