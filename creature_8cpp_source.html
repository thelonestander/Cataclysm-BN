202bool Creature::sees( const Creature &critter ) const
203{
204 // Creatures always see themselves (simplifies drawing).
205 if( &critter == this ) {
206 return true;
207 }
208
209 if( critter.is_hallucination() ) {
210 // hallucinations are imaginations of the player character, npcs or monsters don't hallucinate.
211 // Invisible hallucinations would be pretty useless (nobody would see them at all), therefor
212 // the player will see them always.
213 return is_player();
214 }
215
216 if( !fov_3d && !debug_mode && posz() != critter.posz() ) {
217 return false;
218 }
219
220 // This check is ridiculously expensive so defer it to after everything else.
221 auto visible = []( const Character * ch ) {
222 return ch == nullptr || !ch->is_invisible();
223 };
224
225 const Character *ch = critter.as_character();
226 const int wanted_range = rl_dist( pos(), critter.pos() );
227 // Can always see adjacent monsters on the same level.
228 // We also bypass lighting for vertically adjacent monsters, but still check for floors.
229 if( wanted_range <= 1 && ( posz() == critter.posz() || g->m.sees( pos(), critter.pos(), 1 ) ) ) {
230 return visible( ch );
231 } else if( ( wanted_range > 1 && critter.digging() ) ||
232 ( critter.has_flag( MF_NIGHT_INVISIBILITY ) && g->m.light_at( critter.pos() ) <= lit_level::LOW ) ||
233 ( critter.is_underwater() && !is_underwater() && g->m.is_divable( critter.pos() ) ) ||
234 ( g->m.has_flag_ter_or_furn( TFLAG_HIDE_PLACE, critter.pos() ) &&
235 !( std::abs( posx() - critter.posx() ) <= 1 && std::abs( posy() - critter.posy() ) <= 1 &&
236 std::abs( posz() - critter.posz() ) <= 1 ) ) ) {
237 return false;
238 }
239 if( ch != nullptr ) {
240 if( ch->movement_mode_is( CMM_CROUCH ) ) {
241 const int coverage = g->m.obstacle_coverage( pos(), critter.pos() );
242 if( coverage < 30 ) {
243 return sees( critter.pos(), critter.is_avatar() ) && visible( ch );
244 }
245 float size_modifier = 1.0;
246 switch( ch->get_size() ) {
247 case MS_TINY:
248 size_modifier = 2.0;
249 break;
250 case MS_SMALL:
251 size_modifier = 1.4;
252 break;
253 case MS_MEDIUM:
254 break;
255 case MS_LARGE:
256 size_modifier = 0.6;
257 break;
258 case MS_HUGE:
259 size_modifier = 0.15;
260 break;
261 default:
262 break;
263 }
264 const int vision_modifier = 30 - 0.5 * coverage * size_modifier;
265 if( vision_modifier > 1 ) {
266 return sees( critter.pos(), critter.is_avatar(), vision_modifier ) && visible( ch );
267 }
268 return false;
269 }
270 }
271 return sees( critter.pos(), critter.is_avatar() ) && visible( ch );
272}

274bool Creature::sees( const tripoint &t, bool is_avatar, int range_mod ) const
275{
276 if( !fov_3d && posz() != t.z ) {
277 return false;
278 }
279
280 const int range_cur = sight_range( g->m.ambient_light_at( t ) );
281 const int range_day = sight_range( default_daylight_level() );
282 const int range_night = sight_range( 0 );
283 const int range_max = std::max( range_day, range_night );
284 const int range_min = std::min( range_cur, range_max );
285 const int wanted_range = rl_dist( pos(), t );
286 if( wanted_range <= range_min ||
287 ( wanted_range <= range_max &&
288 g->m.ambient_light_at( t ) > g->natural_light_level( t.z ) ) ) {
289 int range = 0;
290 if( g->m.ambient_light_at( t ) > g->natural_light_level( t.z ) ) {
291 range = MAX_VIEW_DISTANCE;
292 } else {
293 range = range_min;
294 }
295 if( has_effect( effect_no_sight ) ) {
296 range = 1;
297 }
298 if( range_mod > 0 ) {
299 range = std::min( range, range_mod );
300 }
301 if( is_avatar ) {
302 // Special case monster -> player visibility, forcing it to be symmetric with player vision.
303 const float player_visibility_factor = g->u.visibility() / 100.0f;
304 int adj_range = std::floor( range * player_visibility_factor );
305 return adj_range >= wanted_range &&
306 g->m.get_cache_ref( pos().z ).seen_cache[pos().x][pos().y] > LIGHT_TRANSPARENCY_SOLID;
307 } else {
308 return g->m.sees( pos(), t, range );
309 }
310 } else {
311 return false;
312 }

330Creature *Creature::auto_find_hostile_target( int range, int &boo_hoo, int area )
331{
332 Creature *target = nullptr;
333 player &u = g->u; // Could easily protect something that isn't the player
334 constexpr int hostile_adj = 2; // Priority bonus for hostile targets
335 const int iff_dist = ( range + area ) * 3 / 2 + 6; // iff check triggers at this distance
336 // iff safety margin (degrees). less accuracy, more paranoia
337 units::angle iff_hangle = units::from_degrees( 15 + area );
338 float best_target_rating = -1.0f; // bigger is better
339 units::angle u_angle = {}; // player angle relative to turret
340 boo_hoo = 0; // how many targets were passed due to IFF. Tragically.
341 bool self_area_iff = false; // Need to check if the target is near the vehicle we're a part of
342 bool area_iff = false; // Need to check distance from target to player
343 bool angle_iff = true; // Need to check if player is in a cone between us and target
344 int pldist = rl_dist( pos(), g->u.pos() );
345 vehicle *in_veh = is_fake() ? veh_pointer_or_null( g->m.veh_at( pos() ) ) : nullptr;
346 if( pldist < iff_dist && sees( g->u ) ) {
347 area_iff = area > 0;
348 angle_iff = true;
349 // Player inside vehicle won't be hit by shots from the roof,
350 // so we can fire "through" them just fine.
351 const optional_vpart_position vp = g->m.veh_at( u.pos() );
352 if( in_veh && veh_pointer_or_null( vp ) == in_veh && vp->is_inside() ) {
353 angle_iff = false; // No angle IFF, but possibly area IFF
354 } else if( pldist < 3 ) {
355 // granularity increases with proximity
356 iff_hangle = ( pldist == 2 ? 30_degrees : 60_degrees );
357 }
358 u_angle = coord_to_angle( pos(), u.pos() );
359 }
360
361 if( area > 0 && in_veh != nullptr ) {
362 self_area_iff = true;
363 }
364
365 std::vector<Creature *> targets = g->get_creatures_if( [&]( const Creature & critter ) {
366 if( critter.is_monster() ) {
367 // friendly to the player, not a target for us
368 return static_cast<const monster *>( &critter )->friendly == 0;
369 }
370 if( critter.is_npc() ) {
371 // friendly to the player, not a target for us
372 return static_cast<const npc *>( &critter )->get_attitude() == NPCATT_KILL;
373 }
374 // TODO: what about g->u?
375 return false;
376 } );
377 for( auto &m : targets ) {
378 if( !sees( *m ) ) {
379 // can't see nor sense it
380 if( is_fake() && in_veh ) {
381 // If turret in the vehicle then
382 // Hack: trying yo avoid turret LOS blocking by frames bug by trying to see target from vehicle boundary
383 // Or turret wallhack for turret's car
384 // TODO: to visibility checking another way, probably using 3D FOV
385 std::vector<tripoint> path_to_target = line_to( pos(), m->pos() );
386 path_to_target.insert( path_to_target.begin(), pos() );
387
388 // Getting point on vehicle boundaries and on line between target and turret
389 bool continueFlag = true;
390 do {
391 const optional_vpart_position vp = g->m.veh_at( path_to_target.back() );
392 vehicle *const veh = vp ? &vp->vehicle() : nullptr;
393 if( in_veh == veh ) {
394 continueFlag = false;
395 } else {
396 path_to_target.pop_back();
397 }
398 } while( continueFlag );
399
400 tripoint oldPos = pos();
401 setpos( path_to_target.back() ); //Temporary moving targeting npc on vehicle boundary postion
402 bool seesFromVehBound = sees( *m ); // And look from there
403 setpos( oldPos );
404 if( !seesFromVehBound ) {
405 continue;
406 }
407 } else {
408 continue;
409 }
410 }
411 int dist = rl_dist( pos(), m->pos() ) + 1; // rl_dist can be 0
412 if( dist > range + 1 || dist < area ) {
413 // Too near or too far
414 continue;
415 }
416 // Prioritize big, armed and hostile stuff
417 float mon_rating = m->power_rating();
418 float target_rating = mon_rating / dist;
419 if( mon_rating + hostile_adj <= 0 ) {
420 // We wouldn't attack it even if it was hostile
421 continue;
422 }
423
424 if( in_veh != nullptr && veh_pointer_or_null( g->m.veh_at( m->pos() ) ) == in_veh ) {
425 // No shooting stuff on vehicle we're a part of
426 continue;
427 }
428 if( area_iff && rl_dist( u.pos(), m->pos() ) <= area ) {
429 // Player in AoE
430 boo_hoo++;
431 continue;
432 }
433 // Hostility check can be expensive, but we need to inform the player of boo_hoo
434 // only when the target is actually "hostile enough"
435 bool maybe_boo = false;
436 if( angle_iff ) {
437 units::angle tangle = coord_to_angle( pos(), m->pos() );
438 units::angle diff = units::fabs( u_angle - tangle );
439 // Player is in the angle and not too far behind the target
440 if( ( diff + iff_hangle > 360_degrees || diff < iff_hangle ) &&
441 ( dist * 3 / 2 + 6 > pldist ) ) {
442 maybe_boo = true;
443 }
444 }
445 if( !maybe_boo && ( ( mon_rating + hostile_adj ) / dist <= best_target_rating ) ) {
446 // "Would we skip the target even if it was hostile?"
447 // Helps avoid (possibly expensive) attitude calculation
448 continue;
449 }
450 if( m->attitude_to( u ) == A_HOSTILE ) {
451 target_rating = ( mon_rating + hostile_adj ) / dist;
452 if( maybe_boo ) {
453 boo_hoo++;
454 continue;
455 }
456 }
457 if( target_rating <= best_target_rating || target_rating <= 0 ) {
458 continue; // Handle this late so that boo_hoo++ can happen
459 }
460 // Expensive check for proximity to vehicle
461 if( self_area_iff && overlaps_vehicle( in_veh->get_points(), m->pos(), area ) ) {
462 continue;
463 }
464
465 target = m;
466 best_target_rating = target_rating;
467 }
468 return target;
469}

516 const damage_instance &dam, dealt_damage_instance &dealt_dam )

523 // melee attack will start off as targeted at mount

525 monster *mons = dynamic_cast<monster *>( this );
526 if( mons && mons->mounted_player ) {

528 one_in( std::max( 2, mons->get_size() - mons->mounted_player->get_size() ) ) ) {

532 }
533 }

535 bodypart_id bp_hit = convert_bp( select_body_part( source, hit_spread ) ).id();
536 const body_part bp_token = bp_hit->token;
537 block_hit( source, bp_hit, d );
538
539 on_hit( source, bp_hit ); // trigger on-gethit events
540 dealt_dam = deal_damage( source, bp_hit, d );
541 dealt_dam.bp_hit = bp_token;
542}

644/**
645 * Attempts to harm a creature with a projectile.
646 *
647 * @param source Pointer to the creature who shot the projectile.
648 * @param attack A structure describing the attack and its results.
649 * @param print_messages enables message printing by default.
650 */
651void Creature::deal_projectile_attack( Creature *source, dealt_projectile_attack &attack )
652{
653 const bool magic = attack.proj.has_effect( ammo_effect_magic );
654 const bool targetted_crit_allowed = !attack.proj.has_effect( ammo_effect_NO_CRIT );
655 const double missed_by = attack.missed_by;
656 if( missed_by >= 1.0 && !magic ) {
657 // Total miss
658 return;
659 }
660 // If carrying a rider, there is a chance the hits may hit rider instead.
661 if( has_effect( effect_ridden ) ) {
662 monster *mons = dynamic_cast<monster *>( this );
663 if( mons && mons->mounted_player ) {
664 if( !mons->has_flag( MF_MECH_DEFENSIVE ) &&
665 one_in( std::max( 2, mons->get_size() - mons->mounted_player->get_size() ) ) ) {
666 mons->mounted_player->deal_projectile_attack( source, attack );
667 return;
668 }
669 }
670 }
671 const projectile &proj = attack.proj;
672 dealt_damage_instance &dealt_dam = attack.dealt_dam;
673
674 const bool u_see_this = g->u.sees( *this );
675
676 const int avoid_roll = dodge_roll();
677 // Do dice(10, speed) instead of dice(speed, 10) because speed could potentially be > 10000
678 const int diff_roll = dice( 10, proj.speed );
679 // Partial dodge, capped at [0.0, 1.0], added to missed_by
680 const double dodge_rescaled = avoid_roll / static_cast<double>( diff_roll );
681 const double goodhit = missed_by + std::max( 0.0, std::min( 1.0, dodge_rescaled ) );
682
683 if( goodhit >= 1.0 && !magic ) {
684 attack.missed_by = 1.0; // Arbitrary value
685 // "Avoid" rather than "dodge", because it includes removing self from the line of fire
686 // rather than just Matrix-style bullet dodging
687 if( source != nullptr && g->u.sees( *source ) ) {
688 add_msg_player_or_npc(
689 m_warning,
690 _( "You avoid %s projectile!" ),
691 _( "<npcname> avoids %s projectile." ),
692 source->disp_name( true ) );
693 } else {
694 add_msg_player_or_npc(
695 m_warning,
696 _( "You avoid an incoming projectile!" ),
697 _( "<npcname> avoids an incoming projectile." ) );
698 }
699 return;
700 }
701
702 // Bounce applies whether it does damage or not.
703 if( proj.has_effect( ammo_effect_BOUNCE ) ) {
704 add_effect( effect_bounced, 1_turns );
705 }
706
707 bodypart_id bp_hit;
708 double hit_value = missed_by + rng_float( -0.5, 0.5 );
709 if( targetted_crit_allowed || magic ) { //default logic for selecting bodypart
710 if( goodhit < accuracy_critical && hit_value <= 0.2 ) {
711 bp_hit = bodypart_str_id( "head" );
712 } else if( hit_value <= 0.4 || magic ) {
713 bp_hit = bodypart_str_id( "torso" );
714 } else if( one_in( 4 ) ) {
715 if( one_in( 2 ) ) {
716 bp_hit = bodypart_str_id( "leg_l" );
717 } else {
718 bp_hit = bodypart_str_id( "leg_r" );
719 }
720 } else {
721 if( one_in( 2 ) ) {
722 bp_hit = bodypart_str_id( "arm_l" );
723 } else {
724 bp_hit = bodypart_str_id( "arm_r" );
725 }
726 }
727 } else { // no crit logic for selecting bodypart
728 if( hit_value <= 0.4 && !one_in( 4 ) ) {
729 bp_hit = one_in( 3 ) ? bodypart_str_id( "head" ) : bodypart_str_id( "torso" );
730 } else if( one_in( 4 ) ) {
731 if( one_in( 2 ) ) {
732 bp_hit = bodypart_str_id( "leg_l" );
733 } else {
734 bp_hit = bodypart_str_id( "leg_r" );
735 }
736 } else {
737 if( one_in( 2 ) ) {
738 bp_hit = bodypart_str_id( "arm_l" );
739 } else {
740 bp_hit = bodypart_str_id( "arm_r" );
741 }
742 }
743 }
744
745
746 double damage_mult = 1.0;
747
748 ranged::hit_tier ht = ranged::hit_tier::normal;
749 if( magic ) {
750 damage_mult *= rng_float( 0.9, 1.1 );
751 } else if( targetted_crit_allowed && goodhit < accuracy_critical ) {
752 ht = ranged::hit_tier::critical;
753 damage_mult *= 1.5;
754 } else if( goodhit < accuracy_standard ) {
755 damage_mult *= rng_float( 0.9, 1.1 );
756 } else if( goodhit < accuracy_grazing ) {
757 ht = ranged::hit_tier::grazing;
758 damage_mult *= 0.5;
759 }
760
761 attack.missed_by = goodhit;
762
763 // copy it, since we're mutating
764 damage_instance impact = proj.impact;
765 if( damage_mult > 0.0f && proj.has_effect( ammo_effect_NO_DAMAGE_SCALING ) ) {
766 damage_mult = 1.0f;
767 }
768
769 impact.mult_damage( damage_mult );
770
771 if( proj.has_effect( ammo_effect_NOGIB ) ) {
772 float dmg_ratio = static_cast<float>( impact.total_damage() ) / get_hp_max( bp_hit );
773 if( dmg_ratio > 1.25f ) {
774 impact.mult_damage( 1.0f / dmg_ratio );
775 }
776 }
777
778 dealt_dam = deal_damage( source, bp_hit, impact );
779 dealt_dam.bp_hit = bp_hit->token;
780
781 // Apply ammo effects to target.
782 if( proj.has_effect( ammo_effect_TANGLE ) ) {
783 monster *z = dynamic_cast<monster *>( this );
784 player *n = dynamic_cast<player *>( this );
785 // if its a tameable animal, its a good way to catch them if they are running away, like them ranchers do!
786 // we assume immediate success, then certain monster types immediately break free in monster.cpp move_effects()
787 if( z ) {
788 const item &drop_item = proj.get_drop();
789 if( !drop_item.is_null() ) {
790 z->add_effect( effect_tied, 1_turns, num_bp );
791 z->tied_item = cata::make_value<item>( drop_item );
792 } else {
793 add_msg( m_debug, "projectile with TANGLE effect, but no drop item specified" );
794 }
795 } else if( n && !is_immune_effect( effect_downed ) ) {
796 // no tied up effect for people yet, just down them and stun them, its close enough to the desired effect.
797 // we can assume a person knows how to untangle their legs eventually and not panic like an animal.
798 add_effect( effect_downed, 1_turns );
799 // stunned to simulate staggering around and stumbling trying to get the entangled thing off of them.
800 add_effect( effect_stunned, rng( 3_turns, 8_turns ) );
801 }
802 }
803 if( proj.has_effect( ammo_effect_INCENDIARY ) ) {
804 if( made_of( material_id( "veggy" ) ) || made_of_any( cmat_flammable ) ) {
805 add_effect( effect_onfire, rng( 2_turns, 6_turns ), bp_hit->token );
806 } else if( made_of_any( cmat_flesh ) && one_in( 4 ) ) {
807 add_effect( effect_onfire, rng( 1_turns, 4_turns ), bp_hit->token );
808 }
809 } else if( proj.has_effect( ammo_effect_IGNITE ) ) {
810 if( made_of( material_id( "veggy" ) ) || made_of_any( cmat_flammable ) ) {
811 add_effect( effect_onfire, 6_turns, bp_hit->token );
812 } else if( made_of_any( cmat_flesh ) ) {
813 add_effect( effect_onfire, 10_turns, bp_hit->token );
814 }
815 }
816
817 if( bp_hit == bodypart_str_id( "head" ) && proj.has_effect( ammo_effect_BLINDS_EYES ) ) {
818 // TODO: Change this to require bp_eyes
819 add_env_effect( effect_blind, bp_eyes, 5, rng( 3_turns, 10_turns ) );
820 }
821
822 if( proj.has_effect( ammo_effect_APPLY_SAP ) ) {
823 add_effect( effect_sap, 1_turns * dealt_dam.total_damage() );
824 }
825 if( proj.has_effect( ammo_effect_PARALYZEPOISON ) && dealt_dam.total_damage() > 0 ) {
826 add_msg_if_player( m_bad, _( "You feel poison coursing through your body!" ) );
827 add_effect( effect_paralyzepoison, 5_minutes );
828 }
829
830 int stun_strength = 0;
831 if( proj.has_effect( ammo_effect_BEANBAG ) ) {
832 stun_strength = 4;
833 }
834 if( proj.has_effect( ammo_effect_LARGE_BEANBAG ) ) {
835 stun_strength = 16;
836 }
837 if( stun_strength > 0 ) {
838 switch( get_size() ) {
839 case MS_TINY:
840 stun_strength *= 4;
841 break;
842 case MS_SMALL:
843 stun_strength *= 2;
844 break;
845 case MS_MEDIUM:
846 default:
847 break;
848 case MS_LARGE:
849 stun_strength /= 2;
850 break;
851 case MS_HUGE:
852 stun_strength /= 4;
853 break;
854 }
855 add_effect( effect_stunned, 1_turns * rng( stun_strength / 2, stun_strength ) );
856 }
857
858 if( u_see_this ) {
859 if( damage_mult == 0 ) {
860 if( source != nullptr ) {
861 add_msg( source->is_player() ? _( "You miss!" ) : _( "The shot misses!" ) );
862 }
863 } else {
864 ranged::print_dmg_msg( *this, source, dealt_dam, ht );
865 }
866 }
867
868 check_dead_state();
869 attack.hit_critter = this;
870 attack.missed_by = goodhit;
871}

873dealt_damage_instance Creature::deal_damage( Creature *source, bodypart_id bp,
874 const damage_instance &dam )
875{
876 if( is_dead_state() ) {
877 return dealt_damage_instance();
878 }
879 int total_damage = 0;
880 int total_pain = 0;
881 damage_instance d = dam; // copy, since we will mutate in absorb_hit
882
883 dealt_damage_instance dealt_dams;
884 absorb_hit( bp, d );
885
886 // Add up all the damage units dealt
887 for( const auto &it : d.damage_units ) {
888 int cur_damage = 0;
889 deal_damage_handle_type( it, bp, cur_damage, total_pain );
890 if( cur_damage > 0 ) {
891 dealt_dams.dealt_dams[ it.type ] += cur_damage;
892 total_damage += cur_damage;
893 }
894 }
895
896 mod_pain( total_pain );
897
898 apply_damage( source, bp, total_damage );
899 return dealt_dams;
900}
901void Creature::deal_damage_handle_type( const damage_unit &du, bodypart_id bp, int &damage,
902 int &pain )
903{
904 // Handles ACIDPROOF, electric immunity etc.
905 if( is_immune_damage( du.type ) ) {
906 return;
907 }
908
909 // Apply damage multiplier from skill, critical hits or grazes after all other modifications.
910 const int adjusted_damage = du.amount * du.damage_multiplier;
911 if( adjusted_damage <= 0 ) {
912 return;
913 }
914
915 float div = 4.0f;
916
917 switch( du.type ) {
918 case DT_BASH:
919 // Bashing damage is less painful
920 div = 5.0f;
921 break;
922
923 case DT_HEAT:
924 // heat damage sets us on fire sometimes
925 if( rng( 0, 100 ) < adjusted_damage ) {
926 add_effect( effect_onfire, rng( 1_turns, 3_turns ), bp->token );
927 }
928 break;
929
930 case DT_ELECTRIC:
931 // Electrical damage adds a major speed/dex debuff
932 add_effect( effect_zapped, 1_turns * std::max( adjusted_damage, 2 ) );
933 break;
934
935 case DT_ACID:
936 // Acid damage and acid burns are more painful
937 div = 3.0f;
938 break;
939
940 default:
941 break;
942 }
943
944 on_damage_of_type( adjusted_damage, du.type, bp );
945
946 damage += adjusted_damage;
947 pain += roll_remainder( adjusted_damage / div );
948}

987void Creature::add_effect( const efftype_id &eff_id, const time_duration &dur,
988 const bodypart_str_id &base_bp,
989 int intensity, bool force, bool deferred )
990{
991 // Check our innate immunity
992 if( !force && is_immune_effect( eff_id ) ) {
993 return;
994 }
995 if( eff_id == efftype_id( "knockdown" ) && ( has_effect( effect_ridden ) ||
996 has_effect( effect_riding ) ) ) {
997 monster *mons = dynamic_cast<monster *>( this );
998 if( mons && mons->mounted_player ) {
999 mons->mounted_player->forced_dismount();
1000 }
1001 }
1002
1003 if( !eff_id.is_valid() ) {
1004 debugmsg( "Invalid effect, ID: %s", eff_id.c_str() );
1005 return;
1006 }
1007 const effect_type &type = eff_id.obj();
1008
1009 bodypart_str_id bp = base_bp;
1010 // Mutate to a main (HP'd) body_part if necessary.
1011 if( type.get_main_parts() ) {
1012 bp = bp->main_part;
1013 }
1014
1015 bool found = false;
1016 // Check if we already have it
1017 auto matching_map = effects->find( eff_id );
1018 if( matching_map != effects->end() ) {
1019 auto &bodyparts = matching_map->second;
1020 auto found_effect = bodyparts.find( bp );
1021 if( found_effect != bodyparts.end() && !found_effect->second.is_removed() ) {
1022 found = true;
1023 effect &e = found_effect->second;
1024 const int prev_int = e.get_intensity();
1025 // If we do, mod the duration, factoring in the mod value
1026 e.mod_duration( time_duration::from_turns( static_cast<int>(
1027 to_turns<int64_t>( dur ) * e.get_dur_add_perc() / 100 ) ) );
1028 // Limit to max duration
1029 if( e.get_max_duration() > 0_turns && e.get_duration() > e.get_max_duration() ) {
1030 e.set_duration( e.get_max_duration() );
1031 }
1032 // int_dur_factor overrides all other intensity settings
1033 // ...but it's handled in set_duration, so explicitly do nothing here
1034 if( e.get_int_dur_factor() > 0_turns ) {
1035 // Set intensity if value is given
1036 } else if( intensity > 0 ) {
1037 e.set_intensity( intensity );
1038 // Else intensity uses the type'd step size if it already exists
1039 } else if( e.get_int_add_val() != 0 ) {
1040 e.mod_intensity( e.get_int_add_val() );
1041 }
1042
1043 // Bound intensity by [1, max intensity]
1044 if( e.get_intensity() < 1 ) {
1045 add_msg( m_debug, "Bad intensity, ID: %s", e.get_id().c_str() );
1046 e.set_intensity( 1 );
1047 } else if( e.get_intensity() > e.get_max_intensity() ) {
1048 e.set_intensity( e.get_max_intensity() );
1049 }
1050 if( e.get_intensity() != prev_int ) {
1051 on_effect_int_change( e.get_id(), e.get_intensity(), e.get_bp() );
1052 }
1053 }
1054 }
1055
1056 if( !found ) {
1057 // If we don't already have it then add a new one
1058
1059 // Then check if the effect is blocked by another
1060 for( const auto &elem : *effects ) {
1061 for( auto &_effect_it : elem.second ) {
1062 if( !_effect_it.second.is_removed() )
1063 for( const auto &blocked_effect : _effect_it.second.get_blocks_effects() ) {
1064 if( blocked_effect == eff_id ) {
1065 // The effect is blocked by another, return
1066 return;
1067 }
1068 }
1069 }
1070 }
1071
1072 // Now we can make the new effect for application
1073 effect e( &type, dur, bp, intensity, calendar::turn );
1074 // Bound to max duration
1075 if( e.get_max_duration() > 0_turns && e.get_duration() > e.get_max_duration() ) {
1076 e.set_duration( e.get_max_duration() );
1077 }
1078
1079 // Force intensity if it is duration based
1080 if( e.get_int_dur_factor() != 0_turns ) {
1081 // + 1 here so that the lowest is intensity 1, not 0
1082 e.set_intensity( e.get_duration() / e.get_int_dur_factor() + 1 );
1083 }
1084 // Bound new effect intensity by [1, max intensity]
1085 if( e.get_intensity() < 1 ) {
1086 add_msg( m_debug, "Bad intensity, ID: %s", e.get_id().c_str() );
1087 e.set_intensity( 1 );
1088 } else if( e.get_intensity() > e.get_max_intensity() ) {
1089 e.set_intensity( e.get_max_intensity() );
1090 }
1091 ( *effects )[eff_id][bp] = e;
1092 if( Character *ch = as_character() ) {
1093 g->events().send<event_type::character_gains_effect>( ch->getID(), eff_id );
1094 if( is_player() && !type.get_apply_message().empty() ) {
1095 add_msg( type.gain_game_message_type(), _( type.get_apply_message() ) );
1096 }
1097 }
1098 on_effect_int_change( e.get_id(), e.get_intensity(), e.get_bp() );
1099 // Perform any effect addition effects.
1100 // only when not deferred
1101 if( !deferred ) {
1102 process_one_effect( e, true );
1103 }
1104 }

1137 //Effect doesn't exist, so do nothing

1153 if( !bp ) {

1156 if( !e.is_removed() ) {
1157 on_effect_int_change( e.get_id(), 0, e.get_bp() );

1167 // Must be below the set_removed above or we'll get an infinite loop
1168 if( ch != nullptr && eff_id == effect_sleep ) {
1169 ch->wake_up();
1170 }

1255 process_effects_internal();
1256
1257 // id's and body_part's of all effects to be removed. If we ever get player or
1258 // monster specific removals these will need to be moved down to that level and then
1259 // passed in to this function.
1260 std::vector<std::pair<efftype_id, bodypart_str_id>> to_remove;
1261
1262 // Decay/removal of effects
1263 for( auto &elem : *effects ) {
1264 for( auto &_it : elem.second ) {
1265 if( _it.second.is_removed() ) {
1266 to_remove.emplace_back( elem.first, _it.first );
1267 continue;
1268 }
1269 // Add any effects that others remove to the removal list
1270 for( const efftype_id &removed_effect : _it.second.get_removes_effects() ) {
1271 to_remove.emplace_back( removed_effect, bodypart_str_id::NULL_ID() );
1272 }
1273 effect &e = _it.second;
1274 const int prev_int = e.get_intensity();
1275 // Run decay effects, marking effects for removal as necessary.
1276 if( e.decay( calendar::turn, is_player() ) ) {
1277 to_remove.emplace_back( elem.first, _it.first );
1278 }
1279
1280 if( e.get_intensity() != prev_int && e.get_duration() > 0_turns ) {
1281 on_effect_int_change( e.get_id(), e.get_intensity(), e.get_bp() );
1282 }
1283 }
1284 }
1285
1286 // Run the on-remove effects
1287 for( const std::pair<efftype_id, bodypart_str_id> &r : to_remove ) {
1288 remove_effect( r.first, r.second );
1289 }
1290 // Actually remove effects. This should be the last thing done in process_effects().
1291 for( const std::pair<efftype_id, bodypart_str_id> &r : to_remove ) {
1292 if( !r.second ) {
1293 effects->erase( r.first );
1294 } else {
1295 ( *effects )[r.first].erase( r.second );
1296 // If there are no more effects of a given type remove the type map
1297 if( ( *effects )[r.first].empty() ) {
1298 effects->erase( r.first );
1299 }
1300 }
1301 }
1302}
