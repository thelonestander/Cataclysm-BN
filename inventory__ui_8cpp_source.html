355size_t inventory_column::next_selectable_index( size_t index, scroll_direction dir ) const
356{
357 if( entries.empty() ) {
358 return index;
359 }
360 // limit index to the space of the size of entries
361 index = index % entries.size();
362 size_t new_index = index;
363 do {
364 // 'new_index' incremented by 'dir' using division remainder (number of entries) to loop over the entries.
365 // Negative step '-k' (backwards) is equivalent to '-k + N' (forward), where:
366 // N = entries.size() - number of elements,
367 // k = |step| - absolute step (k <= N).
368 new_index = ( new_index + static_cast<int>( dir ) + entries.size() ) % entries.size();
369 } while( new_index != index && !entries[new_index].is_selectable() );
370
371 if( !entries[new_index].is_selectable() ) {
372 return static_cast<size_t>( -1 );
373 }
374
375 return new_index;
376}

480void inventory_column::set_width( const size_t new_width,
481 const std::vector<inventory_column *> &all_columns )
482{
483 reset_width( all_columns );
484 int width_gap = get_width() - new_width;
485 // Now adjust the width if we must
486 while( width_gap != 0 ) {
487 const int step = width_gap > 0 ? -1 : 1;
488 // Should return true when lhs < rhs
489 const auto cmp_for_expansion = []( const cell_t &lhs, const cell_t &rhs ) {
490 return lhs.visible() && lhs.gap() < rhs.gap();
491 };
492 // Should return true when lhs < rhs
493 const auto cmp_for_shrinking = []( const cell_t &lhs, const cell_t &rhs ) {
494 if( !lhs.visible() ) {
495 return false;
496 }
497 if( rhs.gap() <= min_cell_gap ) {
498 return lhs.current_width < rhs.current_width;
499 } else {
500 return lhs.gap() < rhs.gap();
501 }
502 };
503
504 const auto &cell = step > 0
505 ? std::min_element( cells.begin(), cells.end(), cmp_for_expansion )
506 : std::max_element( cells.begin(), cells.end(), cmp_for_shrinking );
507
508 if( cell == cells.end() || !cell->visible() ) {
509 break; // This is highly unlikely to happen, but just in case
510 }
511
512 cell->current_width += step;
513 width_gap += step;
514 }
515 reserved_width = new_width;
516}

531void inventory_column::expand_to_fit( const inventory_entry &entry )
532{
533 if( !entry ) {
534 return;
535 }
536
537 // Don't use cell cache here since the entry may not yet be placed into the vector of entries.
538 const std::string denial = preset.get_denial( entry );
539
540 for( size_t i = 0, num = denial.empty() ? cells.size() : 1; i < num; ++i ) {
541 auto &cell = cells[i];
542
543 cell.real_width = std::max( cell.real_width, get_entry_cell_width( entry, i ) );
544
545 // Don't reveal the cell for headers and stubs
546 if( cell.visible() || ( entry.is_item() && !preset.is_stub_cell( entry, i ) ) ) {
547 const size_t cell_gap = i > 0 ? normal_cell_gap : 0;
548 cell.current_width = std::max( cell.current_width, cell_gap + cell.real_width );
549 }
550 }
551
552 if( !denial.empty() ) {
553 reserved_width = std::max( get_entry_cell_width( entry, 0 ) + min_denial_gap + utf8_width( denial,
554 true ),
555 reserved_width );
556 }

707void inventory_column::add_entry( const inventory_entry &entry )
708{
709 if( std::find( entries.begin(), entries.end(), entry ) != entries.end() ) {
710 debugmsg( "Tried to add a duplicate entry." );
711 return;
712 }
713 const auto iter = std::find_if( entries.rbegin(), entries.rend(),
714 [ &entry ]( const inventory_entry & cur ) {
715 const item_category *cur_cat = cur.get_category_ptr();
716 const item_category *new_cat = entry.get_category_ptr();
717
718 return cur_cat == new_cat || ( cur_cat != nullptr && new_cat != nullptr
719 && ( *cur_cat == *new_cat || *cur_cat < *new_cat ) );
720 } );
721 entries.insert( iter.base(), entry );
722 entries_cell_cache.clear();
723 expand_to_fit( entry );
724 paging_is_valid = false;

740void inventory_column::prepare_paging( const std::string &filter )
741{
742 if( paging_is_valid ) {
743 return;
744 }
745
746 const auto filter_fn = filter_from_string<inventory_entry>(
747 filter, [this]( const std::string & filter ) {
748 return preset.get_filter( filter );
749 } );
750
751 // FIXME: toggled status of multiselect menu resets when filtering the menu
752 // First, remove all non-items
753 const auto new_end = std::remove_if( entries.begin(),
754 entries.end(), [&filter_fn]( const inventory_entry & entry ) {
755 return !entry.is_item() || !filter_fn( entry );
756 } );
757 entries.erase( new_end, entries.end() );
758 // Then sort them with respect to categories
759 auto from = entries.begin();
760 while( from != entries.end() ) {
761 auto to = std::next( from );
762 while( to != entries.end() && from->get_category_ptr() == to->get_category_ptr() ) {
763 to->update_cache();
764 std::advance( to, 1 );
765 }
766 if( ordered_categories.count( from->get_category_ptr()->get_id().c_str() ) == 0 ) {
767 std::sort( from, to, [ this ]( const inventory_entry & lhs, const inventory_entry & rhs ) {
768 if( lhs.is_selectable() != rhs.is_selectable() ) {
769 return lhs.is_selectable(); // Disabled items always go last
770 }
771 return preset.sort_compare( lhs, rhs );
772 } );
773 }
774 from = to;
775 }
776 // Recover categories
777 const item_category *current_category = nullptr;
778 for( auto iter = entries.begin(); iter != entries.end(); ++iter ) {
779 if( iter->get_category_ptr() == current_category ) {
780 continue;
781 }
782 current_category = iter->get_category_ptr();
783 iter = entries.insert( iter, inventory_entry( current_category ) );
784 expand_to_fit( *iter );
785 }
786 // Determine the new height.
787 entries_per_page = height;
788 if( entries.size() > entries_per_page ) {
789 entries_per_page -= 1; // Make room for the page number.
790 for( size_t i = entries_per_page - 1; i < entries.size(); i += entries_per_page ) {
791 auto iter = std::next( entries.begin(), i );
792 if( iter->is_category() ) {
793 // The last item on the page must not be a category.
794 entries.insert( iter, inventory_entry() );
795 } else {
796 // The first item on the next page must be a category.
797 iter = std::next( iter );
798 if( iter != entries.end() && iter->is_item() ) {
799 entries.insert( iter, inventory_entry( iter->get_category_ptr() ) );
800 }
801 }
802 }
803 }
804 entries_cell_cache.clear();
805 paging_is_valid = true;
806 if( entries_unfiltered.empty() ) {
807 entries_unfiltered = entries;
808 }
809 // Select the uppermost possible entry
810 select( selected_index, selected_index ? scroll_direction::BACKWARD : scroll_direction::FORWARD );
811}

859void inventory_column::draw( const catacurses::window &win, point pos ) const
860{
861 if( !visible() ) {
862 return;
863 }
864
865 const auto available_cell_width = [ this ]( size_t index, size_t cell_index ) {
866 const size_t displayed_width = cells[cell_index].current_width;
867 const size_t real_width = get_entry_cell_width( index, cell_index );
868
869 return displayed_width > real_width ? displayed_width - real_width : 0;
870 };
871
872 // Do the actual drawing
873 for( size_t index = page_offset, line = 0; index < entries.size() &&
874 line < entries_per_page; ++index, ++line ) {
875 const auto &entry = entries[index];
876 const auto &entry_cell_cache = get_entry_cell_cache( index );
877
878 if( !entry ) {
879 continue;
880 }
881
882 int x1 = pos.x + get_entry_indent( entry );
883 int x2 = pos.x + std::max( static_cast<int>( reserved_width - get_cells_width() ), 0 );
884 int yy = pos.y + line;
885
886 const bool selected = active && is_selected( entry );
887
888 if( selected && visible_cells() > 1 ) {
889 for( int hx = x1, hx_max = pos.x + get_width(); hx < hx_max; ++hx ) {
890 mvwputch( win, point( hx, yy ), h_white, ' ' );
891 }
892 }
893
894 const std::string &denial = entry_cell_cache.denial;
895
896 if( !denial.empty() ) {
897 const size_t max_denial_width = std::max( static_cast<int>( get_width() - ( min_denial_gap +
898 get_entry_cell_width( index, 0 ) ) ), 0 );
899 const size_t denial_width = std::min( max_denial_width, static_cast<size_t>( utf8_width( denial,
900 true ) ) );
901
902 trim_and_print( win, point( pos.x + get_width() - denial_width, yy ), denial_width, c_red, denial );
903 }
904
905 const size_t count = denial.empty() ? cells.size() : 1;
906
907 for( size_t cell_index = 0; cell_index < count; ++cell_index ) {
908 if( !cells[cell_index].visible() ) {
909 continue; // Don't show empty cells
910 }
911
912 if( line != 0 && cell_index != 0 && entry.is_category() ) {
913 break; // Don't show duplicated titles
914 }
915
916 x2 += cells[cell_index].current_width;
917
918 size_t text_width = utf8_width( entry_cell_cache.text[cell_index], true );
919 size_t text_gap = cell_index > 0 ? std::max( cells[cell_index].gap(), min_cell_gap ) : 0;
920 size_t available_width = x2 - x1 - text_gap;
921
922 if( text_width > available_width ) {
923 // See if we can steal some of the needed width from an adjacent cell
924 if( cell_index == 0 && count >= 2 ) {
925 available_width += available_cell_width( index, 1 );
926 } else if( cell_index > 0 ) {
927 available_width += available_cell_width( index, cell_index - 1 );
928 }
929 text_width = std::min( text_width, available_width );
930 }
931
932 if( text_width > 0 ) {
933 const int text_x = cell_index == 0 ? x1 : x2 -
934 text_width; // Align either to the left or to the right
935 const std::string &text = entry_cell_cache.text[cell_index];
936
937 if( entry.is_item() && ( selected || !entry.is_selectable() ) ) {
938 trim_and_print( win, point( text_x, yy ), text_width, selected ? h_white : c_dark_gray,
939 remove_color_tags( text ) );
940 } else {
941 trim_and_print( win, point( text_x, yy ), text_width, entry_cell_cache.color, text );
942 }
943 }
944
945 x1 = x2;
946 }
947
948 if( entry.is_item() ) {
949 int xx = pos.x;
950 if( entry.get_invlet() != '\0' ) {
951 mvwputch( win, point( pos.x, yy ), entry.get_invlet_color(), entry.get_invlet() );
952 }
953 xx += 2;
954 if( get_option<bool>( "ITEM_SYMBOLS" ) ) {
955 const nc_color color = entry.any_item()->color();
956 mvwputch( win, point( xx, yy ), color, entry.any_item()->symbol() );
957 xx += 2;
958 }
959 if( allows_selecting() && activatable() && multiselect ) {
960 if( entry.chosen_count == 0 ) {
961 mvwputch( win, point( xx, yy ), c_dark_gray, '-' );
962 } else if( entry.chosen_count >= entry.get_available_count() ) {
963 mvwputch( win, point( xx, yy ), c_light_green, '+' );
964 } else {
965 mvwputch( win, point( xx, yy ), c_light_green, '#' );
966 }
967 }
968 }
969 }
970
971 if( pages_count() > 1 ) {
972 mvwprintw( win, pos + point( 0, height - 1 ), _( "Page %d/%d" ), page_index() + 1, pages_count() );
973 }
974}

1483std::vector<std::string> inventory_selector::get_stats() const
1484{
1485 // Stats consist of arrays of cells.
1486 const size_t num_stats = 2;
1487 const std::array<stat, num_stats> stats = get_raw_stats();
1488 // Streams for every stat.
1489 std::array<std::string, num_stats> lines;
1490 std::array<size_t, num_stats> widths;
1491 // Add first cells and spaces after them.
1492 for( size_t i = 0; i < stats.size(); ++i ) {
1493 lines[i] += string_format( "%s", stats[i][0] ) + " ";
1494 }
1495 // Now add the rest of the cells and align them to the right.
1496 for( size_t j = 1; j < stats.front().size(); ++j ) {
1497 // Calculate actual cell width for each stat.
1498 std::transform( stats.begin(), stats.end(), widths.begin(),
1499 [j]( const stat & elem ) {
1500 return utf8_width( elem[j], true );
1501 } );
1502 // Determine the max width.
1503 const size_t max_w = *std::max_element( widths.begin(), widths.end() );
1504 // Align all stats in this cell with spaces.
1505 for( size_t i = 0; i < stats.size(); ++i ) {
1506 if( max_w > widths[i] ) {
1507 lines[i] += std::string( max_w - widths[i], ' ' );
1508 }
1509 lines[i] += string_format( "%s", stats[i][j] );
1510 }
1511 }
1512 // Construct the final result.
1513 return std::vector<std::string>( lines.begin(), lines.end() );
1514}
