486cata::optional<zone_type_id> zone_manager::query_type() const
487{
488 const auto &types = get_manager().get_types();
489 std::vector<std::pair<zone_type_id, zone_type>> types_vec;
490 std::copy( types.begin(), types.end(),
491 std::back_inserter<std::vector<std::pair<zone_type_id, zone_type>>>( types_vec ) );
492 std::sort( types_vec.begin(), types_vec.end(),
493 []( const std::pair<zone_type_id, zone_type> &lhs, const std::pair<zone_type_id, zone_type> &rhs ) {
494 return localized_compare( lhs.second.name(), rhs.second.name() );
495 } );
496
497 uilist as_m;
498 as_m.desc_enabled = true;
499 as_m.text = _( "Select zone type:" );
500
501 size_t i = 0;
502 for( const auto &pair : types_vec ) {
503 const auto &type = pair.second;
504
505 as_m.addentry_desc( i++, true, MENU_AUTOASSIGN, type.name(), type.desc() );
506 }
507
508 as_m.query();
509 if( as_m.ret < 0 ) {
510 return {};
511 }
512 size_t index = as_m.ret;
513
514 auto iter = types_vec.begin();
515 std::advance( iter, index );
516
517 return iter->first;
518}

1049void zone_manager::rotate_zones( map &target_map, const int turns )
1050{
1051 if( turns == 0 ) {
1052 return;
1053 }
1054 const tripoint a_start = target_map.getabs( tripoint_zero );
1055 const tripoint a_end = target_map.getabs( tripoint( 23, 23, 0 ) );
1056 const point dim( 24, 24 );
1057 for( zone_data &zone : zones ) {
1058 const tripoint z_start = zone.get_start_point();
1059 const tripoint z_end = zone.get_end_point();
1060 if( ( a_start.x <= z_start.x && a_start.y <= z_start.y ) &&
1061 ( a_end.x > z_start.x && a_end.y >= z_start.y ) &&
1062 ( a_start.x <= z_end.x && a_start.y <= z_end.y ) &&
1063 ( a_end.x >= z_end.x && a_end.y >= z_end.y ) &&
1064 ( a_start.z == z_start.z )
1065 ) {
1066 tripoint z_l_start3 = target_map.getlocal( z_start );
1067 tripoint z_l_end3 = target_map.getlocal( z_end );
1068 // don't rotate centered squares
1069 if( z_l_start3.x == z_l_start3.y && z_l_end3.x == z_l_end3.y && z_l_start3.x + z_l_end3.x == 23 ) {
1070 continue;
1071 }
1072 point z_l_start = z_l_start3.xy().rotate( turns, dim );
1073 point z_l_end = z_l_end3.xy().rotate( turns, dim );
1074 point new_z_start = target_map.getabs( z_l_start );
1075 point new_z_end = target_map.getabs( z_l_end );
1076 tripoint first = tripoint( std::min( new_z_start.x, new_z_end.x ),
1077 std::min( new_z_start.y, new_z_end.y ), a_start.z );
1078 tripoint second = tripoint( std::max( new_z_start.x, new_z_end.x ),
1079 std::max( new_z_start.y, new_z_end.y ), a_end.z );
1080 zone.set_position( std::make_pair( first, second ), false );
1081 }
1082 }
