22/**
23 * This is a wrapper for implementing the pointer-to-implementation technique,
24 * see for example http://en.cppreference.com/w/cpp/language/pimpl
25 *
26 * It's a container for one element of type @p T, which is *always* present
27 * (one can not create a @ref pimpl instance without a contained @ref T instance).
28 *
29 * The contained @ref T can be access like a pointer via @ref operator-> and @ref operator.
30 *
31 * The constructor forwards all its arguments to the constructor of @ref T.
32 */
33template <typename T>
34class pimpl : private std::unique_ptr<T>
35{
36 public:
37 // Original second template arguments was `std::enable_if<std::is_constructible<T, Args>::value>`,
38 // but this caused errors in few compilers when `T` was not a complete type.
39 // The new argument serves the same purpose: this constructor should *not* be available when the
40 // argument is a `pimpl` itself (the other copy constructors should be used instead).
41 explicit pimpl() : std::unique_ptr<T>( new T() ) { }
42 template < typename P, typename ...Args,
43 typename = typename std::enable_if < !is_pimpl<P>::value >::type >
44 explicit pimpl( P && head, Args &&
