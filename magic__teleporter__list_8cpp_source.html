61static cata::optional<tripoint> find_valid_teleporters_omt( const tripoint_abs_omt &omt_pt )
62{
63 // this is the top left hand square of the global absolute coordinate
64 // of the overmap terrain we want to try to teleport to.
65 // an OMT is SEEX * SEEY in size
66 const tripoint_abs_sm sm_pt = project_to<coords::sm>( omt_pt );
67 tinymap checker;
68 // TODO: fix point types
69 checker.load( sm_pt.raw(), true );
70 for( const tripoint &p : checker.points_on_zlevel() ) {
71 if( checker.has_flag_furn( "TRANSLOCATOR", p ) ) {
72 return checker.getabs( p );
73 }
74 }
75 return cata::nullopt;
76}

95void teleporter_list::translocate( const std::set<tripoint> &targets )
96{
97 if( known_teleporters.empty() ) {
98 // we can't go somewhere if we don't know how to get there!
99 add_msg( m_bad, _( "No translocator target known." ) );
100 return;
101 }
102 cata::optional<tripoint_abs_omt> omt_dest = choose_teleport_location();
103 if( !omt_dest ) {
104 add_msg( _( "Teleport canceled." ) );
105 return;
106 }
107
108 bool valid_targets = false;
109 for( const tripoint &pt : targets ) {
110 Character *you = g->critter_at<Character>( pt );
111
112 if( you && you->is_avatar() ) {
113 valid_targets = true;
114 if( !place_avatar_overmap( *you, *omt_dest ) ) {
115 add_msg( _( "Failed to teleport. Teleporter obstructed or destroyed." ) );
116 deactivate_teleporter( *omt_dest, pt );
117 }
118 }
119 }
120
121 if( !valid_targets ) {
122 add_msg( _( "No valid targets to teleport." ) );
123 }
124}

202 teleport_selector.addentry( index, true, 0, gate.second );

204 index_pairs.emplace( index, gate.first );

207 teleporter_callback cb( index_pairs );

209 teleport_selector.w_width_setup = 38 + column_width;
210 teleport_selector.pad_right_setup = 33;
211 teleport_selector.title = _( "Choose Translocator Gate" );
212
213 teleport_selector.query();
214
215 if( teleport_selector.ret >= 0 ) {
