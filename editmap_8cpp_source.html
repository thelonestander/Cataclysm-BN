66static std::vector<std::string> fld_string( const std::string &str, int width )
67{
68 std::vector<std::string> lines;
69 if( width < 1 ) {
70 lines.push_back( str );
71 return lines;
72 }
73
74 int linepos = width;
75 int linestart = 0;
76 int crpos = -2;
77 while( linepos < static_cast<int>( str.length() ) || crpos != -1 ) {
78 crpos = str.find( '\n', linestart );
79 if( crpos != -1 && crpos <= linepos ) {
80 lines.push_back( str.substr( linestart, crpos - linestart ) );
81 linepos = crpos + width + 1;
82 linestart = crpos + 1;
83 } else {
84 int spacepos = str.rfind( ',', linepos );
85 if( spacepos == -1 ) {
86 spacepos = str.find( ',', linepos );
87 }
88 if( spacepos < linestart ) {
89 spacepos = linestart + width;
90 if( spacepos < static_cast<int>( str.length() ) ) {
91 lines.push_back( str.substr( linestart, width ) );
92 linepos = spacepos + width;
93 linestart = spacepos;
94 }
95 } else {
96 lines.push_back( str.substr( linestart, spacepos - linestart ) );
97 linepos = spacepos + width;
98 linestart = spacepos;
99 }
100 }
101 }
102 lines.push_back( str.substr( linestart ) );
103 return lines;
104}

838static ter_id get_alt_ter( bool isvert, ter_id sel_ter )
839{
840 std::map<std::string, std::string> alts;
841 alts["_v"] = "_h";
842 alts["_vertical"] = "_horizontal";
843 alts["_v_alarm"] = "_h_alarm";
844 const std::string tersid = sel_ter.obj().id.str();
845 const int sidlen = tersid.size();
846 for( std::map<std::string, std::string>::const_iterator it = alts.begin(); it != alts.end();
847 ++it ) {
848 const std::string suffix = isvert ? it->first : it->second;
849 const int slen = suffix.size();
850 if( sidlen > slen && tersid.substr( sidlen - slen, slen ) == suffix ) {
851 const std::string asuffix = isvert ? it->second : it->first;
852 const std::string terasid = tersid.substr( 0, sidlen - slen ) + asuffix;
853 const ter_str_id tid( terasid );
854
855 if( tid.is_valid() ) {
856 return tid.id();
857 }
858 }
859 }
860 return undefined_ter_id;
861}

999template<>
1000void apply<ter_t>( const ter_t &t, const shapetype editshape, const tripoint &target,
1001 const tripoint &origin, const std::vector<tripoint> &target_list )
1002{
1003 bool isvert = false;
1004 bool ishori = false;
1005 bool doalt = false;
1006 ter_id teralt = undefined_ter_id;
1007 int alta = -1;
1008 int altb = -1;
1009 const ter_id sel_ter = t.id.id();
1010 if( editshape == editmap_rect ) {
1011 if( t.symbol() == LINE_XOXO || t.symbol() == '|' ) {
1012 isvert = true;
1013 teralt = get_alt_ter( isvert, sel_ter );
1014 } else if( t.symbol() == LINE_OXOX || t.symbol() == '-' ) {
1015 ishori = true;
1016 teralt = get_alt_ter( isvert, sel_ter );
1017 }
1018 if( teralt != undefined_ter_id ) {
1019 if( isvert ) {
1020 alta = target.y;
1021 altb = origin.y;
1022 } else {
1023 alta = target.x;
1024 altb = origin.x;
1025 }
1026 doalt = true;
1027 }
1028 }
1029
1030 for( auto &elem : target_list ) {
1031 ter_id wter = sel_ter;
1032 if( doalt ) {
1033 if( isvert && ( elem.y == alta || elem.y == altb ) ) {
1034 wter = teralt;
1035 } else if( ishori && ( elem.x == alta || elem.x == altb ) ) {
1036 wter = teralt;
1037 }
1038 }
1039 g->m.ter_set( elem, wter );
1040 }

1519 * Calculate target_list based on origin and target class variables, and shapetype.
1520 */
1521void editmap::recalc_target( shapetype shape )
1522{
1523 const int z = target.z;
1524 target_list.clear();
1525 switch( shape ) {
1526 case editmap_circle: {
1527 int radius = rl_dist( origin, target );
1528 for( const tripoint &p : g->m.points_in_radius( origin, radius ) ) {
1529 if( rl_dist( p, origin ) <= radius ) {
1530 if( editmap_boundaries.contains( p ) ) {
1531 target_list.push_back( p );
1532 }
1533 }
1534 }
1535 }
1536 break;
1537 case editmap_rect_filled:
1538 case editmap_rect:
1539 int sx;
1540 int sy;
1541 int ex;
1542 int ey;
1543 if( target.x < origin.x ) {
1544 sx = target.x;
1545 ex = origin.x;
1546 } else {
1547 sx = origin.x;
1548 ex = target.x;
1549 }
1550 if( target.y < origin.y ) {
1551 sy = target.y;
1552 ey = origin.y;
1553 } else {
1554 sy = origin.y;
1555 ey = target.y;
1556 }
1557 for( int x = sx; x <= ex; x++ ) {
1558 for( int y = sy; y <= ey; y++ ) {
1559 if( shape == editmap_rect_filled || x == sx || x == ex || y == sy || y == ey ) {
1560 const tripoint p( x, y, z );
1561 if( editmap_boundaries.contains( p ) ) {
1562 target_list.push_back( p );
1563 }
1564 }
1565 }
1566 }
1567 break;
1568 case editmap_line:

1574/*
1575 * Shift 'var' (i.e., part of a coordinate plane) by 'shift'.
1576 * If the result is not >= min and < 'max', constrain the result and adjust 'shift',
1577 * so it can adjust subsequent points of a set consistently.
1578 */
1579static int limited_shift( int var, int &shift, int min, int max )
1580{
1581 if( var + shift < min ) {
1582 shift = min - var;
1583 } else if( var + shift >= max ) {
1584 shift = shift + ( max - 1 - ( var + shift ) );
1585 }
1586 return var += shift;
1587}

1589/*
1590 * Move point 'editmap.target' via keystroke. 'moveorigin' determines if point 'editmap.origin' is moved as well:
1591 * 0: no, 1: yes, -1 (or none): as per bool 'editmap.moveall'.
1592 * if input or ch are not valid movement keys, do nothing and return false
1593 */
1594bool editmap::move_target( const std::string &action, int moveorigin )
1595{
1596 tripoint mp;
1597 bool move_origin = moveorigin == 1 ? true :
1598 moveorigin == 0 ? false : moveall;
1599 if( eget_direction( mp, action ) ) {
1600 target.x = limited_shift( target.x, mp.x, 0, MAPSIZE_X );
1601 target.y = limited_shift( target.y, mp.y, 0, MAPSIZE_Y );
1602 target.z = limited_shift( target.z, mp.z, -OVERMAP_DEPTH, OVERMAP_HEIGHT + 1 );
1603 if( move_origin ) {
1604 origin += mp;
1605 }
1606 return true;
1607 }
1608 return false;
1609}
