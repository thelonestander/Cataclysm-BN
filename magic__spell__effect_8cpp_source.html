60struct line_iterable {
61 const std::vector<point> &delta_line;
62 point cur_origin;
63 point delta;
64 size_t index;
65
66 line_iterable( const point &origin, const point &delta, const std::vector<point> &dline )
67 : delta_line( dline ), cur_origin( origin ), delta( delta ), index( 0 ) {}
68
69 point get() const {
70 return cur_origin + delta_line[index];
71 }
72 // Move forward along point set, wrap around and move origin forward if necessary
73 void next() {
74 index = ( index + 1 ) % delta_line.size();
75 cur_origin = cur_origin + delta * ( index == 0 );
76 }
77 // Move back along point set, wrap around and move origin backward if necessary
78 void prev() {
79 cur_origin = cur_origin - delta * ( index == 0 );
80 index = ( index + delta_line.size() - 1 ) % delta_line.size();
81 }
82 void reset( const point &origin ) {
83 cur_origin = origin;
84 index = 0;
85 }
86};

227std::set<tripoint> spell_effect::spell_effect_line( const spell &, const tripoint &source,
228 const tripoint &target, const int aoe_radius, const bool ignore_walls )
229{
230 const point delta = ( target - source ).xy();
231 const int dist = square_dist( point_zero, delta );
232 // Early out to prevent unnecessary calculations
233 if( dist == 0 ) {
234 return std::set<tripoint>();
235 }
236 // Clockwise Perpendicular of Delta vector
237 const point delta_perp( -delta.y, delta.x );
238
239 const point abs_delta = delta.abs();
240 // Primary axis of delta vector
241 const point axis_delta = abs_delta.x > abs_delta.y ? point( delta.x, 0 ) : point( 0, delta.y );
242 // Clockwise Perpendicular of axis vector
243 const point cw_perp_axis( -axis_delta.y, axis_delta.x );
244 const point unit_cw_perp_axis( sgn( cw_perp_axis.x ), sgn( cw_perp_axis.y ) );
245 // bias leg length toward cw side if uneven
246 int ccw_len = aoe_radius / 2;
247 int cw_len = aoe_radius - ccw_len;
248
249 if( !trigdist ) {
250 ccw_len = ( ccw_len * ( abs_delta.x + abs_delta.y ) ) / dist;
251 cw_len = ( cw_len * ( abs_delta.x + abs_delta.y ) ) / dist;
252 }
253
254 // is delta aligned with, cw, or ccw of primary axis
255 int delta_side = spell_detail::side_of( point_zero, axis_delta, delta );
256
257 bool ( *test )( const tripoint & ) = ignore_walls ? test_always_true : test_passable;
258
259 // Canonical path from source to target, offset to local space
260 std::vector<point> path_to_target = line_to( point_zero, delta );
261 // Remove endpoint,
262 path_to_target.pop_back();
263 // and insert startpoint. Path is now prepared for wrapped iteration
264 path_to_target.insert( path_to_target.begin(), point_zero );
265
266 spell_detail::line_iterable base_line( point_zero, delta, path_to_target );
267
268 std::set<tripoint> result;
269
270 // Add midline points (source -> target )
271 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
272
273 // Add cw and ccw legs
274 if( delta_side == 0 ) { // delta is already axis aligned, only need straight lines
275 // cw leg
276 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
277 base_line.reset( p );
278 if( !test( source + p ) ) {
279 break;
280 }
281
282 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
283 }
284 // ccw leg
285 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
286 base_line.reset( p );
287 if( !test( source + p ) ) {
288 break;
289 }
290
291 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
292 }
293 } else if( delta_side == 1 ) { // delta is cw of primary axis
294 // ccw leg is behind perp axis
295 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
296 base_line.reset( p );
297
298 // forward until in
299 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) == 1 ) {
300 base_line.next();
301 }
302 if( !test( source + p ) ) {
303 break;
304 }
305 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
306 }
307 // cw leg is before perp axis
308 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
309 base_line.reset( p );
310
311 // move back
312 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) != 1 ) {
313 base_line.prev();
314 }
315 base_line.next();
316 if( !test( source + p ) ) {
317 break;
318 }
319 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
320 }
321 } else if( delta_side == -1 ) { // delta is ccw of primary axis
322 // ccw leg is before perp axis
323 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
324 base_line.reset( p );
325
326 // move back
327 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) != 1 ) {
328 base_line.prev();
329 }
330 base_line.next();
331 if( !test( source + p ) ) {
332 break;
333 }
334 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
335 }
336 // cw leg is behind perp axis
337 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
338 base_line.reset( p );
339
340 // forward until in
341 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) == 1 ) {
342 base_line.next();
343 }
344 if( !test( source + p ) ) {
345 break;
346 }
347 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
348 }
349 }
350
351 result.erase( source );
352 return result;
353}

521// Adds node to a search tree. Returns true if new node is allocated.
522bool area_expander::enqueue( const tripoint &from, const tripoint &to, float cost )
523{
524 if( contains( to ) ) {
525 // We will modify existing node if its cost is lower.
526 int index = area_search[to];
527 node &node = area[index];
528 if( cost < node.cost ) {
529 node.from = from;
530 node.cost = cost;
531 }
532 return false;

535 area.push_back( {to, from, cost} );
536 frontier.push( index );
537 area_search[to] = index;
538 return true;

541// Run wave propagation
542int area_expander::run( const tripoint &center )
543{
544 enqueue( center, center, 0.0 );
545
546 static constexpr std::array<int, 8> x_offset = {{ -1, 1, 0, 0, 1, -1, -1, 1 }};
547 static constexpr std::array<int, 8> y_offset = {{ 0, 0, -1, 1, -1, 1, -1, 1 }};
548
549 // Number of nodes expanded.
550 int expanded = 0;
551
552 while( !frontier.empty() ) {
553 int best_index = frontier.top();
554 frontier.pop();
555 node &best = area[best_index];
556
557 for( size_t i = 0; i < 8; i++ ) {
558 tripoint pt = best.position + point( x_offset[ i ], y_offset[ i ] );
559
560 if( g->m.impassable( pt ) && !g->m.has_flag( "THIN_OBSTACLE", pt ) ) {
561 continue;
562 }
563
564 float center_range = static_cast<float>( rl_dist( center, pt ) );
565 if( max_range > 0 && center_range > max_range ) {
566 continue;
567 }
568
569 if( max_expand > 0 && expanded > max_expand && contains( pt ) ) {
570 continue;
571 }
572
573 float delta_range = trig_dist( best.position, pt );
574
575 if( enqueue( best.position, pt, best.cost + delta_range ) ) {
576 expanded++;
577 }
578 }
579 }
580 return expanded;
581}

832 } else {
833 add_msg( m_bad, "failed to place monster" );
834 }
835 // whether or not we succeed in spawning a monster, we don't want to try this tripoint again
