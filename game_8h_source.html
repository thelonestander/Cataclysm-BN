362 template<typename T>
363 class non_dead_range
364 {
365 public:
366 std::vector<weak_ptr_fast<T>> items;
367
368 class iterator
369 {
370 private:
371 bool valid();
372 public:
373 std::vector<weak_ptr_fast<T>> &items;
374 typename std::vector<weak_ptr_fast<T>>::iterator iter;
375 shared_ptr_fast<T> current;
376
377 iterator( std::vector<weak_ptr_fast<T>> &i,
378 const typename std::vector<weak_ptr_fast<T>>::iterator t ) : items( i ), iter( t ) {
379 while( iter != items.end() && !valid() ) {
380 ++iter;
381 }
382 }
383 iterator( const iterator & ) = default;
384 iterator &operator=( const iterator & ) = default;
385
386 bool operator==( const iterator &rhs ) const {
387 return iter == rhs.iter;
388 }
389 bool operator!=( const iterator &rhs ) const {
390 return !operator==( rhs );
391 }
392 iterator &operator++() {
393 do {
394 ++iter;
395 } while( iter != items.end() && !valid() );
396 return *this;
397 }
398 T &operator*() const {
399 return *current;
400 }
401 };
402 iterator begin() {
403 return iterator( items, items.begin() );
404 }
405 iterator end() {
406 return iterator( items, items.end() );
407 }
