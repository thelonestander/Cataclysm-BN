153static item_location inv_internal( player &u, const inventory_selector_preset &preset,
154 const std::string &title, int radius,
155 const std::string &none_message,
156 const std::string &hint = std::string() )
157{
158 inventory_pick_selector inv_s( u, preset );
159
160 inv_s.set_title( title );
161 inv_s.set_hint( hint );
162 inv_s.set_display_stats( false );
163
164 std::pair<size_t, size_t> init_pair;
165 bool init_selection = false;
166 std::string init_filter;
167 bool has_init_filter = false;
168
169 const std::vector<activity_id> consuming {
170 ACT_EAT_MENU,
171 ACT_CONSUME_FOOD_MENU,
172 ACT_CONSUME_DRINK_MENU,
173 ACT_CONSUME_MEDS_MENU };
174
175 if( u.has_activity( consuming ) && u.activity.values.size() >= 2 ) {
176 init_pair.first = u.activity.values[0];
177 init_pair.second = u.activity.values[1];
178 init_selection = true;
179 }
180 if( u.has_activity( consuming ) && !u.activity.str_values.empty() ) {
181 init_filter = u.activity.str_values[0];
182 has_init_filter = true;
183 }
184
185 do {
186 u.inv.restack( u );
187
188 inv_s.clear_items();
189 inv_s.add_character_items( u );
190 inv_s.add_nearby_items( radius );
191
192 if( has_init_filter ) {
193 inv_s.set_filter( init_filter );
194 has_init_filter = false;
195 }
196 // Set position after filter to keep cursor at the right position
197 if( init_selection ) {
198 inv_s.select_position( init_pair );
199 init_selection = false;
200 }
201
202 if( inv_s.empty() ) {
203 const std::string msg = none_message.empty()
204 ? _( "You don't have the necessary item at hand." )
205 : none_message;
206 popup( msg, PF_GET_KEY );
207 return item_location();
208 }
209
210 item_location location = inv_s.execute();
211
212 if( inv_s.keep_open ) {
213 inv_s.keep_open = false;
214 continue;
215 }
216
217 if( u.has_activity( consuming ) ) {
218 u.activity.values.clear();
219 init_pair = inv_s.get_selection_position();
220 u.activity.values.push_back( init_pair.first );
221 u.activity.values.push_back( init_pair.second );
222 u.activity.str_values.clear();
223 u.activity.str_values.emplace_back( inv_s.get_filter() );
224 }
225
226 return location;
227
228 } while( true );
229}

632 bool sort_compare( const inventory_entry &lhs, const inventory_entry &rhs ) const override {
633 time_duration time_a = get_time_left( lhs.any_item() );
634 time_duration time_b = get_time_left( rhs.any_item() );
635 int order_a = get_order( lhs.any_item(), time_a );
636 int order_b = get_order( rhs.any_item(), time_b );
637
638 return order_a < order_b
639 || ( order_a == order_b && time_a < time_b )
640 || ( order_a == order_b && time_a == time_b &&
641 inventory_selector_preset::sort_compare( lhs, rhs ) );
642 }
