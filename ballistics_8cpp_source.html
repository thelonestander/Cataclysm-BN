60static void drop_or_embed_projectile( const dealt_projectile_attack &attack )
61{
62 const auto &proj = attack.proj;
63 const item &drop_item = proj.get_drop();
64 if( drop_item.is_null() ) {
65 return;
66 }
67
68 const tripoint &pt = attack.end_point;

70 if( proj.has_effect( ammo_effect_SHATTER_SELF ) ) {
71 // Drop the contents, not the thrown item
72 if( g->u.sees( pt ) ) {
73 add_msg( _( "The %s shatters!" ), drop_item.tname() );
74 }
75
76 // copies the drop item to spill the contents
77 item( drop_item ).spill_contents( pt );
78
79 // TODO: Non-glass breaking

81 sounds::sound( pt, 16, sounds::sound_t::combat, _( "glass breaking!" ), false, "bullet_hit",

85
86 if( proj.has_effect( ammo_effect_BURST ) ) {
87 // Drop the contents, not the thrown item
88 if( g->u.sees( pt ) ) {
89 add_msg( _( "The %s bursts!" ), drop_item.tname() );
90 }
91
92 // copies the drop item to spill the contents
93 item( drop_item ).spill_contents( pt );
94
95 // TODO: Sound
96 return;
97 }
98
99 // Copy the item
100 item dropped_item = drop_item;
101
102 monster *mon = dynamic_cast<monster *>( attack.hit_critter );
103
104 // We can only embed in monsters
105 bool mon_there = mon != nullptr && !mon->is_dead_state();
106 // And if we actually want to embed
107 bool embed = mon_there && !proj.has_effect( ammo_effect_NO_EMBED ) &&
108 !proj.has_effect( ammo_effect_TANGLE );
109 // Don't embed in small creatures
110 if( embed ) {
111 const m_size critter_size = mon->get_size();
112 const units::volume vol = dropped_item.volume();
113 embed = embed && ( critter_size > MS_TINY || vol < 250_ml );
114 embed = embed && ( critter_size > MS_SMALL || vol < 500_ml );
115 // And if we deal enough damage
116 // Item volume bumps up the required damage too
117 embed = embed &&
118 ( attack.dealt_dam.type_damage( DT_CUT ) / 2 ) +
119 attack.dealt_dam.type_damage( DT_STAB ) >
120 attack.dealt_dam.type_damage( DT_BASH ) +
121 vol * 3 / 250_ml + rng( 0, 5 );
122 }
123
124 if( embed ) {
125 mon->add_item( dropped_item );
126 if( g->u.sees( *mon ) ) {
127 add_msg( _( "The %1$s embeds in %2$s!" ), dropped_item.tname(), mon->disp_name() );
128 }
129 } else {
130 bool do_drop = true;
131 // monsters that are able to be tied up will store the item another way
132 // see monexamine.cpp tie_or_untie()
133 // if they aren't friendly they will try and break out of the net/bolas/lasso
134 // players and NPCs just get the downed effect, and item is dropped.
135 // TODO: storing the item on player until they recover from downed
136 if( proj.has_effect( ammo_effect_TANGLE ) && mon_there ) {
137 do_drop = false;
138 }
139 if( proj.has_effect( ammo_effect_ACT_ON_RANGED_HIT ) ) {
140 // Don't drop if it exploded
141 do_drop = !dropped_item.process( nullptr, attack.end_point, true );
142 }
143
144 if( do_drop ) {
145 g->m.add_item_or_charges( attack.end_point, dropped_item );
146 }
147
148 if( proj.has_effect( ammo_effect_HEAVY_HIT ) ) {
149 if( g->m.has_flag( flag_LIQUID, pt ) ) {
150 sounds::sound( pt, 10, sounds::sound_t::combat, _( "splash!" ), false, "bullet_hit", "hit_water" );
151 } else {
152 sounds::sound( pt, 8, sounds::sound_t::combat, _( "thud." ), false, "bullet_hit", "hit_wall" );
153 }
154 const trap &tr = g->m.tr_at( pt );
155 if( tr.triggered_by_item( dropped_item ) ) {
156 tr.trigger( pt, nullptr, &dropped_item );
157 }
158 }

174projectile_attack_aim projectile_attack_roll( const dispersion_sources &dispersion, double range,
175 double target_size )
176{
177 projectile_attack_aim aim;
178
179 // dispersion is a measure of the dispersion of shots due to the gun + shooter characteristics
180 // i.e. it is independent of any particular shot
181
182 // shot_dispersion is the actual dispersion for this particular shot, i.e.
183 // the error angle between where the shot was aimed and where this one actually went
184 // NB: some cases pass dispersion == 0 for a "never misses" shot e.g. bio_magnet,
185 aim.dispersion = dispersion.roll();
186
187 // an isosceles triangle is formed by the intended and actual target tiles
188 aim.missed_by_tiles = iso_tangent( range, units::from_arcmin( aim.dispersion ) );
189
190 // fraction we missed a monster target by (0.0 = perfect hit, 1.0 = miss)
191 if( target_size > 0.0 ) {
192 aim.missed_by = std::min( 1.0, aim.missed_by_tiles / target_size );
193 } else {
194 // Special case 0 size targets, just to be safe from 0.0/0.0 NaNs
195 aim.missed_by = 1.0;
196 }
197
198 return aim;
199}

202 const tripoint &target_arg, const dispersion_sources &dispersion,

234 const bool do_draw_line = proj.has_effect( ammo_effect_DRAW_AS_LINE );

243 ( g->critter_at( target_arg ) == nullptr && g->m.passable( target_arg ) );

258 rad += ( one_in( 2 ) ? 1 : -1 ) * dispersion_angle;

263 range + rng( -offset, offset ) :

266

268 target.y = source.y + roll_remainder( new_range * sin( rad ) );
269

271 target.x = source.x + sgn( dx );
272 target.y = source.y + sgn( dy );
273 }
274
275 // Don't extend range further, miss here can mean hitting the ground near the target
276 range = rl_dist( source, target );
277 extend_to_range = range;
278
279 sfx::play_variant_sound( "bullet_hit", "hit_wall", sfx::get_heard_volume( target ),
280 sfx::get_heard_angle( target ) );
281 // TODO: Z dispersion
282 // If we missed, just draw a straight line.
283 trajectory = line_to( source, target );
284 } else {

451 }
452

454
455 apply_ammo_effects( tp, proj.get_ammo_effects() );
456 const auto &expl = proj.get_custom_explosion();
457 if( expl ) {
458 explosion_handler::explosion( tp, expl );
459 }
460
461 // TODO: Move this outside now that we have hit point in return values?
462 if( proj.has_effect( ammo_effect_BOUNCE ) ) {
463 // Add effect so the shooter is not targeted itself.
464 if( origin && !origin->has_effect( effect_bounced ) ) {
465 origin->add_effect( effect_bounced, 1_turns );
466 }

468 // search for creatures in radius 4 around impact site

494double hit_chance( const dispersion_sources &dispersion, double range, double target_size,
495 double missed_by )
496{
497 if( range <= 0 ) {
498 return 1.0;
499 }
500
501 double missed_by_tiles = missed_by * target_size;
502
503 // T = (2*D**2 * (1 - cos V)) ** 0.5 (from iso_tangent)
504 // cos V = 1 - T**2 / (2*D**2)
505 double cosV = 1 - missed_by_tiles * missed_by_tiles / ( 2 * range * range );
506 double needed_dispersion = ( cosV < -1.0 ? M_PI : acos( cosV ) ) * 180 * 60 / M_PI;
507
508 return normal_cdf( needed_dispersion, dispersion.avg(), dispersion.avg() / 2 );
509}
