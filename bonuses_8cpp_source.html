113void bonus_container::load( const JsonArray &jarr, const bool mult )
114{
115 for( const JsonObject &qualifiers : jarr ) {
116 const affected_stat as = affected_stat_from_string( qualifiers.get_string( "stat" ) );
117 if( as == affected_stat::NONE ) {
118 qualifiers.throw_error( "Invalid affected stat", "stat" );
119 }
120
121 damage_type dt = DT_NULL;
122 if( needs_damage_type( as ) ) {
123 dt = dt_by_name( qualifiers.get_string( "type" ) );
124 if( dt == DT_NULL ) {
125 qualifiers.throw_error( "Invalid damage type", "type" );
126 }
127 }
128
129 const affected_type at( as, dt );
130
131 auto &selected = mult ? bonuses_mult : bonuses_flat;
132 selected[at].emplace_back( qualifiers );
133 }
134}

202std::string bonus_container::get_description() const
203{
204 std::string dump;
205 for( const auto &boni : bonuses_mult ) {
206 std::string type = string_from_affected_stat( boni.first.get_stat() );
207
208 if( needs_damage_type( boni.first.get_stat() ) ) {
209 //~ %1$s: damage type, %2$s: damage-related bonus name
210 type = string_format( pgettext( "type of damage", "%1$s %2$s" ),
211 name_by_dt( boni.first.get_damage_type() ), type );
212 }
213
214 for( const auto &sf : boni.second ) {
215 if( sf.stat ) {
216 //~ %1$s: bonus name, %2$d: bonus percentage, %3$s: stat name

219 } else {
220 //~ %1$s: bonus name, %2$d: bonus percentage
221 dump += string_format( pgettext( "martial art bonus", "* %1$s: <stat>%2$d%%</stat>" ),

224 dump += "\n";
225 }
