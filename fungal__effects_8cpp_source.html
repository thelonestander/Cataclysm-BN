65void fungal_effects::fungalize( const tripoint &p, Creature *origin, double spore_chance )
66{
67 if( monster *const mon_ptr = g->critter_at<monster>( p ) ) {
68 monster &critter = *mon_ptr;
69 if( gm.u.sees( p ) &&
70 !critter.type->in_species( FUNGUS ) ) {
71 add_msg( _( "The %s is covered in tiny spores!" ), critter.name() );
72 }
73 if( !critter.make_fungus() ) {
74 // Don't insta-kill non-fungables. Jabberwocks, for example
75 critter.add_effect( effect_stunned, rng( 1_turns, 3_turns ) );
76 critter.apply_damage( origin, bodypart_id( "torso" ), rng( 25, 50 ) );
77 }
78 } else if( gm.u.pos() == p && gm.u.get_effect_int( effect_fungus ) < 3 ) {
79 // TODO: Make this accept NPCs when they understand fungals
80 player &pl = gm.u;
81 ///\EFFECT_DEX increases chance of knocking fungal spores away with your TAIL_CATTLE
82 ///\EFFECT_MELEE increases chance of knocking fungal sports away with your TAIL_CATTLE
83 if( pl.has_trait( trait_TAIL_CATTLE ) &&
84 one_in( 20 - pl.dex_cur - pl.get_skill_level( skill_melee ) ) ) {
85 pl.add_msg_if_player(
86 _( "The spores land on you, but you quickly swat them off with your tail!" ) );
87 return;
88 }
89 // Spores hit the player--is there any hope?
90 bool hit = false;
91 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_head, 3, 9_minutes, bp_head );
92 hit |= one_in( 2 ) && pl.add_env_effect( effect_spores, bp_torso, 3, 9_minutes, bp_torso );
93 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_arm_l, 3, 9_minutes, bp_arm_l );
94 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_arm_r, 3, 9_minutes, bp_arm_r );
95 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_leg_l, 3, 9_minutes, bp_leg_l );
96 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_leg_r, 3, 9_minutes, bp_leg_r );
97 if( hit ) {
98 add_msg( m_warning, _( "You're covered in tiny spores!" ) );
99 }
100 } else if( gm.num_creatures() < 250 && x_in_y( spore_chance, 1.0 ) ) { // Spawn a spore
101 if( monster *const spore = gm.place_critter_at( mon_spore, p ) ) {
102 monster *origin_mon = dynamic_cast<monster *>( origin );
103 if( origin_mon != nullptr ) {
104 spore->make_ally( *origin_mon );
105 } else if( origin != nullptr && origin->is_player() &&
106 gm.u.has_trait( trait_THRESH_MYCUS ) ) {
107 spore->friendly = 1000;
108 }
109 }
110 } else {
111 spread_fungus( p );
112 }

139void fungal_effects::spread_fungus_one_tile( const tripoint &p, const int growth )
140{
141 bool converted = false;
142 // Terrain conversion
143 if( m.has_flag_ter( flag_DIGGABLE, p ) ) {
144 if( x_in_y( growth * 10, 100 ) ) {
145 m.ter_set( p, t_fungus );
146 converted = true;
147 }
148 } else if( m.has_flag( flag_FLAT, p ) ) {
149 if( m.has_flag( TFLAG_INDOORS, p ) ) {
150 if( x_in_y( growth * 10, 500 ) ) {
151 m.ter_set( p, t_fungus_floor_in );
152 converted = true;
153 }
154 } else if( m.has_flag( TFLAG_SUPPORTS_ROOF, p ) ) {
155 if( x_in_y( growth * 10, 1000 ) ) {
156 m.ter_set( p, t_fungus_floor_sup );
157 converted = true;
158 }
159 } else {
160 if( x_in_y( growth * 10, 2500 ) ) {
161 m.ter_set( p, t_fungus_floor_out );
162 converted = true;
163 }
164 }
165 } else if( m.has_flag( flag_SHRUB, p ) ) {
166 if( x_in_y( growth * 10, 200 ) ) {
167 m.ter_set( p, t_shrub_fungal );
168 converted = true;
169 } else if( x_in_y( growth, 1000 ) ) {
170 m.ter_set( p, t_marloss );
171 converted = true;
172 }
173 } else if( m.has_flag( flag_THIN_OBSTACLE, p ) ) {
174 if( x_in_y( growth * 10, 150 ) ) {
175 m.ter_set( p, t_fungus_mound );
176 converted = true;
177 }
178 } else if( m.has_flag( flag_YOUNG, p ) ) {
179 if( x_in_y( growth * 10, 500 ) ) {
180 if( m.get_field_intensity( p, fd_fungal_haze ) != 0 ) {
181 if( x_in_y( growth * 10, 800 ) ) { // young trees are vulnerable
182 m.ter_set( p, t_fungus );

184 add_msg( m_warning, _( "The young tree blooms forth into a fungal blossom!" ) );
185 }
186 } else if( x_in_y( growth * 10, 400 ) ) {
187 m.ter_set( p, t_marloss_tree );
188 }
189 } else {
190 m.ter_set( p, t_tree_fungal_young );
191 }
192 converted = true;
193 }
194 } else if( m.has_flag( flag_TREE, p ) ) {
195 if( one_in( 10 ) ) {
196 if( m.get_field_intensity( p, fd_fungal_haze ) != 0 ) {
197 if( x_in_y( growth * 10, 100 ) ) {
198 m.ter_set( p, t_fungus );

206 m.ter_set( p, t_tree_fungal );
207 }

209 }
210 } else if( m.has_flag( flag_WALL, p ) && m.has_flag( flag_FLAMMABLE, p ) ) {
211 if( x_in_y( growth * 10, 5000 ) ) {
212 m.ter_set( p, t_fungus_wall );
213 converted = true;

216 // Furniture conversion
217 if( converted ) {
218 if( m.has_flag( flag_FLOWER, p ) ) {
219 m.furn_set( p, f_flower_fungal );

227 // Replace the (already existing) seed
228 // Can't use item_stack::only_item() since there might be fertilizer

234 dbg( DL::Warn ) << "No seed item in the PLANT terrain at position " << p;
235 } else {

242void fungal_effects::spread_fungus( const tripoint &p )
243{
244 int growth = 1;
245 for( const tripoint &tmp : g->m.points_in_radius( p, 1 ) ) {
246 if( tmp == p ) {
247 continue;
248 }
249 if( m.has_flag( flag_FUNGUS, tmp ) ) {
250 growth += 1;
251 }
252 }
253
254 if( !m.has_flag_ter( flag_FUNGUS, p ) ) {
255 spread_fungus_one_tile( p, growth );
256 } else {
257 // Everything is already fungus
258 if( growth == 9 ) {
259 return;
260 }
261 for( const tripoint &dest : g->m.points_in_radius( p, 1 ) ) {
262 // One spread on average
263 if( !m.has_flag( flag_FUNGUS, dest ) && one_in( 9 - growth ) ) {
264 //growth chance is 100 in X simplified
265 spread_fungus_one_tile( dest, 10 );
266 }
267 }
268 }
