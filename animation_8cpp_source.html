180void draw_custom_explosion_curses( game &g,
181 const std::list< std::map<tripoint, explosion_tile> > &layers )
182{
183 // calculate screen offset relative to player + view offset position
184 const tripoint center = g.u.pos() + g.u.view_offset;
185 const tripoint topleft( center.x - getmaxx( g.w_terrain ) / 2,
186 center.y - getmaxy( g.w_terrain ) / 2, 0 );
187
188 explosion_animation anim;
189
190 auto last_layer_it = layers.begin();
191 shared_ptr_fast<game::draw_callback_t> explosion_cb =
192 make_shared_fast<game::draw_callback_t>( [&]() {
193 for( auto it = layers.begin(); it != std::next( last_layer_it ); ++it ) {
194 for( const auto &pr : *it ) {
195 // update tripoint in relation to top left corner of curses window
196 // mvwputch already filters out of bounds coordinates
197 const tripoint p = pr.first - topleft;
198 const explosion_neighbors ngh = pr.second.neighborhood;
199 const nc_color col = pr.second.color;
200
201 switch( ngh ) {
202 // '^', 'v', '<', '>'
203 case N_NORTH:
204 mvwputch( g.w_terrain, p.xy(), col, '^' );
205 break;
206 case N_SOUTH:
207 mvwputch( g.w_terrain, p.xy(), col, 'v' );
208 break;
209 case N_WEST:
210 mvwputch( g.w_terrain, p.xy(), col, '<' );
211 break;
212 case N_EAST:
213 mvwputch( g.w_terrain, p.xy(), col, '>' );
214 break;
215 // '|' and '-'
216 case N_NORTH | N_SOUTH:
217 case N_NORTH | N_SOUTH | N_WEST:
218 case N_NORTH | N_SOUTH | N_EAST:
219 mvwputch( g.w_terrain, p.xy(), col, '|' );
220 break;
221 case N_WEST | N_EAST:
222 case N_WEST | N_EAST | N_NORTH:
223 case N_WEST | N_EAST | N_SOUTH:
224 mvwputch( g.w_terrain, p.xy(), col, '-' );
225 break;
226 // '/' and '\'
227 case N_NORTH | N_WEST:
228 case N_SOUTH | N_EAST:
229 mvwputch( g.w_terrain, p.xy(), col, '/' );
230 break;
231 case N_SOUTH | N_WEST:
232 case N_NORTH | N_EAST:
233 mvwputch( g.w_terrain, p.xy(), col, '\\' );
234 break;
235 case N_NO_NEIGHBORS:
236 mvwputch( g.w_terrain, p.xy(), col, '*' );
237 break;
238 case N_WEST | N_EAST | N_NORTH | N_SOUTH:
239 break;
240 }
241 }
242 }
243 } );
244 g.add_draw_callback( explosion_cb );
245
246 for( last_layer_it = layers.begin(); last_layer_it != layers.end(); ++last_layer_it ) {
247 if( is_layer_visible( *last_layer_it ) ) {
248 anim.progress();
249 }
250 }
251}

301void explosion_handler::draw_custom_explosion( const tripoint &,
302 const std::map<tripoint, nc_color> &all_area,
303 const std::string &exp_name )
304{
305#if !defined(TILES)
306 ( void )exp_name;
307#endif
308 if( test_mode ) {
309 // avoid segfault from null tilecontext in tests
310 return;
311 }
312
313 constexpr explosion_neighbors all_neighbors = N_NORTH | N_SOUTH | N_WEST | N_EAST;
314 // We will "shell" the explosion area
315 // Each phase will strip a single layer of points
316 // A layer contains all points that have less than 4 neighbors in cardinal directions
317 // Layers will first be generated, then drawn in inverse order
318
319 // Start by getting rid of everything except current z-level
320 std::map<tripoint, explosion_tile> neighbors;
321#if defined(TILES)
322 if( !use_tiles ) {
323 for( const auto &pr : all_area ) {
324 const tripoint relative_point = relative_view_pos( g->u, pr.first );
325 if( relative_point.z == 0 ) {
326 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
327 }
328 }
329 } else {
330 // In tiles mode, the coordinates have to be absolute
331 const tripoint view_center = relative_view_pos( g->u, g->u.pos() );
332 for( const auto &pr : all_area ) {
333 // Relative point is only used for z level check
334 const tripoint relative_point = relative_view_pos( g->u, pr.first );
335 if( relative_point.z == view_center.z ) {
336 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
337 }
338 }
339 }
340#else
341 for( const auto &pr : all_area ) {
342 const tripoint relative_point = relative_view_pos( g->u, pr.first );
343 if( relative_point.z == 0 ) {
344 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
345 }
346 }
347#endif
348
349 // Searches for a neighbor, sets the neighborhood flag on current point and on the neighbor
350 const auto set_neighbors = [&]( const tripoint & pos,
351 explosion_neighbors & ngh,
352 explosion_neighbors here,
353 explosion_neighbors there ) {
354 if( ( ngh & here ) == N_NO_NEIGHBORS ) {
355 auto other = neighbors.find( pos );
356 if( other != neighbors.end() ) {
357 ngh = ngh | here;
358 other->second.neighborhood = other->second.neighborhood | there;
359 }
360 }
361 };
362
363 // If the point we are about to remove has a neighbor in a given direction
364 // unset that neighbor's flag that our current point is its neighbor
365 const auto unset_neighbor = [&]( const tripoint & pos,
366 const explosion_neighbors ngh,
367 explosion_neighbors here,
368 explosion_neighbors there ) {
369 if( ( ngh & here ) != N_NO_NEIGHBORS ) {
370 auto other = neighbors.find( pos );
371 if( other != neighbors.end() ) {
372 other->second.neighborhood = ( other->second.neighborhood | there ) ^ there;
373 }
374 }
375 };
376
377 // Find all neighborhoods
378 for( auto &pr : neighbors ) {
379 const tripoint &pt = pr.first;
380 explosion_neighbors &ngh = pr.second.neighborhood;
381
382 set_neighbors( pt + point_west, ngh, N_WEST, N_EAST );
383 set_neighbors( pt + point_east, ngh, N_EAST, N_WEST );
384 set_neighbors( pt + point_north, ngh, N_NORTH, N_SOUTH );
385 set_neighbors( pt + point_south, ngh, N_SOUTH, N_NORTH );
386 }
387
388 // We need to save the layers because we will draw them in reverse order
389 std::list< std::map<tripoint, explosion_tile> > layers;
390 while( !neighbors.empty() ) {
391 std::map<tripoint, explosion_tile> layer;
392 bool changed = false;
393 // Find a layer that can be drawn
394 for( const auto &pr : neighbors ) {
395 if( pr.second.neighborhood != all_neighbors ) {
396 changed = true;
397 layer.insert( pr );
398 }
399 }
400 if( !changed ) {
401 // An error, but a minor one - let it slide
402 return;
403 }
404 // Remove the layer from the area to process
405 for( const auto &pr : layer ) {
406 const tripoint &pt = pr.first;
407 const explosion_neighbors ngh = pr.second.neighborhood;
408
409 unset_neighbor( pt + point_west, ngh, N_WEST, N_EAST );
410 unset_neighbor( pt + point_east, ngh, N_EAST, N_WEST );
411 unset_neighbor( pt + point_north, ngh, N_NORTH, N_SOUTH );
412 unset_neighbor( pt + point_south, ngh, N_SOUTH, N_NORTH );
413 neighbors.erase( pr.first );
414 }
415
416 layers.push_front( std::move( layer ) );
417 }
418
419#if defined(TILES)
420 if( !use_tiles ) {
421 draw_custom_explosion_curses( *g, layers );
422 return;
423 }
424
425 explosion_animation anim;
426 // We need to draw all explosions up to now
427 std::map<tripoint, explosion_tile> combined_layer;
428
429 shared_ptr_fast<game::draw_callback_t> explosion_cb =
430 make_shared_fast<game::draw_callback_t>( [&]() {
431 tilecontext->init_custom_explosion_layer( combined_layer, exp_name );
432 } );
433 g->add_draw_callback( explosion_cb );
434
435 for( const auto &layer : layers ) {
436 combined_layer.insert( layer.begin(), layer.end() );
437 if( is_layer_visible( layer ) ) {
438 anim.progress();
439 }
440 }
441
442 tilecontext->void_custom_explosion();
443#else
444 draw_custom_explosion_curses( *g, layers );
445#endif
446}

523{
524 const tripoint init_pos = relative_view_pos( u, center );
525 // Only show animation if initially visible

527 shared_ptr_fast<game::draw_callback_t> hit_cb = make_shared_fast<game::draw_callback_t>( [&]() {
528 // In case the window is resized during waiting, we always re-calculate the animation position
529 const tripoint pos = relative_view_pos( u, center );
530 if( pos.z == 0 && is_valid_in_w_terrain( pos.xy() ) ) {
531 mvwprintz( g->w_terrain, pos.xy(), cColor, cTile );
532 }
533 } );
534 g->add_draw_callback( hit_cb );
535
536 ui_manager::redraw();
537 inp_mngr.set_timeout( get_option<int>( "ANIMATION_DELAY" ) );
538 // Skip input (if any), because holding down a key with nanosleep can get yourself killed
539 inp_mngr.get_input_event();

628 bool noreveal )

631 for( const tripoint &p : ret ) {
632 const auto critter = g.critter_at( p, true );
633
634 // NPCs and monsters get drawn with inverted colors
635 if( critter && g.u.sees( *critter ) ) {
636 critter->draw( g.w_terrain, center, true );
637 } else if( noreveal && !g.u.sees( p ) ) {

643 const int j = p.y + getmaxy( w ) / 2 - center.y;
644 mvwputch( w, point( k, j ), col, sym );
645 } else {
646 // This function reveals tile at p and writes it to the player's memory

975bucketed_points bucket_by_distance( const tripoint &origin,
976 const std::map<tripoint, double> &to_bucket )
977{
978 std::map<int, one_bucket> by_distance;
979 for( const std::pair<const tripoint, double> &pv : to_bucket ) {
980 int dist = trig_dist_squared( origin, pv.first );
981 by_distance[dist].emplace_back( pv.first, pv.second );
982 }
983 bucketed_points buckets;
984 for( const std::pair<const int, one_bucket> &bc : by_distance ) {
985 buckets.emplace_back( bc.second );
986 }
987 return buckets;
988}
989
990bucketed_points optimal_bucketing( const bucketed_points &buckets, size_t max_buckets )
991{
992 if( buckets.size() <= max_buckets ) {
993 return buckets;
994 }
995 assert( max_buckets > 1 );
996
997 std::vector<size_t> sizes = {};
998 for( const one_bucket &bc : buckets ) {
999 sizes.emplace_back( bc.size() );
1000 }
1001
1002 bucketed_points optimal = buckets;
1003 // TODO: Good algorithm here, this one is a greedy finder of smallest adjacent size sums
1004 for( size_t i = 0; i < buckets.size() - max_buckets; i++ ) {
1005 auto smallest = sizes.begin();
1006 size_t smallest_sum = *smallest + *( smallest + 1 );
1007 for( auto iter = sizes.begin() + 1; ( iter + 1 ) != sizes.end(); iter++ ) {
1008 size_t sum = *iter + *( iter + 1 );
1009 if( sum < smallest_sum ) {
1010 smallest = iter;
1011 smallest_sum = sum;
1012 }
1013 }
1014
1015 size_t distance = std::distance( sizes.begin(), smallest );
1016 sizes[distance] += sizes[distance + 1];
1017 sizes.erase( smallest + 1 );
1018 auto left_bucket = std::next( optimal.begin(), distance );
1019 auto right_bucket = std::next( left_bucket );
1020 left_bucket->insert( left_bucket->end(), right_bucket->begin(), right_bucket->end() );
1021 optimal.erase( right_bucket );
1022 }
1023
1024 return optimal;
1025}

1027static void draw_cone_aoe_curses( const tripoint &, const bucketed_points &waves )
1028{
1029 // Calculate screen offset relative to player + view offset position
1030 const avatar &u = get_avatar();
1031 const tripoint center = u.pos() + u.view_offset;
1032 const tripoint topleft( center.x - catacurses::getmaxx( g->w_terrain ) / 2,
1033 center.y - catacurses::getmaxy( g->w_terrain ) / 2, 0 );
1034
1035 auto it = waves.begin();
1036 shared_ptr_fast<game::draw_callback_t> wave_cb =
1037 make_shared_fast<game::draw_callback_t>( [&]() {
1038 // All the buckets up until now
1039 for( auto inner_it = waves.begin(); inner_it != std::next( it ); inner_it++ ) {
1040 for( const point_with_value &pr : *inner_it ) {
1041 // update tripoint in relation to top left corner of curses window
1042 // mvwputch already filters out of bounds coordinates
1043 const tripoint p = pr.pt - topleft;
1044 int intensity = ( pr.val >= 1.0 ) + ( pr.val >= 0.5 ) + ( inner_it == it );
1045 nc_color col;
1046 switch( intensity ) {
1047 case 3:
1048 col = c_red;
1049 break;
1050 case 2:
1051 col = c_yellow;
1052 break;
1053 case 1:
1054 col = c_white;
1055 break;
1056 default:
1057 col = c_dark_gray;
1058 break;
1059 }
1060
1061 // TODO: Prettier
1062 mvwputch( g->w_terrain, p.xy(), col, '*' );
1063 }
1064 }
1065 } );
1066 g->add_draw_callback( wave_cb );
1067
1068 wave_animation anim;
1069 for( it = waves.begin(); it != waves.end(); it++ ) {
1070 anim.progress();
1071 }
1072}
