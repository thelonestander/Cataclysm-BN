435bool avatar_action::ramp_move( avatar &you, map &m, const tripoint &dest_loc )
436{
437 if( dest_loc.z != you.posz() ) {
438 // No recursive ramp_moves
439 return false;
440 }
441
442 // We're moving onto a tile with no support, check if it has a ramp below
443 if( !m.has_floor_or_support( dest_loc ) ) {
444 tripoint below( dest_loc.xy(), dest_loc.z - 1 );
445 if( m.has_flag( TFLAG_RAMP, below ) ) {
446 // But we're moving onto one from above
447 const tripoint dp = dest_loc - you.pos();
448 move( you, m, tripoint( dp.xy(), -1 ) );
449 // No penalty for misaligned stairs here
450 // Also cheaper than climbing up
451 return true;
452 }
453
454 return false;
455 }
456
457 if( !m.has_flag( TFLAG_RAMP, you.pos() ) ||
458 m.passable( dest_loc ) ) {
459 return false;
460 }
461
462 // Try to find an aligned end of the ramp that will make our climb faster
463 // Basically, finish walking on the stairs instead of pulling self up by hand
464 bool aligned_ramps = false;
465 for( const tripoint &pt : m.points_in_radius( you.pos(), 1 ) ) {
466 if( rl_dist( pt, dest_loc ) < 2 && m.has_flag( flag_RAMP_END, pt ) ) {
467 aligned_ramps = true;
468 break;
469 }
470 }
471
472 const tripoint above_u( you.posx(), you.posy(), you.posz() + 1 );
473 if( m.has_floor_or_support( above_u ) ) {
474 add_msg( m_warning, _( "You can't climb here - there's a ceiling above." ) );
475 return false;
476 }
477
478 const tripoint dp = dest_loc - you.pos();
479 const tripoint old_pos = you.pos();
480 move( you, m, tripoint( dp.xy(), 1 ) );
481 // We can't just take the result of the above function here
482 if( you.pos() != old_pos ) {
483 you.moves -= 50 + ( aligned_ramps ? 0 : 50 );
484 }
485
486 return true;
487}

580{

845void avatar_action::plthrow( avatar &you, item_location loc,
846 const cata::optional<tripoint> &blind_throw_from_pos )
847{
848 if( you.has_active_mutation( trait_SHELL2 ) ) {
849 add_msg( m_info, _( "You can't effectively throw while you're in your shell." ) );
850 return;
851 }
852 if( you.is_mounted() ) {
853 monster *mons = get_player_character().mounted_creature.get();
854 if( mons->has_flag( MF_RIDEABLE_MECH ) ) {
855 if( !mons->check_mech_powered() ) {
856 add_msg( m_bad, _( "Your %s refuses to move as its batteries have been drained." ),
857 mons->get_name() );
858 return;
859 }
860 }
861 }
862
863 if( !loc ) {
864 loc = game_menus::inv::titled_menu( you, _( "Throw item" ),
865 _( "You don't have any items to throw." ) );
866 }
867
868 if( !loc ) {
869 add_msg( _( "Never mind." ) );
870 return;
871 }
872 // make a copy and get the original.
873 // the copy is thrown and has its and the originals charges set appropiately
874 // or deleted from inventory if its charges(1) or not stackable.
875 item thrown = *loc.get_item();
876 int range = you.throw_range( thrown );
877 if( range < 0 ) {
878 add_msg( m_info, _( "You don't have that item." ) );
879 return;
880 } else if( range == 0 ) {
881 add_msg( m_info, _( "That is too heavy to throw." ) );
882 return;
883 }
884
885 if( you.is_wielding( *loc ) && loc->has_flag( flag_NO_UNWIELD ) ) {
886 // pos == -1 is the weapon, NO_UNWIELD is used for bio_claws_weapon
887 add_msg( m_info, _( "That's part of your body, you can't throw that!" ) );
888 return;
889 }
890
891 if( you.has_effect( effect_relax_gas ) ) {
892 if( one_in( 5 ) ) {
893 add_msg( m_good, _( "You concentrate mightily, and your body obeys!" ) );
894 } else {
895 you.moves -= rng( 2, 5 ) * 10;
896 add_msg( m_bad, _( "You can't muster up the effort to throw anything…" ) );
897 return;
898 }
899 }
900 // if you're wearing the item you need to be able to take it off
901 if( you.is_wearing( loc->typeId() ) ) {
902 ret_val<bool> ret = you.can_takeoff( *loc );
903 if( !ret.success() ) {
904 add_msg( m_info, "%s", ret.c_str() );
905 return;
906 }
907 }
908 // you must wield the item to throw it
909 // But only if you don't have enough free hands
910 int usable_hands = you.get_working_arm_count() -
911 ( you.is_armed() ? 1 : 0 ) -
912 ( you.weapon.is_two_handed( you ) ? 1 : 0 );
913 if( !you.is_wielding( *loc ) &&
914 ( usable_hands < ( loc->is_two_handed( you ) ? 2 : 1 ) ) ) {
915 if( !you.wield( *loc ) ) {
916 add_msg( m_info, _( "You do not have enough free hands to throw %s without wielding it." ),
917 loc->tname() );
918 return;
919 }
920
921 loc = item_location( you, &you.weapon );
922 }
923
924 // Shift our position to our "peeking" position, so that the UI
925 // for picking a throw point lets us target the location we couldn't
926 // otherwise see.
927 const tripoint original_player_position = you.pos();
928 if( blind_throw_from_pos ) {
929 you.setpos( *blind_throw_from_pos );
930 }
931
932 target_handler::trajectory trajectory = target_handler::mode_throw( you, *loc,
933 blind_throw_from_pos.has_value() );
934
935 // If we previously shifted our position, put ourselves back now that we've picked our target.
936 if( blind_throw_from_pos ) {
937 you.setpos( original_player_position );
938 }
939
940 if( trajectory.empty() ) {
941 return;
942 }
943
944 if( &*loc != &you.weapon ) {
945 // This is to represent "implicit offhand wielding"
946 int extra_cost = you.item_handling_cost( *loc, true, INVENTORY_HANDLING_PENALTY / 2 );
947 you.mod_moves( -extra_cost );
948 }
949
950 if( loc->count_by_charges() && loc->charges > 1 ) {
951 loc->mod_charges( -1 );
952 thrown.charges = 1;
953 } else {
954 loc.remove_item();
955 }
956 you.throw_item( trajectory.back(), thrown, blind_throw_from_pos );
957 g->reenter_fullscreen();
958}
