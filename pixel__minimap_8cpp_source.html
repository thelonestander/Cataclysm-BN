126// a texture pool to avoid recreating textures every time player changes their view
127// at most 142 out of 144 textures can be in use due to regular player movement
128// (moving from submap corner to new corner) with MAPSIZE = 11
129// textures are dumped when the player moves more than one submap in one update
130// (teleporting, z-level change) to prevent running out of the remaining pool
131class pixel_minimap::shared_texture_pool
132{
133 public:
134 shared_texture_pool( const std::function<SDL_Texture_Ptr()> &generator ) {
135 const size_t pool_size = ( MAPSIZE + 1 ) * ( MAPSIZE + 1 );
136
137 texture_pool.reserve( pool_size );
138 inactive_index.reserve( pool_size );
139
140 for( size_t i = 0; i < pool_size; ++i ) {
141 texture_pool.emplace_back( generator() );
142 inactive_index.push_back( i );
143 }
144 }
145
146 //reserves a texture from the inactive group and returns tracking info
147 SDL_Texture_Ptr request_tex( size_t &index ) {
148 if( inactive_index.empty() ) {
149 debugmsg( "Ran out of available textures in the pool." );
150 //shouldn't be happening, but minimap will just be default color instead of crashing
151 return nullptr;
152 }
153 index = inactive_index.back();
154 inactive_index.pop_back();
155 return std::move( texture_pool[index] );
156 }
157
158 //releases the provided texture back into the inactive pool to be used again
159 //called automatically in the submap cache destructor
160 void release_tex( size_t index, SDL_Texture_Ptr &&ptr ) {
161 inactive_index.push_back( index );
162 texture_pool[index] = std::move( ptr );
163 }
164
165 private:
166 std::vector<SDL_Texture_Ptr> texture_pool;
167 std::vector<size_t> inactive_index;
168};

170struct pixel_minimap::submap_cache {
171 //the color stored for each submap tile
172 std::array<SDL_Color, SEEX *SEEY> minimap_colors = {};
173 //checks if the submap has been looked at by the minimap routine
174 bool touched;
175 //the texture updates are drawn to
176 SDL_Texture_Ptr chunk_tex;
177 //the submap being handled
178 size_t texture_index;
179 //the list of updates to apply to the texture
180 //reduces render target switching to once per submap
181 std::vector<point> update_list;
182 //flag used to indicate that the texture needs to be cleared before first use
183 bool ready;
184 shared_texture_pool &pool;
185
186 //reserve the SEEX * SEEY submap tiles
187 submap_cache( shared_texture_pool &pool ) :
188 touched( false ),
189 ready( false ),
190 pool( pool ) {
191 chunk_tex = pool.request_tex( texture_index );
192 }
193
194 //handle the release of the borrowed texture
195 ~submap_cache() {
196 pool.release_tex( texture_index, std::move( chunk_tex ) );
197 }
198
199 submap_cache( submap_cache && ) = default;
200
201 SDL_Color &color_at( const point &p ) {
202 assert( p.x < SEEX );
203 assert( p.y < SEEY );
204
205 return minimap_colors[p.y * SEEX + p.x];
206 }
207};

232void pixel_minimap::prepare_cache_for_updates( const tripoint &center )
233{
234 const tripoint new_center_sm = g->m.get_abs_sub() + ms_to_sm_copy( center );
235 const tripoint center_sm_diff = cached_center_sm - new_center_sm;
236
237 //invalidate the cache if the game shifted more than one submap in the last update, or if z-level changed.
238 if( std::abs( center_sm_diff.x ) > 1 ||
239 std::abs( center_sm_diff.y ) > 1 ||
240 std::abs( center_sm_diff.z ) > 0 ) {
241 cache.clear();
242 } else {
243 for( auto &mcp : cache ) {
244 mcp.second.touched = false;
245 }
246 }
247
248 cached_center_sm = new_center_sm;
249}

260//draws individual updates to the submap cache texture
261//the render target will be set back to display_buffer after all submaps are updated
262void pixel_minimap::flush_cache_updates()
263{
264 for( auto &mcp : cache ) {
265 if( mcp.second.update_list.empty() ) {
266 continue;
267 }
268
269 SetRenderTarget( renderer, mcp.second.chunk_tex );
270
271 if( !mcp.second.ready ) {
272 mcp.second.ready = true;
273
274 SetRenderDrawColor( renderer, 0x00, 0x00, 0x00, 0x00 );
275 RenderClear( renderer );
276
277 for( int y = 0; y < SEEY; ++y ) {
278 for( int x = 0; x < SEEX; ++x ) {
279 const point tile_pos = projector->get_tile_pos( { x, y }, { SEEX, SEEY } );
280 const point tile_size = projector->get_tile_size();
281
282 const SDL_Rect rect = SDL_Rect{ tile_pos.x, tile_pos.y, tile_size.x, tile_size.y };
283
284 geometry->rect( renderer, rect, SDL_Color() );
285 }
286 }
287 }
288
289 for( const point &p : mcp.second.update_list ) {
290 const point tile_pos = projector->get_tile_pos( p, { SEEX, SEEY } );
291 const SDL_Color tile_color = mcp.second.color_at( p );
292
293 if( pixel_size.x == 1 && pixel_size.y == 1 ) {
294 SetRenderDrawColor( renderer, tile_color.r, tile_color.g, tile_color.b, tile_color.a );
295 RenderDrawPoint( renderer, tile_pos );
296 } else {
297 geometry->rect( renderer, tile_pos, pixel_size.x, pixel_size.y, tile_color );
298 }
299 }
300
301 mcp.second.update_list.clear();
302 }

305void pixel_minimap::update_cache_at( const tripoint &sm_pos )
306{
307 const level_cache &access_cache = g->m.access_cache( sm_pos.z );
308 const bool nv_goggle = g->u.get_vision_modes()[NV_GOGGLES];
309
310 submap_cache &cache_item = get_cache_at( g->m.get_abs_sub() + sm_pos );
311 const tripoint ms_pos = sm_to_ms_copy( sm_pos );
312
313 cache_item.touched = true;
314
315 for( int y = 0; y < SEEY; ++y ) {
316 for( int x = 0; x < SEEX; ++x ) {
317 const tripoint p = ms_pos + tripoint{ x, y, 0 };
318 const lit_level lighting = access_cache.visibility_cache[p.x][p.y];
319
320 SDL_Color color;
321
322 if( lighting == lit_level::BLANK || lighting == lit_level::DARK ) {
323 // TODO: Map memory?
324 color = { 0x00, 0x00, 0x00, 0xFF };
325 } else {
326 color = get_map_color_at( p );
327
328 //color terrain according to lighting conditions
329 if( nv_goggle ) {
330 if( lighting == lit_level::LOW ) {
331 color = color_pixel_nightvision( color );
332 } else if( lighting != lit_level::DARK && lighting != lit_level::BLANK ) {
333 color = color_pixel_overexposed( color );
334 }
335 } else if( lighting == lit_level::LOW ) {
336 color = color_pixel_grayscale( color );
337 }
338
339 color = adjust_color_brightness( color, settings.brightness );
340 }
341
342 SDL_Color &current_color = cache_item.color_at( { x, y } );
343
344 if( current_color != color ) {
345 current_color = color;
346 cache_item.update_list.push_back( { x, y } );
347 }
348 }
349 }

377void pixel_minimap::set_screen_rect( const SDL_Rect &screen_rect )
378{
379 if( this->screen_rect == screen_rect && main_tex && tex_pool && projector ) {
380 return;
381 }
382
383 this->screen_rect = screen_rect;
384
385 projector = create_projector( screen_rect );
386 pixel_size = get_pixel_size( projector->get_tile_size(), settings.mode );
387
388 const point size_on_screen = projector->get_tiles_size( total_tiles_count );
389
390 if( settings.scale_to_fit ) {
391 main_tex_clip_rect = SDL_Rect{ 0, 0, size_on_screen.x, size_on_screen.y };
392 screen_clip_rect = fit_rect_inside( main_tex_clip_rect, screen_rect );
393
394 SDL_SetHint( SDL_HINT_RENDER_SCALE_QUALITY, "1" );
395 main_tex = create_cache_texture( renderer, size_on_screen.x, size_on_screen.y );
396 SDL_SetHint( SDL_HINT_RENDER_SCALE_QUALITY, "0" );
397
398 } else {
399 const point d( ( size_on_screen.x - screen_rect.w ) / 2, ( size_on_screen.y - screen_rect.h ) / 2 );
400
401 main_tex_clip_rect = SDL_Rect{
402 std::max( d.x, 0 ),
403 std::max( d.y, 0 ),
404 size_on_screen.x - 2 * std::max( d.x, 0 ),
405 size_on_screen.y - 2 * std::max( d.y, 0 )
406 };
407
408 screen_clip_rect = SDL_Rect{
409 screen_rect.x - std::min( d.x, 0 ),
410 screen_rect.y - std::min( d.y, 0 ),
411 main_tex_clip_rect.w,
412 main_tex_clip_rect.h
413 };
414
415 main_tex = create_cache_texture( renderer, size_on_screen.x, size_on_screen.y );
416 }
417
418 cache.clear();
419
420 const point chunk_size = projector->get_tiles_size( { SEEX, SEEY } );
421
422 const auto chunk_texture_generator = [&chunk_size, this]() {
423 SDL_Texture_Ptr result = create_cache_texture( renderer, chunk_size.x, chunk_size.y );
424 SetTextureBlendMode( result, SDL_BLENDMODE_BLEND );
425 return result;
426 };
427
428 tex_pool = std::make_unique<shared_texture_pool>( chunk_texture_generator );
429}

455void pixel_minimap::render_cache( const tripoint &center )
456{
457 const tripoint sm_center = g->m.get_abs_sub() + ms_to_sm_copy( center );
458 const tripoint sm_offset = tripoint{
459 total_tiles_count.x / SEEX / 2,
460 total_tiles_count.y / SEEY / 2, 0
461 };
462
463 point ms_offset = center.xy();
464 ms_to_sm_remain( ms_offset );
465 ms_offset = point{ SEEX / 2, SEEY / 2 } - ms_offset;
466
467 for( const auto &elem : cache ) {
468 if( !elem.second.touched ) {
469 continue; // What you gonna do with all that junk?
470 }
471
472 const tripoint rel_pos = elem.first - sm_center;
473
474 if( std::abs( rel_pos.x ) > sm_offset.x + 1 ||
475 std::abs( rel_pos.y ) > sm_offset.y + 1 ||
476 rel_pos.z != 0 ) {
477 continue;
478 }
479
480 const tripoint sm_pos = rel_pos + sm_offset;
481 const tripoint ms_pos = sm_to_ms_copy( sm_pos ) + ms_offset;
482
483 const SDL_Rect chunk_rect = projector->get_chunk_rect( ms_pos.xy(), { SEEX, SEEY } );
484
485 RenderCopy( renderer, elem.second.chunk_tex, nullptr, &chunk_rect );
486 }

489void pixel_minimap::render_critters( const tripoint &center )
490{
491 //handles the enemy faction red highlights
492 //this value should be divisible by 200
493 const int indicator_length = settings.beacon_blink_interval * 200; //default is 2000 ms, 2 seconds
494
495 int flicker = 100;
496 int mixture = 0;
497
498 if( indicator_length > 0 ) {
499 const float t = get_animation_phase( 2 * indicator_length );
500 const float s = std::sin( 2 * M_PI * t );
501
502 flicker = lerp_clamped( 25, 100, std::abs( s ) );
503 mixture = lerp_clamped( 0, 100, std::max( s, 0.0f ) );
504 }
505
506 const level_cache &access_cache = g->m.access_cache( center.z );
507
508 const int start_x = center.x - total_tiles_count.x / 2;
509 const int start_y = center.y - total_tiles_count.y / 2;
510 const point beacon_size = {
511 std::max<int>( projector->get_tile_size().x *settings.beacon_size / 2, 2 ),
512 std::max<int>( projector->get_tile_size().y *settings.beacon_size / 2, 2 )
513 };
514
515 cached_has_animated_beacons = false;
516 for( int y = 0; y < total_tiles_count.y; y++ ) {
517 for( int x = 0; x < total_tiles_count.x; x++ ) {
518 const tripoint p = tripoint{ start_x + x, start_y + y, center.z };
519 const lit_level lighting = access_cache.visibility_cache[p.x][p.y];
520
521 if( lighting == lit_level::DARK || lighting == lit_level::BLANK ) {
522 continue;
523 }
524
525 const auto critter = g->critter_at( p, true );
526
527 if( critter == nullptr || !g->u.sees( *critter ) ) {
528 continue;
529 }
530
531 const point critter_pos = projector->get_tile_pos( { x, y }, total_tiles_count );
532 const SDL_Rect critter_rect = SDL_Rect{ critter_pos.x, critter_pos.y, beacon_size.x, beacon_size.y };
533 const SDL_Color critter_color = get_critter_color( critter, flicker, mixture );
534 cached_has_animated_beacons = cached_has_animated_beacons || is_critter_animated( critter );
535
536 draw_beacon( critter_rect, critter_color );
537 }
538 }

562void pixel_minimap::draw_beacon( const SDL_Rect &rect, const SDL_Color &color )
563{
564 for( int x = -rect.w, x_max = rect.w; x <= x_max; ++x ) {
565 for( int y = -rect.h + std::abs( x ), y_max = rect.h - std::abs( x ); y <= y_max; ++y ) {
566 const int divisor = 2 * ( std::abs( y ) == rect.h - std::abs( x ) ? 1 : 0 ) + 1;
567
568 SetRenderDrawColor( renderer, color.r / divisor, color.g / divisor, color.b / divisor, 0xFF );
569 RenderDrawPoint( renderer, point( rect.x + x, rect.y + y ) );
