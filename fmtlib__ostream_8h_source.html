24template <class Char> class formatbuf : public std::basic_streambuf<Char>
25{
26 private:
27 using int_type = typename std::basic_streambuf<Char>::int_type;
28 using traits_type = typename std::basic_streambuf<Char>::traits_type;
29
30 buffer<Char> &buffer_;
31
32 public:
33 formatbuf( buffer<Char> &buf ) : buffer_( buf ) {}
34
35 protected:
36 // The put-area is actually always empty. This makes the implementation
37 // simpler and has the advantage that the streambuf and the buffer are always
38 // in sync and sputc never writes into uninitialized memory. The obvious
39 // disadvantage is that each call to sputc always results in a (virtual) call
40 // to overflow. There is no disadvantage here for sputn since this always
41 // results in a call to xsputn.
42
43 int_type overflow( int_type ch = traits_type::eof() ) FMT_OVERRIDE {
44 if( !traits_type::eq_int_type( ch, traits_type::eof() ) ) {
45 buffer_.push_back( static_cast<Char>( ch ) );
46 }
47 return ch;
48 }
49
50 std::streamsize xsputn( const Char *s, std::streamsize count ) FMT_OVERRIDE {
51 buffer_.append( s, s + count );
52 return count;
53 }
54};

96// Write the content of buf to os.
97template <typename Char>
98void write_buffer( std::basic_ostream<Char> &os, buffer<Char> &buf )
99{
100 const Char *buf_data = buf.data();
101 using unsigned_streamsize = std::make_unsigned<std::streamsize>::type;
102 unsigned_streamsize size = buf.size();
103 unsigned_streamsize max_size = to_unsigned( max_value<std::streamsize>() );
104 do {
105 unsigned_streamsize n = size <= max_size ? size : max_size;
106 os.write( buf_data, static_cast<std::streamsize>( n ) );
107 buf_data += n;
108 size -= n;
109 } while( size != 0 );
110}

112template <typename Char, typename T>
113void format_value( buffer<Char> &buf, const T &value,
114 locale_ref loc = locale_ref() )
115{
116 formatbuf<Char> format_buf( buf );
117 std::basic_ostream<Char> output( &format_buf );
118#if !defined(FMT_STATIC_THOUSANDS_SEPARATOR)
119 if( loc ) {
120 output.imbue( loc.get<std::locale>() );
121 }
122#endif
123 output << value;
124 output.exceptions( std::ios_base::failbit | std::ios_base::badbit );
125 buf.try_resize( buf.size() );
126}
