103void gate_data::check() const
104{

106 const ter_str_id winch_tid( id.str() );
107
108 if( !winch_tid.is_valid() ) {

110 } else if( winch_tid->examine != controls_gate ) {

124 }
125 }

156// A gate handle is adjacent to a wall section, and next to that wall section on one side or
157// another is the gate. There may be a handle on the other side, but this is optional.
158// The gate continues until it reaches a non-floor tile, so they can be arbitrary length.
159//
160// | !|! -++-++- !|++++-
161// + + ! +

168void gates::open_gate( const tripoint &pos )
169{
170 const gate_id gid = get_gate_id( pos );
171
172 if( !gates_data.is_valid( gid ) ) {
173 return;
174 }
175
176 const gate_data &gate = gates_data.obj( gid );
177
178 bool open = false;
179 bool close = false;
180 bool fail = false;
181
182 for( const point &wall_offset : four_adjacent_offsets ) {
183 const tripoint wall_pos = pos + wall_offset;
184
185 if( !gate.is_suitable_wall( wall_pos ) ) {
186 continue;
187 }
188
189 for( const point &gate_offset : four_adjacent_offsets ) {
190 const tripoint gate_pos = wall_pos + gate_offset;
191
192 if( gate_pos == pos ) {
193 continue; // Never comes back
194 }
195
196 if( !open ) { // Closing the gate...
197 tripoint cur_pos = gate_pos;
198 while( g->m.ter( cur_pos ) == gate.floor.id() ) {
199 fail = !g->forced_door_closing( cur_pos, gate.door.id(), gate.bash_dmg ) || fail;
200 close = !fail;
201 cur_pos += gate_offset;
202 }
203 }
204
205 if( !close ) { // Opening the gate...
206 tripoint cur_pos = gate_pos;
207 while( true ) {
208 const ter_id ter = g->m.ter( cur_pos );
209
210 if( ter == gate.door.id() ) {
211 g->m.ter_set( cur_pos, gate.floor.id() );
212 open = !fail;
213 } else if( ter != gate.floor.id() ) {
214 break;
215 }
216 cur_pos += gate_offset;
217 }
218 }
219 }
220 }
221
222 if( g->u.sees( pos ) ) {
223 if( open ) {
224 add_msg( gate.open_message );
225 } else if( close ) {
226 add_msg( gate.close_message );
227 } else if( fail ) {
228 add_msg( gate.fail_message );
229 } else {
230 add_msg( _( "Nothing happens." ) );
231 }
232 }
233}
