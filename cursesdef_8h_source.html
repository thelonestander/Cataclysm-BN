14/**
15 * Contains the curses interface used by the whole game.
16 *
17 * All code (except platform/OS/build type specific code) should use functions/
18 * types from this namespace only. The @ref input_manager and @ref input_context
19 * should be used for user input.
20 *
21 * There are currently (Nov 2017) two implementations for most of this interface:
22 * - ncurses (mostly in ncurses_def.cpp). The interface originates from there,
23 * so it's mostly just forwarding to ncurses functions of the same name.
24 * - our own curses library @ref cata_cursesport (mostly in cursesport.cpp),
25 * cursesport.h contains the structures and some functions specific to that.
26 *
27 * A few (system specific) functions are implemented in three versions:
28 * - ncurses (ncurses_def.cpp),
29 * - Windows curses (no SDL, using @ref cata_cursesport, see wincurses.cpp), and
30 * - SDL curses (using @ref cata_cursesport, see sdltiles.cpp).
31 *
32 * As this interface is derived from ncurses, refer to documentation of that.
33 *
34 * The interface is in a separate namespace we can link with the ncurses library,
35 * which exports its functions globally.
36 */

44/**
45 * A wrapper over a pointer to a curses window.
46 * All curses functions here receive/return and operate on such a @ref window.
47 * The objects have shared ownership of the contained pointer (behaves like a
48 * shared pointer).
49 *
50 * Use @ref newwin to get a new instance. Use the default constructor to get
51 * an invalid window pointer (test for this via @ref operator bool()).
52 * Use the @ref operator= to reset the pointer.
53 */
54class window
55{
56 private:
57 std::shared_ptr<void> native_window;
58
59 public:
60 window() = default;
61 window( std::shared_ptr<void> ptr ) : native_window( std::move( ptr ) ) {
62 }
63 template<typename T = void>
64 T * get() const {
65 return static_cast<T *>( native_window.get() );
66 }
67 explicit operator bool() const {
68 return native_window.operator bool();
69 }
70 bool operator==( const window &rhs ) const {
71 return native_window.get() == rhs.native_window.get();
72 }
73 std::weak_ptr<void> weak_ptr() const {
74 return native_window;
75 }
76};
