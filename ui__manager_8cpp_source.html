103// This function does two things:
104// 1. Ensure that any UI that would be overwritten by redrawing a lower invalidated
105// UI also gets redrawn.
106// 2. Optimize the invalidated flag so completely occluded UIs will not be redrawn.
107//
108// The current implementation may still invalidate UIs that in fact do not need to
109// be redrawn, but all UIs that need to be redrawn are guaranteed be invalidated.
110void ui_adaptor::invalidation_consistency_and_optimization()
111{
112 // Only ensure consistency and optimize for UIs not disabled by another UI
113 // with `disable_uis_below`, since if a UI is disabled, it does not get
114 // resized or redrawn, so the invalidation flag is not cleared, and including
115 // the disabled UI in the following calculation would unnecessarily
116 // invalidate any upper intersecting UIs.
117 auto rfirst = ui_stack.crbegin();
118 for( ; rfirst != ui_stack.crend(); ++rfirst ) {
119 if( rfirst->get().disabling_uis_below ) {
120 break;
121 }
122 }
123 const auto first = rfirst == ui_stack.crend() ? ui_stack.cbegin() : std::prev( rfirst.base() );
124 for( auto it_upper = first; it_upper < ui_stack.cend(); ++it_upper ) {
125 const ui_adaptor &ui_upper = it_upper->get();
126 for( auto it_lower = first; it_lower < it_upper; ++it_lower ) {
127 const ui_adaptor &ui_lower = it_lower->get();
128 if( !ui_upper.invalidated && ui_lower.invalidated &&
129 overlap( ui_upper.dimensions, ui_lower.dimensions ) ) {
130 // invalidated by lower invalidated UIs
131 ui_upper.invalidated = true;
132 }
133 if( ui_upper.invalidated && ui_lower.invalidated &&
134 contains( ui_upper.dimensions, ui_lower.dimensions ) ) {
135 // fully obscured lower UIs do not need to be redrawn.
136 ui_lower.invalidated = false;
137 // Note: we don't need to re-test ui_lower from earlier iterations
138 // during which ui_upper.invalidated hadn't yet been determined to
139 // be true, because if the ui_lower would be obscured by ui_upper,
140 // it implies that ui_lower would overlap with ui_upper, by which
141 // we would have already determined ui_upper.invalidated to be true
142 // then.
143 }
144 }
145 }

148void ui_adaptor::invalidate_ui() const
149{
150 if( invalidated ) {
151 return;
152 }
153 auto it = ui_stack.cbegin();
154 for( ; it < ui_stack.cend(); ++it ) {
155 if( &it->get() == this ) {
156 break;
157 }
158 }
159 if( it == ui_stack.end() ) {
160 return;
161 }
162 // If an upper UI occludes this UI then nothing gets redrawn
163 for( auto it_upper = std::next( it ); it_upper < ui_stack.cend(); ++it_upper ) {
164 if( contains( it_upper->get().dimensions, dimensions ) ) {
165 return;
166 }
167 }
168 // Always mark this UI for redraw even if it is below another UI with
169 // `disable_uis_below`, so when the UI with `disable_uis_below` is removed,
170 // this UI is correctly marked for redraw.
171 invalidated = true;
172 invalidation_consistency_and_optimization();
173}

182void ui_adaptor::invalidate( const rectangle<point> &rect, const bool reenable_uis_below )
183{
184 if( rect.p_min.x >= rect.p_max.x || rect.p_min.y >= rect.p_max.y ) {
185 if( reenable_uis_below ) {
186 invalidation_consistency_and_optimization();
187 }
188 return;
189 }
190 // Always invalidate every UI, even if it is below another UI with
191 // `disable_uis_below`, so when the UI with `disable_uis_below` is removed,
192 // UIs below are correctly marked for redraw.
193 for( auto it_upper = ui_stack.cbegin(); it_upper < ui_stack.cend(); ++it_upper ) {
194 const ui_adaptor &ui_upper = it_upper->get();
195 if( !ui_upper.invalidated && overlap( ui_upper.dimensions, rect ) ) {
196 // invalidated by `rect`
197 ui_upper.invalidated = true;
198 }
199 }
200 invalidation_consistency_and_optimization();
201}
