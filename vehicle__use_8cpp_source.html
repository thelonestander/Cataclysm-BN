163 if( std::abs( veh->velocity ) < braking_power ) {
164 veh->stop();
165 } else {

498 for( int p : speciality ) {
499 if( part_flag( p, "SECURITY" ) && !parts[ p ].is_broken() ) {

503 }

508 int skill = g->u.get_skill_level( skill_mechanics );
509 int percent_controls = 70 / ( 1 + skill );

512

514 damage_direct( c, part_info( c ).durability / 4 );

516 if( parts[ c ].removed || parts[ c ].is_broken() ) {
517 g->u.controlling_vehicle = false;

522 }
523 }
524 if( percent_alarm > rand ) {
525 damage_direct( s, part_info( s ).durability / 5 );
526 // chance to disable alarm immediately, or disable on destruction
527 if( percent_alarm / 4 > rand || parts[ s ].is_broken() ) {
528 is_alarm_on = false;
529 }
530 }
531 add_msg( ( is_alarm_on ) ? _( "The alarm keeps going." ) : _( "The alarm stops." ) );
532 } else {

885double vehicle::engine_cold_factor( const int e ) const
886{
887 if( !part_info( engines[e] ).has_flag( "E_COLD_START" ) ) {
888 return 0.0;

892 if( !parts[ engines[ e ] ].faults().count( fault_glowplug ) ) {
893 eff_temp = std::min( eff_temp, 20 );
894 }

899int vehicle::engine_start_time( const int e ) const
900{
901 if( !is_engine_on( e ) || part_info( engines[e] ).has_flag( "E_STARTS_INSTANTLY" ) ||
902 !engine_fuel_left( e ) ) {
903 return 0;
904 }
905
906 const double dmg = parts[engines[e]].damage_percent();
907
908 // non-linear range [100-1000]; f(0.0) = 100, f(0.6) = 250, f(0.8) = 500, f(0.9) = 1000
909 // diesel engines with working glow plugs always start with f = 0.6 (or better)
910 const double cold = 100 / tanh( 1 - std::min( engine_cold_factor( e ), 0.9 ) );
911
912 // watts to old vhp = watts / 373
913 // divided by magic 16 = watts / 6000
914 const double watts_per_time = 6000;
915 return part_vpower_w( engines[ e ], true ) / watts_per_time + 100 * dmg + cold;
916}

918bool vehicle::start_engine( const int e )
919{
920 if( !is_engine_on( e ) ) {
921 return false;
922 }
923
924 const vpart_info &einfo = part_info( engines[e] );
925 vehicle_part &eng = parts[ engines[ e ] ];
926
927 bool out_of_fuel = false;
928 if( einfo.fuel_type != fuel_type_none && engine_fuel_left( e ) <= 0 ) {
929 for( const itype_id &fuel_id : einfo.engine_fuel_opts() ) {
930 if( fuel_left( fuel_id ) > 0 ) {
931 eng.fuel_set( fuel_id );
932 break;
933 }
934 }
935 out_of_fuel = true;
936 }
937
938 if( out_of_fuel ) {
939 if( einfo.fuel_type == fuel_type_muscle ) {
940 // Muscle engines cannot start with broken limbs
941 if( einfo.has_flag( "MUSCLE_ARMS" ) && ( g->u.get_working_arm_count() < 2 ) ) {
942 add_msg( _( "You cannot use %s with a broken arm." ), eng.name() );
943 return false;
944 } else if( einfo.has_flag( "MUSCLE_LEGS" ) && ( g->u.get_working_leg_count() < 2 ) ) {
945 add_msg( _( "You cannot use %s with a broken leg." ), eng.name() );
946 return false;
947 }
948 } else {
949 add_msg( _( "Looks like the %1$s is out of %2$s." ), eng.name(),
950 item::nname( einfo.fuel_type ) );
951 return false;
952 }
953 }
954
955 const double dmg = parts[engines[e]].damage_percent();
956 const int engine_power = std::abs( part_epower_w( engines[e] ) );
957 const double cold_factor = engine_cold_factor( e );
958 const int start_moves = engine_start_time( e );
959
960 const tripoint pos = global_part_pos3( engines[e] );
961 if( einfo.engine_backfire_threshold() ) {
962 if( ( 1 - dmg ) < einfo.engine_backfire_threshold() &&
963 one_in( einfo.engine_backfire_freq() ) ) {
964 backfire( e );
965 } else {
966 sounds::sound( pos, start_moves / 10, sounds::sound_t::movement,
967 string_format( _( "the %s bang as it starts" ), eng.name() ), true, "vehicle",
968 "engine_bangs_start" );
969 }
970 }
971
972 // Immobilizers need removing before the vehicle can be started
973 if( eng.faults().count( fault_immobiliser ) ) {
974 sounds::sound( pos, 5, sounds::sound_t::alarm,
975 string_format( _( "the %s making a long beep" ), eng.name() ), true, "vehicle",
976 "fault_immobiliser_beep" );
977 return false;
978 }
979
980 // Engine with starter motors can fail on both battery and starter motor
981 if( eng.faults_potential().count( fault_starter ) ) {
982 if( eng.faults().count( fault_starter ) ) {
983 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::alarm,
984 string_format( _( "the %s clicking once" ), eng.name() ), true, "vehicle",
985 "engine_single_click_fail" );
986 return false;
987 }
988 // TODO: start_moves is in moves, but it's an integer, convert it to some time class
989 const int start_draw_bat = power_to_energy_bat( engine_power *
990 ( 1.0 + dmg / 2 + cold_factor / 5 ) * 10,
991 1_turns * start_moves / 100 );
992 if( discharge_battery( start_draw_bat, true ) != 0 ) {
993 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::alarm,
994 string_format( _( "the %s rapidly clicking" ), eng.name() ), true, "vehicle",
995 "engine_multi_click_fail" );
996 return false;
997 }
998 }
999
1000 // Engines always fail to start with faulty fuel pumps
1001 if( eng.faults().count( fault_pump ) || eng.faults().count( fault_diesel ) ) {
1002 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1003 string_format( _( "the %s quickly stuttering out." ), eng.name() ), true, "vehicle",
1004 "engine_stutter_fail" );
1005 return false;
1006 }
1007
1008 // Damaged non-electric engines have a chance of failing to start
1009 if( !is_engine_type( e, fuel_type_battery ) && x_in_y( dmg * 100, 120 ) ) {
1010 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1011 string_format( _( "the %s clanking and grinding" ), eng.name() ), true, "vehicle",
1012 "engine_clanking_fail" );
1013 return false;
1014 }
1015 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1016 string_format( _( "the %s starting" ), eng.name() ) );
1017
1018 if( sfx::has_variant_sound( "engine_start", eng.info().get_id().str() ) ) {
1019 sfx::play_variant_sound( "engine_start", eng.info().get_id().str(),
1020 eng.info().engine_noise_factor() );
1021 } else if( is_engine_type( e, fuel_type_muscle ) ) {
1022 sfx::play_variant_sound( "engine_start", "muscle", eng.info().engine_noise_factor() );
1023 } else if( is_engine_type( e, fuel_type_wind ) ) {
1024 sfx::play_variant_sound( "engine_start", "wind", eng.info().engine_noise_factor() );
1025 } else if( is_engine_type( e, fuel_type_battery ) ) {
1026 sfx::play_variant_sound( "engine_start", "electric", eng.info().engine_noise_factor() );
1027 } else {
1028 sfx::play_variant_sound( "engine_start", "combustion", eng.info().engine_noise_factor() );
1029 }
1030 return true;
1031}

1055void vehicle::start_engines( const bool take_control, const bool autodrive )
1056{
1057 bool has_engine = std::any_of( engines.begin(), engines.end(), [&]( int idx ) {
1058 return parts[ idx ].enabled && !parts[ idx ].is_broken();
1059 } );
1060
1061 // if no engines enabled then enable all before trying to start the vehicle
1062 if( !has_engine ) {
1063 for( auto idx : engines ) {
1064 if( !parts[ idx ].is_broken() ) {
1065 parts[ idx ].enabled = true;
1066 }
1067 }
1068 }
1069
1070 int start_time = 0;
1071 // record the first usable engine as the referenced position checked at the end of the engine starting activity
1072 bool has_starting_engine_position = false;
1073 tripoint starting_engine_position;
1074 for( size_t e = 0; e < engines.size(); ++e ) {
1075 if( !has_starting_engine_position && !parts[ engines[ e ] ].is_broken() &&
1076 parts[ engines[ e ] ].enabled ) {
1077 starting_engine_position = global_part_pos3( engines[ e ] );
1078 has_starting_engine_position = true;
1079 }
1080 has_engine = has_engine || is_engine_on( e );
1081 start_time = std::max( start_time, engine_start_time( e ) );
1082 }
1083
1084 if( !has_starting_engine_position ) {
1085 starting_engine_position = global_pos3();
1086 }
1087
1088 if( !has_engine ) {
1089 add_msg( m_info, _( "The %s doesn't have an engine!" ), name );
1090 return;
1091 }
1092
1093 if( take_control && !g->u.controlling_vehicle ) {
1094 g->u.controlling_vehicle = true;
1095 add_msg( _( "You take control of the %s." ), name );
1096 }
1097 if( !autodrive ) {
1098 g->u.assign_activity( ACT_START_ENGINES, start_time );
1099 g->u.activity.placement = starting_engine_position - g->u.pos();
1100 g->u.activity.values.push_back( take_control );
1101 }
