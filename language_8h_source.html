8/**
9 * Contains information on a language supported by the game.
10 *
11 * Fields here mirror JSON fields in the language definitions file.
12 */
13struct language_info {
14 /**
15 * Language id.
16 *
17 * Usually in form of `ll` or `ll_CC`
18 * (ll - language, CC - country code).
19 *
20 * Used for autodetection on Linux, and also for resolving paths to
21 * localized files (title screen ascii art, lisst of character/town names).
22 */
23 std::string id;
24
25 /**
26 * Native language name.
27 */
28 std::string name;
29
30 /**
31 * Locale to use on non-Windows platforms.
32 *
33 * If missing, falls back to `en_US.UTF-8`, then to user locale, then to C.
34 * If this language is used as 'System language', this field is ignored
35 * and the game uses user locale (or, if failed, C).
36 *
37 * On Windows, always uses user locale with code page 1252.
38 */
39 std::string locale;
40
41 /**
42 * List of grammatical genders. Default should be first.
43 *
44 * Current choices are m (male), f (female), n (neuter).
45 *
46 * As appropriate we might add e.g. a (animate) or c (common).
47 * New genders must be added to all_genders in lang/extract_json_strings.py
48 * and src/language.cpp.
49 *
50 * The primary purpose of this is for NPC dialogue which might depend on
51 * gender. Only add genders to the extent needed by such translations.
52 * They are likely only needed if they affect the first and second
53 * person. For example, one gender suffices for English even though
54 * third person pronouns differ.
55 *
56 * Defaults to `[ "n" ]` if left empty.
57 */
58 std::vector<std::string> genders;
59
60 /**
61 * OSX autodetection correction (optional field).
62 *
63 * Allows the code to handle special case for Simplified/Traditional Chinese.
64 * Simplified/Traditional is actually denoted by the region code in older
65 * iterations of the language codes, whereas now (at least on OS X)
66 * region is distinct. That is, the game expects 'zh_CN' but OS X might
67 * give 'zh-Hans-CN'.
68 */
69 std::string osx;
70
71 /**
72 * List of language code identifiers (LCIDs) for Windows lang autodetection.
73 *
74 * See `Windows Language Code Identifier (LCID) Reference`
75 * https://docs.microsoft.com/en-us/openspecs/windows_protocols/ms-lcid/70feba9f-294e-491e-b6eb-56532684c37f
76 */
77 std::vector<int> lcids;
78};
