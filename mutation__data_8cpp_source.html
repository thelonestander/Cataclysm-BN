231static mut_attack load_mutation_attack( const JsonObject &jo )
232{
233 mut_attack ret;
234 jo.read( "attack_text_u", ret.attack_text_u );
235 jo.read( "attack_text_npc", ret.attack_text_npc );
236 jo.read( "required_mutations", ret.required_mutations );
237 jo.read( "blocker_mutations", ret.blocker_mutations );
238 jo.read( "hardcoded_effect", ret.hardcoded_effect );
239
240 if( jo.has_string( "body_part" ) ) {
241 ret.bp = get_body_part_token( jo.get_string( "body_part" ) );
242 }
243
244 jo.read( "chance", ret.chance );
245
246 if( jo.has_array( "base_damage" ) ) {
247 ret.base_damage = load_damage_instance( jo.get_array( "base_damage" ) );
248 } else if( jo.has_object( "base_damage" ) ) {
249 JsonObject jo_dam = jo.get_object( "base_damage" );
250 ret.base_damage = load_damage_instance( jo_dam );
251 }
252
253 if( jo.has_array( "strength_damage" ) ) {
254 ret.strength_damage = load_damage_instance( jo.get_array( "strength_damage" ) );
255 } else if( jo.has_object( "strength_damage" ) ) {
256 JsonObject jo_dam = jo.get_object( "strength_damage" );
257 ret.strength_damage = load_damage_instance( jo_dam );
258 }
259
260 if( ret.attack_text_u.empty() || ret.attack_text_npc.empty() ) {
261 jo.throw_error( "Attack message unset" );
262 }
263
264 if( !ret.hardcoded_effect && ret.base_damage.empty() && ret.strength_damage.empty() ) {
265 jo.throw_error( "Damage unset" );
266 } else if( ret.hardcoded_effect && ( !ret.base_damage.empty() || !ret.strength_damage.empty() ) ) {
267 jo.throw_error( "Damage and hardcoded effect are both set (must be one, not both)" );
268 }
269
270 if( ret.chance <= 0 ) {
271 jo.throw_error( "Chance of procing must be set and positive" );
272 }
273
274 return ret;
275}

828void mutation_branch::add_entry( Trait_group &tg, const JsonObject &obj )
829{
830 std::unique_ptr<Trait_creation_data> ptr;
831 int probability = obj.get_int( "prob", 100 );
832 JsonArray jarr;
833
834 if( obj.has_member( "collection" ) ) {
835 ptr = std::make_unique<Trait_group_collection>( probability );
836 jarr = obj.get_array( "collection" );
837 } else if( obj.has_member( "distribution" ) ) {
838 ptr = std::make_unique<Trait_group_distribution>( probability );
839 jarr = obj.get_array( "distribution" );
840 }
841
842 if( ptr ) {
843 Trait_group &tg2 = dynamic_cast<Trait_group &>( *ptr );
844 for( const JsonObject job2 : jarr ) {
845 add_entry( tg2, job2 );
846 }
847 tg.add_entry( std::move( ptr ) );
848 return;
849 }
850
851 if( obj.has_member( "trait" ) ) {
852 trait_id id( obj.get_string( "trait" ) );
853 ptr = std::make_unique<Single_trait_creator>( id, probability );
854 } else if( obj.has_member( "group" ) ) {
855 ptr = std::make_unique<Trait_group_creator>( trait_group::Trait_group_tag(
856 obj.get_string( "group" ) ),
857 probability );
858 }
859
860 if( !ptr ) {
861 return;
862 }
863
864 tg.add_entry( std::move( ptr ) );
865}
