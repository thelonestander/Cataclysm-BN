149static bool has_pre_terrain( const construction &con, const tripoint &p )

152 return true;
153 }

155 if( con.pre_is_furniture ) {

891void place_construction( const std::string &desc )
892{
893 const inventory &total_inv = g->u.crafting_inventory();
894
895 std::vector<construction *> cons = constructions_by_desc( desc );
896 std::map<tripoint, const construction *> valid;
897 for( const tripoint &p : g->m.points_in_radius( g->u.pos(), 1 ) ) {
898 for( const auto *con : cons ) {
899 if( p != g->u.pos() && can_construct( *con, p ) && player_can_build( g->u, total_inv, *con ) ) {
900 valid[ p ] = con;
901 }
902 }
903 }
904
905 shared_ptr_fast<game::draw_callback_t> draw_valid = make_shared_fast<game::draw_callback_t>( [&]() {
906 for( auto &elem : valid ) {
907 g->m.drawsq( g->w_terrain, elem.first, drawsq_params().highlight( true ).show_items( true ) );
908 }
909 } );
910 g->add_draw_callback( draw_valid );
911
912 const cata::optional<tripoint> pnt_ = choose_adjacent( _( "Construct where?" ) );
913 if( !pnt_ ) {
914 return;
915 }
916 const tripoint pnt = *pnt_;
917
918 if( valid.find( pnt ) == valid.end() ) {
919 cons.front()->explain_failure( pnt );
920 return;
921 }
922 // Maybe there is already a partial_con on an existing trap, that isn't caught by the usual trap-checking.
923 // because the pre-requisite construction is already a trap anyway.
924 // This shouldn't normally happen, unless it's a spike pit being built on a pit for example.
925 partial_con *pre_c = g->m.partial_con_at( pnt );
926 if( pre_c ) {
927 add_msg( m_info,
928 _( "There is already an unfinished construction there, examine it to continue working on it" ) );
929 return;
930 }
931 std::list<item> used;
932 const construction &con = *valid.find( pnt )->second;
933 // create the partial construction struct
934 partial_con pc;
935 pc.id = con.id;
936 pc.counter = 0;
937 // Set the trap that has the examine function
938 // Special handling for constructions that take place on existing traps.
939 // Basically just don't add the unfinished construction trap.
940 // TODO: handle this cleaner, instead of adding a special case to pit iexamine.
941 if( g->m.tr_at( pnt ).loadid == tr_null ) {
942 g->m.trap_set( pnt, tr_unfinished_construction );
943 }
944 // Use up the components
945 for( const auto &it : con.requirements->get_components() ) {
946 std::list<item> tmp = g->u.consume_items( it, 1, is_crafting_component );
947 used.splice( used.end(), tmp );
948 }
949 pc.components = used;
950 g->m.partial_con_set( pnt, pc );
951 for( const auto &it : con.requirements->get_tools() ) {
952 g->u.consume_tools( it );
953 }
954 g->u.assign_activity( ACT_BUILD );
955 g->u.activity.placement = g->m.getabs( pnt );
956}

958void complete_construction( player *p )
959{
960 if( !finalized ) {
961 debugmsg( "complete_construction called before finalization" );
962 return;
963 }
964 map &here = get_map();
965 const tripoint terp = here.getlocal( p->activity.placement );
966 partial_con *pc = here.partial_con_at( terp );
967 if( !pc ) {
968 debugmsg( "No partial construction found at activity placement in complete_construction()" );
969 if( here.tr_at( terp ).loadid == tr_unfinished_construction ) {
970 here.remove_trap( terp );
971 }
972 if( p->is_npc() ) {
973 npc *guy = dynamic_cast<npc *>( p );
974 guy->current_activity_id = activity_id::NULL_ID();
975 guy->revert_after_activity();
976 guy->set_moves( 0 );

981 const auto award_xp = [&]( player & c ) {
982 for( const auto &pr : built.required_skills ) {
983 c.practice( pr.first, static_cast<int>( ( 10 + 15 * pr.second ) * ( 1 + built.time / 180000.0 ) ),
984 static_cast<int>( pr.second * 1.25 ) );
985 }
986 };
987
988 award_xp( *p );
989 // Friendly NPCs gain exp from assisting or watching...
990 // TODO: NPCs watching other NPCs do stuff and learning from it
991 if( p->is_player() ) {

1001 }
1002 }

1004 here.remove_trap( terp );
1005 }
1006 here.partial_con_remove( terp );

1008 if( built.post_flags.count( "keep_items" ) == 0 ) {
1009 // Move any items that have found their way onto the construction site.
1010 std::vector<tripoint> dump_spots;
1011 for( const tripoint &pt : here.points_in_radius( terp, 1 ) ) {
1012 if( here.can_put_items( pt ) && pt != terp ) {
1013 dump_spots.push_back( pt );
1014 }
1015 }

1017 tripoint dump_spot = random_entry( dump_spots );
1018 map_stack items = here.i_at( terp );

1020 here.add_item_or_charges( dump_spot, *it );
1021 it = items.erase( it );
1022 }
1023 } else {
1024 debugmsg( "No space to displace items from construction finishing" );
1025 }
1026 }
1027 // Make the terrain change
1028 if( !built.post_terrain.empty() ) {
1029 if( built.post_is_furniture ) {
1030 here.furn_set( terp, furn_str_id( built.post_terrain ) );
1031 active_tile_data *active = active_tiles::furn_at<active_tile_data>(
1032 tripoint_abs_ms( here.getabs( terp ) ) );
1033 if( active != nullptr ) {
1034 active->set_last_updated( calendar::turn );
1035 }
1036 } else {
1037 const ter_id new_ter = ter_str_id( built.post_terrain );
1038 here.ter_set( terp, new_ter );
1039 const tripoint above = terp + tripoint_above;
1040 // TODO: What to do if tile above has no floor, but isn't open air?
1041 if( new_ter->roof && here.ter( above ) == t_open_air ) {
1042 here.ter_set( above, new_ter->roof );
1043 }
1044 }
1045 }
1046
1047 // Spawn byproducts
1048 if( built.byproduct_item_group ) {
1049 here.spawn_items( p->pos(), item_group::items_from( built.byproduct_item_group, calendar::turn ) );
1050 }
1051
1052 add_msg( m_info, _( "%s finished construction: %s." ), p->disp_name(), _( built.description ) );
1053 // clear the activity
1054 p->activity.set_to_null();
1055
1056 // This comes after clearing the activity, in case the function interrupts
1057 // activities
1058 built.post_special( terp );
1059 // npcs will automatically resume backlog, players wont.
1060 if( p->is_player() && !p->backlog.empty() &&
1061 p->backlog.front().id() == ACT_MULTIPLE_CONSTRUCTION ) {
1062 p->backlog.clear();
1063 p->assign_activity( ACT_MULTIPLE_CONSTRUCTION );
1064 }

1723void finalize_constructions()
1724{
1725 std::vector<item_comp> frame_items;
1726 for( const auto &e : vpart_info::all() ) {
1727 const vpart_info &vp = e.second;
1728 if( !vp.has_flag( flag_INITIAL_PART ) ) {
1729 continue;
1730 }
1731 frame_items.push_back( item_comp( vp.item, 1 ) );
1732 }
1733
1734 if( frame_items.empty() ) {
1735 debugmsg( "No valid frames detected for vehicle construction" );
1736 }
1737
1738 for( construction &con : constructions ) {
1739 if( con.vehicle_start ) {
1740 const_cast<requirement_data &>( con.requirements.obj() ).get_components().push_back( frame_items );
1741 }
1742 bool is_valid_construction_category = false;
1743 for( const construction_category &cc : construction_categories::get_all() ) {
1744 if( con.category == cc.id ) {
1745 is_valid_construction_category = true;
1746 break;
1747 }
1748 }
1749 if( !is_valid_construction_category ) {
1750 debugmsg( "Invalid construction category (%s) defined for construction (%s)", con.category.str(),
1751 con.description );
1752 }
1753 requirement_data requirements_ = std::accumulate( con.reqs_using.begin(), con.reqs_using.end(),
1754 *con.requirements,
1755 []( const requirement_data & lhs, const std::pair<requirement_id, int> &rhs ) {
1756 return lhs + ( *rhs.first * rhs.second );
1757 } );
1758
1759 requirement_data::save_requirement( requirements_, con.requirements );
1760 con.reqs_using.clear();
1761 }
1762
1763 constructions.erase( std::remove_if( constructions.begin(), constructions.end(),
1764 [&]( const construction & c ) {
1765 return c.requirements->is_blacklisted();
1766 } ), constructions.end() );
1767
1768 construction_id_map.clear();
1769 for( size_t i = 0; i < constructions.size(); i++ ) {
1770 constructions[ i ].id = construction_id( i );
1771 construction_id_map.emplace( constructions[i].str_id, constructions[i].id );
1772 }
1773
1774 finalized = true;
1775}

1777void get_build_reqs_for_furn_ter_ids( const std::pair<std::map<ter_id, int>,
1778 std::map<furn_id, int>> &changed_ids,
1779 build_reqs &total_reqs )
1780{
1781 if( !finalized ) {
1782 debugmsg( "get_build_reqs_for_furn_ter_ids called before finalization" );
1783 return;
1784 }
1785 std::map<construction_id, int> total_builds;
1786
1787 // iteratively recurse through the pre-terrains until the pre-terrain is empty, adding
1788 // the constructions to the total_builds map
1789 const auto add_builds = [&total_builds]( const construction & build, int count ) {
1790 if( total_builds.find( build.id ) == total_builds.end() ) {
1791 total_builds[build.id] = 0;
1792 }
1793 total_builds[build.id] += count;
1794 std::string build_pre_ter = build.pre_terrain;
1795 while( !build_pre_ter.empty() ) {
1796 for( const construction &pre_build : constructions ) {
1797 if( pre_build.category == construction_category_REPAIR ) {
1798 continue;
1799 }
1800 if( pre_build.post_terrain == build_pre_ter ) {
1801 if( total_builds.find( pre_build.id ) == total_builds.end() ) {
1802 total_builds[pre_build.id] = 0;
