571bool check_special_rule( const std::vector<material_id> &materials, const std::string &rule )
572{
573 char type = ' ';
574 std::vector<std::string> filter;
575 if( rule[1] == ':' ) {
576 type = rule[0];
577 filter = string_split( rule.substr( 2 ), ',' );
578 }
579
580 if( filter.empty() || materials.empty() ) {
581 return false;
582 }
583
584 if( type == 'm' ) {
585 return std::any_of( materials.begin(), materials.end(), [&filter]( const material_id & mat ) {
586 return std::any_of( filter.begin(), filter.end(), [&mat]( const std::string & search ) {
587 return lcmatch( mat->name(), search );
588 } );
589 } );
590
591 } else if( type == 'M' ) {
592 return std::all_of( materials.begin(), materials.end(), [&filter]( const material_id & mat ) {
593 return std::any_of( filter.begin(), filter.end(), [&mat]( const std::string & search ) {
594 return lcmatch( mat->name(), search );
595 } );
596 } );
597 }
598
599 return false;
600}

626void rule_list::create_rule( cache &map_items, const item &it )
627{
628 const std::string to_match = it.tname( 1, false );
629
630 for( const rule &elem : *this ) {
631 if( !elem.bActive ) {
632 continue;
633 } else if( !check_special_rule( it.made_of(), elem.sRule ) &&
634 !wildcard_match( to_match, elem.sRule ) ) {
635 continue;
636 }
637
638 map_items[ to_match ] = elem.bExclude ? RULE_BLACKLISTED : RULE_WHITELISTED;
639 }

642void player_settings::refresh_map_items( cache &map_items ) const
643{
644 //process include/exclude in order of rules, global first, then character specific
645 //if a specific item is being added, all the rules need to be checked now
646 //may have some performance issues since exclusion needs to check all items also
647 global_rules.refresh_map_items( map_items );
648 character_rules.refresh_map_items( map_items );
649}
650
651void rule_list::refresh_map_items( cache &map_items ) const
652{
653 for( const rule &elem : *this ) {
654 if( elem.sRule.empty() || !elem.bActive ) {
655 continue;
656 }
657
658 if( !elem.bExclude ) {
659 //Check include patterns against all itemfactory items
660 for( const itype *e : item_controller->all() ) {
661 const std::string &cur_item = e->nname( 1 );
662
663 if( !check_special_rule( e->materials, elem.sRule ) && !wildcard_match( cur_item, elem.sRule ) ) {
664 continue;
665 }
666
667 map_items[ cur_item ] = RULE_WHITELISTED;
668 map_items.temp_items[ cur_item ] = e;
669 }
670 } else {
671 //only re-exclude items from the existing mapping for now
672 //new exclusions will process during pickup attempts
673 for( auto &map_item : map_items ) {
674 if( !check_special_rule( map_items.temp_items[ map_item.first ]->materials, elem.sRule ) &&
675 !wildcard_match( map_item.first, elem.sRule ) ) {
676 continue;
677 }
678
679 map_items[ map_item.first ] = RULE_BLACKLISTED;
680 }
681 }
682 }
