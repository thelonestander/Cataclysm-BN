488SDL_Rect get_android_render_rect( float DisplayBufferWidth, float DisplayBufferHeight )
489{
490 // If the display buffer aspect ratio is wider than the display,
491 // draw it at the top of the screen so it doesn't get covered up
492 // by the virtual keyboard. Otherwise just center it.
493 SDL_Rect dstrect;
494 float DisplayBufferAspect = DisplayBufferWidth / ( float )DisplayBufferHeight;
495 float WindowHeightLessShortcuts = ( float )WindowHeight;
496 if( !get_option<bool>( "ANDROID_SHORTCUT_OVERLAP" ) && quick_shortcuts_enabled ) {
497 WindowHeightLessShortcuts -= get_option<int>( "ANDROID_SHORTCUT_HEIGHT" );
498 }
499 float WindowAspect = WindowWidth / ( float )WindowHeightLessShortcuts;
500 if( WindowAspect < DisplayBufferAspect ) {
501 dstrect.x = 0;
502 dstrect.y = 0;
503 dstrect.w = WindowWidth;
504 dstrect.h = WindowWidth / DisplayBufferAspect;
505 } else {
506 dstrect.x = 0.5f * ( WindowWidth - ( WindowHeightLessShortcuts * DisplayBufferAspect ) );
507 dstrect.y = 0;
508 dstrect.w = WindowHeightLessShortcuts * DisplayBufferAspect;
509 dstrect.h = WindowHeightLessShortcuts;
510 }
511
512 // Make sure the destination rectangle fits within the visible area
513 if( get_option<bool>( "ANDROID_KEYBOARD_SCREEN_SCALE" ) && has_visible_display_frame ) {
514 int vdf_right = visible_display_frame.x + visible_display_frame.w;
515 int vdf_bottom = visible_display_frame.y + visible_display_frame.h;
516 if( vdf_right < dstrect.x + dstrect.w ) {
517 dstrect.w = vdf_right - dstrect.x;
518 }
519 if( vdf_bottom < dstrect.y + dstrect.h ) {
520 dstrect.h = vdf_bottom - dstrect.y;
521 }
522 }
523 return dstrect;
524}

681static cata::optional<std::pair<tripoint_abs_omt, std::string>> get_mission_arrow(
682 const inclusive_cuboid<tripoint> &overmap_area, const tripoint_abs_omt &center )
683{
684 if( get_avatar().get_active_mission() == nullptr ) {
685 return cata::nullopt;
686 }
687 if( !get_avatar().get_active_mission()->has_target() ) {
688 return cata::nullopt;
689 }
690 const tripoint_abs_omt mission_target = get_avatar().get_active_mission_target();
691
692 std::string mission_arrow_variant;
693 if( overmap_area.contains( mission_target.raw() ) ) {
694 mission_arrow_variant = "mission_cursor";
695 return std::make_pair( mission_target, mission_arrow_variant );
696 }
697
698 inclusive_rectangle<point> area_flat( overmap_area.p_min.xy(), overmap_area.p_max.xy() );
699 if( area_flat.contains( mission_target.raw().xy() ) ) {
700 int area_z = center.z();
701 if( mission_target.z() > area_z ) {
702 mission_arrow_variant = "mission_arrow_up";
703 } else {
704 mission_arrow_variant = "mission_arrow_down";
705 }
706 return std::make_pair( tripoint_abs_omt( mission_target.xy(), area_z ), mission_arrow_variant );
707 }
708
709 const std::vector<tripoint> traj = line_to( center.raw(),
710 tripoint( mission_target.raw().xy(), center.raw().z ) );
711
712 if( traj.empty() ) {
713 debugmsg( "Failed to gen overmap mission trajectory %s %s",
714 center.to_string(), mission_target.to_string() );
715 return cata::nullopt;
716 }
717
718 tripoint arr_pos = traj[0];
719 for( auto it = traj.rbegin(); it != traj.rend(); it++ ) {
720 if( overmap_area.contains( *it ) ) {
721 arr_pos = *it;
722 break;
723 }
724 }
725
726 const int north_border_y = ( overmap_area.p_max.y - overmap_area.p_min.y ) / 3;
727 const int south_border_y = north_border_y * 2;
728 const int west_border_x = ( overmap_area.p_max.x - overmap_area.p_min.x ) / 3;
729 const int east_border_x = west_border_x * 2;
730
731 tripoint north_pmax( overmap_area.p_max );
732 north_pmax.y = overmap_area.p_min.y + north_border_y;
733 tripoint south_pmin( overmap_area.p_min );
734 south_pmin.y += south_border_y;
735 tripoint west_pmax( overmap_area.p_max );
736 west_pmax.x = overmap_area.p_min.x + west_border_x;
737 tripoint east_pmin( overmap_area.p_min );
738 east_pmin.x += east_border_x;
739
740 const inclusive_cuboid<tripoint> north_sector( overmap_area.p_min, north_pmax );
741 const inclusive_cuboid<tripoint> south_sector( south_pmin, overmap_area.p_max );
742 const inclusive_cuboid<tripoint> west_sector( overmap_area.p_min, west_pmax );
743 const inclusive_cuboid<tripoint> east_sector( east_pmin, overmap_area.p_max );
744
745 mission_arrow_variant = "mission_arrow_";
746 if( north_sector.contains( arr_pos ) ) {
747 mission_arrow_variant += 'n';
748 } else if( south_sector.contains( arr_pos ) ) {
749 mission_arrow_variant += 's';
750 }
751 if( west_sector.contains( arr_pos ) ) {
752 mission_arrow_variant += 'w';
753 } else if( east_sector.contains( arr_pos ) ) {
754 mission_arrow_variant += 'e';
755 }
756
757 return std::make_pair( tripoint_abs_omt( arr_pos ), mission_arrow_variant );
758}

760std::string cata_tiles::get_omt_id_rotation_and_subtile(
761 const tripoint_abs_omt &omp, int &rota, int &subtile )
762{
763 auto oter_at = []( const tripoint_abs_omt & p ) {
764 const oter_id &cur_ter = overmap_buffer.ter( p );
765
766 if( !uistate.overmap_show_forest_trails &&
767 is_ot_match( "forest_trail", cur_ter, ot_match_type::type ) ) {
768 return oter_id( "forest" );
769 }
770
771 return cur_ter;
772 };
773
774 oter_id ot_id = oter_at( omp );
775 const oter_t &ot = *ot_id;
776 oter_type_id ot_type_id = ot.get_type_id();
777 oter_type_t ot_type = *ot_type_id;
778
779 if( ot_type.has_connections() ) {
780 // This would be for connected terrain
781
782 // get terrain neighborhood
783 const oter_type_id neighborhood[4] = {
784 oter_at( omp + point_south )->get_type_id(),
785 oter_at( omp + point_east )->get_type_id(),
786 oter_at( omp + point_west )->get_type_id(),
787 oter_at( omp + point_north )->get_type_id()
788 };
789
790 char val = 0;
791
792 // populate connection information
793 for( int i = 0; i < 4; ++i ) {
794 if( ot_type.connects_to( neighborhood[i] ) ) {
795 val += 1 << i;
796 }
797 }
798
799 get_rotation_and_subtile( val, rota, subtile );
800 } else {
801 // 'Regular', nonlinear terrain only needs to worry about rotation, not
802 // subtile
803 ot.get_rotation_and_subtile( rota, subtile );
804 }
805
806 return ot_type_id.id().str();
807}

809static point draw_string( Font &font,
810 const SDL_Renderer_Ptr &renderer,
811 const GeometryRenderer_Ptr &geometry,
812 const std::string &str,
813 point p,
814 const unsigned char color )
815{
816 const char *cstr = str.c_str();
817 int len = str.length();
818 while( len > 0 ) {
819 const uint32_t ch32 = UTF8_getch( &cstr, &len );
820 const std::string ch = utf32_to_utf8( ch32 );
821 font.OutputChar( renderer, geometry, ch, p, color );
822 p.x += mk_wcwidth( ch32 ) * font.width;
823 }
824 return p;
825}

1230static bool draw_window( Font_Ptr &font, const catacurses::window &w, const point &offset )
1231{
1232 if( scaling_factor > 1 ) {
1233 SDL_RenderSetLogicalSize( renderer.get(), WindowWidth / scaling_factor,
1234 WindowHeight / scaling_factor );
1235 }
1236
1237 cata_cursesport::WINDOW *const win = w.get<cata_cursesport::WINDOW>();
1238 //Keeping track of the last drawn window
1239 const cata_cursesport::WINDOW *winBuffer = static_cast<cata_cursesport::WINDOW *>
1240 ( ::winBuffer.lock().get() );
1241 if( !fontScaleBuffer ) {
1242 fontScaleBuffer = tilecontext->get_tile_width();
1243 }
1244 const int fontScale = tilecontext->get_tile_width();
1245 //This creates a problem when map_font is different from the regular font
1246 //Specifically when showing the overmap
1247 //And in some instances of screen change, i.e. inventory.
1248 bool oldWinCompatible = false;
1249
1250 // clear the oversized buffer proportionally
1251 invalidate_framebuffer_proportion( win );
1252
1253 // use the oversize buffer when dealing with windows that can have a different font than the main text font
1254 bool use_oversized_framebuffer = g && ( w == g->w_terrain || w == g->w_overmap );
1255
1256 std::vector<curseline> &framebuffer = use_oversized_framebuffer ? oversized_framebuffer :
1257 terminal_framebuffer;
1258
1259 /*
1260 Let's try to keep track of different windows.
1261 A number of windows are coexisting on the screen, so don't have to interfere.
1262
1263 g->w_terrain, g->w_minimap, g->w_HP, g->w_status, g->w_status2, g->w_messages,
1264 g->w_location, and g->w_minimap, can be buffered if either of them was
1265 the previous window.
1266
1267 g->w_overmap and g->w_omlegend are likewise.
1268
1269 Everything else works on strict equality because there aren't yet IDs for some of them.
1270 */
1271 if( g && ( w == g->w_terrain || w == g->w_minimap ) ) {
1272 if( winBuffer == g->w_terrain || winBuffer == g->w_minimap ) {
1273 oldWinCompatible = true;
1274 }
1275 } else if( g && ( w == g->w_overmap || w == g->w_omlegend ) ) {
1276 if( winBuffer == g->w_overmap || winBuffer == g->w_omlegend ) {
1277 oldWinCompatible = true;
1278 }
1279 } else {
1280 if( win == winBuffer ) {
1281 oldWinCompatible = true;
1282 }
1283 }
1284
1285 // TODO: Get this from UTF system to make sure it is exactly the kind of space we need
1286 static const std::string space_string = " ";
1287
1288 bool update = false;
1289 for( int j = 0; j < win->height; j++ ) {
1290 if( !win->line[j].touched ) {
1291 continue;
1292 }
1293
1294 const int fby = win->pos.y + j;
1295 if( fby >= static_cast<int>( framebuffer.size() ) ) {
1296 // prevent indexing outside the frame buffer. This might happen for some parts of the window. FIX #28953.
1297 break;
1298 }
1299
1300 update = true;
1301 win->line[j].touched = false;
1302 for( int i = 0; i < win->width; i++ ) {
1303 const int fbx = win->pos.x + i;
1304 if( fbx >= static_cast<int>( framebuffer[fby].chars.size() ) ) {
1305 // prevent indexing outside the frame buffer. This might happen for some parts of the window.
1306 break;
1307 }
1308
1309 const cursecell &cell = win->line[j].chars[i];
1310
1311 const int drawx = offset.x + i * font->width;
1312 const int drawy = offset.y + j * font->height;
1313 if( drawx + font->width > WindowWidth || drawy + font->height > WindowHeight ) {
1314 // Outside of the display area, would not render anyway
1315 continue;
1316 }
1317
1318 // Avoid redrawing an unchanged tile by checking the framebuffer cache
1319 // TODO: handle caching when drawing normal windows over graphical tiles
1320 cursecell &oldcell = framebuffer[fby].chars[fbx];
1321
1322 if( oldWinCompatible && cell == oldcell && fontScale == fontScaleBuffer ) {
1323 continue;
1324 }
1325 oldcell = cell;
1326
1327 if( cell.ch.empty() ) {
1328 continue; // second cell of a multi-cell character
1329 }
1330
1331 // Spaces are used a lot, so this does help noticeably
1332 if( cell.ch == space_string ) {
1333 geometry->rect( renderer, point( drawx, drawy ), font->width, font->height,
1334 color_as_sdl( cell.BG ) );
1335 continue;
1336 }
1337 const int codepoint = UTF8_getch( cell.ch );
1338 const catacurses::base_color FG = cell.FG;
1339 const catacurses::base_color BG = cell.BG;
1340 int cw = ( codepoint == UNKNOWN_UNICODE ) ? 1 : utf8_width( cell.ch );
1341 if( cw < 1 ) {
1342 // utf8_width() may return a negative width
1343 continue;
1344 }
1345 bool use_draw_ascii_lines_routine = get_option<bool>( "USE_DRAW_ASCII_LINES_ROUTINE" );
1346 unsigned char uc = static_cast<unsigned char>( cell.ch[0] );
1347 switch( codepoint ) {
1348 case LINE_XOXO_UNICODE:
1349 uc = LINE_XOXO_C;
1350 break;
1351 case LINE_OXOX_UNICODE:
1352 uc = LINE_OXOX_C;
1353 break;
1354 case LINE_XXOO_UNICODE:
1355 uc = LINE_XXOO_C;
1356 break;
1357 case LINE_OXXO_UNICODE:
1358 uc = LINE_OXXO_C;
1359 break;
1360 case LINE_OOXX_UNICODE:
1361 uc = LINE_OOXX_C;
1362 break;
1363 case LINE_XOOX_UNICODE:
1364 uc = LINE_XOOX_C;
1365 break;
1366 case LINE_XXXO_UNICODE:
1367 uc = LINE_XXXO_C;
1368 break;
1369 case LINE_XXOX_UNICODE:
1370 uc = LINE_XXOX_C;
1371 break;
1372 case LINE_XOXX_UNICODE:
1373 uc = LINE_XOXX_C;
1374 break;
1375 case LINE_OXXX_UNICODE:
1376 uc = LINE_OXXX_C;
1377 break;
1378 case LINE_XXXX_UNICODE:
1379 uc = LINE_XXXX_C;
1380 break;
1381 case UNKNOWN_UNICODE:
1382 use_draw_ascii_lines_routine = true;
1383 break;
1384 default:
1385 use_draw_ascii_lines_routine = false;
1386 break;
1387 }
1388 geometry->rect( renderer, point( drawx, drawy ), font->width * cw, font->height,
1389 color_as_sdl( BG ) );
1390 if( use_draw_ascii_lines_routine ) {
1391 font->draw_ascii_lines( renderer, geometry, uc, point( drawx, drawy ), FG );
1392 } else {
1393 font->OutputChar( renderer, geometry, cell.ch, point( drawx, drawy ), FG );
1394 }
1395 }
1396 }
1397 win->draw = false; //We drew the window, mark it as so
1398 //Keeping track of last drawn window and tilemode zoom level
1399 ::winBuffer = w.weak_ptr();
1400 fontScaleBuffer = tilecontext->get_tile_width();
1401
1402 return update;

1593static int HandleDPad()
1594{
1595 // Check if we have a gamepad d-pad event.
1596 if( SDL_JoystickGetHat( joystick, 0 ) != SDL_HAT_CENTERED ) {
1597 // When someone tries to press a diagonal, they likely will
1598 // press a single direction first. Wait a few milliseconds to
1599 // give them time to press both of the buttons for the diagonal.
1600 int button = SDL_JoystickGetHat( joystick, 0 );
1601 int lc = ERR;
1602 if( button == SDL_HAT_LEFT ) {
1603 lc = JOY_LEFT;
1604 } else if( button == SDL_HAT_DOWN ) {
1605 lc = JOY_DOWN;
1606 } else if( button == SDL_HAT_RIGHT ) {
1607 lc = JOY_RIGHT;
1608 } else if( button == SDL_HAT_UP ) {
1609 lc = JOY_UP;
1610 } else if( button == SDL_HAT_LEFTUP ) {
1611 lc = JOY_LEFTUP;
1612 } else if( button == SDL_HAT_LEFTDOWN ) {
1613 lc = JOY_LEFTDOWN;
1614 } else if( button == SDL_HAT_RIGHTUP ) {
1615 lc = JOY_RIGHTUP;
1616 } else if( button == SDL_HAT_RIGHTDOWN ) {
1617 lc = JOY_RIGHTDOWN;
1618 }
1619
1620 if( delaydpad == std::numeric_limits<Uint32>::max() ) {
1621 delaydpad = SDL_GetTicks() + dpad_delay;
1622 queued_dpad = lc;
1623 }
1624
1625 // Okay it seems we're ready to process.
1626 if( SDL_GetTicks() > delaydpad ) {
1627
1628 if( lc != ERR ) {
1629 if( dpad_continuous && ( lc & lastdpad ) == 0 ) {
1630 // Continuous movement should only work in the same or similar directions.
1631 dpad_continuous = false;
1632 lastdpad = lc;
1633 return 0;
1634 }
1635
1636 last_input = input_event( lc, CATA_INPUT_GAMEPAD );
1637 lastdpad = lc;
1638 queued_dpad = ERR;
1639
1640 if( !dpad_continuous ) {
1641 delaydpad = SDL_GetTicks() + 200;
1642 dpad_continuous = true;
1643 } else {
1644 delaydpad = SDL_GetTicks() + 60;
1645 }
1646 return 1;
1647 }
1648 }
1649 } else {
1650 dpad_continuous = false;
1651 delaydpad = std::numeric_limits<Uint32>::max();
1652
1653 // If we didn't hold it down for a while, just
1654 // fire the last registered press.
1655 if( queued_dpad != ERR ) {
1656 last_input = input_event( queued_dpad, CATA_INPUT_GAMEPAD );
1657 queued_dpad = ERR;
1658 return 1;
1659 }
1660 }
1661
1662 return 0;
1663}

1737/**

1739 * allows the input_manager to only consider those.
1740 * @return 0 if the input can not be translated (unknown key?),
1741 * -1 when a ALT+number sequence has been started,
1742 * or something that a call to ncurses getch would return.
1743 */
1744static int sdl_keysym_to_curses( const SDL_Keysym &keysym )
1745{
1746
1747 const std::string diag_mode = get_option<std::string>( "DIAG_MOVE_WITH_MODIFIERS_MODE" );
1748
1749 if( diag_mode == "mode1" ) {
1750 if( keysym.mod & KMOD_CTRL && sdl_keycode_is_arrow( keysym.sym ) ) {
1751 return handle_arrow_combo( keysym.sym );
1752 } else {
1753 end_arrow_combo();
1754 }
1755 }

1758 //Shift + Cursor Arrow (diagonal clockwise)
1759 if( keysym.mod & KMOD_SHIFT ) {
1760 switch( keysym.sym ) {
1761 case SDLK_LEFT:
1762 return inp_mngr.get_first_char_for_action( "LEFTUP" );
1763 case SDLK_RIGHT:
1764 return inp_mngr.get_first_char_for_action( "RIGHTDOWN" );
1765 case SDLK_UP:
1766 return inp_mngr.get_first_char_for_action( "RIGHTUP" );
1767 case SDLK_DOWN:
1768 return inp_mngr.get_first_char_for_action( "LEFTDOWN" );
1769 }
1770 }
1771 //Ctrl + Cursor Arrow (diagonal counter-clockwise)
1772 if( keysym.mod & KMOD_CTRL ) {

2572 if( dist > ( get_option<float>( "ANDROID_DEADZONE_RANGE" )*std::max( WindowWidth,
2573 WindowHeight ) ) ) {

2576 last_input = input_event( delta_x > delta_y ? KEY_RIGHT : KEY_DOWN, CATA_INPUT_KEYBOARD );
2577 } else if( delta_x < 0 && delta_y >= 0 ) {
2578 last_input = input_event( -delta_x > delta_y ? KEY_LEFT : KEY_DOWN, CATA_INPUT_KEYBOARD );
2579 } else if( delta_x >= 0 && delta_y < 0 ) {
2580 last_input = input_event( delta_x > -delta_y ? KEY_RIGHT : KEY_UP, CATA_INPUT_KEYBOARD );
2581 } else if( delta_x < 0 && delta_y < 0 ) {
2582 last_input = input_event( -delta_x > -delta_y ? KEY_LEFT : KEY_UP, CATA_INPUT_KEYBOARD );
2583 }
2584 } else {

2587 // swipe right
2588 last_input = input_event( KEY_RIGHT, CATA_INPUT_KEYBOARD );

2590 if( delta_y < 0 ) {
2591 // swipe up-right
2592 last_input = input_event( JOY_RIGHTUP, CATA_INPUT_GAMEPAD );

2594 // swipe down-right
2595 last_input = input_event( JOY_RIGHTDOWN, CATA_INPUT_GAMEPAD );
2596 }
2597 } else {
2598 if( delta_y < 0 ) {
2599 // swipe up
2600 last_input = input_event( KEY_UP, CATA_INPUT_KEYBOARD );
2601 } else {
2602 // swipe down
2603 last_input = input_event( KEY_DOWN, CATA_INPUT_KEYBOARD );
2604 }
2605 }
2606 } else {
2607 if( std::abs( delta_y ) < -delta_x * 0.5f ) {

2609 last_input = input_event( KEY_LEFT, CATA_INPUT_KEYBOARD );

2614
2615 } else {
2616 // swipe down-left

2618 }

2641 ticks - last_tap_time < static_cast<uint32_t>( get_option<int>( "ANDROID_INITIAL_DELAY" ) ) ) {

2649 }
