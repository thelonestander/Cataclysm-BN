214 }
215 // flood_buf is filled with correct tiles; now copy them back to g->m
216 for( const tripoint &p : g->m.points_on_zlevel() ) {
217 g->m.ter_set( p, flood_buf[p.x][p.y] );

240void timed_event::per_turn()
241{
242 switch( type ) {
243 case TIMED_EVENT_WANTED: {
244 // About once every 5 minutes. Suppress in classic zombie mode.
245 if( g->get_levz() >= 0 && one_in( 50 ) && !get_option<bool>( "DISABLE_ROBOT_RESPONSE" ) ) {
246 point place = g->m.random_outdoor_tile();
247 if( place.x == -1 && place.y == -1 ) {
248 // We're safely indoors!
249 return;
250 }
251 g->place_critter_at( mon_eyebot, tripoint( place, g->u.posz() ) );
252 if( g->u.sees( tripoint( place, g->u.posz() ) ) ) {
253 add_msg( m_warning, _( "An eyebot swoops down nearby!" ) );
254 }
255 // One eyebot per trigger is enough, really
256 when = calendar::turn;
257 }
258 }
259 break;
260
261 case TIMED_EVENT_SPAWN_WYRMS:
262 if( g->get_levz() >= 0 ) {
263 when -= 1_turns;
264 return;
265 }
266 if( calendar::once_every( time_duration::from_seconds( rng( 2, 3 ) ) ) && !g->u.is_deaf() ) {
267 add_msg( m_warning, _( "You hear screeches from the rock above and around you!" ) );
268 }
269 break;
270
271 case TIMED_EVENT_AMIGARA:
272 if( calendar::once_every( time_duration::from_seconds( rng( 2, 3 ) ) ) ) {
273 add_msg( m_warning, _( "The entire cavern shakes!" ) );
274 }
275 break;
276
277 case timed_event_type::AMIGARA_WHISPERS: {
278 bool faults = false;
279 for( const tripoint &p : g->m.points_on_zlevel() ) {
280 if( g->m.ter( p ) == t_fault ) {
281 faults = true;
282 break;
283 }
284 }
285
286 if( calendar::once_every( time_duration::from_seconds( 10 ) ) && faults ) {
287 add_msg( m_info, "You hear someone whispering \"%s\"",
288 SNIPPET.random_from_category( "amigara_whispers" ).value_or( translation() ) );
289 }
290 }
291 break;
292
293 case TIMED_EVENT_TEMPLE_OPEN:
294 if( calendar::once_every( time_duration::from_seconds( rng( 2, 3 ) ) ) ) {
295 add_msg( m_warning, _( "The earth rumbles." ) );
296 }
297 break;
298
299 default:
300 // Nothing happens for other events
301 break;
302 }

305void timed_event_manager::process()
306{
307 for( auto it = events.begin(); it != events.end(); ) {
308 it->per_turn();
309 if( it->when <= calendar::turn ) {
310 it->actualize();
311 it = events.erase( it );

313 it++;
314 }
