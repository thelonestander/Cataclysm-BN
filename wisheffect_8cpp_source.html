420void effect_edit_menu( Creature &c )
421{
422 bool not_dirty = false;
423 bool stay_in_menu = true;
424 int last_selected = 0;
425 while( stay_in_menu ) {
426 const effects_map &em = c.get_all_effects();
427 uilist menu;
428 menu.input_category = "edit_creature_effects";
429 std::map<bodypart_str_id, std::vector<const effect *>> bp_effects;
430 const std::vector<bodypart_id> &all_bps = c.get_all_body_parts();
431 for( const auto &pr : em ) {
432 for( const auto &inner : pr.second ) {
433 bp_effects[inner.first].push_back( &inner.second );
434 if( !inner.first.is_null() &&
435 std::count( all_bps.begin(), all_bps.end(), inner.first ) == 0 ) {
436 debugmsg( "Creature %s has effect on body part %s, not in anatomy",
437 c.get_name(), inner.first.str() );
438 }
439 }
440 }
441
442 std::vector<entry_data> meta;
443 size_t i = 0;
444 const auto add_effect_entry = [&menu, &i]( const effect & eff ) {
445 // TODO: Columns
446 std::string effect_description = string_format(
447 "- %s %d, %d s",
448 eff.get_id().str(),
449 eff.get_intensity(),
450 to_seconds<int>( eff.get_duration() ) );
451 menu.addentry( i++, true, 0, effect_description );
452 };
453 menu.addentry( i++, true, -1, _( "Global" ) );
454 auto global_iter = bp_effects.find( bodypart_str_id::NULL_ID() );
455 meta.emplace_back( bodypart_str_id::NULL_ID() );
456 if( global_iter != bp_effects.end() ) {
457 for( const effect *eff : global_iter->second ) {
458 add_effect_entry( *eff );
459 meta.emplace_back( bodypart_str_id::NULL_ID(), *eff );
460 }
461
462 }
463 for( const bodypart_id &bp : all_bps ) {
464 menu.addentry( i++, true, -1, bp.id().str() );
465 meta.emplace_back( bp.id() );
466 auto iter = bp_effects.find( bp.id() );
467 if( iter != bp_effects.end() ) {
468 for( const effect *eff : iter->second ) {
469 add_effect_entry( *eff );
470 meta.emplace_back( bp.id(), *eff );
471 }
472 }
473 }
474
475 const auto on_creature_changed = [&not_dirty]() {
476 not_dirty = false;
477 };
478 effect_edit_callback callback( c, meta, on_creature_changed );
479 register_callback( menu, callback );
480 not_dirty = true;
481 while( not_dirty ) {
482 // Have to do this because menus don't support entry removal
483 menu.set_selected( last_selected );
484 menu.query( false );
485 if( menu.ret == UILIST_CANCEL ) {
486 stay_in_menu = false;
487 break;
488 }
489 last_selected = menu.selected;
490 }
491 }
