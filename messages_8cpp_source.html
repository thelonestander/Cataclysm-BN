200 /** Check if the current message needs to be prevented (hidden) or not from being displayed in the side bar.
201 * @param message The message to be checked.
202 */
203 void hide_message_in_cooldown( game_message &message ) {
204 message.cooldown_hidden = false;
205
206 if( message_cooldown <= 0 || message.turn() <= calendar::start_of_cataclysm ) {
207 return;
208 }
209
210 // We look for **exactly the same** message string in the cooldown templates
211 // If there is one, this means the same message was already displayed.
212 const auto cooldown_it = std::find_if( cooldown_templates.begin(), cooldown_templates.end(),
213 [&message]( game_message & m_cooldown ) -> bool {
214 return m_cooldown.message == message.message;
215 } );
216 if( cooldown_it == cooldown_templates.end() ) {
217 // nothing found, not in cooldown.
218 return;
219 }
220
221 // note: from this point the current message (`message`) has the same string than one of the active cooldown template messages (`cooldown_it`).
222
223 // check how much times this message has been seen during its cooldown.
224 // If it's only one time, then no need to hide it.
225 if( cooldown_it->cooldown_seen == 1 ) {
226 return;
227 }
228
229 // check if it's the message that started the cooldown timer.
230 if( message.turn() == cooldown_it->turn() ) {
231 return;
232 }
233
234 // current message turn.
235 const auto cm_turn = to_turn<int>( message.turn() );
236 // maximum range of the cooldown timer.
237 const auto max_cooldown_range = to_turn<int>( cooldown_it->turn() ) + message_cooldown;
238 // If the current message is in the cooldown range then hide it.
239 if( cm_turn <= max_cooldown_range ) {
240 message.cooldown_hidden = true;
241 }
242 }

244 std::vector<std::pair<std::string, std::string>> recent_messages( size_t count ) const {

256 } );

259 }

261 /** Refresh the cooldown timers, removing elapsed ones and making new ones if needed.
262 * @param message The current message that needs to be checked.
263 * @param flags Flags pertaining to the message.
264 */
265 void refresh_cooldown( const game_message &message, const game_message_flags flags ) {
266 // is cooldown used? (also checks for messages arriving here at game initialization: we don't care about them).
267 if( message_cooldown <= 0 || message.turn() <= calendar::start_of_cataclysm ) {
268 return;
269 }
270
271 // housekeeping: remove any cooldown message with an expired cooldown time from the cooldown queue.
272 const auto now = calendar::turn;
273 for( auto it = cooldown_templates.begin(); it != cooldown_templates.end(); ) {
274 // number of turns elapsed since the cooldown started.
275 const auto turns = to_turns<int>( now - it->turn() );
276 if( turns >= message_cooldown ) {
277 // time elapsed! remove it.
278 it = cooldown_templates.erase( it );
279 } else {
280 ++it;
281 }
282 }
283
284 // do not hide messages which bypasses cooldown.
285 if( ( flags & gmf_bypass_cooldown ) != 0 ) {
286 return;
287 }
288
289 // Is the message string already in the cooldown queue?
290 // If it's not we must put it in the cooldown queue now, otherwise just increment the number of times we have seen it.
291 const auto cooldown_message_it = std::find_if( cooldown_templates.begin(),
292 cooldown_templates.end(), [&message]( game_message & cooldown_message ) -> bool {
293 return cooldown_message.message == message.message;
294 } );
295 if( cooldown_message_it == cooldown_templates.end() ) {
296 // push current message to cooldown message templates.
297 cooldown_templates.emplace_back( message );
298 } else {
299 // increment the number of time we have seen this message.
300 cooldown_message_it->cooldown_seen++;
301 }
302 }

648void Messages::dialog::do_filter( const std::string &filter_str )
649{
650 // Split the search string into type and text
651 bool has_type_filter = false;
652 game_message_type filter_type = m_neutral;
653 std::string filter_text;
654 const auto colon = filter_str.find( ':' );
655 if( colon != std::string::npos ) {
656 has_type_filter = msg_type_from_name( filter_type, filter_str.substr( 0, colon ) );
657 filter_text = filter_str.substr( colon + 1 );
658 } else {
659 filter_text = filter_str;
660 }
661
662 // Start filtering the log
663 folded_filtered.clear();
664 for( size_t folded_ind = 0; folded_ind < folded_all.size(); ) {
665 const size_t msg_ind = folded_all[folded_ind].first;
666 const game_message &msg = player_messages.history( msg_ind );
667 const bool match = ( !has_type_filter || filter_type == msg.type ) &&
668 ci_find_substr( remove_color_tags( msg.get_with_count() ), filter_text ) >= 0;
669
670 // Always advance the index, but only add to filtered list if the original message matches
671 for( ; folded_ind < folded_all.size() && folded_all[folded_ind].first == msg_ind; ++folded_ind ) {
672 if( match ) {
673 folded_filtered.emplace_back( folded_ind );
674 }
675 }
676 }
677
678 // Reset view
679 if( log_from_top || max_lines > folded_filtered.size() ) {
680 offset = 0;
681 } else {
682 offset = folded_filtered.size() - max_lines;
683 }

697 if( !filter.canceled() ) {
698 const std::string &new_filter_str = filter.text();
699 if( new_filter_str != filter_str ) {
700 filter_str = new_filter_str;
701
702 do_filter( filter_str );
703 }
704 } else {
705 filter.text( filter_str );
706 }
707 } else {
708 const std::string &action = ctxt.handle_input();
709 if( action == "DOWN" && offset + max_lines < folded_filtered.size() ) {
710 ++offset;
711 } else if( action == "UP" && offset > 0 ) {
712 --offset;
713 } else if( action == "PAGE_DOWN" ) {
714 if( offset + max_lines * 2 <= folded_filtered.size() ) {
715 offset += max_lines;
716 } else if( max_lines <= folded_filtered.size() ) {
717 offset = folded_filtered.size() - max_lines;
718 } else {
719 offset = 0;
720 }
721 } else if( action == "PAGE_UP" ) {
722 if( offset >= max_lines ) {
723 offset -= max_lines;
724 } else {
725 offset = 0;
726 }
727 } else if( action == "FILTER" ) {
728 filtering = true;
729 if( filter_sentry ) {
730 enable_ime();
731 } else {
732 // this implies enable_ime() and ensures that the ime mode is always
733 // restored when closing the dialog if at least filtered once
734 filter_sentry.emplace();
735 }
736 } else if( action == "RESET_FILTER" ) {
737 filter_str.clear();
738 filter.text( filter_str );
739 do_filter( filter_str );
740 } else if( action == "QUIT" ) {
741 canceled = true;
742 }
743 }

781 const auto &col_name = get_all_colors().get_name( msgtype_to_color( it->first ) );
782 auto next_it = std::next( it );
783 // Skip m_debug outside debug mode
784 if( !debug_mode && next_it != type_list.end() && next_it->first == m_debug ) {
785 next_it = std::next( next_it );
786 }
787 if( next_it != type_list.end() ) {
788 //~ the 2nd %s is a type name, this is used to format a list of type names
789 type_text += string_format( pgettext( "message log", "<color_%s>%s</color>, " ),
790 col_name, pgettext( "message type", it->second ) );
791 } else {
792 //~ the 2nd %s is a type name, this is used to format the last type name in a list of type names
793 type_text += string_format( pgettext( "message log", "<color_%s>%s</color>." ),
794 col_name, pgettext( "message type", it->second ) );
795 }
796 }
797 }

808void Messages::display_messages( const catacurses::window &ipk_target, const int left,
809 const int top, const int right, const int bottom )
810{
811 if( !size() ) {
812 return;
813 }
814
815 const int maxlength = right - left;
816 int line = log_from_top ? top : bottom;
817
818 if( log_from_top ) {
819 for( int i = size() - 1; i >= 0; --i ) {
820 if( line > bottom ) {
821 break;
822 }
823
824 const game_message &m = player_messages.messages[i];
825 if( message_exceeds_ttl( m ) ) {
826 break;
827 }
828
829 const nc_color col = m.get_color( player_messages.curmes );
830 std::string message_text = m.get_with_count();
831 if( !m.is_recent( player_messages.curmes ) ) {
832 message_text = remove_color_tags( message_text );
833 }
834
835 for( const std::string &folded : foldstring( message_text, maxlength ) ) {
836 if( line > bottom ) {
837 break;
838 }
839 // Redrawing line to ensure new messages similar to previous
840 // messages will not be missed by screen readers
841 wredrawln( ipk_target, line, 1 );
842 nc_color col_out = col;
843 print_colored_text( ipk_target, point( left, line++ ), col_out, col, folded );
844 }
845 }
846 } else {
847 for( int i = size() - 1; i >= 0; --i ) {
848 if( line < top ) {
849 break;
850 }
851
852 const game_message &m = player_messages.messages[i];
853 if( message_exceeds_ttl( m ) ) {
854 break;
855 }
856
857 if( m.is_in_cooldown() ) {
858 // message is still (or was at some point) into a cooldown period.
859 continue;
860 }
861
862 const nc_color col = m.get_color( player_messages.curmes );
863 std::string message_text = m.get_with_count();
864 if( !m.is_recent( player_messages.curmes ) ) {
865 message_text = remove_color_tags( message_text );
866 }
867
868 const auto folded_strings = foldstring( message_text, maxlength );
869 const auto folded_rend = folded_strings.rend();
870 for( auto string_iter = folded_strings.rbegin();
871 string_iter != folded_rend && line >= top; ++string_iter, line-- ) {
872 // Redrawing line to ensure new messages similar to previous
873 // messages will not be missed by screen readers
874 wredrawln( ipk_target, line, 1 );
875 nc_color col_out = col;
876 print_colored_text( ipk_target, point( left, line ), col_out, col, *string_iter );
877 }
878 }
879 }
880
881 player_messages.curmes = calendar::turn;
882}
