269 void destroy_all_data( const node_pointer_type last_endpoint_node ) noexcept {
270 if( block_pointer == nullptr ) {
271 return;
272 }
273
274 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ||
275 !std::is_trivially_destructible<node_pointer_type>::value ) {
276 clear( last_endpoint_node ); // If clear has already been called, last_endpoint_node will already be == block_pointer->nodes, so no work will occur
277 }
278
279 const group_pointer_type end_group = block_pointer + size;
280 for( group_pointer_type current_group = block_pointer; current_group != end_group;
281 ++current_group ) {
282 LIST_DESTROY( group_allocator_type, group_allocator_pair, current_group );
283 }
284
285 LIST_DEALLOCATE( group_allocator_type, group_allocator_pair, block_pointer,
286 group_allocator_pair.capacity );
287 blank();
288 }

290 void clear( const node_pointer_type last_endpoint_node ) noexcept {
291 for( group_pointer_type current_group = block_pointer; current_group != last_endpoint_group;
292 ++current_group ) {
293 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ||
294 !std::is_trivially_destructible<node_pointer_type>::value ) {
295 const node_pointer_type end = current_group->beyond_end;
296
297 if( ( end - current_group->nodes ) != current_group->number_of_elements ) {
298 // If there are erased nodes present in the group
299 for( node_pointer_type current_node = current_group->nodes; current_node != end; ++current_node ) {
300 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
301 if( current_node->next != nullptr ) { // ie. is not part of free list
302 LIST_DESTROY( element_allocator_type, element_allocator_pair, &( current_node->element ) );
303 }
304 }
305
306 if LIST_CONSTEXPR( !std::is_trivially_destructible<node_pointer_type>::value ) {
307 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->next ) );
308 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->previous ) );
309 }
310 }
311 } else {
312 for( node_pointer_type current_node = current_group->nodes; current_node != end; ++current_node ) {
313 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
314 LIST_DESTROY( element_allocator_type, element_allocator_pair, &( current_node->element ) );
315 }
316
317 if LIST_CONSTEXPR( !std::is_trivially_destructible<node_pointer_type>::value ) {
318 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->next ) );
319 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->previous ) );
320 }
321 }
322 }
323 }
324
325 current_group->free_list_head = nullptr;
326 current_group->number_of_elements = 0;
327 }
328
329 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ||
330 !std::is_trivially_destructible<node_pointer_type>::value ) {
331 if( ( last_endpoint_node - last_endpoint_group->nodes ) !=
332 last_endpoint_group->number_of_elements ) {
333 // If there are erased nodes present in the group
334 for( node_pointer_type current_node = last_endpoint_group->nodes;
335 current_node != last_endpoint_node; ++current_node ) {
336 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
337 if( current_node->next != nullptr ) {
338 // is not part of free list ie. element has not already had it's destructor called
339 LIST_DESTROY( element_allocator_type, element_allocator_pair, &( current_node->element ) );
340 }
341 }
342
343 if LIST_CONSTEXPR( !std::is_trivially_destructible<node_pointer_type>::value ) {
344 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->next ) );
345 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->previous ) );
346 }
347 }
348 } else {
349 for( node_pointer_type current_node = last_endpoint_group->nodes;
350 current_node != last_endpoint_node; ++current_node ) {
351 if LIST_CONSTEXPR( !std::is_trivially_destructible<element_type>::value ) {
352 LIST_DESTROY( element_allocator_type, element_allocator_pair, &( current_node->element ) );
353 }
354
355 if LIST_CONSTEXPR( !std::is_trivially_destructible<node_pointer_type>::value ) {
356 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->next ) );
357 LIST_DESTROY( node_pointer_allocator_type, ( *this ), &( current_node->previous ) );
358 }
359 }
360 }
361 }
362
363 last_endpoint_group->free_list_head = nullptr;
364 last_endpoint_group->number_of_elements = 0;
365 last_searched_group = last_endpoint_group = block_pointer;
366 }

368 void expand_capacity( const size_type new_capacity ) { // used by add_new and append
369 group_pointer_type const old_block = block_pointer;
370 block_pointer = LIST_ALLOCATE( group_allocator_type, group_allocator_pair, new_capacity, nullptr );
371
372 if LIST_CONSTEXPR( std::is_trivially_copyable<node_pointer_type>::value &&
373 std::is_trivially_destructible<node_pointer_type>::value ) {
374 // Dereferencing here in order to deal with smart pointer situations i.e. obtaining the raw pointer from the smart pointer
375 // reinterpret_cast necessary to deal with GCC 8 warnings
376 std::memcpy( static_cast<void *>( &*block_pointer ), static_cast<void *>( &*old_block ),
377 sizeof( group ) * size );
378 } else if LIST_CONSTEXPR( std::is_move_constructible<node_pointer_type>::value ) {
379 std::uninitialized_copy( std::make_move_iterator( old_block ),
380 std::make_move_iterator( old_block + size ), block_pointer );
381 } else {
382 // If allocator supplies non-trivial pointers it becomes necessary to destroy the group. uninitialized_copy will not work in this context as the copy constructor for "group" is overridden in C++03/98. The = operator for "group" has been overridden to make the following work:
383 const group_pointer_type beyond_end = old_block + size;
384 group_pointer_type current_new_group = block_pointer;
385
386 for( group_pointer_type current_group = old_block; current_group != beyond_end; ++current_group ) {
387 *( current_new_group++ ) = *( current_group );
388
389 current_group->nodes = nullptr;
390 current_group->beyond_end = nullptr;
391 LIST_DESTROY( group_allocator_type, group_allocator_pair, current_group );
392 }
393 }
394
395 // correct pointer post-reallocation
396 last_searched_group = block_pointer + ( last_searched_group - old_block );
397 LIST_DEALLOCATE( group_allocator_type, group_allocator_pair, old_block,
398 group_allocator_pair.capacity );
399 group_allocator_pair.capacity = new_capacity;
400 }

429 void remove( group_pointer_type const group_to_erase ) noexcept {
430 if( last_searched_group >= group_to_erase && last_searched_group != block_pointer ) {
431 --last_searched_group;
432 }
433
434 element_allocator_pair.capacity -= static_cast<size_type>( group_to_erase->beyond_end -
435 group_to_erase->nodes );
436
437 LIST_DESTROY( group_allocator_type, group_allocator_pair, group_to_erase );
438
439 if LIST_CONSTEXPR( std::is_trivially_copyable<node_pointer_type>::value &&
440 std::is_trivially_destructible<node_pointer_type>::value ) {
441 // Dereferencing here in order to deal with smart pointer situations ie. obtaining the raw pointer from the smart pointer
442 std::memmove( static_cast<void *>( &*group_to_erase ), static_cast<void *>( &*group_to_erase + 1 ),
443 sizeof( group ) * ( --size - static_cast<size_type>( &*group_to_erase - &*block_pointer ) ) );
444 } else if LIST_CONSTEXPR( std::is_move_constructible<node_pointer_type>::value ) {
445 std::move( group_to_erase + 1, block_pointer + size--, group_to_erase );
446 } else {
447 group_pointer_type back = block_pointer + size--;
448 std::copy( group_to_erase + 1, back--, group_to_erase );
449
450 back->nodes = nullptr;
451 back->beyond_end = nullptr;
452 LIST_DESTROY( group_allocator_type, group_allocator_pair, back );
453 }

456 void move_to_back( group_pointer_type const group_to_erase ) {
457 if( last_searched_group >= group_to_erase && last_searched_group != block_pointer ) {
458 --last_searched_group;
459 }
460
461 group *temp_group = LIST_ALLOCATE( group_allocator_type, group_allocator_pair, 1, nullptr );
462
463 if LIST_CONSTEXPR( std::is_trivially_copyable<node_pointer_type>::value &&
464 std::is_trivially_destructible<node_pointer_type>::value ) {
465 std::memcpy( static_cast<void *>( &*temp_group ), static_cast<void *>( &*group_to_erase ),
466 sizeof( group ) );
467 std::memmove( static_cast<void *>( &*group_to_erase ), static_cast<void *>( &*group_to_erase + 1 ),
468 sizeof( group ) * ( ( size - 1 ) - static_cast<size_type>( &*group_to_erase - &*block_pointer ) ) );
469 std::memcpy( static_cast<void *>( &*( block_pointer + size - 1 ) ),
470 static_cast<void *>( &*temp_group ), sizeof( group ) );
471 } else if LIST_CONSTEXPR( std::is_move_constructible<node_pointer_type>::value ) {
472 LIST_CONSTRUCT( group_allocator_type, group_allocator_pair, temp_group,
473 std::move( *group_to_erase ) );
474 std::move( group_to_erase + 1, block_pointer + size, group_to_erase );
475 *( block_pointer + size - 1 ) = std::move( *temp_group );
476
477 if LIST_CONSTEXPR( !std::is_trivially_destructible<node_pointer_type>::value ) {
478 LIST_DESTROY( group_allocator_type, group_allocator_pair, temp_group );
479 }
480 } else {
481 LIST_CONSTRUCT( group_allocator_type, group_allocator_pair, temp_group, group() );
482
483 *temp_group = *group_to_erase;
484 std::copy( group_to_erase + 1, block_pointer + size, group_to_erase );
485 *( block_pointer + --size ) = *temp_group;
486
487 temp_group->nodes = nullptr;
488 LIST_DESTROY( group_allocator_type, group_allocator_pair, temp_group );
489 }
490
491 LIST_DEALLOCATE( group_allocator_type, group_allocator_pair, temp_group, 1 );
492 }

494 // In working implementation this cannot throw
495 group_pointer_type get_nearest_freelist_group( const node_pointer_type location_node ) noexcept {
496 const group_pointer_type beyond_end_group = last_endpoint_group + 1;
497 group_pointer_type left = last_searched_group - 1, right = last_searched_group + 1,
498 freelist_group = nullptr;
499 bool right_not_beyond_back = ( right < beyond_end_group );
500 bool left_not_beyond_front = ( left >= block_pointer );
501
502 // ie. location is within last_search_group
503 if( location_node >= last_searched_group->nodes &&
504 location_node < last_searched_group->beyond_end ) {
505 // if last_searched_group has previously-erased nodes
506 if( last_searched_group->free_list_head != nullptr ) {
507 return last_searched_group;
508 }
509 } else { // search for the node group which location_node is located within, using last_searched_group as a starting point and searching left and right. Try and find the closest node group with reusable erased-element locations along the way:
510 group_pointer_type closest_freelist_left = ( last_searched_group->free_list_head == nullptr ) ?
511 nullptr :
512 last_searched_group, closest_freelist_right = ( last_searched_group->free_list_head == nullptr ) ?
513 nullptr : last_searched_group;
514
515 while( true ) {
516 if( right_not_beyond_back ) {
517 // location_node's group is found
518 if( ( location_node < right->beyond_end ) && ( location_node >= right->nodes ) ) {
519 // group has erased nodes, reuse them:
520 if( right->free_list_head != nullptr ) {
521 last_searched_group = right;
522 return right;
523 }
524
525 difference_type left_distance;
526
527 if( closest_freelist_right != nullptr ) {
528 last_searched_group = right;
529 left_distance = right - closest_freelist_right;
530
531 if( left_distance <= 2 ) { // ie. this group is close enough to location_node's group
532 return closest_freelist_right;
533 }
534
535 freelist_group = closest_freelist_right;
536 } else {
537 last_searched_group = right;
538 left_distance = right - left;
539 }
540
541 // Otherwise find closest group with freelist - check an equal distance on the right to the distance we've checked on the left:
542 const group_pointer_type end_group = ( ( ( right + left_distance ) > beyond_end_group ) ?
543 beyond_end_group : ( right + left_distance - 1 ) );
544
545 while( ++right != end_group ) {
546 if( right->free_list_head != nullptr ) {
547 return right;
548 }
549 }
550
551 if( freelist_group != nullptr ) {
552 return freelist_group;
553 }
554
555 right_not_beyond_back = ( right < beyond_end_group );
556 break; // group with reusable erased nodes not found yet, continue searching in loop below
557 }
558
559 // location_node's group not found, but a reusable location found
560 if( right->free_list_head != nullptr ) {
561 if( ( closest_freelist_right == nullptr ) & ( closest_freelist_left == nullptr ) ) {
562 closest_freelist_left = right;
563 }
564
565 closest_freelist_right = right;
566 }
567
568 right_not_beyond_back = ( ++right < beyond_end_group );
569 }
570
571 if( left_not_beyond_front ) {
572 if( ( location_node >= left->nodes ) && ( location_node < left->beyond_end ) ) {
573 if( left->free_list_head != nullptr ) {
574 last_searched_group = left;
575 return left;
576 }
577
578 difference_type right_distance;
579
580 if( closest_freelist_left != nullptr ) {
581 last_searched_group = left;
582 right_distance = closest_freelist_left - left;
583
584 if( right_distance <= 2 ) {
585 return closest_freelist_left;
586 }
587
588 freelist_group = closest_freelist_left;
589 } else {
590 last_searched_group = left;
591 right_distance = right - left;
592 }
593
594 // Otherwise find closest group with freelist:
595 const group_pointer_type end_group = ( ( ( left - right_distance ) < block_pointer ) ? block_pointer
596 - 1 : ( left - right_distance ) + 1 );
597
598 while( --left != end_group ) {
599 if( left->free_list_head != nullptr ) {
600 return left;
601 }
602 }
603
604 if( freelist_group != nullptr ) {
605 return freelist_group;
606 }
607
608 left_not_beyond_front = ( left >= block_pointer );
609 break;
610 }
611
612 if( left->free_list_head != nullptr ) {
613 if( ( closest_freelist_left == nullptr ) & ( closest_freelist_right == nullptr ) ) {
614 closest_freelist_right = left;
615 }
616
617 closest_freelist_left = left;
618 }
619
620 left_not_beyond_front = ( --left >= block_pointer );
621 }
622 }
623 }
624
625 // The node group which location_node is located within, is known at this point. Continue searching outwards from this group until a group is found with a reusable location:
626 while( true ) {
627 if( right_not_beyond_back ) {
628 if( right->free_list_head != nullptr ) {
629 return right;
630 }
631
632 right_not_beyond_back = ( ++right < beyond_end_group );
633 }
634
635 if( left_not_beyond_front ) {
636 if( left->free_list_head != nullptr ) {
637 return left;
638 }
639
640 left_not_beyond_front = ( --left >= block_pointer );
641 }
642 }
643
644 // Will never reach here on functioning implementations
645 }

647 void swap( group_vector &source ) LIST_NOEXCEPT_SWAP( group_allocator_type ) {
648 if LIST_CONSTEXPR(
649 std::is_trivial<group_pointer_type>::value ) { // if all pointer types are trivial we can just copy using memcpy - faster - avoids constructors/destructors etc
650 char temp[sizeof( group_vector )];
651 std::memcpy( static_cast<void *>( &temp ), static_cast<void *>( this ), sizeof( group_vector ) );
652 std::memcpy( static_cast<void *>( this ), static_cast<void *>( &source ), sizeof( group_vector ) );
653 std::memcpy( static_cast<void *>( &source ), static_cast<void *>( &temp ), sizeof( group_vector ) );
654 } else {
655 const group_pointer_type swap_last_endpoint_group = last_endpoint_group,
656 swap_block_pointer = block_pointer, swap_last_searched_group = last_searched_group;
657 const size_type swap_size = size, swap_element_capacity = element_allocator_pair.capacity,
658 swap_capacity = group_allocator_pair.capacity;
659
660 last_endpoint_group = source.last_endpoint_group;
661 block_pointer = source.block_pointer;
662 last_searched_group = source.last_searched_group;
663 size = source.size;
664 element_allocator_pair.capacity = source.element_allocator_pair.capacity;
665 group_allocator_pair.capacity = source.group_allocator_pair.capacity;
666
667 source.last_endpoint_group = swap_last_endpoint_group;
668 source.block_pointer = swap_block_pointer;
669 source.last_searched_group = swap_last_searched_group;
670 source.size = swap_size;
671 source.element_allocator_pair.capacity = swap_element_capacity;
672 source.group_allocator_pair.capacity = swap_capacity;
673 }

676 void trim_trailing_groups() noexcept {
677 const group_pointer_type beyond_last = block_pointer + size;
678
679 for( group_pointer_type current_group = last_endpoint_group + 1; current_group != beyond_last;
680 ++current_group ) {
681 element_allocator_pair.capacity -= static_cast<size_type>( current_group->beyond_end -
682 current_group->nodes );
683 LIST_DESTROY( group_allocator_type, group_allocator_pair, current_group );
684 }
685
686 size -= static_cast<size_type>( beyond_last - ( last_endpoint_group + 1 ) );
687 }

689 void append( group_vector &source ) {
690 source.trim_trailing_groups();
691 trim_trailing_groups();
692
693 if( size + source.size > group_allocator_pair.capacity ) {
694 expand_capacity( size + source.size );
695 }
696
697 if LIST_CONSTEXPR( std::is_trivially_copyable<node_pointer_type>::value &&
698 std::is_trivially_destructible<node_pointer_type>::value ) {
699 // &* in order to deal with smart pointer situations ie. obtaining the raw pointer from the smart pointer
700 std::memcpy( static_cast<void *>( &*block_pointer + size ),
701 static_cast<void *>( &*source.block_pointer ), sizeof( group ) * source.size );
702 } else if LIST_CONSTEXPR( std::is_move_constructible<node_pointer_type>::value ) {
703 std::uninitialized_copy( std::make_move_iterator( source.block_pointer ),
704 std::make_move_iterator( source.block_pointer + source.size ), block_pointer + size );
705 } else {
706 group_pointer_type current_new_group = block_pointer + size;
707 const group_pointer_type beyond_end_source = source.block_pointer + source.size;
708
709 for( group_pointer_type current_group = source.block_pointer; current_group != beyond_end_source;
710 ++current_group ) {
711 *( current_new_group++ ) = *( current_group );
712
713 current_group->nodes = nullptr;
714 current_group->beyond_end = nullptr;
715 LIST_DESTROY( group_allocator_type, source.group_allocator_pair, current_group );
716 }
717 }
718
719 LIST_DEALLOCATE( group_allocator_type, source.group_allocator_pair, source.block_pointer,
720 source.group_allocator_pair.capacity );
721 size += source.size;
722 last_endpoint_group = block_pointer + size - 1;
723 element_allocator_pair.capacity += source.element_allocator_pair.capacity;
724 source.blank();
725 }

1520 // Fill insert
1521 iterator insert( iterator position, const size_type number_of_elements,
1522 const element_type &element ) {
1523 if( number_of_elements == 0 ) {
1524 return end_iterator;
1525 } else if( number_of_elements == 1 ) {
1526 return insert( position, element );
1527 }
1528
1529 if( node_pointer_allocator_pair.total_number_of_elements == 0 && last_endpoint != nullptr &&
1530 ( static_cast<size_type>( groups.block_pointer->beyond_end - groups.block_pointer->nodes ) <
1531 number_of_elements ) &&
1532 ( static_cast<size_type>( groups.block_pointer->beyond_end - groups.block_pointer->nodes ) <
1533 LIST_BLOCK_MAX ) ) {
1534 reset();
1535 }
1536
1537 if( groups.block_pointer == nullptr ) { // ie. Uninitialized list
1538 if( number_of_elements > LIST_BLOCK_MAX ) {
1539 size_type multiples = number_of_elements / LIST_BLOCK_MAX;
1540 const group_size_type remainder = static_cast<group_size_type>( number_of_elements -
1541 ( multiples++ * LIST_BLOCK_MAX ) ); // ++ to aid while loop below
1542
1543 // Create and fill first group:
1544 if( remainder != 0 ) { // make sure smallest block is first
1545 if( remainder >= LIST_BLOCK_MIN ) {
1546 groups.initialize( remainder );
1547 end_node.next = end_node.previous = last_endpoint = begin_iterator.node_pointer =
1548 groups.last_endpoint_group->nodes;
1549 group_fill_position( element, remainder, end_iterator.node_pointer );
1550 } else {
1551 // Create first group as BLOCK_MIN size then subtract difference between BLOCK_MIN and remainder from next group:
1552 groups.initialize( LIST_BLOCK_MIN );
1553 end_node.next = end_node.previous = last_endpoint = begin_iterator.node_pointer =
1554 groups.last_endpoint_group->nodes;
1555 group_fill_position( element, LIST_BLOCK_MIN, end_iterator.node_pointer );
1556
1557 groups.add_new( LIST_BLOCK_MAX - ( LIST_BLOCK_MIN - remainder ) );
1558 end_node.previous = last_endpoint = groups.last_endpoint_group->nodes;
1559 group_fill_position( element, LIST_BLOCK_MAX - ( LIST_BLOCK_MIN - remainder ),
1560 end_iterator.node_pointer );
1561 --multiples;
1562 }
1563 } else {
1564 groups.initialize( LIST_BLOCK_MAX );
1565 end_node.next = end_node.previous = last_endpoint = begin_iterator.node_pointer =
1566 groups.last_endpoint_group->nodes;
1567 group_fill_position( element, LIST_BLOCK_MAX, end_iterator.node_pointer );
1568 --multiples;
1569 }
1570
1571 while( --multiples != 0 ) {
1572 groups.add_new( LIST_BLOCK_MAX );
1573 end_node.previous = last_endpoint = groups.last_endpoint_group->nodes;
1574 group_fill_position( element, LIST_BLOCK_MAX, end_iterator.node_pointer );
1575 }
1576
1577 } else {
1578 groups.initialize( ( number_of_elements < LIST_BLOCK_MIN ) ? LIST_BLOCK_MIN :
1579 static_cast<group_size_type>( number_of_elements ) ); // Construct first group
1580 end_node.next = end_node.previous = last_endpoint = begin_iterator.node_pointer =
1581 groups.last_endpoint_group->nodes;
1582 group_fill_position( element, static_cast<group_size_type>( number_of_elements ),
1583 end_iterator.node_pointer );
1584 }
1585
1586 node_pointer_allocator_pair.total_number_of_elements = number_of_elements;
1587 return begin_iterator;
1588 } else {
1589 // Insert first element, then use up any erased nodes:
1590 size_type remainder = number_of_elements - 1;
1591 const iterator return_iterator = insert( position, element );
1592
1593 while( node_allocator_pair.number_of_erased_nodes != 0 ) {
1594 insert( position, element );
1595 --node_allocator_pair.number_of_erased_nodes;
1596
1597 if( --remainder == 0 ) {
1598 return return_iterator;
1599 }
1600 }
1601
1602 node_pointer_allocator_pair.total_number_of_elements += remainder;
1603
1604 // then use up remainder of last_endpoint_group:
1605 const group_size_type remaining_nodes_in_group = static_cast<group_size_type>
1606 ( groups.last_endpoint_group->beyond_end - last_endpoint );
1607
1608 if( remaining_nodes_in_group != 0 ) {
1609 if( remaining_nodes_in_group < remainder ) {
1610 group_fill_position( element, remaining_nodes_in_group, position.node_pointer );
1611 remainder -= remaining_nodes_in_group;
1612 } else {
1613 group_fill_position( element, static_cast<group_size_type>( remainder ), position.node_pointer );
1614 return return_iterator;
1615 }
1616 }
1617
1618 // use up trailing groups:
1619 while( ( groups.last_endpoint_group != ( groups.block_pointer + groups.size - 1 ) ) &
1620 ( remainder != 0 ) ) {
1621 last_endpoint = ( ++groups.last_endpoint_group )->nodes;
1622 const group_size_type group_size = static_cast<group_size_type>
1623 ( groups.last_endpoint_group->beyond_end - groups.last_endpoint_group->nodes );
1624
1625 if( group_size < remainder ) {
1626 group_fill_position( element, group_size, position.node_pointer );
1627 remainder -= group_size;
1628 } else {
1629 group_fill_position( element, static_cast<group_size_type>( remainder ), position.node_pointer );
1630 return return_iterator;
1631 }
1632 }
1633
1634 size_type multiples = remainder / static_cast<size_type>( LIST_BLOCK_MAX );
1635 remainder -= multiples * LIST_BLOCK_MAX;
1636
1637 while( multiples-- != 0 ) {
1638 groups.add_new( LIST_BLOCK_MAX );
1639 last_endpoint = groups.last_endpoint_group->nodes;
1640 group_fill_position( element, LIST_BLOCK_MAX, position.node_pointer );
1641 }
1642
1643 if( remainder !=
1644 0 ) { // Bit annoying to create a large block to house a lower number of elements, but beats the alternatives
1645 groups.add_new( LIST_BLOCK_MAX );
1646 last_endpoint = groups.last_endpoint_group->nodes;
1647 group_fill_position( element, static_cast<group_size_type>( remainder ), position.node_pointer );
1648 }
1649
1650 return return_iterator;
1651 }

1694 // Single erase:
1695 // if uninitialized/invalid iterator supplied, function could generate an exception, hence no noexcept
1696 iterator erase( const const_iterator it ) {
1697 assert( node_pointer_allocator_pair.total_number_of_elements != 0 );
1698 assert( it.node_pointer != nullptr );
1699 assert( it.node_pointer != end_iterator.node_pointer );
1700
1701 if LIST_CONSTEXPR( !( std::is_trivially_destructible<element_type>::value ) ) {
1702 LIST_DESTROY( element_allocator_type, ( *this ),
1703 &( it.node_pointer->element ) ); // Destruct element
1704 }
1705
1706 --node_pointer_allocator_pair.total_number_of_elements;
1707 ++node_allocator_pair.number_of_erased_nodes;
1708
1709 group_pointer_type node_group = groups.last_searched_group;
1710
1711 // find nearest group with reusable (erased element) memory location:
1712 if( ( it.node_pointer < node_group->nodes ) || ( it.node_pointer >= node_group->beyond_end ) ) {
1713 // Search left and right:
1714 const group_pointer_type beyond_end_group = groups.last_endpoint_group + 1;
1715 group_pointer_type left = node_group - 1;
1716 bool right_not_beyond_back = ( ++node_group < beyond_end_group );
1717 bool left_not_beyond_front = ( left >= groups.block_pointer );
1718
1719 while( true ) {
1720 if( right_not_beyond_back ) {
1721 if( ( it.node_pointer < node_group->beyond_end ) &&
1722 ( it.node_pointer >= node_group->nodes ) ) { // usable location found
1723 break;
1724 }
1725
1726 right_not_beyond_back = ( ++node_group < beyond_end_group );
1727 }
1728
1729 if( left_not_beyond_front ) {
1730 if( ( it.node_pointer >= left->nodes ) &&
1731 ( it.node_pointer < left->beyond_end ) ) { // usable location found
1732 node_group = left;
1733 break;
1734 }
1735
1736 left_not_beyond_front = ( --left >= groups.block_pointer );
1737 }
1738 }
1739
1740 groups.last_searched_group = node_group;
1741 }
1742
1743 it.node_pointer->next->previous = it.node_pointer->previous;
1744 it.node_pointer->previous->next = it.node_pointer->next;
1745
1746 if( it.node_pointer == begin_iterator.node_pointer ) {
1747 begin_iterator.node_pointer = it.node_pointer->next;
1748 }
1749
1750 const iterator return_iterator( it.node_pointer->next );
1751
1752 // ie. group is not empty yet, add node to free list
1753 if( --( node_group->number_of_elements ) != 0 ) {
1754 // next == NULL so that destructor can detect the free list item as opposed to non-free-list item
1755 it.node_pointer->next = nullptr;
1756 it.node_pointer->previous = node_group->free_list_head;
1757 node_group->free_list_head = it.node_pointer;
1758 return return_iterator;
1759 } else if( node_group != groups.last_endpoint_group-- ) {
1760 // remove group (and decrement active back group)
1761 const group_size_type group_size = static_cast<group_size_type>( node_group->beyond_end -
1762 node_group->nodes );
1763 node_allocator_pair.number_of_erased_nodes -= group_size;
1764
1765 if LIST_CONSTEXPR( !( std::is_trivially_destructible<node_pointer_type>::value ) ) {
1766 destroy_all_node_pointers( node_group, node_group->beyond_end );
1767 }
1768
1769 node_group->free_list_head = nullptr;
