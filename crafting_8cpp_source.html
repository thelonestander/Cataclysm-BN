131float player::lighting_craft_speed_multiplier( const recipe &rec ) const
132{
133 // negative is bright, 0 is just bright enough, positive is dark, +7.0f is pitch black
134 float darkness = fine_detail_vision_mod() - 4.0f;
135 if( darkness <= 0.0f ) {
136 return 1.0f; // it's bright, go for it
137 }
138 bool rec_blind = rec.has_flag( flag_BLIND_HARD ) || rec.has_flag( flag_BLIND_EASY );
139 if( darkness > 0 && !rec_blind ) {
140 return 0.0f; // it's dark and this recipe can't be crafted in the dark
141 }
142 if( rec.has_flag( flag_BLIND_EASY ) ) {
143 // 100% speed in well lit area at skill+0
144 // 25% speed in pitch black at skill+0
145 // skill+2 removes speed penalty
146 return 1.0f - ( darkness / ( 7.0f / 0.75f ) ) * std::max( 0,
147 2 - exceeds_recipe_requirements( rec ) ) / 2.0f;
148 }
149 if( rec.has_flag( flag_BLIND_HARD ) && exceeds_recipe_requirements( rec ) >= 2 ) {
150 // 100% speed in well lit area at skill+2
151 // 25% speed in pitch black at skill+2
152 // skill+8 removes speed penalty
153 return 1.0f - ( darkness / ( 7.0f / 0.75f ) ) * std::max( 0,
154 8 - exceeds_recipe_requirements( rec ) ) / 6.0f;
155 }
156 return 0.0f; // it's dark and you could craft this if you had more skill
157}

159float player::morale_crafting_speed_multiplier( const recipe &rec ) const
160{
161 int morale = get_morale_level();
162 if( morale >= 0 ) {
163 // No bonus for being happy yet
164 return 1.0f;
165 }
166
167 // Harder jobs are more frustrating, even when skilled
168 // For each skill where skill=difficulty, multiply effective morale by 200%
169 float morale_mult = std::max( 1.0f, 2.0f * rec.difficulty / std::max( 1,
170 get_skill_level( rec.skill_used ) ) );
171 for( const auto &pr : rec.required_skills ) {
172 morale_mult *= std::max( 1.0f, 2.0f * pr.second / std::max( 1, get_skill_level( pr.first ) ) );
173 }
174
175 // Halve speed at -50 effective morale, quarter at -150
176 float morale_effect = 1.0f + ( morale_mult * morale ) / -50.0f;
177
178 return 1.0f / morale_effect;
179}

181template<typename T>
182static float lerped_multiplier( const T &value, const T &low, const T &high )
183{
184 // No effect if less than allowed value
185 if( value < low ) {
186 return 1.0f;
187 }
188 // Bottom out at 25% speed
189 if( value > high ) {
190 return 0.25f;
191 }
192 // Linear interpolation between high and low
193 // y = y0 + ( x - x0 ) * ( y1 - y0 ) / ( x1 - x0 )
194 return 1.0f + ( value - low ) * ( 0.25f - 1.0f ) / ( high - low );
195}

198{
199 float multiplier = 0.0f;
200 units::mass allowed_mass;
201 units::volume allowed_volume;
202
203 const units::mass &craft_mass = craft.weight();
204 const units::volume &craft_volume = craft.volume();
205
206 // The whole block below is so ugly because all the benches have different structs with same content
207 switch( bench.type ) {
208 case bench_type::hands: {
209 const furn_t &f = string_id<furn_t>( "f_fake_bench_hands" ).obj();
210 multiplier = f.workbench->multiplier;
211 allowed_mass = f.workbench->allowed_mass;
212 allowed_volume = f.workbench->allowed_volume;
213 }
214 break;
215 case bench_type::ground: {
216 // Ground - we can always use this, but it's bad
217 const furn_t &f = string_id<furn_t>( "f_ground_crafting_spot" ).obj();
218 multiplier = f.workbench->multiplier;
219 allowed_mass = f.workbench->allowed_mass;
220 allowed_volume = f.workbench->allowed_volume;
221 }
222 break;
223 case bench_type::furniture:
224 if( g->m.furn( bench.position ).obj().workbench ) {
225 // Furniture workbench
226 const furn_t &f = g->m.furn( bench.position ).obj();
227 multiplier = f.workbench->multiplier;
228 allowed_mass = f.workbench->allowed_mass;
229 allowed_volume = f.workbench->allowed_volume;
230 } else {
231 return 0.0f;
232 }
233 break;
234 case bench_type::vehicle:
235 if( const cata::optional<vpart_reference> vp = g->m.veh_at(
236 bench.position ).part_with_feature( "WORKBENCH", true ) ) {
237 // Vehicle workbench
238 const vpart_info &vp_info = vp->part().info();
239 if( const cata::optional<vpslot_workbench> &wb_info = vp_info.get_workbench_info() ) {
240 multiplier = wb_info->multiplier;

242 allowed_volume = wb_info->allowed_volume;
243 } else {
244 debugmsg( "part '%S' with WORKBENCH flag has no workbench info", vp->part().name() );
245 return 0.0f;

251 return 0.0f;
252 }

254
255 multiplier *= lerped_multiplier( craft_mass, allowed_mass, 1000_kilogram );
256 multiplier *= lerped_multiplier( craft_volume, allowed_volume, 1000_liter );
257
258 return multiplier;

277float crafting_speed_multiplier( const player &p, const item &craft, const bench_location &bench )
278{
279 if( !craft.is_craft() ) {
280 debugmsg( "Can't calculate crafting speed multiplier of non-craft '%s'", craft.tname() );
281 return 1.0f;
282 }
283
284 const recipe &rec = craft.get_making();
285
286 const float light_multi = p.lighting_craft_speed_multiplier( rec );
287 const float bench_multi = workbench_crafting_speed_multiplier( craft, bench );
288 const float morale_multi = p.morale_crafting_speed_multiplier( rec );
289
290 const float total_multi = light_multi * bench_multi * morale_multi;
291
292 if( light_multi <= 0.0f ) {
293 p.add_msg_if_player( m_bad, _( "You can no longer see well enough to keep crafting." ) );
294 return 0.0f;
295 }
296 if( bench_multi <= 0.1f || ( bench_multi <= 0.33f && total_multi <= 0.2f ) ) {
297 p.add_msg_if_player( m_bad, _( "The %s is too large and/or heavy to work on. You may want to"
298 " use a workbench or a smaller batch size" ), craft.tname() );
299 return 0.0f;
300 }
301 if( morale_multi <= 0.2f || ( morale_multi <= 0.33f && total_multi <= 0.2f ) ) {
302 p.add_msg_if_player( m_bad, _( "Your morale is too low to continue crafting." ) );
303 return 0.0f;
304 }
305
306 // If we're working below 20% speed, just give up
307 if( total_multi <= 0.2f ) {
308 p.add_msg_if_player( m_bad, _( "You are too frustrated to continue and just give up." ) );
309 return 0.0f;
310 }
311
312 if( calendar::once_every( 1_hours ) && total_multi < 0.75f ) {
313 if( light_multi <= 0.5f ) {
314 p.add_msg_if_player( m_bad, _( "You can't see well and are working slowly." ) );
315 }
316 if( bench_multi <= 0.5f ) {
317 p.add_msg_if_player( m_bad,
318 _( "The %s is to large and/or heavy to work on comfortably. You are"
319 " working slowly." ), craft.tname() );
320 }
321 if( morale_multi <= 0.5f ) {
322 p.add_msg_if_player( m_bad, _( "You can't focus and are working slowly." ) );
323 }
324 }
325
326 return total_multi;

407bool player::check_eligible_containers_for_crafting( const recipe &rec, int batch_size ) const
408{
409 std::vector<const item *> conts = get_eligible_containers_for_crafting();
410 const std::vector<item> res = rec.create_results( batch_size );
411 const std::vector<item> bps = rec.create_byproducts( batch_size );
412 std::vector<item> all;
413 all.reserve( res.size() + bps.size() );
414 all.insert( all.end(), res.begin(), res.end() );
415 all.insert( all.end(), bps.begin(), bps.end() );
416
417 for( const item &prod : all ) {
418 if( !prod.made_of( LIQUID ) ) {
419 continue;
420 }
421
422 // we go through half-filled containers first, then go through empty containers if we need
423 std::sort( conts.begin(), conts.end(), item_ptr_compare_by_charges );
424
425 int charges_to_store = prod.charges;
426 for( const item *cont : conts ) {
427 if( charges_to_store <= 0 ) {
428 break;
429 }
430
431 if( !cont->is_container_empty() ) {
432 if( cont->contents.front().typeId() == prod.typeId() ) {
433 charges_to_store -= cont->get_remaining_capacity_for_liquid( cont->contents.front(), true );
434 }
435 } else {
436 charges_to_store -= cont->get_remaining_capacity_for_liquid( prod, true );
437 }
438 }
439
440 // also check if we're currently in a vehicle that has the necessary storage
441 if( charges_to_store > 0 ) {
442 if( optional_vpart_position vp = g->m.veh_at( pos() ) ) {
443 const itype_id &ftype = prod.typeId();
444 int fuel_cap = vp->vehicle().fuel_capacity( ftype );
445 int fuel_amnt = vp->vehicle().fuel_left( ftype );
446
447 if( fuel_cap >= 0 ) {
448 int fuel_space_left = fuel_cap - fuel_amnt;
449 charges_to_store -= fuel_space_left;
450 }
451 }
452 }
453
454 if( charges_to_store > 0 ) {
455 if( !query_yn(
456 _( "You don't have anything in which to store %s and may have to pour it out or consume it as soon as it is prepared! Proceed?" ),
457 prod.tname() ) ) {
458 return false;
459 }
460 }
461 }
462
463 return true;
464}

616// @param offset is the index of the created item in the range [0, batch_size-1],
617// it makes sure that the used items are distributed equally among the new items.
618static void set_components( std::list<item> &components, const std::list<item> &used,
619 const int batch_size, const size_t offset )
620{
621 if( batch_size <= 1 ) {
622 components.insert( components.begin(), used.begin(), used.end() );
623 return;
624 }
625 // This count does *not* include items counted by charges!
626 size_t non_charges_counter = 0;
627 for( auto &tmp : used ) {
628 if( tmp.count_by_charges() ) {
629 components.push_back( tmp );
630 // This assumes all (count-by-charges) items of the same type have been merged into one,
631 // which has a charges value that can be evenly divided by batch_size.
632 components.back().charges = tmp.charges / batch_size;
633 } else {
634 if( ( non_charges_counter + offset ) % batch_size == 0 ) {
635 components.push_back( tmp );
636 }
637 non_charges_counter++;
638 }

762void player::craft_skill_gain( const item &craft, const int &multiplier )
763{
764 if( !craft.is_craft() ) {
765 debugmsg( "craft_skill_check() called on non-craft '%s.' Aborting.", craft.tname() );
766 return;
767 }
768
769 const recipe &making = craft.get_making();
770 const int batch_size = craft.charges;
771
772 std::vector<npc *> helpers = get_crafting_helpers();
773
774 if( making.skill_used ) {
775 // Normalize experience gain to crafting time, giving a bonus for longer crafting
776 const double batch_mult = batch_size + base_time_to_craft( making, batch_size ) / 30000.0;
777 // This is called after every 5% crafting progress, so divide by 20
778 // TODO: Don't multiply, instead divide the crafting time into more "learn bits"
779 const int base_practice = roll_remainder( ( making.difficulty * 15 + 10 ) * batch_mult /
780 20.0 ) * multiplier;
781 const int skill_cap = static_cast<int>( making.difficulty * 1.25 );
782 practice( making.skill_used, base_practice, skill_cap, true );
783
784 // NPCs assisting or watching should gain experience...
785 for( auto &helper : helpers ) {
786 //If the NPC can understand what you are doing, they gain more exp
787 if( helper->get_skill_level( making.skill_used ) >= making.difficulty ) {
788 helper->practice( making.skill_used, roll_remainder( base_practice / 2.0 ),
789 skill_cap );
790 if( batch_size > 1 && one_in( 3 ) ) {
791 add_msg( m_info, _( "%s assists with crafting…" ), helper->name );
792 }
793 if( batch_size == 1 && one_in( 3 ) ) {
794 add_msg( m_info, _( "%s could assist you with a batch…" ), helper->name );
795 }
796 // NPCs around you understand the skill used better
797 } else {
798 helper->practice( making.skill_used, roll_remainder( base_practice / 10.0 ),
799 skill_cap );
800 if( one_in( 3 ) ) {
801 add_msg( m_info, _( "%s watches you craft…" ), helper->name );
802 }
803 }
804 }

808double player::crafting_success_roll( const recipe &making ) const
809{
810 int secondary_dice = 0;
811 int secondary_difficulty = 0;
812 for( const auto &pr : making.required_skills ) {
813 secondary_dice += get_skill_level( pr.first );
814 secondary_difficulty += pr.second;
815 }
816
817 // # of dice is 75% primary skill, 25% secondary (unless secondary is null)
818 int skill_dice;
819 if( secondary_difficulty > 0 ) {
820 skill_dice = get_skill_level( making.skill_used ) * 3 + secondary_dice;
821 } else {
822 skill_dice = get_skill_level( making.skill_used ) * 4;
823 }
824
825 for( const npc *np : get_crafting_helpers() ) {
826 if( np->get_skill_level( making.skill_used ) >=
827 get_skill_level( making.skill_used ) ) {
828 // NPC assistance is worth half a skill level
829 skill_dice += 2;
830 add_msg_if_player( m_info, _( "%s helps with crafting…" ), np->name );
831 break;
832 }
833 }
834
835 // farsightedness can impose a penalty on electronics and tailoring success
836 // it's equivalent to a 2-rank electronics penalty, 1-rank tailoring
837 if( has_trait( trait_HYPEROPIC ) && !worn_with_flag( flag_FIX_FARSIGHT ) &&
838 !has_effect( effect_contacts ) ) {
839 int main_rank_penalty = 0;
840 if( making.skill_used == skill_electronics ) {
841 main_rank_penalty = 2;
842 } else if( making.skill_used == skill_tailor ) {
843 main_rank_penalty = 1;
844 }
845 skill_dice -= main_rank_penalty * 4;
846 }
847
848 // It's tough to craft with paws. Fortunately it's just a matter of grip and fine-motor,
849 // not inability to see what you're doing
850 for( const trait_id &mut : get_mutations() ) {
851 for( const std::pair<const skill_id, int> &skib : mut->craft_skill_bonus ) {
852 if( making.skill_used == skib.first ) {
853 skill_dice += skib.second;
854 }
855 }
856 }
857
858 // Sides on dice is 16 plus your current intelligence
859 ///\EFFECT_INT increases crafting success chance
860 const int skill_sides = 16 + int_cur;
861
862 int diff_dice;
863 if( secondary_difficulty > 0 ) {
864 diff_dice = making.difficulty * 3 + secondary_difficulty;
865 } else {
866 // Since skill level is * 4 also
867 diff_dice = making.difficulty * 4;
868 }
869
870 const int diff_sides = 24; // 16 + 8 (default intelligence)
871
872 const double skill_roll = dice( skill_dice, skill_sides );
873 const double diff_roll = dice( diff_dice, diff_sides );
874
875 if( diff_roll == 0 ) {
876 // Automatic success
877 return 2;
878 }
879
880 return skill_roll / diff_roll;
881}

918bool item::handle_craft_failure( player &crafter )
919{
920 if( !is_craft() ) {
921 debugmsg( "handle_craft_failure() called on non-craft '%s.' Aborting.", tname() );
922 return false;
923 }
924
925 const double success_roll = crafter.crafting_success_roll( get_making() );
926 const int starting_components = this->components.size();
927 // Destroy at most 75% of the components, always a chance of losing 1 though
928 const size_t max_destroyed = std::max<size_t>( 1, components.size() * 3 / 4 );
929 for( size_t i = 0; i < max_destroyed; i++ ) {
930 // This shouldn't happen
931 if( components.empty() ) {
932 break;
933 }
934 // If we roll success, skip destroying a component
935 if( x_in_y( success_roll, 1.0 ) ) {
936 continue;
937 }
938 destroy_random_component( *this, crafter );
939 }
940 if( starting_components > 0 && this->components.empty() ) {
941 // The craft had components and all of them were destroyed.
942 return true;
943 }
944
945 // Minimum 25% progress lost, average 35%. Falls off exponentially
946 // Loss is scaled by the success roll
947 const double percent_progress_loss = rng_exponential( 0.25, 0.35 ) *
948 ( 1.0 - std::min( success_roll, 1.0 ) );
949 const int progess_loss = item_counter * percent_progress_loss;
950 crafter.add_msg_player_or_npc( _( "You mess up and lose %d%% progress." ),
951 _( "<npcname> messes up and loses %d%% progress." ), progess_loss / 100000 );
952 item_counter = clamp( item_counter - progess_loss, 0, 10000000 );
953
954 set_next_failure_point( crafter );
955
956 // Check if we can consume a new component and continue
957 if( !crafter.can_continue_craft( *this ) ) {
958 crafter.cancel_activity();
959 }
960 return false;
961}

1007void complete_craft( player &p, item &craft, const bench_location & )
1008{
1009 if( !craft.is_craft() ) {
1010 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1011 return;
1012 }
1013
1014 const recipe &making = craft.get_making();
1015 const int batch_size = craft.charges;
1016 std::list<item> &used = craft.components;
1017 const double relative_rot = craft.get_relative_rot();
1018
1019 // Set up the new item, and assign an inventory letter if available
1020 std::vector<item> newits = making.create_results( batch_size );
1021
1022 const bool should_heat = making.hot_result();
1023
1024 bool first = true;
1025 size_t newit_counter = 0;
1026 for( item &newit : newits ) {
1027
1028 // Points to newit unless newit is a non-empty container, then it points to newit's contents.
1029 // Necessary for things like canning soup; sometimes we want to operate on the soup, not the can.
1030 item &food_contained = ( newit.is_container() && !newit.contents.empty() ) ?
1031 newit.contents.back() : newit;
1032
1033 // messages, learning of recipe, food spoilage calculation only once
1034 if( first ) {
1035 first = false;
1036 // TODO: reconsider recipe memorization
1037 if( p.knows_recipe( &making ) ) {
1038 add_msg( _( "You craft %s from memory." ), making.result_name() );
1039 } else {
1040 add_msg( _( "You craft %s using a book as a reference." ), making.result_name() );
1041 // If we made it, but we don't know it,
1042 // we're making it from a book and have a chance to learn it.
1043 // Base expected time to learn is 1000*(difficulty^4)/skill/int moves.
1044 // This means time to learn is greatly decreased with higher skill level,
1045 // but also keeps going up as difficulty goes up.
1046 // Worst case is lvl 10, which will typically take
1047 // 10^4/10 (1,000) minutes, or about 16 hours of crafting it to learn.
1048 int difficulty = p.has_recipe( &making, p.crafting_inventory(), p.get_crafting_helpers() );
1049 ///\EFFECT_INT increases chance to learn recipe when crafting from a book
1050 const double learning_speed =
1051 std::max( p.get_skill_level( making.skill_used ), 1 ) *
1052 std::max( p.get_int(), 1 );
1053 const double time_to_learn = 1000 * 8 * std::pow( difficulty, 4 ) / learning_speed;
1054 if( x_in_y( making.time, time_to_learn ) ) {
1055 p.learn_recipe( &making );
1056 add_msg( m_good, _( "You memorized the recipe for %s!" ),
1057 making.result_name() );
1058 }
1059 }
1060 }
1061
1062 food_contained.inherit_flags( used, making );
1063
1064 for( const std::string &flag : making.flags_to_delete ) {
1065 food_contained.unset_flag( flag );
1066 }
1067
1068 // Don't store components for things made by charges,
1069 // Don't store components for things that can't be uncrafted.
1070 if( recipe_dictionary::get_uncraft( making.result() ) && !food_contained.count_by_charges() &&
1071 making.is_reversible() ) {
1072 // Setting this for items counted by charges gives only problems:
1073 // those items are automatically merged everywhere (map/vehicle/inventory),
1074 // which would either lose this information or merge it somehow.
1075 set_components( food_contained.components, used, batch_size, newit_counter );
1076 newit_counter++;
1077 } else if( food_contained.is_food() && !food_contained.has_flag( flag_NUTRIENT_OVERRIDE ) ) {
1078 // if a component item has "cooks_like" it will be replaced by that item as a component
1079 for( item &comp : used ) {
1080 // only comestibles have cooks_like. any other type of item will throw an exception, so filter those out
1081 if( comp.is_comestible() && !comp.get_comestible()->cooks_like.is_empty() ) {
1082 comp = item( comp.get_comestible()->cooks_like, comp.birthday(), comp.charges );
1083 }
1084 // If this recipe is cooked, components are no longer raw.
1085 if( should_heat ) {
1086 comp.set_flag_recursive( flag_COOKED );
1087 }
1088 }
1089
1090 // use a copy of the used list so that the byproducts don't build up over iterations (#38071)
1091 std::list<item> usedbp = used;
1092 // byproducts get stored as a "component" but with a byproduct flag for consumption purposes
1093 if( making.has_byproducts() ) {
1094 for( item &byproduct : making.create_byproducts( batch_size ) ) {
1095 byproduct.set_flag( flag_BYPRODUCT );
1096 usedbp.push_back( byproduct );
1097 }
1098 }
1099 // store components for food recipes that do not have the override flag
1100 set_components( food_contained.components, usedbp, batch_size, newit_counter );
1101
1102 // store the number of charges the recipe would create with batch size 1.
1103 if( &newit != &food_contained ) { // If a canned/contained item was crafted…
1104 // … the container holds exactly one completion of the recipe, no matter the batch size.
1105 food_contained.recipe_charges = food_contained.charges;
1106 } else { // Otherwise, the item is already stacked so we need to divide by batch size.
1107 newit.recipe_charges = newit.charges / batch_size;
1108 }
1109 newit_counter++;
1110 }
1111
1112 if( food_contained.goes_bad() ) {
1113 food_contained.set_relative_rot( relative_rot );
1114 }
1115
1116 newit.set_owner( p.get_faction()->id );
1117 // If these aren't equal, newit is a container, so finalize its contents too.
1118 if( &newit != &food_contained ) {
1119 food_contained.set_owner( p.get_faction()->id );
1120 }
1121
1122 if( newit.made_of( LIQUID ) ) {
1123 liquid_handler::handle_all_liquid( newit, PICKUP_RANGE );
1124 } else {
1125 set_item_inventory( p, newit );
1126 }
1127 }
1128
1129 if( making.has_byproducts() ) {
1130 std::vector<item> bps = making.create_byproducts( batch_size );
1131 for( auto &bp : bps ) {
1132 if( bp.goes_bad() ) {
1133 bp.set_relative_rot( relative_rot );
1134 }
1135 bp.set_owner( p.get_faction()->id );
1136 if( bp.made_of( LIQUID ) ) {
1137 liquid_handler::handle_all_liquid( bp, PICKUP_RANGE );
1138 } else {
1139 set_item_inventory( p, bp );
1140 }
1141 }
1142 }
1143
1144 p.inv.restack( p );
1145}

1147bool player::can_continue_craft( item &craft )
1148{
1149 if( !craft.is_craft() ) {
1150 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1151 return false;
1152 }
1153 if( has_trait( trait_DEBUG_HS ) ) {
1154 return true;
1155 }
1156
1157 const recipe &rec = craft.get_making();
1158
1159 const requirement_data continue_reqs = craft.get_continue_reqs();
1160
1161 // Avoid building an inventory from the map if we don't have to, as it is expensive
1162 if( !continue_reqs.is_empty() ) {
1163
1164 std::function<bool( const item & )> filter = rec.get_component_filter();
1165 const std::function<bool( const item & )> no_rotten_filter =
1166 rec.get_component_filter( recipe_filter_flags::no_rotten );
1167 // continue_reqs are for all batches at once
1168 const int batch_size = 1;
1169
1170 if( !continue_reqs.can_make_with_inventory( crafting_inventory(), filter, batch_size ) ) {
1171 std::string buffer = _( "You don't have the required components to continue crafting!" );
1172 buffer += "\n";
1173 buffer += continue_reqs.list_missing();
1174 popup( buffer, PF_NONE );
1175 return false;
1176 }
1177
1178 std::string buffer = _( "Consume the missing components and continue crafting?" );
1179 buffer += "\n";
1180 buffer += continue_reqs.list_all();
1181 if( !query_yn( buffer ) ) {
1182 return false;
1183 }
1184
1185 if( continue_reqs.can_make_with_inventory( crafting_inventory(), no_rotten_filter,
1186 batch_size ) ) {
1187 filter = no_rotten_filter;
1188 } else {
1189 if( !query_yn( _( "Some components required to continue are rotten.\n"
1190 "Continue crafting anyway?" ) ) ) {
1191 return false;
1192 }
1193 }
1194
1195 inventory map_inv;
1196 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1197
1198 std::vector<comp_selection<item_comp>> item_selections;
1199 for( const auto &it : continue_reqs.get_components() ) {
1200 comp_selection<item_comp> is = select_item_component( it, batch_size, map_inv, true, filter );
1201 if( is.use_from == cancel ) {
1202 cancel_activity();
1203 add_msg( _( "You stop crafting." ) );
1204 return false;
1205 }
1206 item_selections.push_back( is );
1207 }
1208 for( const auto &it : item_selections ) {
1209 craft.components.splice( craft.components.end(), consume_items( it, batch_size, filter ) );
1210 }
1211 }
1212
1213 if( !craft.has_tools_to_continue() ) {
1214
1215 const std::vector<std::vector<tool_comp>> &tool_reqs = rec.simple_requirements().get_tools();
1216 const int batch_size = craft.charges;
1217
1218 std::vector<std::vector<tool_comp>> adjusted_tool_reqs;
1219 for( const std::vector<tool_comp> &alternatives : tool_reqs ) {
1220 std::vector<tool_comp> adjusted_alternatives;
1221 for( const tool_comp &alternative : alternatives ) {
1222 tool_comp adjusted_alternative = alternative;
1223 if( adjusted_alternative.count > 0 ) {
1224 adjusted_alternative.count *= batch_size;
1225 // Only for the next 5% progress
1226 adjusted_alternative.count = std::max(
1227 crafting::charges_for_continuing( adjusted_alternative.count ), 1 );
1228 }
1229 adjusted_alternatives.push_back( adjusted_alternative );
1230 }
1231 adjusted_tool_reqs.push_back( adjusted_alternatives );
1232 }

1298/* selection of component if a recipe requirement has multiple options (e.g. 'duct tap' or 'welder') */
1299comp_selection<item_comp> player::select_item_component( const std::vector<item_comp> &components,
1300 int batch, inventory &map_inv, bool can_cancel,
1301 const std::function<bool( const item & )> &filter, bool player_inv )
1302{
1303 std::vector<item_comp> player_has;
1304 std::vector<item_comp> map_has;
1305 std::vector<item_comp> mixed;
1306
1307 comp_selection<item_comp> selected;
1308
1309 for( const auto &component : components ) {
1310 itype_id type = component.type;
1311 int count = ( component.count > 0 ) ? component.count * batch : std::abs( component.count );
1312
1313 if( item::count_by_charges( type ) && count > 0 ) {
1314 int map_charges = map_inv.charges_of( type, INT_MAX, filter );
1315
1316 // If map has infinite charges, just use them
1317 if( map_charges == item::INFINITE_CHARGES ) {
1318 selected.use_from = use_from_map;
1319 selected.comp = component;
1320 return selected;
1321 }
1322 if( player_inv ) {
1323 int player_charges = charges_of( type, INT_MAX, filter );
1324 bool found = false;
1325 if( player_charges >= count ) {
1326 player_has.push_back( component );
1327 found = true;
1328 }
1329 if( map_charges >= count ) {
1330 map_has.push_back( component );
1331 found = true;
1332 }
1333 if( !found && player_charges + map_charges >= count ) {
1334 mixed.push_back( component );
1335 }
1336 } else {
1337 if( map_charges >= count ) {
1338 map_has.push_back( component );
1339 }
1340 }
1341 } else { // Counting by units, not charges
1342
1343 // Can't use pseudo items as components
1344 if( player_inv ) {
1345 bool found = false;
1346 if( has_amount( type, count, false, filter ) ) {
1347 player_has.push_back( component );
1348 found = true;
1349 }
1350 if( map_inv.has_components( type, count, filter ) ) {
1351 map_has.push_back( component );
1352 found = true;
1353 }
1354 if( !found &&
1355 amount_of( type, false, std::numeric_limits<int>::max(), filter ) +
1356 map_inv.amount_of( type, false, std::numeric_limits<int>::max(), filter ) >= count ) {
1357 mixed.push_back( component );
1358 }
1359 } else {
1360 if( map_inv.has_components( type, count, filter ) ) {
1361 map_has.push_back( component );
1362 }
1363 }
1364 }
1365 }
1366
1367 /* select 1 component to use */
1368 if( player_has.size() + map_has.size() + mixed.size() == 1 ) { // Only 1 choice
1369 if( player_has.size() == 1 ) {
1370 selected.use_from = use_from_player;
1371 selected.comp = player_has[0];
1372 } else if( map_has.size() == 1 ) {
1373 selected.use_from = use_from_map;
1374 selected.comp = map_has[0];
1375 } else {
1376 selected.use_from = use_from_both;
1377 selected.comp = mixed[0];
1378 }
1379 } else if( is_npc() ) {
1380 if( !player_has.empty() ) {
1381 selected.use_from = use_from_player;
1382 selected.comp = player_has[0];
1383 } else if( !map_has.empty() ) {
1384 selected.use_from = use_from_map;
1385 selected.comp = map_has[0];
1386 } else {
1387 debugmsg( "Attempted a recipe with no available components!" );
1388 selected.use_from = cancel;
1389 return selected;
1390 }
1391 } else { // Let the player pick which component they want to use
1392 uilist cmenu;
1393 // Populate options with the names of the items
1394 for( auto &map_ha : map_has ) { // Index 0-(map_has.size()-1)
1395 std::string tmpStr = string_format( _( "%s (%d/%d nearby)" ),
1396 item::nname( map_ha.type ),
1397 ( map_ha.count * batch ),
1398 item::count_by_charges( map_ha.type ) ?
1399 map_inv.charges_of( map_ha.type, INT_MAX, filter ) :
1400 map_inv.amount_of( map_ha.type, false, INT_MAX, filter ) );
1401 cmenu.addentry( tmpStr );
1402 }
1403 for( auto &player_ha : player_has ) { // Index map_has.size()-(map_has.size()+player_has.size()-1)
1404 std::string tmpStr = string_format( _( "%s (%d/%d on person)" ),
1405 item::nname( player_ha.type ),
1406 ( player_ha.count * batch ),
1407 item::count_by_charges( player_ha.type ) ?
1408 charges_of( player_ha.type, INT_MAX, filter ) :
1409 amount_of( player_ha.type, false, INT_MAX, filter ) );
1410 cmenu.addentry( tmpStr );
1411 }
1412 for( auto &component : mixed ) {
1413 // Index player_has.size()-(map_has.size()+player_has.size()+mixed.size()-1)
1414 int available = item::count_by_charges( component.type ) ?
1415 map_inv.charges_of( component.type, INT_MAX, filter ) +
1416 charges_of( component.type, INT_MAX, filter ) :
1417 map_inv.amount_of( component.type, false, INT_MAX, filter ) +
1418 amount_of( component.type, false, INT_MAX, filter );
1419 std::string tmpStr = string_format( _( "%s (%d/%d nearby & on person)" ),
1420 item::nname( component.type ),
1421 component.count * batch,
1422 available );
1423 cmenu.addentry( tmpStr );
1424 }
1425
1426 // Unlike with tools, it's a bad thing if there aren't any components available
1427 if( cmenu.entries.empty() ) {
1428 if( player_inv ) {
1429 if( has_trait( trait_DEBUG_HS ) ) {
1430 selected.use_from = use_from_player;
1431 return selected;
1432 }
1433 }
1434 debugmsg( "Attempted a recipe with no available components!" );
1435 selected.use_from = cancel;
1436 return selected;
1437 }
1438
1439 cmenu.allow_cancel = can_cancel;
1440
1441 // Get the selection via a menu popup
1442 cmenu.title = _( "Use which component?" );
1443 cmenu.query();
1444
1445 if( cmenu.ret < 0 ||
1446 static_cast<size_t>( cmenu.ret ) >= map_has.size() + player_has.size() + mixed.size() ) {
1447 selected.use_from = cancel;
1448 return selected;
1449 }
1450
1451 size_t uselection = static_cast<size_t>( cmenu.ret );
1452 if( uselection < map_has.size() ) {
1453 selected.use_from = usage::use_from_map;
1454 selected.comp = map_has[uselection];
1455 } else if( uselection < map_has.size() + player_has.size() ) {
1456 uselection -= map_has.size();
1457 selected.use_from = usage::use_from_player;
1458 selected.comp = player_has[uselection];
1459 } else {
1460 uselection -= map_has.size() + player_has.size();
1461 selected.use_from = usage::use_from_both;
1462 selected.comp = mixed[uselection];
1463 }
1464 }
1465
1466 return selected;
1467}

1493std::list<item> player::consume_items( map &m, const comp_selection<item_comp> &is, int batch,
1494 const std::function<bool( const item & )> &filter,
1495 const tripoint &origin, int radius )
1496{
1497 std::list<item> ret;
1498
1499 if( has_trait( trait_DEBUG_HS ) ) {
1500 return ret;
1501 }
1502
1503 item_comp selected_comp = is.comp;
1504
1505 const tripoint &loc = origin;
1506 const bool by_charges = item::count_by_charges( selected_comp.type ) && selected_comp.count > 0;
1507 // Count given to use_amount/use_charges, changed by those functions!
1508 int real_count = ( selected_comp.count > 0 ) ? selected_comp.count * batch : std::abs(
1509 selected_comp.count );
1510 // First try to get everything from the map, than (remaining amount) from player
1511 if( is.use_from & use_from_map ) {
1512 if( by_charges ) {
1513 std::list<item> tmp = m.use_charges( loc, radius, selected_comp.type, real_count, filter );
1514 ret.splice( ret.end(), tmp );
1515 } else {
1516 std::list<item> tmp = g->m.use_amount( loc, radius, selected_comp.type, real_count, filter );
1517 remove_ammo( tmp, *this );
1518 ret.splice( ret.end(), tmp );
1519 }
1520 }
1521 if( is.use_from & use_from_player ) {
1522 if( by_charges ) {
1523 std::list<item> tmp = use_charges( selected_comp.type, real_count, filter );
1524 ret.splice( ret.end(), tmp );
1525 } else {
1526 std::list<item> tmp = use_amount( selected_comp.type, real_count, filter );
1527 remove_ammo( tmp, *this );
1528 ret.splice( ret.end(), tmp );
1529 }
1530 }
1531 // condense those items into one
1532 if( by_charges && ret.size() > 1 ) {
1533 std::list<item>::iterator b = ret.begin();
1534 b++;
1535 while( ret.size() > 1 ) {
1536 ret.front().charges += b->charges;
1537 b = ret.erase( b );
1538 }
1539 }
1540 lastconsumed = selected_comp.type;
1541 empty_buckets( *this );
1542 return ret;
1543}

1572static std::vector<avail_tool_comp>
1573find_tool_component( const Character *player_with_inv, const std::vector<tool_comp> &tools,
1574 int batch,
1575 const inventory &map_inv, cost_adjustment charge_mod )
1576{
1577 std::vector<avail_tool_comp> available_tools;
1578
1579 auto calc_charges = [&]( const tool_comp & t ) {
1580 const int full_craft_charges = std::max( 1, t.count * batch );
1581 switch( charge_mod ) {
1582 case cost_adjustment::none:
1583 return std::make_pair( charges_for_complete( full_craft_charges ),
1584 charges_for_complete( full_craft_charges ) );
1585 case cost_adjustment::start_only:
1586 return std::make_pair( charges_for_starting( full_craft_charges ),
1587 charges_for_complete( full_craft_charges ) );
1588 case cost_adjustment::continue_only:
1589 return std::make_pair( charges_for_continuing( full_craft_charges ),
1590 charges_for_complete( full_craft_charges ) );
1591 }
1592
1593 debugmsg( "Invalid tool_charge_mod" );
1594 return std::make_pair( INT_MAX, INT_MAX );
1595 };
1596
1597 bool found_nocharge = false;
1598 // Use charges of any tools that require charges used
1599 for( auto it = tools.begin(); it != tools.end() && !found_nocharge; ++it ) {
1600 itype_id type = it->type;
1601 if( it->count > 0 ) {
1602 const std::pair<int, int> &expected_count = calc_charges( *it );
1603 const int count = expected_count.first;
1604 const int ideal = expected_count.second;
1605 if( player_with_inv ) {
1606 if( player_with_inv->has_charges( type, count ) ) {
1607 int total_charges = player_with_inv->charges_of( type );
1608 comp_selection<tool_comp> sel( use_from_player, *it );
1609 available_tools.emplace_back( sel, total_charges, ideal );
1610 }
1611 }
1612 if( map_inv.has_charges( type, count ) ) {
1613 int total_charges = map_inv.charges_of( type );
1614 comp_selection<tool_comp> sel( use_from_map, *it );
1615 available_tools.emplace_back( sel, total_charges, ideal );
1616 }
1617 } else if( ( player_with_inv && player_with_inv->has_amount( type, 1 ) )
1618 || map_inv.has_tools( type, 1 ) ) {
1619 comp_selection<tool_comp> sel( use_from_none, *it );
1620 available_tools.emplace_back( sel, 0, 0 );
1621 }
1622 }
1623
1624 std::sort( available_tools.begin(), available_tools.end(),
1625 []( const avail_tool_comp & lhs, const avail_tool_comp & rhs ) {
1626 if( lhs.comp.use_from == use_from_none && rhs.comp.use_from != use_from_none ) {
1627 return true;
1628 }
1629 if( rhs.comp.use_from == use_from_none ) {
1630 return false;
1631 }
1632 if( lhs.charges >= lhs.ideal && rhs.charges < rhs.ideal ) {
1633 return true;
1634 }
1635 if( lhs.charges < lhs.ideal && rhs.charges >= rhs.ideal ) {
1636 return false;
1637 }
1638 // We want "bigger" tools first because those are likely to be vehicle mounted
1639 return static_cast<float>( lhs.ideal ) / lhs.charges > static_cast<float>
1640 ( rhs.ideal ) / rhs.charges;
1641 } );
1642 return available_tools;
1643}

1730bool player::craft_consume_tools( item &craft, int mulitplier, bool start_craft )
1731{
1732 if( !craft.is_craft() ) {
1733 debugmsg( "craft_consume_tools() called on non-craft '%s.' Aborting.", craft.tname() );
1734 return false;
1735 }
1736 if( has_trait( trait_DEBUG_HS ) ) {
1737 return true;
1738 }
1739 if( start_craft && mulitplier > 1 ) {
1740 debugmsg( "start_craft is true, but multiplier is %d > 1", mulitplier );
1741 return false;
1742 }
1743
1744 const auto calc_charges = [&craft, &start_craft, &mulitplier]( int charges ) {
1745 int ret = charges;
1746
1747 if( charges <= 0 ) {
1748 return ret;
1749 }
1750
1751 // Account for batch size
1752 int full_cost = charges * craft.charges;
1753
1754 if( start_craft ) {
1755 return crafting::charges_for_starting( full_cost );
1756 } else {
1757 // In case more than 5% progress was accomplished in one turn
1758 return crafting::charges_for_continuing( full_cost ) * mulitplier;
1759 }
1760 };
1761
1762 // First check if we still have our cached selections
1763 const std::vector<comp_selection<tool_comp>> &cached_tool_selections =
1764 craft.get_cached_tool_selections();
1765
1766 inventory map_inv;
1767 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1768
1769 for( const comp_selection<tool_comp> &tool_sel : cached_tool_selections ) {
1770 itype_id type = tool_sel.comp.type;
1771 if( tool_sel.comp.count > 0 ) {
1772 const int count = calc_charges( tool_sel.comp.count );
1773 switch( tool_sel.use_from ) {
1774 case use_from_player:
1775 if( !has_charges( type, count ) ) {
1776 add_msg_player_or_npc(
1777 _( "You have insufficient %s charges and can't continue crafting" ),
1778 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1779 item::nname( type ) );
1780 craft.set_tools_to_continue( false );
1781 return false;
1782 }
1783 break;
1784 case use_from_map:
1785 if( !map_inv.has_charges( type, count ) ) {
1786 add_msg_player_or_npc(
1787 _( "You have insufficient %s charges and can't continue crafting" ),
1788 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1789 item::nname( type ) );
1790 craft.set_tools_to_continue( false );
1791 return false;
1792 }
1793 break;
1794 case use_from_both:
1795 case use_from_none:
1796 case cancel:
1797 case num_usages:
1798 break;
1799 }
1800 } else if( !has_amount( type, 1 ) && !map_inv.has_tools( type, 1 ) ) {
1801 add_msg_player_or_npc(
1802 _( "You no longer have a %s and can't continue crafting" ),
1803 _( "<npcname> no longer has a %s and can't continue crafting" ),
1804 item::nname( type ) );
1805 craft.set_tools_to_continue( false );
1806 return false;
1807 }
1808 }
1809
1810 // We have the selections, so consume them
1811 for( const comp_selection<tool_comp> &tool : cached_tool_selections ) {
1812 comp_selection<tool_comp> to_consume = tool;
1813 to_consume.comp.count = calc_charges( to_consume.comp.count );
1814 consume_tools( to_consume, 1 );
1815 }
1816 return true;
1817}

1855ret_val<bool> player::can_disassemble( const item &obj, const inventory &inv ) const
1856{
1857 const auto &r = recipe_dictionary::get_uncraft( obj.typeId() );
1858
1859 if( !r || obj.has_flag( flag_ETHEREAL_ITEM ) ) {
1860 return ret_val<bool>::make_failure( _( "You cannot disassemble this." ) );
1861 }
1862
1863 // check sufficient light
1864 if( lighting_craft_speed_multiplier( r ) == 0.0f ) {
1865 return ret_val<bool>::make_failure( _( "You can't see to craft!" ) );
1866 }
1867
1868 // refuse to disassemble rotten items
1869 const item *food = obj.get_food();
1870 if( ( obj.goes_bad() && obj.rotten() ) || ( food && food->goes_bad() && food->rotten() ) ) {
1871 return ret_val<bool>::make_failure( _( "It's rotten, I'm not taking that apart." ) );
1872 }
1873
1874 // refuse to disassemble items containing monsters/pets
1875 std::string monster = obj.get_var( "contained_name" );
1876 if( !monster.empty() ) {
1877 return ret_val<bool>::make_failure( _( "You must remove the %s before you can disassemble this." ),
1878 monster );
1879 }
1880
1881 if( !obj.is_ammo() ) { //we get ammo quantity to disassemble later on
1882 if( obj.count_by_charges() && !r.has_flag( flag_UNCRAFT_SINGLE_CHARGE ) ) {
1883 // Create a new item to get the default charges
1884 int qty = r.create_result().charges;
1885 if( obj.charges < qty ) {
1886 auto msg = vgettext( "You need at least %d charge of %s.",
1887 "You need at least %d charges of %s.", qty );
1888 return ret_val<bool>::make_failure( msg, qty, obj.tname() );
1889 }
1890 }
1891 }
1892 const auto &dis = r.disassembly_requirements();
1893
1894 for( const auto &opts : dis.get_qualities() ) {
1895 for( const auto &qual : opts ) {
1896 if( !qual.has( inv, return_true<item> ) ) {
1897 // Here should be no dot at the end of the string as 'to_string()' provides it.
1898 return ret_val<bool>::make_failure( _( "You need %s" ), qual.to_string() );
1899 }
1900 }
1901 }
1902
1903 for( const auto &opts : dis.get_tools() ) {
1904 const bool found = std::any_of( opts.begin(), opts.end(),
1905 [&]( const tool_comp & tool ) {
1906 return ( tool.count <= 0 && inv.has_tools( tool.type, 1 ) ) ||
1907 ( tool.count > 0 && inv.has_charges( tool.type, tool.count ) );
1908 } );
1909
1910 if( !found ) {
1911 const tool_comp &tool_required = opts.front();
1912 if( tool_required.count <= 0 ) {
1913 return ret_val<bool>::make_failure( _( "You need %s." ),
1914 item::nname( tool_required.type ) );
1915 } else {
1916 //~ %1$s: tool name, %2$d: needed charges
1917 return ret_val<bool>::make_failure( vgettext( "You need a %1$s with %2$d charge.",
1918 "You need a %1$s with %2$d charges.", tool_required.count ),
1919 item::nname( tool_required.type ),
1920 tool_required.count );
1921 }
1922 }
1923 }
1924
1925 return ret_val<bool>::make_success();
1926}
