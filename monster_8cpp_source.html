311int monster::next_upgrade_time()
312{
313 if( type->age_grow > 0 ) {

331void monster::try_upgrade( bool pin_time )
332{
333 if( !can_upgrade() ) {
334 return;
335 }
336
337 const int current_day = to_days<int>( calendar::turn - calendar::turn_zero );
338 //This should only occur when a monster is created or upgraded to a new form
339 if( upgrade_time < 0 ) {
340 upgrade_time = next_upgrade_time();
341 if( upgrade_time < 0 ) {
342 return;
343 }
344 if( pin_time || type->age_grow > 0 ) {
345 // offset by today, always true for growing creatures
346 upgrade_time += current_day;
347 } else {
348 // offset by starting season
349 // TODO: revisit this and make it simpler
350 upgrade_time += to_days<int>( calendar::start_of_cataclysm - calendar::turn_zero );
351 }
352 }
353
354 // Here we iterate until we either are before upgrade_time or can't upgrade any more.
355 // This is so that late into game new monsters can 'catch up' with all that half-life
356 // upgrades they'd get if we were simulating whole world.
357 while( true ) {
358 if( upgrade_time > current_day ) {
359 // not yet
360 return;
361 }
362
363 if( type->upgrade_into ) {
364 //If we upgrade into a blacklisted monster, treat it as though we are non-upgradeable
365 if( MonsterGroupManager::monster_is_blacklisted( type->upgrade_into ) ) {
366 return;
367 }
368 poly( type->upgrade_into );
369 } else {
370 const mtype_id &new_type = MonsterGroupManager::GetRandomMonsterFromGroup( type->upgrade_group );
371 if( new_type ) {
372 poly( new_type );
373 }
374 }
375
376 if( !upgrades ) {
377 // upgraded into a non-upgradeable monster
378 return;
379 }
380
381 const int next_upgrade = next_upgrade_time();
382 if( next_upgrade < 0 ) {
383 // hit never_upgrade
384 return;
385 }
386 upgrade_time += next_upgrade;
387 }

390void monster::try_reproduce()
391{
392 if( !reproduces ) {
393 return;
394 }
395 // This can happen if the monster type has changed (from reproducing to non-reproducing monster)
396 if( !type->baby_timer ) {
397 return;
398 }
399
400 if( !baby_timer ) {
401 // Assume this is a freshly spawned monster (because baby_timer is not set yet), set the point when it reproduce to somewhere in the future.
402 baby_timer.emplace( calendar::turn + *type->baby_timer );
403 }
404
405 bool season_spawn = false;
406 bool season_match = true;
407
408 // only 50% of animals should reproduce
409 bool female = one_in( 2 );
410 for( auto &elem : type->baby_flags ) {
411 if( elem == "SUMMER" || elem == "WINTER" || elem == "SPRING" || elem == "AUTUMN" ) {
412 season_spawn = true;
413 }
414 }
415
416 // add a decreasing chance of additional spawns when "catching up" an existing animal
417 int chance = -1;
418 while( true ) {
419 if( *baby_timer > calendar::turn ) {
420 return;
421 }
422
423 if( season_spawn ) {
424 season_match = false;
425 for( auto &elem : type->baby_flags ) {
426 if( ( season_of_year( *baby_timer ) == SUMMER && elem == "SUMMER" ) ||
427 ( season_of_year( *baby_timer ) == WINTER && elem == "WINTER" ) ||
428 ( season_of_year( *baby_timer ) == SPRING && elem == "SPRING" ) ||
429 ( season_of_year( *baby_timer ) == AUTUMN && elem == "AUTUMN" ) ) {
430 season_match = true;
431 }
432 }
433 }
434
435 chance += 2;
436 if( season_match && female && one_in( chance ) ) {
437 int spawn_cnt = rng( 1, type->baby_count );
438 if( type->baby_monster ) {
439 g->m.add_spawn( type->baby_monster, spawn_cnt, pos() );
440 } else {
441 g->m.add_item_or_charges( pos(), item( type->baby_egg, *baby_timer, spawn_cnt ), true );
442 }
443 }
444
445 *baby_timer += *type->baby_timer;
446 }

449void monster::refill_udders()
450{
451 if( type->starting_ammo.empty() ) {
452 debugmsg( "monster %s has no starting ammo to refill udders", get_name() );
453 return;
454 }
455 if( ammo.empty() ) {
456 // legacy animals got empty ammo map, fill them up now if needed.
457 ammo[type->starting_ammo.begin()->first] = type->starting_ammo.begin()->second;
458 }
459 auto current_milk = ammo.find( itype_milk_raw );
460 if( current_milk == ammo.end() ) {
461 current_milk = ammo.find( itype_milk );
462 if( current_milk != ammo.end() ) {
463 // take this opportunity to update milk udders to raw_milk
464 ammo[itype_milk_raw] = current_milk->second;
465 // Erase old key-value from map
466 ammo.erase( current_milk );
467 }
468 }
469 // if we got here, we got milk.
470 if( current_milk->second == type->starting_ammo.begin()->second ) {
471 // already full up
472 return;
473 }
474 if( calendar::turn - udder_timer > 1_days ) {
475 // no point granularizing this really, you milk once a day.
476 ammo.begin()->second = type->starting_ammo.begin()->second;
477 udder_timer = calendar::turn;

1020Creature::Attitude monster::attitude_to( const Creature &other ) const
1021{
1022 const monster *m = other.is_monster() ? static_cast< const monster *>( &other ) : nullptr;
1023 const player *p = other.as_player();
1024 if( m != nullptr ) {
1025 if( m == this ) {
1026 return A_FRIENDLY;
1027 }
1028
1029 auto faction_att = faction.obj().attitude( m->faction );
1030 if( ( friendly != 0 && m->friendly != 0 ) ||
1031 ( friendly == 0 && m->friendly == 0 && faction_att == MFA_FRIENDLY ) ) {
1032 // Friendly (to player) monsters are friendly to each other
1033 // Unfriendly monsters go by faction attitude
1034 return A_FRIENDLY;
1035 } else if( ( friendly == 0 && m->friendly == 0 && faction_att == MFA_HATE ) ) {
1036 // Stuff that hates a specific faction will always attack that faction
1037 return A_HOSTILE;
1038 } else if( ( friendly == 0 && m->friendly == 0 && faction_att == MFA_NEUTRAL ) ||
1039 morale < 0 || anger < 10 ) {
1040 // Stuff that won't attack is neutral to everything
1041 return A_NEUTRAL;
1042 } else {
1043 return A_HOSTILE;
1044 }
1045 } else if( p != nullptr ) {
1046 switch( attitude( const_cast<player *>( p ) ) ) {
1047 case MATT_FRIEND:
1048 case MATT_ZLAVE:
1049 return A_FRIENDLY;
1050 case MATT_FPASSIVE:
1051 case MATT_FLEE:
1052 case MATT_IGNORE:
1053 case MATT_FOLLOW:
1054 return A_NEUTRAL;
1055 case MATT_ATTACK:
1056 return A_HOSTILE;
1057 case MATT_NULL:
1058 case NUM_MONSTER_ATTITUDES:
1059 break;
1060 }
1061 }
1062 // Should not happen!, creature should be either player or monster
1063 return A_NEUTRAL;
1064}

1066monster_attitude monster::attitude( const Character *u ) const
1067{
1068 if( friendly != 0 ) {
1069 if( has_effect( effect_docile ) ) {
1070 return MATT_FPASSIVE;
1071 }
1072 if( u == &g->u ) {
1073 return MATT_FRIEND;
1074 }
1075 // Zombies don't understand not attacking NPCs, but dogs and bots should.
1076 const npc *np = dynamic_cast< const npc * >( u );
1077 if( np != nullptr && np->get_attitude() != NPCATT_KILL && !type->in_species( ZOMBIE ) ) {
1078 return MATT_FRIEND;
1079 }
1080 if( np != nullptr && np->is_hallucination() ) {
1081 return MATT_IGNORE;
1082 }
1083 }
1084 if( effect_cache[FLEEING] ) {
1085 return MATT_FLEE;
1086 }
1087 if( has_effect( effect_pacified ) ) {
1088 return MATT_ZLAVE;
1089 }
1090
1091 int effective_anger = anger;
1092 int effective_morale = morale;
1093
1094 if( u != nullptr ) {
1095 // Those are checked quite often, so avoiding string construction is a good idea
1096 static const string_id<monfaction> faction_bee( "bee" );
1097 if( faction == faction_bee ) {
1098 if( u->has_trait( trait_BEE ) ) {
1099 return MATT_FRIEND;
1100 } else if( u->has_trait( trait_FLOWERS ) ) {
1101 effective_anger -= 10;
1102 }
1103 }
1104
1105 if( type->in_species( FUNGUS ) && ( u->has_trait( trait_THRESH_MYCUS ) ||
1106 u->has_trait( trait_MYCUS_FRIEND ) ) ) {
1107 return MATT_FRIEND;
1108 }
1109
1110 if( effective_anger >= 10 &&
1111 ( ( type->in_species( MAMMAL ) && u->has_trait( trait_PHEROMONE_MAMMAL ) ) ||
1112 ( type->in_species( INSECT ) && u->has_trait( trait_PHEROMONE_INSECT ) ) ) ) {
1113 effective_anger -= 20;
1114 }
1115
1116 if( u->has_trait( trait_TERRIFYING ) ) {
1117 effective_morale -= 10;
1118 }
1119
1120 if( has_flag( MF_ANIMAL ) ) {
1121 if( u->has_trait( trait_ANIMALEMPATH ) ) {
1122 effective_anger -= 10;
1123 if( effective_anger < 10 ) {
1124 effective_morale += 55;
1125 }
1126 } else if( u->has_trait( trait_ANIMALEMPATH2 ) ) {
1127 effective_anger -= 20;
1128 if( effective_anger < 20 ) {
1129 effective_morale += 80;
1130 }
1131 } else if( u->has_trait( trait_ANIMALDISCORD ) ) {
1132 if( effective_anger >= 10 ) {
1133 effective_anger += 10;
1134 }
1135 if( effective_anger < 10 ) {
1136 effective_morale -= 5;
1137 }
1138 } else if( u->has_trait( trait_ANIMALDISCORD2 ) ) {
1139 if( effective_anger >= 20 ) {
1140 effective_anger += 20;
1141 }
1142 if( effective_anger < 20 ) {
1143 effective_morale -= 5;
1144 }
1145 }
1146 }
1147
1148 for( const trait_id &mut : u->get_mutations() ) {
1149 for( const std::pair<const species_id, int> &elem : mut.obj().anger_relations ) {
1150 if( type->in_species( elem.first ) ) {
1151 effective_anger += elem.second;
1152 }
1153 }
1154 }
1155
1156 for( const trait_id &mut : u->get_mutations() ) {
1157 for( const species_id &spe : mut.obj().ignored_by ) {
1158 if( type->in_species( spe ) ) {
1159 return MATT_IGNORE;
1160 }
1161 }
1162 }
1163 }
1164
1165 if( effective_morale < 0 ) {
1166 if( effective_morale + effective_anger > 0 && get_hp() > get_hp_max() / 3 ) {
1167 return MATT_FOLLOW;
1168 }
1169 return MATT_FLEE;
1170 }
1171
1172 if( effective_anger <= 0 ) {
1173 if( get_hp() != get_hp_max() ) {
1174 return MATT_FLEE;
1175 } else {
1176 return MATT_IGNORE;
1177 }
1178 }
1179
1180 if( effective_anger < 10 ) {
1181 return MATT_FOLLOW;
1182 }
1183
1184 return MATT_ATTACK;
1185}

1211 if( morale < type->morale ) {
1212 morale++;
1213 } else {
1214 morale--;
1215 }
