19// Some details about how achievements work
20// ========================================
21//
22// Achievements are built on the stats_tracker, which is in turn built on the
23// event_bus. Each of these layers involves subscription / callback style
24// interfaces, so the code flow may now be obvious. Here's a quick outline of
25// the execution flow to help clarify how it all fits together.
26//
27// * Various core game code paths generate events via the event bus.
28// * The stats_traecker subscribes to the event bus, and receives these events.
29// * Events contribute to event_multisets managed by the stats_tracker.
30// * (In the docs, these event_multisets are described as "event streams").
31// * (Optionally) event_transformations transform these event_multisets into
32// other event_multisets based on json-defined transformation rules. These
33// are also managed by stats_tracker.
34// * event_statistics monitor these event_multisets and summarize them into
35// single values. These are also managed by stats_tracker.
36// * Each achievement requirement has a corresponding requirement_watcher which
37// is alerted to statistic changes via the stats_tracker's watch interface.
38// * Each requirement_watcher notifies its achievement_tracker (of which there
39// is one per achievement) of the requirement's status on each change to a
40// statistic.
41// * The achievement_tracker keeps track of which requirements are currently
42// satisfied and which are not.
43// * When all the requirements are satisfied, the achievement_tracker tells the
44// achievement_tracker (only one of these exists per game).
45// * The achievement_tracker calls the achievement_attained_callback it was
46// given at construction time. This hooks into the actual game logic (e.g.
47// telling the player they just got an achievement).

434std::string achievement_state::ui_text( const achievement *ach ) const
435{
436 // First: the achievement name and description
437 nc_color c = color_from_completion( completion );
438 std::string result = colorize( ach->name(), c ) + "\n";
439 if( !ach->description().empty() ) {
440 result += " " + colorize( ach->description(), c ) + "\n";
441 }
442
443 if( completion == achievement_completion::completed ) {
444 std::string message = string_format(
445 _( "Completed %s" ), to_string( last_state_change ) );
446 result += " " + colorize( message, c ) + "\n";
447 } else {
448 // Next: the time constraint
449 if( ach->time_constraint() ) {
450 result += " " + ach->time_constraint()->ui_text() + "\n";
451 }
452 }
453
454 // Next: the requirements
455 const std::vector<achievement_requirement> &reqs = ach->requirements();
456 // If these two vectors are of different sizes then the definition must
457 // have changed since it was complated / failed, so we don't print any
458 // requirements info.
459 if( final_values.size() == reqs.size() ) {
460 for( size_t i = 0; i < final_values.size(); ++i ) {
461 result += " " + text_for_requirement( reqs[i], final_values[i] ) + "\n";
462 }
463 }
464
465 return result;

483achievement_tracker::achievement_tracker( const achievement &a, achievements_tracker &tracker,
484 stats_tracker &stats ) :
485 achievement_( &a ),
486 tracker_( &tracker )
487{
488 for( const achievement_requirement &req : a.requirements() ) {
489 watchers_.push_back( std::make_unique<requirement_watcher>( *this, req, stats ) );
490 }
491
492 for( const std::unique_ptr<requirement_watcher> &watcher : watchers_ ) {
493 bool is_satisfied = watcher->is_satisfied( stats );
494 sorted_watchers_[is_satisfied].insert( watcher.get() );
495 }

498void achievement_tracker::set_requirement( requirement_watcher *watcher, bool is_satisfied )
499{
500 if( sorted_watchers_[is_satisfied].insert( watcher ).second ) {
501 // Remove from other; check for completion.
502 sorted_watchers_[!is_satisfied].erase( watcher );
503 assert( sorted_watchers_[0].size() + sorted_watchers_[1].size() == watchers_.size() );
504 }
505
506 achievement_completion time_comp =
507 achievement_->time_constraint() ?
508 achievement_->time_constraint()->completed() : achievement_completion::completed;
509
510 if( sorted_watchers_[false].empty() && time_comp == achievement_completion::completed ) {
511 // report_achievement can result in this being deleted, so it must be
512 // the last thing in the function
513 tracker_->report_achievement( achievement_, achievement_completion::completed );
514 return;
515 }
516
517 if( time_comp == achievement_completion::failed ||
518 ( !is_satisfied && watcher->requirement().becomes_false ) ) {
519 // report_achievement can result in this being deleted, so it must be
520 // the last thing in the function
521 tracker_->report_achievement( achievement_, achievement_completion::failed );
522 }

592void achievements_tracker::report_achievement( const achievement *a, achievement_completion comp )
593{
594 assert( comp != achievement_completion::pending );
595 assert( !achievements_status_.count( a->id ) );
596
597 auto tracker_it = trackers_.find( a->id );
598 achievements_status_.emplace(
599 a->id,
600 achievement_state{
601 comp,
602 calendar::turn,
603 tracker_it->second.current_values()
604 }
605 );
606 if( comp == achievement_completion::completed ) {
607 achievement_attained_callback_( a );
