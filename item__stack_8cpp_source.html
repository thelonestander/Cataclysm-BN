107int item_stack::amount_can_fit( const item &it ) const
108{
109 // Without stacking charges, would we violate the count limit?
110 const bool violates_count = size() >= static_cast<size_t>( count_limit() );
111 const item *here = it.count_by_charges() ? stacks_with( it ) : nullptr;
112
113 if( violates_count && !here ) {
114 return 0;

116 // Call max because a tile may have been overfilled to begin with (e.g. #14115)
117 const int ret = std::max( 0, it.charges_per_volume( free_volume() ) );
118 return it.count_by_charges() ? std::min( ret, it.charges ) : ret;
