40// GENERAL GUIDELINES
41// To determine mount position for parts (dx, dy), check this scheme:
42// orthogonal dir left: (Y-)
43// ^
44// back: X- -------> forward dir: X+
45// v
46// orthogonal dir right (Y+)
47//
48// i.e, if you want to add a part to the back from the center of vehicle,
49// use dx = -1, dy = 0;
50// for the part 1 tile forward and two tiles left from the center of vehicle,
51// use dx = 1, dy = -2.
52//
53// Internal parts should be added after external on the same mount point, i.e:
54// part {"x": 0, "y": 1, "part": "seat"}, // put a seat (it's external)
55// part {"x": 0, "y": 1, "part": "controls"}, // put controls for driver here
56// part {"x": 0, "y": 1, "seatbelt"} // also, put a seatbelt here
57// To determine, what parts can be external, and what can not, check
58// vehicle_parts.json
59// If you use wrong config, installation of part will fail

160static void parse_vp_reqs( const JsonObject &obj, const std::string &id, const std::string &key,
161 std::vector<std::pair<requirement_id, int>> &reqs,
162 std::map<skill_id, int> &skills, int &moves )
163{
164
165 if( !obj.has_object( key ) ) {
166 return;
167 }
168 JsonObject src = obj.get_object( key );
169
170 auto sk = src.get_array( "skills" );
171 if( !sk.empty() ) {
172 skills.clear();
173 }
174 for( JsonArray cur : sk ) {
175 skills.emplace( skill_id( cur.get_string( 0 ) ), cur.size() >= 2 ? cur.get_int( 1 ) : 1 );
176 }
177
178 if( src.has_int( "time" ) ) {
179 moves = src.get_int( "time" );
180 } else if( src.has_string( "time" ) ) {
181 moves = to_moves<int>( read_from_json_string<time_duration>( *src.get_raw( "time" ),
182 time_duration::units ) );
183 }
184
185 if( src.has_string( "using" ) ) {
186 reqs = { { requirement_id( src.get_string( "using" ) ), 1 } };
187 } else if( src.has_array( "using" ) ) {
188 reqs.clear();
189 for( JsonArray cur : src.get_array( "using" ) ) {
190 reqs.emplace_back( requirement_id( cur.get_string( 0 ) ), cur.get_int( 1 ) );
191 }
192 } else {
193 reqs.clear();
194 // Construct a requirement to capture "components", "qualities", and
195 // "tools" that might be listed.
196 const requirement_id req_id( string_format( "inline_%s_%s", key.c_str(), id.c_str() ) );
197 requirement_data::load_requirement( src, req_id );
198 reqs.emplace_back( req_id, 1 );
199 }

806static int scale_time( const std::map<skill_id, int> &sk, int mv, const player &p )
807{
808 if( sk.empty() ) {
809 return mv;
810 }
811
812 const int lvl = std::accumulate( sk.begin(), sk.end(), 0, [&p]( int lhs,
813 const std::pair<skill_id, int> &rhs ) {
814 return lhs + std::max( std::min( p.get_skill_level( rhs.first ), MAX_SKILL ) - rhs.second,
815 0 );
816 } );
817 // 10% per excess level (reduced proportionally if >1 skill required) with max 50% reduction
818 // 10% reduction per assisting NPC
819 return mv * ( 1.0 - std::min( static_cast<double>( lvl ) / sk.size() / 10.0, 0.5 ) )
820 * ( 10 - p.get_crafting_helpers( 3 ).size() ) / 10;
821}

960 *Caches a vehicle definition from a JsonObject to be loaded after itypes is initialized.
961 */
962void vehicle_prototype::load( const JsonObject &jo )
963{
964 vehicle_prototype &vproto = vtypes[ vproto_id( jo.get_string( "id" ) ) ];
965 // If there are already parts defined, this vehicle prototype overrides an existing one.
966 // If the json contains a name, it means a completely new prototype (replacing the
967 // original one), therefore the old data has to be cleared.
968 // If the json does not contain a name (the prototype would have no name), it means appending
969 // to the existing prototype (the parts are not cleared).
970 if( !vproto.parts.empty() && jo.has_string( "name" ) ) {
971 vproto = vehicle_prototype();
972 }
973 if( vproto.parts.empty() ) {
974 vproto.name = jo.get_string( "name" );
975 }
976
977 vgroups[vgroup_id( jo.get_string( "id" ) )].add_vehicle( vproto_id( jo.get_string( "id" ) ), 100 );
978
979 const auto add_part_obj = [&]( const JsonObject & part, point pos ) {
980 part_def pt;
981 pt.pos = pos;
982 pt.part = vpart_id( part.get_string( "part" ) );
983
984 assign( part, "ammo", pt.with_ammo, true, 0, 100 );
985 assign( part, "ammo_types", pt.ammo_types, true );
986 assign( part, "ammo_qty", pt.ammo_qty, true, 0 );
987 assign( part, "fuel", pt.fuel, true );
988
989 vproto.parts.push_back( pt );
990 };
991
992 const auto add_part_string = [&]( const std::string & part, point pos ) {
993 part_def pt;
994 pt.pos = pos;
995 pt.part = vpart_id( part );
996 vproto.parts.push_back( pt );
997 };
998
999 if( jo.has_member( "blueprint" ) ) {
1000 // currently unused, read to suppress unvisited members warning
1001 jo.get_array( "blueprint" );
1002 }
1003
1004 for( JsonObject part : jo.get_array( "parts" ) ) {
1005 point pos = point( part.get_int( "x" ), part.get_int( "y" ) );
1006
1007 if( part.has_string( "part" ) ) {
1008 add_part_obj( part, pos );
1009 } else if( part.has_array( "parts" ) ) {
1010 for( const JsonValue entry : part.get_array( "parts" ) ) {
1011 if( entry.test_string() ) {
1012 std::string part_name = entry.get_string();
1013 add_part_string( part_name, pos );
1014 } else {
1015 JsonObject subpart = entry.get_object();
1016 add_part_obj( subpart, pos );
1017 }
1018 }
1019 }
1020 }
1021
1022 for( JsonObject spawn_info : jo.get_array( "items" ) ) {
1023 vehicle_item_spawn next_spawn;
1024 next_spawn.pos.x = spawn_info.get_int( "x" );
1025 next_spawn.pos.y = spawn_info.get_int( "y" );
1026
1027 next_spawn.chance = spawn_info.get_int( "chance" );
1028 if( next_spawn.chance <= 0 || next_spawn.chance > 100 ) {
1029 debugmsg( "Invalid spawn chance in %s (%d, %d): %d%%",
1030 vproto.name, next_spawn.pos.x, next_spawn.pos.y, next_spawn.chance );
1031 }
1032
1033 // constrain both with_magazine and with_ammo to [0-100]
1034 next_spawn.with_magazine = std::max( std::min( spawn_info.get_int( "magazine",
1035 next_spawn.with_magazine ), 100 ), 0 );
1036 next_spawn.with_ammo = std::max( std::min( spawn_info.get_int( "ammo", next_spawn.with_ammo ),
1037 100 ), 0 );
1038
1039 if( spawn_info.has_array( "items" ) ) {
1040 //Array of items that all spawn together (i.e. jack+tire)
1041 spawn_info.read( "items", next_spawn.item_ids, true );
1042 } else if( spawn_info.has_string( "items" ) ) {
1043 //Treat single item as array
1044 next_spawn.item_ids.push_back( itype_id( spawn_info.get_string( "items" ) ) );
1045 }
1046 if( spawn_info.has_array( "item_groups" ) ) {
1047 //Pick from a group of items, just like map::place_items
1048 for( const std::string line : spawn_info.get_array( "item_groups" ) ) {
1049 next_spawn.item_groups.push_back( item_group_id( line ) );
1050 }
1051 } else if( spawn_info.has_string( "item_groups" ) ) {
1052 next_spawn.item_groups.push_back( item_group_id( spawn_info.get_string( "item_groups" ) ) );
1053 }
1054 vproto.item_spawns.push_back( std::move( next_spawn ) );
1055 }

1107 } else {

1110 if( !ammo->ammo && base->gun->ammo.count( ammo->ammo->type ) ) {
1111 debugmsg( "init_vehicles: turret %s has invalid ammo_type %s in %s",
1112 pt.part.c_str(), e.c_str(), id.c_str() );
1113 }
1114 }

1118 }
1119 }
1120 }

1122 if( base->container || base->magazine ) {
1123 if( !pt.fuel.is_valid() ) {
1124 debugmsg( "init_vehicles: tank %s specified invalid fuel in %s", pt.part.c_str(), id.c_str() );
1125 }
1126 } else {
1127 if( !pt.fuel.is_null() ) {
1128 debugmsg( "init_vehicles: non-fuel store part %s with fuel in %s", pt.part.c_str(), id.c_str() );
1129 }
1130 }
1131
1132 if( pt.part.obj().has_flag( "CARGO" ) ) {
1133 cargo_spots.insert( pt.pos );
1134 }
1135 }
1136 blueprint.enable_refresh();
1137
1138 for( auto &i : proto.item_spawns ) {
1139 if( cargo_spots.count( i.pos ) == 0 ) {
1140 debugmsg( "Invalid spawn location (no CARGO vpart) in %s (%d, %d): %d%%",
1141 proto.name, i.pos.x, i.pos.y, i.chance );
1142 }

1145 debugmsg( "unknown item %s in spawn list of %s", j.c_str(), id.c_str() );
1146 }

1149 if( !j.is_valid() ) {
1150 debugmsg( "unknown item group %s in spawn list of %s", j.c_str(), id.c_str() );
1151 }
