73struct bionic_sort_less {
74 bool operator()( const bionic *lhs, const bionic *rhs ) const {
75 const bionic_data &lbd = lhs->info();
76 const bionic_data &rbd = rhs->info();
77
78 switch( uistate.bionic_sort_mode ) {
79 case bionic_ui_sort_mode::nsort:
80 case bionic_ui_sort_mode::NONE:
81 //use installation order
82 return true;
83 case bionic_ui_sort_mode::INVLET:
84 return lhs->invlet < rhs->invlet;
85 case bionic_ui_sort_mode::POWER: {
86 units::energy lbd_sort_power = bionic_sort_power( lbd );
87 units::energy rbd_sort_power = bionic_sort_power( rbd );
88 if( lbd_sort_power != rbd_sort_power ) {
89 return lbd_sort_power < rbd_sort_power;
90 }
91 }
92 /* fallthrough */
93 case bionic_ui_sort_mode::NAME:
94 return localized_compare( lbd.name.translated(), rbd.name.translated() );
95 }
96 return false;
97 }
98};

284//builds the power usage string of a given bionic
285static std::string build_bionic_poweronly_string( const bionic &bio )
286{
287 const bionic_data &bio_data = bio.id.obj();
288 std::vector<std::string> properties;
289
290 if( bio_data.power_activate > 0_kJ ) {
291 properties.push_back( string_format( _( "%s act" ),
292 units::display( bio_data.power_activate ) ) );
293 }
294 if( bio_data.power_deactivate > 0_kJ ) {
295 properties.push_back( string_format( _( "%s deact" ),
296 units::display( bio_data.power_deactivate ) ) );
297 }
298 if( bio_data.charge_time > 0 && bio_data.power_over_time > 0_kJ ) {
299 properties.push_back( bio_data.charge_time == 1
300 ? string_format( _( "%s/turn" ), units::display( bio_data.power_over_time ) )
301 : string_format( _( "%s/%d turns" ), units::display( bio_data.power_over_time ),
302 bio_data.charge_time ) );
303 }
304 if( bio_data.has_flag( STATIC( flag_str_id( "BIONIC_TOGGLED" ) ) ) ) {
305 properties.push_back( bio.powered ? _( "ON" ) : _( "OFF" ) );
306 }
307 if( bio.incapacitated_time > 0_turns ) {
308 properties.push_back( _( "(incapacitated)" ) );
309 }
310 if( !bio.has_flag( flag_SAFE_FUEL_OFF ) && ( !bio.info().fuel_opts.empty() ||
311 bio.info().is_remote_fueled ) ) {
312 properties.push_back( _( "(fuel saving ON)" ) );
313 }
314 if( bio.is_auto_start_on() && ( !bio.info().fuel_opts.empty() || bio.info().is_remote_fueled ) ) {
315 const std::string label = string_format( _( "(auto start < %d %%)" ),
316 static_cast<int>( bio.get_auto_start_thresh() * 100 ) );
317 properties.push_back( label );
318 }
319
320 return enumerate_as_string( properties, enumeration_conjunction::none );
321}

384static void draw_connectors( const catacurses::window &win, const point &start,
385 int last_x, const bionic_id &bio_id, const std::map<bodypart_str_id, size_t> &bp_to_pos )
386{
387 const int LIST_START_Y = 7;
388 // first: pos_y, second: occupied slots
389 std::vector<std::pair<int, size_t>> pos_and_num;
390 for( const std::pair<const bodypart_str_id, int> &elem : bio_id->occupied_bodyparts ) {
391 auto pos = bp_to_pos.find( elem.first );
392 if( pos != bp_to_pos.end() ) {
393 pos_and_num.emplace_back( static_cast<int>( pos->second ) + LIST_START_Y,
394 static_cast<size_t>( elem.second ) );
395 }
396 }
397 if( pos_and_num.empty() || !get_option < bool >( "CBM_SLOTS_ENABLED" ) ) {
398 return;
399 }
400
401 // draw horizontal line from selected bionic
402 const int turn_x = start.x + ( last_x - start.x ) * 2 / 3;
403 mvwputch( win, start, BORDER_COLOR, '>' );
404 // NOLINTNEXTLINE(cata-use-named-point-constants)
405 mvwhline( win, start + point( 1, 0 ), LINE_OXOX, turn_x - start.x - 1 );
406
407 int min_y = start.y;
408 int max_y = start.y;
409 for( const auto &elem : pos_and_num ) {
410 min_y = std::min( min_y, elem.first );
411 max_y = std::max( max_y, elem.first );
412 }
413 if( max_y - min_y > 1 ) {
414 mvwvline( win, point( turn_x, min_y + 1 ), LINE_XOXO, max_y - min_y - 1 );
415 }
416
417 bool move_up = false;
418 bool move_same = false;
419 bool move_down = false;
420 for( const auto &elem : pos_and_num ) {
421 const int y = elem.first;
422 if( !move_up && y < start.y ) {
423 move_up = true;
424 }
425 if( !move_same && y == start.y ) {
426 move_same = true;
427 }
428 if( !move_down && y > start.y ) {
429 move_down = true;
430 }
431
432 // symbol is defined incorrectly for case ( y == start_y ) but
433 // that's okay because it's overlapped by bionic_chr anyway
434 int bp_chr = ( y > start.y ) ? LINE_XXOO : LINE_OXXO;
435 if( ( max_y > y && y > start.y ) || ( min_y < y && y < start.y ) ) {
436 bp_chr = LINE_XXXO;
437 }
438
439 mvwputch( win, point( turn_x, y ), BORDER_COLOR, bp_chr );
440
441 // draw horizontal line to bodypart title
442 mvwhline( win, point( turn_x + 1, y ), LINE_OXOX, last_x - turn_x - 1 );
443 mvwputch( win, point( last_x, y ), BORDER_COLOR, '<' );
444
445 // draw amount of consumed slots by this CBM
446 const std::string fmt_num = string_format( "(%d)", elem.second );
447 mvwprintz( win, point( turn_x + std::max( 1, ( last_x - turn_x - utf8_width( fmt_num ) ) / 2 ), y ),
448 c_yellow, fmt_num );
449 }
450
451 // define and draw a proper intersection character
452 // 001
453 // '-'
454 int bionic_chr = LINE_OXOX;
455 if( move_up && !move_down && !move_same ) {
456 // 100
457 // '_|'
458 bionic_chr = LINE_XOOX;
459 } else if( move_up && move_down && !move_same ) {
460 // 110
461 // '-|'
462 bionic_chr = LINE_XOXX;
463 } else if( move_up && move_down && move_same ) {
464 // 111
465 // '-|-'
466 bionic_chr = LINE_XXXX;
467 } else if( move_up && !move_down && move_same ) {
468 // 101
469 // '_|_'
470 bionic_chr = LINE_XXOX;
471 } else if( !move_up && move_down && !move_same ) {
472 // 010
473 // '^|'
474 bionic_chr = LINE_OOXX;
475 } else if( !move_up && move_down && move_same ) {
476 // 011
477 // '^|^'
478 bionic_chr = LINE_OXXX;
479 }
480 mvwputch( win, point( turn_x, start.y ), BORDER_COLOR, bionic_chr );
481}
