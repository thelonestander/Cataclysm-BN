233/**
234 * Prints a list of descriptions for all parts to the screen inside of a boxed window
235 * @param win The window to draw in.
236 * @param max_y Draw no further than this y-coordinate.
237 * @param width The width of the window.
238 * @param p The index of the part being examined.
239 * @param start_at Which vehicle part to start printing at.
240 * @param start_limit the part index beyond which the display is full
241 */
242void vehicle::print_vparts_descs( const catacurses::window &win, int max_y, int width, int p,
243 int &start_at, int &start_limit ) const
244{
245 if( p < 0 || p >= static_cast<int>( parts.size() ) ) {
246 return;
247 }
248
249 std::vector<int> pl = this->parts_at_relative( parts[p].mount, true );
250 std::string msg;
251
252 int lines = 0;
253 /*
254 * start_at and start_limit interaction is little tricky
255 * start_at and start_limit start at 0 when moving to a new frame
256 * if all the descriptions are displayed in the window, start_limit stays at 0 and
257 * start_at is capped at 0 - so no scrolling at all.
258 * if all the descriptions aren't displayed, start_limit jumps to the last displayed part
259 * and the next scrollthrough can start there - so scrolling down happens.
260 * when the scroll reaches the point where all the remaining descriptions are displayed in
261 * the window, start_limit is set to start_at again.
262 * on the next attempted scrolldown, start_limit is set to the nth item, and start_at is
263 * capped to the nth item, so no more scrolling down.
264 * start_at can always go down, but never below 0, so scrolling up is only possible after
265 * some scrolling down has occurred.
266 * important! the calling function needs to track p, start_at, and start_limit, and set
267 * start_limit to 0 if p changes.
268 */
269 start_at = std::max( 0, std::min( start_at, start_limit ) );
270 if( start_at ) {
271 msg += std::string( "<color_yellow>" ) + "< " + _( "More parts here…" ) + "</color>\n";
272 lines += 1;
273 }
274 for( size_t i = start_at; i < pl.size(); i++ ) {
275 const vehicle_part &vp = parts[ pl [ i ] ];
276 std::string possible_msg;
277 const nc_color name_color = vp.is_broken() ? c_dark_gray : c_light_green;
278 possible_msg += colorize( vp.name(), name_color ) + "\n";
279 const nc_color desc_color = vp.is_broken() ? c_dark_gray : c_light_gray;
280 // -4 = -2 for left & right padding + -2 for "> "
281 int new_lines = 2 + vp.info().format_description( possible_msg, desc_color, width - 4 );
282 if( vp.has_flag( vehicle_part::carrying_flag ) ) {
283 possible_msg += " Carrying a vehicle on a rack.\n";
284 new_lines += 1;
285 }
286 if( vp.has_flag( vehicle_part::carried_flag ) ) {
287 possible_msg += string_format( " Part of a %s carried on a rack.\n",
288 vp.carried_name() );
289 new_lines += 1;
290 }
291
292 possible_msg += "</color>\n";
293 if( lines + new_lines <= max_y ) {
294 msg += possible_msg;
295 lines += new_lines;
296 start_limit = start_at;
297 } else {
298 msg += std::string( "<color_yellow>" ) + _( "More parts here…" ) + " >" + "</color>\n";
299 start_limit = i;
300 break;
301 }
302 }
303 werase( win );
304 // -2 for left & right padding
305 // NOLINTNEXTLINE(cata-use-named-point-constants)
306 fold_and_print( win, point( 1, 0 ), width - 2, c_light_gray, msg );
307 wnoutrefresh( win );
308}

314std::vector<itype_id> vehicle::get_printable_fuel_types() const
315{

324

398void vehicle::print_fuel_indicator( const catacurses::window &win, const point &p,
399 const itype_id &fuel_type,
400 std::map<itype_id, float> fuel_usages,
401 bool verbose, bool desc )
402{
403 const char fsyms[5] = { 'E', '\\', '|', '/', 'F' };
404 nc_color col_indf1 = c_light_gray;
405 int cap = fuel_capacity( fuel_type );
406 int f_left = fuel_left( fuel_type );
407 nc_color f_color = fuel_type->color;
408 // NOLINTNEXTLINE(cata-text-style): not an ellipsis
409 mvwprintz( win, p, col_indf1, "E...F" );
410 int amnt = cap > 0 ? f_left * 99 / cap : 0;
411 int indf = ( amnt / 20 ) % 5;
412 mvwprintz( win, p + point( indf, 0 ), f_color, "%c", fsyms[indf] );
413 if( verbose ) {
414 if( debug_mode ) {
415 mvwprintz( win, p + point( 6, 0 ), f_color, "%d/%d", f_left, cap );
416 } else {
417 mvwprintz( win, p + point( 6, 0 ), f_color, "%d", f_left * 100 / cap );
418 wprintz( win, c_light_gray, "%c", 045 );
419 }
420 }
421 if( desc ) {
422 wprintz( win, c_light_gray, " - %s", item::nname( fuel_type ) );
423 }
424 if( verbose ) {
425 auto fuel_data = fuel_usages.find( fuel_type );
426 int rate = 0;
427 std::string units;
428 if( fuel_data != fuel_usages.end() ) {
429 rate = -consumption_per_hour( fuel_type, fuel_data->second );
430 units = _( "mL" );
431 }
432 if( fuel_type == itype_id( "battery" ) ) {
433 rate += power_to_energy_bat( net_battery_charge_rate_w(), 1_hours );
434 units = _( "kJ" );
435 }
436 if( rate != 0 ) {
437 int tank_use = 0;
438 nc_color tank_color = c_light_green;
439 std::string tank_goal = _( "full" );
440 if( rate > 0 ) {
441 tank_use = cap - f_left;
442 if( !tank_use ) {
443 return;
444 }
445 } else {
446 if( !f_left ) {
447 return;
448 }
449 tank_use = f_left;
450 tank_color = c_light_red;
451 tank_goal = _( "empty" );
452 }
453
454 item fitem( fuel_type );
455 int charges_per_L = fitem.charges_per_volume( 1_liter );
456 if( charges_per_L == 0 || charges_per_L == item::INFINITE_CHARGES ) {
457 return;
458 }
459 float charges_per_mL = charges_per_L / 1000.0f;
460 tank_use = tank_use / charges_per_mL;
461
462 // promote to double so esimate doesn't overflow for high fuel values
463 // 3600 * tank_use overflows signed 32 bit when tank_use is over ~596523
464 double turns = to_turns<double>( 60_minutes );
465 time_duration estimate = time_duration::from_turns( turns * tank_use / std::abs( rate ) );
466
467 if( debug_mode ) {
468 wprintz( win, tank_color, _( ", %d %s(%4.2f%%)/hour, %s until %s" ),
469 rate, units, 100.0 * rate / cap, to_string_clipped( estimate ), tank_goal );
470 } else {
471 wprintz( win, tank_color, _( ", %3.1f%% / hour, %s until %s" ),
472 100.0 * rate / cap, to_string_clipped( estimate ), tank_goal );
473 }
474 }
