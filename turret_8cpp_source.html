258{

275void turret_data::post_fire( player &p, int shots )
276{
277 // remove any temporary recoil adjustments
278 p.remove_effect( effect_on_roof );
279 p.recoil = cached_recoil;
280
281 auto mode = base()->gun_current_mode();
282
283 // handle draining of vehicle tanks and UPS charges, if applicable
284 if( part->info().has_flag( "USE_TANKS" ) ) {
285 veh->drain( ammo_current(), mode->ammo_required() * shots );
286 mode->ammo_unset();

542int vehicle::automatic_fire_turret( vehicle_part &pt )
543{
544 turret_data gun = turret_query( pt );
545
546 int shots = 0;
547
548 if( gun.query() != turret_data::status::ready ) {
549 return shots;
550 }
551
552 // The position of the vehicle part.
553 tripoint pos = global_part_pos3( pt );
554
555 // Create the targeting computer's npc
556 npc cpu = get_targeting_npc( pt );
557
558 int area = max_aoe_size( gun.ammo_effects() );
559 if( area > 0 ) {
560 // Pad a bit for less friendly fire
561 area += area == 1 ? 1 : 2;
562 }
563
564 const bool u_see = g->u.sees( pos );
565 const bool u_hear = !g->u.is_deaf();
566 // The current target of the turret.
567 auto &target = pt.target;
568 if( target.first == target.second ) {
569 // Manual target not set, find one automatically.
570 // BEWARE: Calling turret_data.fire on tripoint min coordinates starts a crash
571 // triggered at `trajectory.insert( trajectory.begin(), source )` at ranged.cpp:236
572 pt.reset_target( pos );
573 int boo_hoo;
574
575 // TODO: calculate chance to hit and cap range based upon this
576 int max_range = 20;
577 int range = std::min( gun.range(), max_range );
578 Creature *auto_target = cpu.auto_find_hostile_target( range, boo_hoo, area );
579 if( auto_target == nullptr ) {
580 if( boo_hoo ) {
581 cpu.name = string_format( pgettext( "vehicle turret", "The %s" ), pt.name() );
582 // check if the player can see or hear then print chooses a message accordingly
583 if( u_see && u_hear ) {
584 add_msg( m_warning, vgettext( "%s points in your direction and emits an IFF warning beep.",
585 "%s points in your direction and emits %d annoyed sounding beeps.",
586 boo_hoo ),
587 cpu.name, boo_hoo );
588 } else if( u_hear ) {
589 add_msg( m_warning, vgettext( "You hear a warning beep.",
590 "You hear %d annoyed sounding beeps.",
591 boo_hoo ), boo_hoo );
592 } else if( u_see ) {
593 add_msg( m_warning, _( "%s points in your direction." ), cpu.name );
594 }
595 }
596 return shots;
597 }
598
599 target.second = auto_target->pos();
600
601 } else {
602 // Target is already set, make sure we didn't move after aiming (it's a bug if we did).
603 if( pos != target.first ) {
604 target.second = target.first;
605 debugmsg( "%s moved after aiming but before it could fire.", cpu.name );
606 return shots;
607 }
608 }
609
610 // Get the turret's target and reset it
611 tripoint targ = target.second;
612 pt.reset_target( pos );
613
614 shots = gun.fire( cpu, targ );
615
616 if( shots && u_see && !g->u.sees( targ ) ) {
617 add_msg( _( "The %1$s fires its %2$s!" ), name, pt.name() );
618 }
619
620 return shots;
