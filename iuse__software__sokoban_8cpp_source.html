36void sokoban_game::parse_level( std::istream &fin )
37{
38 /*
39 # Wall
40 $ Package
41 space Floor
42 . Goal
43 * Package on Goal
44 @ Sokoban
45 + Sokoban on Goal
46 */
47
48 iCurrentLevel = 0;
49 iNumLevel = 0;
50
51 vLevel.clear();
52 vUndo.clear();
53 vLevelDone.clear();
54
55 std::string sLine;
56 while( !fin.eof() ) {
57 safe_getline( fin, sLine );
58
59 if( sLine.substr( 0, 3 ) == "; #" ) {
60 iNumLevel++;
61 continue;
62 } else if( sLine[0] == ';' ) {
63 continue;
64 }
65
66 if( sLine.empty() ) {
67 //Find level start
68 vLevel.resize( iNumLevel + 1 );
69 vLevelDone.resize( iNumLevel + 1 );
70 mLevelInfo[iNumLevel]["MaxLevelY"] = 0;
71 mLevelInfo[iNumLevel]["MaxLevelX"] = 0;
72 mLevelInfo[iNumLevel]["PlayerY"] = 0;
73 mLevelInfo[iNumLevel]["PlayerX"] = 0;
74 continue;
75 }
76
77 if( mLevelInfo[iNumLevel]["MaxLevelX"] < sLine.length() ) {
78 mLevelInfo[iNumLevel]["MaxLevelX"] = sLine.length();
79 }
80
81 for( size_t i = 0; i < sLine.length(); i++ ) {
82 if( sLine[i] == '@' ) {
83 if( mLevelInfo[iNumLevel]["PlayerY"] == 0 && mLevelInfo[iNumLevel]["PlayerX"] == 0 ) {
84 mLevelInfo[iNumLevel]["PlayerY"] = mLevelInfo[iNumLevel]["MaxLevelY"];
85 mLevelInfo[iNumLevel]["PlayerX"] = i;
86 } else {
87 // TODO: describe why it's invalid
88 throw std::runtime_error( "invalid content of sokoban file" );
89 }
90 }
91
92 if( sLine[i] == '.' || sLine[i] == '*' || sLine[i] == '+' ) {
93 vLevelDone[iNumLevel].push_back( std::make_pair( static_cast<int>
94 ( mLevelInfo[iNumLevel]["MaxLevelY"] ), static_cast<int>( i ) ) );
95 }
96
97 vLevel[iNumLevel][mLevelInfo[iNumLevel]["MaxLevelY"]][i] = sLine[i];
98 }
99
100 mLevelInfo[iNumLevel]["MaxLevelY"]++;
101 }
102}
