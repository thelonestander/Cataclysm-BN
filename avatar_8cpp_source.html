334int avatar::time_to_read( const item &book, const player &reader, const player *learner ) const
335{
336 const auto &type = book.type->book;
337 const skill_id &skill = type->skill;
338 // The reader's reading speed has an effect only if they're trying to understand the book as they read it
339 // Reading speed is assumed to be how well you learn from books (as opposed to hands-on experience)
340 const bool try_understand = reader.fun_to_read( book ) ||
341 reader.get_skill_level( skill ) < type->level;
342 int reading_speed = try_understand ? std::max( reader.read_speed(), read_speed() ) : read_speed();
343 if( learner ) {
344 reading_speed = std::max( reading_speed, learner->read_speed() );
345 }
346
347 int retval = type->time * reading_speed;
348 retval *= std::min( fine_detail_vision_mod(), reader.fine_detail_vision_mod() );
349
350 const int effective_int = std::min( { get_int(), reader.get_int(), learner ? learner->get_int() : INT_MAX } );
351 if( type->intel > effective_int && !reader.has_trait( trait_PROF_DICEMASTER ) ) {
352 retval += type->time * ( type->intel - effective_int ) * 100;
353 }
354 return retval;

652static void skim_book_msg( const item &book, avatar &u )
653{
654 if( !book.type->book ) {
655 return;
656 }
657 const auto &reading = book.type->book;
658 const skill_id &skill = reading->skill;
659
660 if( skill && u.get_skill_level_object( skill ).can_train() ) {
661 add_msg( m_info, _( "Can bring your %s skill to %d." ),
662 skill.obj().name(), reading->level );
663 if( reading->req != 0 ) {
664 add_msg( m_info, _( "Requires %s level %d to understand." ),
665 skill.obj().name(), reading->req );
666 }
667 }
668
669 if( reading->intel != 0 ) {
670 add_msg( m_info, _( "Requires intelligence of %d to easily read." ), reading->intel );
671 }
672 //It feels wrong to use a pointer to *this, but I can't find any other player pointers in this method.
673 if( u.book_fun_for( book, u ) != 0 ) {
674 add_msg( m_info, _( "Reading this book affects your morale by %d" ), u.book_fun_for( book, u ) );
675 }
676
677 if( book.type->use_methods.count( "MA_MANUAL" ) ) {
678 const matype_id style_to_learn = martial_art_learned_from( *book.type );
679 add_msg( m_info, _( "You can learn %s style from it." ), style_to_learn->name );
680 add_msg( m_info, _( "This fighting style is %s to learn." ),
681 martialart_difficulty( style_to_learn ) );
682 add_msg( m_info, _( "It would be easier to master if you'd have skill expertise in %s." ),
683 style_to_learn->primary_skill->name() );

685 to_string( time_duration::from_minutes( reading->time ) ) );
686 } else {
687 add_msg( m_info, vgettext( "A chapter of this book takes %d minute to read.",
688 "A chapter of this book takes %d minutes to read.", reading->time ),
689 reading->time );
690 }
691
692 std::vector<std::string> recipe_list;
693 for( const auto &elem : reading->recipes ) {
694 // If the player knows it, they recognize it even if it's not clearly stated.
695 if( elem.is_hidden() && !u.knows_recipe( elem.recipe ) ) {
696 continue;
697 }
698 recipe_list.push_back( elem.name );
699 }
700 if( !recipe_list.empty() ) {
701 std::string recipe_line =
702 string_format( vgettext( "This book contains %1$zu crafting recipe: %2$s",
703 "This book contains %1$zu crafting recipes: %2$s",
704 recipe_list.size() ),
705 recipe_list.size(),
706 enumerate_as_string( recipe_list ) );
707 add_msg( m_info, recipe_line );
708 }
709 if( recipe_list.size() != reading->recipes.size() ) {
710 add_msg( m_info, _( "It might help you figuring out some more recipes." ) );
711 }
712
713 add_msg( _( "You note that you have a copy of %s in your possession." ), book.type_name() );

716void avatar::do_read( item_location loc )
717{
718 if( !loc ) {
719 activity.set_to_null();
720 return;
721 }
722
723 item &book = *loc;
724 const auto &reading = book.type->book;
725 if( !reading ) {
726 activity.set_to_null();
727 return;
728 }
729 const skill_id &skill = reading->skill;
730
731 if( !has_identified( book.typeId() ) ) {
732 // Note that we've read the book.
733 items_identified.insert( book.typeId() );
734 skim_book_msg( book, *this );
735 activity.set_to_null();
736 return;
737 }
738
739 const bool allow_recipes = get_option<bool>( "ALLOW_LEARNING_BOOK_RECIPES" );
740
741 //learners and their penalties
742 std::vector<std::pair<player *, double>> learners;
743 for( size_t i = 0; i < activity.values.size(); i++ ) {
744 player *n = g->find_npc( character_id( activity.values[i] ) );
745 if( n != nullptr ) {
746 const std::string &s = activity.get_str_value( i, "1" );
747 learners.push_back( { n, strtod( s.c_str(), nullptr ) } );
748 }
749 // Otherwise they must have died/teleported or something
750 }
751 learners.push_back( { this, 1.0 } );
752 //whether to continue reading or not
753 bool continuous = false;
754 // NPCs who learned a little about the skill
755 std::set<std::string> little_learned;
756 std::set<std::string> cant_learn;
757 std::list<std::string> out_of_chapters;
758
759 for( auto &elem : learners ) {
760 player *learner = elem.first;
761
762 if( book_fun_for( book, *learner ) != 0 ) {
763 //Fun bonus is no longer calculated here.
764 learner->add_morale( MORALE_BOOK, book_fun_for( book, *learner ) * 5, book_fun_for( book,
765 *learner ) * 15, 1_hours, 30_minutes, true,
766 book.type );
767 }
768
769 book.mark_chapter_as_read( *learner );
770
771 const auto available_recipes = book.get_available_recipes( *learner );
772 std::vector<const recipe *> learnable_recipes;
773 for( const std::pair<const recipe *, int> &p : available_recipes ) {
774 if( allow_recipes && !learner->knows_recipe( p.first ) ) {
775 learnable_recipes.push_back( p.first );
776 learner->learn_recipe( p.first );
777 if( learner->is_player() ) {
778 add_msg( m_info, _( "You memorize a recipe for %s." ), p.first->result_name() );
779 }
780 }
781 }
782
783 if( skill && learner->get_skill_level( skill ) < reading->level &&
784 learner->get_skill_level_object( skill ).can_train() ) {
785 SkillLevel &skill_level = learner->get_skill_level_object( skill );
786 const int originalSkillLevel = skill_level.level();
787
788 // Calculate experience gained
789 /** @EFFECT_INT increases reading comprehension */
790 // Enhanced Memory Banks modestly boosts experience
791 int min_ex = std::max( 1, reading->time / 10 + learner->get_int() / 4 );
792 int max_ex = reading->time / 5 + learner->get_int() / 2 - originalSkillLevel;
793 if( has_active_bionic( bio_memory ) ) {
794 min_ex += 2;
795 }
796
797 min_ex = adjust_for_focus( min_ex );
798 max_ex = adjust_for_focus( max_ex );
799
800 if( max_ex < 2 ) {
801 max_ex = 2;
802 }
803 if( max_ex > 10 ) {
804 max_ex = 10;
805 }
806 if( max_ex < min_ex ) {
807 max_ex = min_ex;
808 }
809
810 min_ex *= ( originalSkillLevel + 1 ) * elem.second;
811 min_ex = std::max( min_ex, 1 );
812 max_ex *= ( originalSkillLevel + 1 ) * elem.second;
813 max_ex = std::max( min_ex, max_ex );
814
815 skill_level.readBook( min_ex, max_ex, reading->level );
816
817 std::string skill_name = skill.obj().name();
818
819 if( skill_level != originalSkillLevel ) {
820 g->events().send<event_type::gains_skill_level>(
821 learner->getID(), skill, skill_level.level() );
822 if( learner->is_player() ) {
823 add_msg( m_good, _( "You increase %s to level %d." ), skill.obj().name(),
824 originalSkillLevel + 1 );
825 } else {
826 add_msg( m_good, _( "%s increases their %s level." ), learner->disp_name(), skill_name );
827 }
828 } else {
829 //skill_level == originalSkillLevel
830 if( activity.index == learner->getID().get_value() ) {
831 continuous = true;
832 }
833 if( learner->is_player() ) {
834 add_msg( m_info, _( "You learn a little about %s! (%d%%)" ), skill_name, skill_level.exercise() );
835 } else {
836 little_learned.insert( learner->disp_name() );
837 }
838 }
839
840 if( learnable_recipes.empty() && ( skill_level == reading->level || !skill_level.can_train() ) ) {
841 if( learner->is_player() ) {
842 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
843 } else {
844 cant_learn.insert( learner->disp_name() );
845 }
846 }
847 } else if( learnable_recipes.empty() && skill ) {
848 if( learner->is_player() ) {
849 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
850 } else {
851 cant_learn.insert( learner->disp_name() );
852 }
853 }
854 }
855 //end for all learners
856
857 if( little_learned.size() == 1 ) {
858 add_msg( m_info, _( "%s learns a little about %s!" ), little_learned.begin()->c_str(),
859 skill.obj().name() );
860 } else if( !little_learned.empty() ) {
861 const std::string little_learned_msg = enumerate_as_string( little_learned );
862 add_msg( m_info, _( "%s learn a little about %s!" ), little_learned_msg, skill.obj().name() );
863 }
864
865 if( !cant_learn.empty() ) {
866 const std::string names = enumerate_as_string( cant_learn );
867 add_msg( m_info, _( "%s can no longer learn from %s." ), names, book.type_name() );
868 }
869 if( !out_of_chapters.empty() ) {
870 const std::string names = enumerate_as_string( out_of_chapters );
871 add_msg( m_info, _( "Rereading the %s isn't as much fun for %s." ),
872 book.type_name(), names );
873 if( out_of_chapters.front() == disp_name() && one_in( 6 ) ) {
874 add_msg( m_info, _( "Maybe you should find something new to read…" ) );
875 }
876 }
877
878 // NPCs can't learn martial arts from manuals (yet).
879 auto m = book.type->use_methods.find( "MA_MANUAL" );
880 if( m != book.type->use_methods.end() ) {
881 const matype_id style_to_learn = martial_art_learned_from( *book.type );
882 skill_id skill_used = style_to_learn->primary_skill;
883 int difficulty = std::max( 1, style_to_learn->learn_difficulty );
884 difficulty = std::max( 1, 20 + difficulty * 2 - get_skill_level( skill_used ) * 2 );
885 add_msg( m_debug, _( "Chance to learn one in: %d" ), difficulty );
886
887 if( one_in( difficulty ) ) {
888 m->second.call( *this, book, false, pos() );
889 continuous = false;
890 } else {
891 if( activity.index == getID().get_value() ) {
892 continuous = true;
893 switch( rng( 1, 5 ) ) {
894 case 1:
895 add_msg( m_info,
896 _( "You train the moves according to the book, but can't get a grasp of the style, so you start from the beginning." ) );
897 break;
898 case 2:
899 add_msg( m_info,
900 _( "This martial art is not easy to grasp. You start training the moves from the beginning." ) );
901 break;
902 case 3:
903 add_msg( m_info,
904 _( "You decide to read the manual and train even more. In martial arts, patience leads to mastery." ) );
905 break;
906 case 4:
907 case 5:
908 add_msg( m_info, _( "You try again. This training will finally pay off." ) );
909 break;
910 }
911 } else {
912 add_msg( m_info, _( "You train for a while." ) );
913 }
914 }
915 }
916
917 if( continuous ) {
918 activity.set_to_null();
919 read( loc, true );
920 if( activity ) {
921 return;
922 }
923 }
924
925 activity.set_to_null();
926}

1226bool avatar::wield( item &target )
1227{
1228 if( is_wielding( target ) ) {
1229 return true;
1230 }
1231
1232 if( !can_wield( target ).success() ) {
1233 return false;
1234 }
1235
1236 if( !unwield() ) {
1237 return false;
1238 }
1239 cached_info.erase( "weapon_value" );
1240 if( target.is_null() ) {
1241 return true;
1242 }
1243
1244 // Query whether to draw an item from a holster when attempting to wield the holster
1245 if( target.get_use( "holster" ) && !target.contents.empty() ) {
1246 //~ %1$s: weapon name, %2$s: holster name
1247 if( query_yn( pgettext( "holster", "Draw %1$s from %2$s?" ), target.get_contained().tname(),
1248 target.tname() ) ) {
1249 invoke_item( &target );
1250 return false;
1251 }
1252 }
1253
1254 // Wielding from inventory is relatively slow and does not improve with increasing weapon skill.
1255 // Worn items (including guns with shoulder straps) are faster but still slower
1256 // than a skilled player with a holster.
1257 // There is an additional penalty when wielding items from the inventory whilst currently grabbed.
1258
1259 bool worn = is_worn( target );
1260 int mv = item_handling_cost( target, true,
1261 worn ? INVENTORY_HANDLING_PENALTY / 2 : INVENTORY_HANDLING_PENALTY );
1262
1263 add_msg( m_debug, "wielding took %d moves", mv );
1264 moves -= mv;
1265
1266 if( has_item( target ) ) {
1267 weapon = i_rem( &target );
1268 } else {
1269 weapon = target;
1270 }
1271
1272 last_item = weapon.typeId();
