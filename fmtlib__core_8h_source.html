357/**
358 An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
359 subset of the API. ``fmt::basic_string_view`` is used for format strings even
360 if ``std::string_view`` is available to prevent issues when a library is
361 compiled with a different ``-std`` option than the client code (which is not
362 recommended).
363 */
364template <typename Char> class basic_string_view
365{
366 private:
367 const Char *data_;
368 size_t size_;
369
370 public:
371 using value_type = Char;
372 using iterator = const Char*;
373
374 constexpr basic_string_view() FMT_NOEXCEPT :
375 data_( nullptr ), size_( 0 ) {}
376
377 /** Constructs a string reference object from a C string and a size. */
378 constexpr basic_string_view( const Char *s, size_t count ) FMT_NOEXCEPT
379 :
380 data_( s ),
381 size_( count ) {}
382
383 /**
384 \rst
385 Constructs a string reference object from a C string computing
386 the size with ``std::char_traits<Char>::length``.
387 \endrst
388 */
389#if __cplusplus >= 201703L // C++17's char_traits::length() is constexpr.
390 FMT_CONSTEXPR
391#endif
392 basic_string_view( const Char *s )
393 : data_( s ), size_( std::char_traits<Char>::length( s ) ) {}
394
395 /** Constructs a string reference from a ``std::basic_string`` object. */
396 template <typename Traits, typename Alloc>
397 FMT_CONSTEXPR basic_string_view(
398 const std::basic_string<Char, Traits, Alloc> &s ) FMT_NOEXCEPT
399 :
400 data_( s.data() ),
401 size_( s.size() ) {}
402
403 template <typename S, FMT_ENABLE_IF( std::is_same<
404 S, detail::std_string_view<Char>>::value )>
405 FMT_CONSTEXPR basic_string_view( S s ) FMT_NOEXCEPT :
406 data_( s.data() ),
407 size_( s.size() ) {}
408
409 /** Returns a pointer to the string data. */
410 constexpr const Char *data() const {
411 return data_;
412 }
413
414 /** Returns the string size. */
415 constexpr size_t size() const {
416 return size_;
417 }
418
419 constexpr iterator begin() const {
420 return data_;
421 }
422 constexpr iterator end() const {
423 return data_ + size_;
424 }
425
426 constexpr const Char &operator[]( size_t pos ) const {
427 return data_[pos];
428 }
429
430 FMT_CONSTEXPR void remove_prefix( size_t n ) {
431 data_ += n;
432 size_ -= n;
433 }
434
435 // Lexicographically compare this string reference to other.
436 int compare( basic_string_view other ) const {
437 size_t str_size = size_ < other.size_ ? size_ : other.size_;
438 int result = std::char_traits<Char>::compare( data_, other.data_, str_size );
439 if( result == 0 ) {
440 result = size_ == other.size_ ? 0 : ( size_ < other.size_ ? -1 : 1 );
441 }
442 return result;
443 }
444
445 friend bool operator==( basic_string_view lhs, basic_string_view rhs ) {
446 return lhs.compare( rhs ) == 0;
447 }
448 friend bool operator!=( basic_string_view lhs, basic_string_view rhs ) {
449 return lhs.compare( rhs ) != 0;
450 }
451 friend bool operator<( basic_string_view lhs, basic_string_view rhs ) {
452 return lhs.compare( rhs ) < 0;
453 }
454 friend bool operator<=( basic_string_view lhs, basic_string_view rhs ) {
455 return lhs.compare( rhs ) <= 0;
456 }
457 friend bool operator>( basic_string_view lhs, basic_string_view rhs ) {
458 return lhs.compare( rhs ) > 0;
459 }
460 friend bool operator>=( basic_string_view lhs, basic_string_view rhs ) {
461 return lhs.compare( rhs ) >= 0;
462 }
463};

574/**
575 \rst
576 Parsing context consisting of a format string range being parsed and an
577 argument counter for automatic indexing.
578
579 You can use one of the following type aliases for common character types:
580
581 +-----------------------+-------------------------------------+
582 | Type | Definition |
583 +=======================+=====================================+
584 | format_parse_context | basic_format_parse_context<char> |
585 +-----------------------+-------------------------------------+
586 | wformat_parse_context | basic_format_parse_context<wchar_t> |
587 +-----------------------+-------------------------------------+
588 \endrst
589 */
590template <typename Char, typename ErrorHandler = detail::error_handler>
591class basic_format_parse_context : private ErrorHandler
592{
593 private:
594 basic_string_view<Char> format_str_;
595 int next_arg_id_;
596
597 public:
598 using char_type = Char;
599 using iterator = typename basic_string_view<Char>::iterator;
600
601 explicit constexpr basic_format_parse_context(
602 basic_string_view<Char> format_str, ErrorHandler eh = {},
603 int next_arg_id = 0 )
604 : ErrorHandler( eh ), format_str_( format_str ), next_arg_id_( next_arg_id ) {}
605
606 /**
607 Returns an iterator to the beginning of the format string range being
608 parsed.
609 */
610 constexpr iterator begin() const FMT_NOEXCEPT {
611 return format_str_.begin();
612 }
613
614 /**
615 Returns an iterator past the end of the format string range being parsed.
616 */
617 constexpr iterator end() const FMT_NOEXCEPT {
618 return format_str_.end();
619 }
620
621 /** Advances the begin iterator to ``it``. */
622 FMT_CONSTEXPR void advance_to( iterator it ) {
623 format_str_.remove_prefix( detail::to_unsigned( it - begin() ) );
624 }
625
626 /**
627 Reports an error if using the manual argument indexing; otherwise returns
628 the next argument index and switches to the automatic indexing.
629 */
630 FMT_CONSTEXPR int next_arg_id() {
631 // Don't check if the argument id is valid to avoid overhead and because it
632 // will be checked during formatting anyway.
633 if( next_arg_id_ >= 0 ) {
634 return next_arg_id_++;
635 }
636 on_error( "cannot switch from manual to automatic argument indexing" );
637 return 0;
638 }
639
640 /**
641 Reports an error if using the automatic argument indexing; otherwise
642 switches to the manual indexing.
643 */
644 FMT_CONSTEXPR void check_arg_id( int ) {
645 if( next_arg_id_ > 0 ) {
646 on_error( "cannot switch from automatic to manual argument indexing" );
647 } else {
648 next_arg_id_ = -1;
649 }
650 }
651
652 FMT_CONSTEXPR void check_arg_id( basic_string_view<Char> ) {}
653
654 FMT_CONSTEXPR void on_error( const char *message ) {
655 ErrorHandler::on_error( message );
656 }
657
658 constexpr ErrorHandler error_handler() const {
659 return *this;
660 }
661};

737 virtual void grow( size_t capacity ) = 0;

779 // Tries resizing the buffer to contain *count* elements. If T is a POD type
780 // the new elements may not be initialized.
781 void try_resize( size_t count ) {
782 try_reserve( count );
783 size_ = count <= capacity_ ? count : capacity_;
784 }
785
786 // Tries increasing the buffer capacity to *new_capacity*. It can increase the
787 // capacity by a smaller amount than requested but guarantees there is space
788 // for at least one additional element either by increasing the capacity or by
789 // flushing the buffer if it is full.
790 void try_reserve( size_t new_capacity ) {
791 if( new_capacity > capacity_ ) {
792 grow( new_capacity );
793 }
794 }

797 try_reserve( size_ + 1 );
798 ptr_[size_++] = value;
799 }

1479/**
1480 \rst
1481 Visits an argument dispatching to the appropriate visit method based on
1482 the argument type. For example, if the argument type is ``double`` then
1483 ``vis(value)`` will be called with the value of type ``double``.
1484 \endrst

1486template <typename Visitor, typename Context>
1487FMT_CONSTEXPR_DECL FMT_INLINE auto visit_format_arg(
1488 Visitor &&vis, const basic_format_arg<Context> &arg ) -> decltype( vis( 0 ) )
1489{
1490 using char_type = typename Context::char_type;
1491 switch( arg.type_ ) {
1492 case detail::type::none_type:
1493 break;
1494 case detail::type::int_type:
1495 return vis( arg.value_.int_value );
1496 case detail::type::uint_type:
1497 return vis( arg.value_.uint_value );
1498 case detail::type::long_long_type:
1499 return vis( arg.value_.long_long_value );
1500 case detail::type::ulong_long_type:
1501 return vis( arg.value_.ulong_long_value );
1502#if FMT_USE_INT128
1503 case detail::type::int128_type:
1504 return vis( arg.value_.int128_value );
1505 case detail::type::uint128_type:
1506 return vis( arg.value_.uint128_value );
1507#else
1508 case detail::type::int128_type:
1509 case detail::type::uint128_type:
1510 break;
1511#endif
1512 case detail::type::bool_type:
1513 return vis( arg.value_.bool_value );
1514 case detail::type::char_type:
1515 return vis( arg.value_.char_value );
1516 case detail::type::float_type:
1517 return vis( arg.value_.float_value );
1518 case detail::type::double_type:
1519 return vis( arg.value_.double_value );
1520 case detail::type::long_double_type:
1521 return vis( arg.value_.long_double_value );
1522 case detail::type::cstring_type:
1523 return vis( arg.value_.string.data );
1524 case detail::type::string_type:
1525 return vis( basic_string_view<char_type>( arg.value_.string.data,
1526 arg.value_.string.size ) );
1527 case detail::type::pointer_type:
1528 return vis( arg.value_.pointer );
1529 case detail::type::custom_type:
1530 return vis( typename basic_format_arg<Context>::handle( arg.value_.custom ) );
1531 }
1532 return vis( monostate() );
1533}

1878#if FMT_GCC_VERSION && FMT_GCC_VERSION < 409
1879// Workaround a GCC template argument substitution bug.
1880 : public basic_format_args<Context>

1888 detail::mapped_type_constant<T, Context>::value;

1906 std::vector<detail::named_arg_info<char_type>> named_info_;

1930 void emplace_arg( const detail::named_arg<char_type, T> &arg ) {
1931 if( named_info_.empty() ) {
1932 constexpr const detail::named_arg_info<char_type> *zero_ptr{nullptr};
1933 data_.insert( data_.begin(), {zero_ptr, 0} );
1934 }
1935 data_.emplace_back( detail::make_arg<Context>( detail::unwrap( arg.value ) ) );
1936 auto pop_one = []( std::vector<basic_format_arg<Context>> *data ) {
1937 data->pop_back();
1938 };
1939 std::unique_ptr<std::vector<basic_format_arg<Context>>, decltype( pop_one )>
1940 guard{&data_, pop_one};
1941 named_info_.push_back( {arg.name, static_cast<int>( data_.size() - 2u )} );
1942 data_[0].value_.named_args = {named_info_.data(), named_info_.size()};
1943 guard.release();

1955 **Example**::
1956

1958 store.push_back(42);

1961 std::string result = fmt::vformat("{} and {} and {}", store);
1962 \endrst

1977
1978 **Example**::
1979
1980 fmt::dynamic_format_arg_store<fmt::format_context> store;
1981 char str[] = "1234567890";
1982 store.push_back(std::cref(str));

1985 store.push_back(std::cref(a1));

1991 assert(result == "X234567890 and 42");

1995 static_assert(
1996 detail::is_named_arg<typename std::remove_cv<T>::type>::value ||

1998 "objects of built-in types and string views are always copied" );
1999 emplace_arg( arg.get() );

2040/**
2041 \rst
2042 A view of a collection of formatting arguments. To avoid lifetime issues it
2043 should only be used as a parameter type in type-erased functions such as
2044 ``vformat``::
2045
2046 void vlog(string_view format_str, format_args args); // OK
2047 format_args args = make_format_args(42); // Error: dangling reference
2048 \endrst
2049 */
2050template <typename Context> class basic_format_args
2051{
2052 public:
2053 using size_type = int;
2054 using format_arg = basic_format_arg<Context>;
2055
2056 private:
2057 // A descriptor that contains information about formatting arguments.
2058 // If the number of arguments is less or equal to max_packed_args then
2059 // argument types are passed in the descriptor. This reduces binary code size
2060 // per formatting function call.
2061 unsigned long long desc_;
2062 union {
2063 // If is_packed() returns true then argument values are stored in values_;
2064 // otherwise they are stored in args_. This is done to improve cache
2065 // locality and reduce compiled code size since storing larger objects
2066 // may require more code (at least on x86-64) even if the same amount of
2067 // data is actually copied to stack. It saves ~10% on the bloat test.
2068 const detail::value<Context> *values_;
2069 const format_arg *args_;
2070 };
2071
2072 bool is_packed() const {
2073 return ( desc_ & detail::is_unpacked_bit ) == 0;
2074 }
2075 bool has_named_args() const {
