482template<typename C, typename F>
483void load_season_array( const JsonObject &jo, const std::string &key, C &container, F load_func )
484{
485 if( !jo.has_member( key ) ) {
486 // Throw 'member not found' error
487 jo.get_member( key );
488
489 } else if( jo.has_string( key ) ) {
490 container.fill( load_func( jo.get_string( key ) ) );
491
492 } else if( jo.has_array( key ) ) {
493 auto arr = jo.get_array( key );
494 if( arr.size() == 1 ) {
495 container.fill( load_func( arr.get_string( 0 ) ) );
496
497 } else if( arr.size() == container.size() ) {
498 for( auto &e : container ) {
499 e = load_func( arr.next_string() );
500 }
501
502 } else {
503 jo.throw_error( "Incorrect number of entries", key );
504 }
505
506 } else {
507 jo.throw_error( "Expected string or array", key );
508 }
509}

516void map_data_common_t::load_symbol( const JsonObject &jo )
517{
518 if( jo.has_member( "copy-from" ) && looks_like.empty() ) {
519 looks_like = jo.get_string( "copy-from" );
520 }
521 jo.read( "looks_like", looks_like );
522
523 load_season_array( jo, "symbol", symbol_, [&jo]( const std::string & str ) {
524 if( str == "LINE_XOXO" ) {
525 return LINE_XOXO;
526 } else if( str == "LINE_OXOX" ) {
527 return LINE_OXOX;
528 } else if( str.length() != 1 ) {
529 jo.throw_error( "Symbol string must be exactly 1 character long.", "symbol" );
530 }
531 return static_cast<int>( str[0] );
532 } );
533
534 const bool has_color = jo.has_member( "color" );
535 const bool has_bgcolor = jo.has_member( "bgcolor" );
536 if( has_color && has_bgcolor ) {
537 jo.throw_error( "Found both color and bgcolor, only one of these is allowed." );
538 } else if( has_color ) {
539 load_season_array( jo, "color", color_, []( const std::string & str ) {
540 // has to use a lambda because of default params
541 return color_from_string( str );
542 } );
543 } else if( has_bgcolor ) {
544 load_season_array( jo, "bgcolor", color_, bgcolor_from_string );
545 } else {
546 jo.throw_error( R"(Missing member: one of: "color", "bgcolor" must exist.)" );
547 }
