76/**
77 * Returns a random entry in the container.
78 * The container must have a `size()` function and must support iterators as usual.
79 * For empty containers it returns the given default object.
80 * `C` is the container type,
81 * `D` is the type of the default value (which may differ from the return type)
82 * `V` is the type of the elements in the container.
83 * Note that this function does not return a reference because the default value could be
84 * a temporary object that is not valid after this function has left:
85 * \code random_entry( vect, std::string("default") ); \endcode
86 */
87template<typename C, typename D, typename V = typename C::value_type>
88inline V random_entry( const C &container, D default_value )
89{
90 if( container.empty() ) {
91 return default_value;
92 }
93 auto iter = container.begin();
94 std::advance( iter, rng( 0, container.size() - 1 ) );
95 return *iter;
96}
97/**
98 * Same as above but returns a reference to the chosen entry (if the container
99 * is not empty) or an empty `optional` object (if the container is empty).
100 * This function handles empty containers without requiring an instance of the
101 * contained type when container is empty.
102 */
103template<typename C>
104inline auto random_entry_opt( C &container ) ->
105cata::optional<decltype( std::ref( *container.begin() ) )>
106{
107 if( container.empty() ) {
108 return cata::nullopt;
109 }
110 auto iter = container.begin();
111 std::advance( iter, rng( 0, container.size() - 1 ) );
112 return std::ref( *iter );
113}
114/**
115 * Same as above, but returns a default constructed value if the container
116 * is empty.
117 */
118template<typename C, typename V = typename C::value_type>
119inline V random_entry( const C &container )
120{
121 if( container.empty() ) {
122 return V();
123 }
124 auto iter = container.begin();
125 std::advance( iter, rng( 0, container.size() - 1 ) );
126 return *iter;
127}
