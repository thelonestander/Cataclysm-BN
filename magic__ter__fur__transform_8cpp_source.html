60static void load_transform_results( const JsonObject &jsi, const std::string &json_key,
61 weighted_int_list<T> &list )
62{
63 if( jsi.has_string( json_key ) ) {
64 list.add( T( jsi.get_string( json_key ) ), 1 );
65 return;
66 }
67 for( const JsonValue entry : jsi.get_array( json_key ) ) {
68 if( entry.test_array() ) {
69 JsonArray inner = entry.get_array();
70 list.add( T( inner.get_string( 0 ) ), inner.get_int( 1 ) );
71 } else {
72 list.add( T( entry.get_string() ), 1 );
73 }
74 }
