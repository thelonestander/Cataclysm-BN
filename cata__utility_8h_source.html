150/**
151 * Linear interpolation: returns first argument if t is 0, second if t is 1, otherwise proportional to t.
152 * Does not clamp t, meaning it can return values lower than min (if t<0) or higher than max (if t>1).
153 */
154template<typename T>
155constexpr T lerp( const T &min, const T &max, float t )

175/**
176 * @brief Class used to access a list as if it were circular.
177 *
178 * Some times you just want to have a list loop around on itself.
179 * This wrapper class allows you to do that. It requires the list to exist
180 * separately, but that also means any changes to the list get propagated (both ways).
181 */
182template<typename T>
183class list_circularizer
184{
185 private:
186 unsigned int _index = 0;
187 std::vector<T> *_list;
188 public:
189 /** Construct list_circularizer from an existing std::vector. */
190 list_circularizer( std::vector<T> &_list ) : _list( &_list ) {
191 }
192
193 /** Advance list to next item, wrapping back to 0 at end of list */
194 void next() {
195 _index = ( _index == _list->size() - 1 ? 0 : _index + 1 );
196 }
197
198 /** Advance list to previous item, wrapping back to end at zero */
199 void prev() {
200 _index = ( _index == 0 ? _list->size() - 1 : _index - 1 );
201 }
202
203 /** Return list element at the current location */
204 T &cur() const {
205 // list could be null, but it would be a design time mistake and really, the callers fault.
206 return ( *_list )[_index];
207 }
208};
