220void body_part_type::load( const JsonObject &jo, const std::string & )
221{
222 mandatory( jo, was_loaded, "id", id );
223
224 mandatory( jo, was_loaded, "name", name );
225 // This is NOT the plural of `name`; it's a name refering to the pair of
226 // bodyparts which this bodypart belongs to, and thus should not be implemented
227 // using "vgettext" or "translation::make_plural". Otherwise, in languages
228 // without plural forms, translation of this string would indicate it
229 // to be a left or right part, while it is not.
230 optional( jo, was_loaded, "name_multiple", name_multiple );
231
232 mandatory( jo, was_loaded, "accusative", accusative );
233 // same as the above comment
234 optional( jo, was_loaded, "accusative_multiple", accusative_multiple );

292void body_part_type::check() const
293{
294 const auto &under_token = get_bp( token );
295 if( this != &under_token ) {
296 debugmsg( "Body part %s has duplicate token %d, mapped to %s", id.c_str(), token,
297 under_token.id.c_str() );
298 }
299
300 if( !id.is_null() && main_part.is_null() ) {
301 debugmsg( "Body part %s has unset main part", id.c_str() );
302 }
303
304 if( !id.is_null() && opposite_part.is_null() ) {
305 debugmsg( "Body part %s has unset opposite part", id.c_str() );
306 }
307
308 if( !main_part.is_valid() ) {
309 debugmsg( "Body part %s has invalid main part %s.", id.c_str(), main_part.c_str() );
310 }
311
312 if( !opposite_part.is_valid() ) {
313 debugmsg( "Body part %s has invalid opposite part %s.", id.c_str(), opposite_part.c_str() );
314 }
