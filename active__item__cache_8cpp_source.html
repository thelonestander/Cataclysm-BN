75std::vector<item_reference> active_item_cache::get_for_processing()
76{
77 std::vector<item_reference> items_to_process;
78 for( std::pair<const int, std::list<item_reference>> &kv : active_items ) {
79 // Rely on iteration logic to make sure the number is sane.
80 int num_to_process = kv.second.size() / kv.first;
81 std::list<item_reference>::iterator it = kv.second.begin();
82 for( ; it != kv.second.end() && num_to_process >= 0; ) {
83 if( it->item_ref ) {
84 items_to_process.push_back( *it );
85 --num_to_process;
86 ++it;
87 } else {
88 // The item has been destroyed, so remove the reference from the cache
89 it = kv.second.erase( it );
90 }
91 }
92 // Rotate the returned items to the end of their list so that the items that weren't
93 // returned this time will be first in line on the next call
94 kv.second.splice( kv.second.end(), kv.second, kv.second.begin(), it );
95 }
96 return items_to_process;
97}
