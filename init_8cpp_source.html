132shared_ptr_fast<std::istream> DynamicDataLoader::get_cached_stream( const std::string &path )
133{
134 assert( !finalized && "Cannot open data file after finalization." );
135 assert( stream_cache && "Stream cache is only available during finalization" );
136 shared_ptr_fast<std::istringstream> cached = stream_cache->cache.get( path, nullptr );
137 // Create a new stream if the file is not opened yet, or if some code is still
138 // using the previous stream (in such case, `cached` and `stream_cache` have
139 // two references to the stream, hence the test for > 2).
140 if( !cached ) {
141 cached = make_shared_fast<std::istringstream>( read_entire_file( path ) );
142 } else if( cached.use_count() > 2 ) {
143 cached = make_shared_fast<std::istringstream>( cached->str() );
144 }
145 stream_cache->cache.insert( 8, path, cached );
146 return cached;
147}

149void DynamicDataLoader::load_deferred( deferred_json &data )
150{
151 while( !data.empty() ) {
152 const size_t n = data.size();
153 auto it = data.begin();
154 for( size_t idx = 0; idx != n; ++idx ) {
155 if( !it->first.path ) {
156 debugmsg( "JSON source location has null path, data may load incorrectly" );
157 } else {
158 try {
159 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
160 JsonIn jsin( *stream, it->first );
161 JsonObject jo = jsin.get_object();
162 load_object( jo, it->second );
163 } catch( const JsonError &err ) {
164 debugmsg( "(json-error)\n%s", err.what() );
165 }
166 }
167 ++it;
168 }
169 data.erase( data.begin(), it );
170 if( data.size() == n ) {
171 for( const auto &elem : data ) {
172 if( !elem.first.path ) {
173 debugmsg( "JSON source location has null path when reporting circular dependency" );
174 } else {
175 try {
176 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
177 JsonIn jsin( *stream, elem.first );
178 jsin.error( "JSON contains circular dependency, this object is discarded" );
179 } catch( const JsonError &err ) {
180 debugmsg( "(json-error)\n%s", err.what() );
181 }
182 }
183 }
184 data.clear();
185 return; // made no progress on this cycle so abort
186 }
187 }

452void DynamicDataLoader::load_data_from_path( const std::string &path, const std::string &src,
453 loading_ui &ui )
454{
455 assert( !finalized && "Can't load additional data after finalization. Must be unloaded first." );
456 // We assume that each folder is consistent in itself,
457 // and all the previously loaded folders.
458 // E.g. the core might provide a vpart "frame-x"
459 // the first loaded mode might provide a vehicle that uses that frame
460 // But not the other way round.
461
462 // get a list of all files in the directory
463 str_vec files = get_files_from_path( ".json", path, true, true );
464 if( files.empty() ) {
465 std::ifstream tmp( path.c_str(), std::ios::in );
466 if( tmp ) {
467 // path is actually a file, don't checking the extension,
468 // assume we want to load this file anyway
469 files.push_back( path );
470 }
471 }
472 // iterate over each file
473 for( auto &files_i : files ) {
474 const std::string &file = files_i;
475 // open the file as a stream
476 cata_ifstream infile = std::move( cata_ifstream().mode( cata_ios_mode::binary ).open( file ) );
477 // and stuff it into ram
478 std::istringstream iss(
479 std::string(
480 ( std::istreambuf_iterator<char>( *infile ) ),
481 std::istreambuf_iterator<char>()
482 )
483 );
484 try {
485 // parse it
486 JsonIn jsin( iss, file );
487 load_all_from_json( jsin, src, ui, path, file );
488 } catch( const JsonError &err ) {
489 throw std::runtime_error( err.what() );
490 }
491 }
492}

494void DynamicDataLoader::load_all_from_json( JsonIn &jsin, const std::string &src, loading_ui &,
495 const std::string &base_path, const std::string &full_path )
496{
497 // TEMPORARY until 0.G: Remove single object support for consistency
498 if( jsin.test_object() ) {
499 // find type and dispatch single object
500 JsonObject jo = jsin.get_object();
501 load_object( jo, src, base_path, full_path );
502 jo.finish();
503 // if there's anything else in the file, it's an error.
504 jsin.eat_whitespace();
505 if( jsin.good() ) {
506 jsin.error( string_format( "expected single-object file but found '%c'", jsin.peek() ) );
507 }
508 } else if( jsin.test_array() ) {
509 jsin.start_array();
510 // find type and dispatch each object until array close
511 while( !jsin.end_array() ) {
512 JsonObject jo = jsin.get_object();
513 load_object( jo, src, base_path, full_path );
514 jo.finish();
515 }
516 } else {
517 // not an object or an array?
518 jsin.error( "expected object or array" );
519 }
