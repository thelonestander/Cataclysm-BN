463void npc::check_or_use_weapon_cbm( const bionic_id &cbm_id )
464{
465 // if we're already using a bio_weapon, keep using it
466 if( cbm_weapon_index >= 0 ) {
467 return;
468 }
469 const float allowed_ratio = static_cast<int>( rules.cbm_reserve ) / 100.0f;
470 const units::energy free_power = get_power_level() - get_max_power_level() * allowed_ratio;
471 if( free_power <= 0_J ) {
472 return;
473 }
474
475 int index = 0;
476 bool found = false;
477 for( bionic &i : *my_bionics ) {
478 if( i.id == cbm_id && !i.powered ) {
479 found = true;
480 break;
481 }
482 index += 1;
483 }
484 if( !found ) {
485 return;
486 }
487 bionic &bio = ( *my_bionics )[index];
488
489 if( bio.info().has_flag( flag_BIONIC_GUN ) ) {
490 const item cbm_weapon = item( bio.info().fake_item );
491 bool not_allowed = !rules.has_flag( ally_rule::use_guns ) ||
492 ( rules.has_flag( ally_rule::use_silent ) && !cbm_weapon.is_silent() );
493 if( is_player_ally() && not_allowed ) {
494 return;
495 }
496
497 const int ups_charges = charges_of( itype_UPS );
498 int ammo_count = weapon.ammo_remaining();
499 const int ups_drain = weapon.get_gun_ups_drain();
500 if( ups_drain > 0 ) {
501 ammo_count = std::min( ammo_count, ups_charges / ups_drain );
502 }
503 const int cbm_ammo = free_power / bio.info().power_activate;
504
505 if( weapon_value( weapon, ammo_count ) < weapon_value( cbm_weapon, cbm_ammo ) ) {
506 real_weapon = weapon;
507 weapon = cbm_weapon;
508 cbm_weapon_index = index;
509 }
510 } else if( bio.info().has_flag( flag_BIONIC_WEAPON ) && !weapon.has_flag( flag_NO_UNWIELD ) &&
511 free_power > bio.info().power_activate ) {
512 if( is_armed() ) {
513 stow_item( weapon );
514 }
515 if( get_player_character().sees( pos() ) ) {
516 add_msg( m_info, _( "%s activates their %s." ), disp_name(), bio.info().name );
517 }
518
519 weapon = item( bio.info().fake_item );
520 mod_power_level( -bio.info().power_activate );
521 bio.powered = true;
522 cbm_weapon_index = index;
523 }

1474 float effective_efficiency = fuel_efficiency;

1494/**

1496 * @param bio the bionic that is meant to be recharged.
1497 * @param amount the amount of power that is to be spent recharging the bionic.
1498 * @param factor multiplies the power cost per turn.
1499 * @param rate divides the number of turns we may charge (rate of 2 discharges in half the time).
1500 * @return indicates whether we successfully charged the bionic.
1501 */
1502static bool attempt_recharge( Character &p, bionic &bio, units::energy &amount, int factor = 1,
1503 int rate = 1 )
1504{

1507 bool recharged = false;
1508

1511 // Set the recharging cost and charge the bionic.
1512 amount = power_cost;
1513 // This is our first turn of charging, so subtract a turn from the recharge delay.
1514 bio.charge_timer = info.charge_time - rate;
1515 recharged = true;
1516 }
1517 }
1518
1519 return recharged;

1882// bionic manipulation chance of success
1883int bionic_manip_cos( float adjusted_skill, int bionic_difficulty )
1884{
1885 if( g->u.has_trait( trait_DEBUG_BIONICS ) ) {
1886 return 100;
1887 }
1888
1889 int chance_of_success = 0;
1890 // we will base chance_of_success on a ratio of skill and difficulty
1891 // when skill=difficulty, this gives us 1. skill < difficulty gives a fraction.
1892 float skill_difficulty_parameter = static_cast<float>( adjusted_skill /
1893 ( 4.0 * bionic_difficulty ) );
1894
1895 // when skill == difficulty, chance_of_success is 50%. Chance of success drops quickly below that
1896 // to reserve bionics for characters with the appropriate skill. For more difficult bionics, the
1897 // curve flattens out just above 80%
1898 chance_of_success = static_cast<int>( ( 100 * skill_difficulty_parameter ) /
1899 ( skill_difficulty_parameter + std::sqrt( 1 / skill_difficulty_parameter ) ) );
1900
1901 return chance_of_success;

2394void Character::bionics_install_failure( const std::string &installer,
2395 int difficulty, int success, float adjusted_skill )
2396{
2397 // "success" should be passed in as a negative integer representing how far off we
2398 // were for a successful install. We use this to determine consequences for failing.
2399 success = std::abs( success );
2400
2401 // failure level is decided by how far off the character was from a successful install, and
2402 // this is scaled up or down by the ratio of difficulty/skill. At high skill levels (or low
2403 // difficulties), only minor consequences occur. At low skill levels, severe consequences
2404 // are more likely.
2405 int failure_level = static_cast<int>( std::sqrt( success * 4.0 * difficulty / adjusted_skill ) );
2406 int fail_type = ( failure_level > 5 ? 5 : failure_level );
2407
2408 if( installer != "NOT_MED" ) {
2409 //~"Complications" is USian medical-speak for "unintended damage from a medical procedure".
2410 add_msg( m_neutral, _( "%s training helps to minimize the complications." ),
2411 installer );
2412 // In addition to the bonus, medical residents know enough OR protocol to avoid botching.
2413 // Take MD and be immune to faulty bionics.
2414 if( fail_type > 3 ) {
2415 fail_type = rng( 1, 3 );
2416 }
2417 }
2418
2419 switch( fail_type ) {
2420 case 0:
2421 case 1:
2422 do_damage_for_bionic_failure( 2, 5 );
2423 break;
2424 case 2:
2425 case 3:
2426 do_damage_for_bionic_failure( 5, difficulty * 5 );
2427 break;
2428 case 4:
2429 case 5: {
2430 std::vector<bionic_id> valid;
2431 std::copy_if( begin( faulty_bionics ), end( faulty_bionics ), std::back_inserter( valid ),
2432 [&]( const bionic_id & id ) {
2433 return !has_bionic( id );
2434 } );
2435
2436 // We've got all the bad bionics!
2437 if( valid.empty() ) {
2438 if( has_max_power() ) {
2439 units::energy old_power = get_max_power_level();
2440 add_msg( m_bad, _( "%s lose power capacity!" ), disp_name() );
2441 set_max_power_level( units::from_kilojoule( rng( 0,
2442 units::to_kilojoule( get_max_power_level() ) - 25 ) ) );
2443 if( get_max_power_level() < units::from_kilojoule( 0 ) ) {
2444 set_max_power_level( units::from_kilojoule( 0 ) );
2445 }
2446 if( is_player() ) {
2447 g->memorial().add(
2448 pgettext( "memorial_male", "Lost %d units of power capacity." ),
2449 pgettext( "memorial_female", "Lost %d units of power capacity." ),
2450 units::to_kilojoule( old_power - get_max_power_level() ) );
2451 }
2452 // If no faults available and no power capacity, downgrade to second-worst complication.
2453 } else {
2454 do_damage_for_bionic_failure( 5, difficulty * 5 );
2455 break;
2456 }
2457 } else {
2458 const bionic_id &id = random_entry( valid );
2459 add_bionic( id );
2460 g->events().send<event_type::installs_faulty_cbm>( getID(), id );
2461 add_msg( m_bad,
2462 _( "Complication in installation caused a malfunction - %s. Uninstall it to clear the malfunction." ),
2463 id.obj().name );
2464 }
2465 }
2466 break;
2467 }
