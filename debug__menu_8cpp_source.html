302 * This allows to have some menu elements at the same time in the main menu and in the debug menu.

305static int debug_menu_uilist( bool display_all_entries = true )
306{
307 std::vector<uilist_entry> menu = {
308 { uilist_entry( 1, true, 'i', _( "Info…" ) ) },
309 };

314 { uilist_entry( 2, true, 's', _( "Spawning…" ) ) },
315 { uilist_entry( 3, true, 'p', _( "Player…" ) ) },
316 { uilist_entry( 6, true, 'v', _( "Vehicle…" ) ) },

318 { uilist_entry( 5, true, 'm', _( "Map…" ) ) },
319 };
320
321 // insert debug-only menu right after "Info".
322 menu.insert( menu.begin() + 1, debug_menu.begin(), debug_menu.end() );
323 }
324
325 std::string msg;
326 if( display_all_entries ) {
327 msg = _( "Debug Functions - Using these will cheat not only the game, but yourself.\nYou won't grow. You won't improve.\nTaking this shortcut will gain you nothing. Your victory will be hollow.\nNothing will be risked and nothing will be gained." );
328 } else {

330 }
331

333 const int group = uilist( msg, menu );
334
335 int action;
336
337 switch( group ) {
338 case DEBUG_QUIT_NOSAVE:
339 action = DEBUG_QUIT_NOSAVE;
