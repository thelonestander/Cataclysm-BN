116void overmapbuffer::fix_mongroups( overmap &new_overmap )
117{
118 for( auto it = new_overmap.zg.begin(); it != new_overmap.zg.end(); ) {
119 auto &mg = it->second;
120 // spawn related code simply sets population to 0 when they have been
121 // transformed into spawn points on a submap, the group can then be removed
122 if( mg.empty() ) {
123 new_overmap.zg.erase( it++ );
124 continue;
125 }
126 // Inside the bounds of the overmap?
127 if( mg.pos.x() >= 0 && mg.pos.y() >= 0 && mg.pos.x() < OMAPX * 2 &&
128 mg.pos.y() < OMAPY * 2 ) {
129 ++it;
130 continue;
131 }
132 point_abs_sm smabs = project_combine( new_overmap.pos(), mg.pos.xy() );
133 point_abs_om omp;
134 point_om_sm sm_rem;
135 std::tie( omp, sm_rem ) = project_remain<coords::om>( smabs );
136 if( !has( omp ) ) {
137 // Don't generate new overmaps, as this can be called from the
138 // overmap-generating code.
139 ++it;
140 continue;
141 }
142 overmap &om = get( omp );
143 mg.pos = tripoint_om_sm( sm_rem, mg.pos.z() );
144 om.add_mon_group( mg );
145 new_overmap.zg.erase( it++ );
146 }

149void overmapbuffer::fix_npcs( overmap &new_overmap )
150{
151 // First step: move all npcs that are located outside of the given overmap
152 // into a separate container. After that loop, new_overmap.npcs is no
153 // accessed anymore!
154 decltype( overmap::npcs ) to_relocate;
155 for( auto it = new_overmap.npcs.begin(); it != new_overmap.npcs.end(); ) {
156 npc &np = **it;
157 const tripoint_abs_omt npc_omt_pos = np.global_omt_location();
158 const point_abs_om npc_om_pos = project_to<coords::om>( npc_omt_pos.xy() );
159 const point_abs_om loc = new_overmap.pos();
160 if( npc_om_pos == loc ) {
161 // Nothing to do
162 ++it;
163 continue;
164 }
165 to_relocate.push_back( *it );
166 it = new_overmap.npcs.erase( it );
167 }
168 // Second step: put them back where they belong. This step involves loading
169 // new overmaps (via `get`), which does in turn call this function for the
170 // newly loaded overmaps. This in turn may move NPCs from the second overmap
171 // back into the first overmap. This messes up the iteration of it. The
172 // iteration is therefore done in a separate step above (which does *not*
173 // involve loading new overmaps).
174 for( auto &ptr : to_relocate ) {
175 npc &np = *ptr;
176 const tripoint_abs_omt npc_omt_pos = np.global_omt_location();
177 const point_abs_om npc_om_pos = project_to<coords::om>( npc_omt_pos.xy() );
178 const point_abs_om loc = new_overmap.pos();
179 if( !has( npc_om_pos ) ) {
180 // This can't really happen without save editing
181 // We have no sane option here, just place the NPC on the edge
182 debugmsg( "NPC %s is out of bounds, on non-generated overmap %s",
183 np.name, loc.to_string() );
184 point_abs_sm npc_sm = project_to<coords::sm>( npc_om_pos );
185 point_abs_sm min = project_to<coords::sm>( loc );
186 point_abs_sm max =
187 project_to<coords::sm>( loc + point_south_east ) - point_south_east;
188 npc_sm.x() = clamp( npc_sm.x(), min.x(), max.x() );
189 npc_sm.y() = clamp( npc_sm.y(), min.y(), max.y() );
190 // TODO: fix point types
191 np.spawn_at_sm( tripoint_abs_sm( npc_sm, np.posz() ).raw() );
192 new_overmap.npcs.push_back( ptr );
193 continue;
194 }
195
196 // Simplest case: just move the pointer
197 get( npc_om_pos ).insert_npc( ptr );
198 }

258overmap *overmapbuffer::get_existing( const point_abs_om &p )
259{
260 if( last_requested_overmap && last_requested_overmap->pos() == p ) {
261 return last_requested_overmap;
262 }
263 const auto it = overmaps.find( p );
264 if( it != overmaps.end() ) {
265 return last_requested_overmap = it->second.get();
266 }
267 if( known_non_existing.count( p ) > 0 ) {
268 // This overmap does not exist on disk (this has already been
269 // checked in a previous call of this function).
270 return nullptr;
271 }
272 if( file_exist( terrain_filename( p ) ) ) {
273 // File exists, load it normally (the get function
274 // indirectly call overmap::open to do so).
275 return &get( p );
276 }
277 // File does not exist (or not readable which is essentially
278 // the same for our usage). A second call of this function with
279 // the same coordinates will not check the file system, and
280 // return early.
281 // If the overmap had been created in the mean time, the previous
282 // loop would have found and returned it.
283 known_non_existing.insert( p );
284 return nullptr;
285}

506std::vector<mongroup *> overmapbuffer::monsters_at( const tripoint_abs_omt &p )
507{
508 // (x,y) are overmap terrain coordinates, they spawn 2x2 submaps,
509 // but monster groups are defined with submap coordinates.
510 tripoint_abs_sm p_sm = project_to<coords::sm>( p );
511 std::vector<mongroup *> result;
512 for( const point &offset : std::array<point, 4> { { { point_zero }, { point_south }, { point_east }, { point_south_east } } } ) {
513 std::vector<mongroup *> tmp = groups_at( p_sm + offset );
514 result.insert( result.end(), tmp.begin(), tmp.end() );
515 }
516 return result;
517}

519std::vector<mongroup *> overmapbuffer::groups_at( const tripoint_abs_sm &p )
520{
521 std::vector<mongroup *> result;
522 point_om_sm sm_within_om;
523 point_abs_om omp;
524 std::tie( omp, sm_within_om ) = project_remain<coords::om>( p.xy() );
525 if( !has( omp ) ) {
526 return result;
527 }
528 overmap &om = get( omp );
529 auto groups_range = om.zg.equal_range( tripoint_om_sm( sm_within_om, p.z() ) );
530 for( auto it = groups_range.first; it != groups_range.second; ++it ) {
531 mongroup &mg = it->second;
532 if( mg.empty() ) {
533 continue;
534 }
535 result.push_back( &mg );
536 }
537 return result;
538}

683bool overmapbuffer::reveal( const tripoint_abs_omt &center, int radius,
684 const std::function<bool( const oter_id & )> &filter )
685{
686 int radius_squared = radius * radius;
687 bool result = false;
688 for( int i = -radius; i <= radius; i++ ) {
689 for( int j = -radius; j <= radius; j++ ) {
690 const tripoint_abs_omt p = center + point( i, j );
691 if( seen( p ) ) {
692 continue;
693 }
694 if( trigdist && i * i + j * j > radius_squared ) {
695 continue;
696 }
697 if( !filter( ter( p ) ) ) {
698 continue;
699 }
700 result = true;
701 set_seen( p, true );
702 }
703 }
704 return result;

823std::vector<tripoint_abs_omt> overmapbuffer::get_travel_path(
824 const tripoint_abs_omt &src, const tripoint_abs_omt &dest, overmap_path_params params )
825{
826 if( src == overmap::invalid_tripoint || dest == overmap::invalid_tripoint ) {
827 return {};
828 }
829
830 const pf::omt_scoring_fn estimate = [&]( tripoint_abs_omt pos ) {
831 const int cur_cost = pos == src ? 0 : get_terrain_cost( pos, params );
832 if( cur_cost < 0 ) {
833 return pf::omt_score::rejected;
834 }
835 return pf::omt_score( cur_cost, is_ramp( pos ) );
836 };
837
838 constexpr int radius = 4 * OMAPX; // radius of search in OMTs = 4 overmaps
839 const pf::simple_path<tripoint_abs_omt> path = pf::find_overmap_path( src, dest, radius, estimate );
840 return path.points;

843bool overmapbuffer::reveal_route( const tripoint_abs_omt &source, const tripoint_abs_omt &dest,
844 const omt_route_params &params )
845{
846 // Maximal radius of search (in overmaps)
847 static const int RADIUS = 4;
848 // half-size of the area to search in
849 static const point_rel_omt O( RADIUS * OMAPX, RADIUS * OMAPY );
850
851 if( source == overmap::invalid_tripoint || dest == overmap::invalid_tripoint ) {
852 return false;
853 }
854
855 // Local source - center of the local area
856 const point_rel_omt start( O );
857 // To convert local coordinates to global ones
858 const tripoint_abs_omt base = source - start;
859 // Local destination - relative to base
860 const point_rel_omt finish = ( dest - base ).xy();
861
862 const auto get_ter_at = [&]( const point_rel_omt & p ) {
863 return ter( base + p );
864 };
865
866 const oter_id oter = get_ter_at( start );
867 const auto connection = overmap_connections::guess_for( oter );
868
869 if( !connection ) {
870 return false;
871 }
872
873 const pf::two_node_scoring_fn<point_rel_omt> estimate =
874 [&]( pf::directed_node<point_rel_omt> cur, cata::optional<pf::directed_node<point_rel_omt>> ) {
875 int cost = 0;
876 const oter_id oter = get_ter_at( cur.pos );
877 if( !connection->has( oter ) ) {
878 if( params.road_only ) {
879 return pf::node_score::rejected;
880 }
881 if( is_river( oter ) ) {
882 return pf::node_score::rejected; // Can't walk on water
883 }
884 // Allow going slightly off-road to overcome small obstacles (e.g. craters),
885 // but heavily penalize that to make roads preferable
886 cost = 250;
887 }
888 return pf::node_score( cost, manhattan_dist( finish, cur.pos ) );
889 };
890
891 // TODO: use overmapbuffer::get_travel_path() with appropriate params instead
892 // TODO: tick params.popup
893 const auto path = pf::greedy_path( start, finish, O * 2, estimate );
894
895 for( const auto &node : path.nodes ) {
896 reveal( base + node.pos, params.radius );
897 }
898 return !path.nodes.empty();
899}

958 if( params.existing_only ) {
959 for( const std::pair<std::string, ot_match_type> &elem : params.types ) {
960 type_matches = check_ot_existing( elem.first, elem.second, location );
961 if( type_matches ) {
962 break;
963 }
964 }
965 } else {

967 type_matches = check_ot( elem.first, elem.second, location );
968 if( type_matches ) {

974 return false;
975 }
976

978 return false;
979 }
980 if( params.cant_see && seen( location ) ) {
981 return false;
982 }
983

985 bool meets_om_special = false;

988 } else {
989 meets_om_special = check_overmap_special_type( *params.om_special, location );
990 }
991 if( !meets_om_special ) {
992 return false;
993 }
994 }
995
996 return true;

1009tripoint_abs_omt overmapbuffer::find_closest( const tripoint_abs_omt &origin,
1010 const omt_find_params &params )
1011{
1012 // Check the origin before searching adjacent tiles!
1013 if( params.min_distance == 0 && is_findable_location( origin, params ) ) {
1014 return origin;
1015 }
1016
1017 // By default search overmaps within a radius of 4,
1018 // i.e. C = current overmap, X = overmaps searched:
1019 // XXXXXXXXX
1020 // XXXXXXXXX
1021 // XXXXXXXXX
1022 // XXXXXXXXX
1023 // XXXXCXXXX
1024 // XXXXXXXXX
1025 // XXXXXXXXX
1026 // XXXXXXXXX
1027 // XXXXXXXXX
1028 //
1029 // See overmap::place_specials for how we attempt to insure specials are placed within this
1030 // range. The actual number is 5 because 1 covers the current overmap,
1031 // and each additional one expends the search to the next concentric circle of overmaps.
1032 const int min_dist = params.min_distance;
1033 const int max_dist = params.search_range ? params.search_range : OMAPX * 5;
1034
1035 std::vector<tripoint_abs_omt> result;
1036 cata::optional<int> found_dist;
1037
1038 size_t num_overmaps = overmaps.size();
1039 size_t counter = 0;
1040
1041 for( const point_abs_omt &loc_xy : closest_points_first( origin.xy(), min_dist, max_dist ) ) {
1042 const int dist_xy = square_dist( origin.xy(), loc_xy );
1043
1044 if( found_dist && *found_dist < dist_xy ) {
1045 break;
1046 }
1047
1048 for( int z = -OVERMAP_DEPTH; z <= OVERMAP_HEIGHT; z++ ) {
1049 const tripoint_abs_omt loc( loc_xy, z );
1050 const int dist = square_dist( origin, loc );
1051
1052 if( found_dist && *found_dist < dist ) {
1053 continue;
1054 }
1055
1056 if( is_findable_location( loc, params ) ) {
1057 found_dist = dist;
1058 result.push_back( loc );
1059 }
1060
1061 counter += 1;
1062 if( params.popup && ( num_overmaps != overmaps.size() || counter == 512 ) ) {
1063 params.popup->refresh();
1064 num_overmaps = overmaps.size();
1065 counter = 0;
1066 }
1067 }
1068 }
1069
1070 return random_entry( result, overmap::invalid_tripoint );
1071}

1074 const omt_find_params &params )

1180std::vector<overmap *> overmapbuffer::get_overmaps_near( const tripoint_abs_sm &location,
1181 const int radius )
1182{
1183 // Grab the corners of a square around the target location at distance radius.
1184 // Convert to overmap coordinates and iterate from the minimum to the maximum.
1185 const point_abs_om start =
1186 project_to<coords::om>( location.xy() + point( -radius, -radius ) );
1187 const point_abs_om end =
1188 project_to<coords::om>( location.xy() + point( radius, radius ) );
1189 const point_rel_om offset = end - start;
1190
1191 std::vector<overmap *> result;
1192 result.reserve( ( offset.x() + 1 ) * ( offset.y() + 1 ) );
1193
1194 for( int x = start.x(); x <= end.x(); ++x ) {
1195 for( int y = start.y(); y <= end.y(); ++y ) {
1196 if( overmap *existing_om = get_existing( point_abs_om( x, y ) ) ) {
1197 result.emplace_back( existing_om );
1198 }
1199 }
1200 }
1201
1202 // Sort the resulting overmaps so that the closest ones are first.
1203 const tripoint_abs_om center = project_to<coords::om>( location );
1204 std::sort( result.begin(), result.end(), [&center]( const overmap * lhs,
1205 const overmap * rhs ) {
1206 const tripoint_abs_om lhs_pos( lhs->pos(), 0 );
1207 const tripoint_abs_om rhs_pos( rhs->pos(), 0 );
1208 return trig_dist( center, lhs_pos ) < trig_dist( center, rhs_pos );
1209 } );
1210
1211 return result;
1212}

1586 const tripoint_abs_omt &center, int radius )
1587{
1588 // First find the requested special. If it doesn't exist, we're done here.
1589 bool found = false;
1590 overmap_special special;
1591 for( const auto &elem : overmap_specials::get_all() ) {
1592 if( elem.id == special_id ) {
1593 special = elem;
1594 found = true;
1595 break;
1596 }
1597 }
1598 if( !found ) {
1599 return false;
1600 }
1601
1602 // Force our special to occur just once when we're spawning it here.
1603 special.occurrences.min = 1;
1604 special.occurrences.max = 1;
1605
1606 // Figure out the longest side of the special for purposes of determining our sector size
1607 // when attempting placements.
1608 const auto calculate_longest_side = [&special]() {
1609 auto min_max_x = std::minmax_element( special.terrains.begin(),
1610 special.terrains.end(), []( const overmap_special_terrain & lhs,
1611 const overmap_special_terrain & rhs ) {
1612 return lhs.p.x < rhs.p.x;
1613 } );
1614
1615 auto min_max_y = std::minmax_element( special.terrains.begin(),
1616 special.terrains.end(), []( const overmap_special_terrain & lhs,
1617 const overmap_special_terrain & rhs ) {
1618 return lhs.p.y < rhs.p.y;
1619 } );
1620
1621 const int special_longest_side = std::max( min_max_x.second->p.x - min_max_x.first->p.x,
1622 min_max_y.second->p.y - min_max_y.first->p.y ) + 1;
1623
1624 // If our longest side is greater than the OMSPEC_FREQ, just use that instead.
1625 return std::min( special_longest_side, OMSPEC_FREQ );
1626 };
1627
1628 const int longest_side = calculate_longest_side();
1629
1630 // Predefine our sectors to search in.
1631 om_special_sectors sectors = get_sectors( longest_side );
1632
1633 // Get all of the overmaps within the defined radius of the center.
1634 for( const auto &om : get_overmaps_near(
1635 project_to<coords::sm>( center ), omt_to_sm_copy( radius ) ) ) {
1636
1637 // Build an overmap_special_batch for the special on this overmap.
1638 std::vector<const overmap_special *> specials;
1639 specials.push_back( &special );

1641
1642 // Filter the sectors to those which are in in range of our center point, so
1643 // that we don't end up creating specials in areas that are outside of our radius,
1644 // since the whole point is to create a special that is within the parameters.
1645 std::vector<point_om_omt> sector_points_in_range;
1646 std::copy_if( sectors.sectors.begin(), sectors.sectors.end(),
1647 std::back_inserter( sector_points_in_range ), [&]( point_om_omt & p ) {
1648 const point_abs_omt global_sector_point = project_combine( om->pos(), p );
1649 // We'll include this sector if it's within our radius. We reduce the radius by
1650 // the length of the longest side of our special so that we don't end up in a
1651 // scenario where one overmap terrain of the special is within the radius but the
1652 // rest of it is outside the radius (due to size, rotation, etc), which would
1653 // then result in us placing the special but then not finding it later if we
1654 // search using the same radius value we used in placing it.
1655 return square_dist( global_sector_point, center.xy() ) <= radius - longest_side;
1656 } );
1657 om_special_sectors sectors_in_range {sector_points_in_range, sectors.sector_width};

1659 // Attempt to place the specials using our batch and sectors. We
1660 // require they be placed in unexplored terrain right now.
1661 om->place_specials_pass( batch, sectors_in_range, true, true );

1664 // maximum number of instances so first check if its been erased.
1665 if( batch.empty() ) {

1667 }

1670 for( const auto &special_placement : batch ) {

1672 // It was placed, lets get outta here.

1674 }
