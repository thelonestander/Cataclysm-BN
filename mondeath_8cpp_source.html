125static void scatter_chunks( const itype_id &chunk_name, int chunk_amt, monster &z, int distance,
126 int pile_size = 1 )
127{
128 // can't have less than one item in a pile or it would cause an infinite loop
129 pile_size = std::max( pile_size, 1 );
130 // can't have more items in a pile than total items
131 pile_size = std::min( chunk_amt, pile_size );
132 distance = std::abs( distance );
133 const item chunk( chunk_name, calendar::turn, pile_size );
134 for( int i = 0; i < chunk_amt; i += pile_size ) {
135 bool drop_chunks = true;
136 tripoint tarp( z.pos() + point( rng( -distance, distance ), rng( -distance, distance ) ) );
137 const auto traj = line_to( z.pos(), tarp );
138
139 for( size_t j = 0; j < traj.size(); j++ ) {
140 tarp = traj[j];
141 if( one_in( 2 ) && z.bloodType() ) {
142 g->m.add_splatter( z.bloodType(), tarp );
143 } else if( z.gibType() ) {
144 g->m.add_splatter( z.gibType(), tarp, rng( 1, j + 1 ) );
145 }
146 if( g->m.impassable( tarp ) ) {
147 g->m.bash( tarp, distance );
148 if( g->m.impassable( tarp ) ) {
149 // Target is obstacle, not destroyed by bashing,
150 // stop trajectory in front of it, if this is the first
151 // point (e.g. wall adjacent to monster), don't drop anything on it
152 if( j > 0 ) {
153 tarp = traj[j - 1];
154 } else {
155 drop_chunks = false;
156 }
157 break;
158 }
159 }
160 }
161 if( drop_chunks ) {
162 g->m.add_item_or_charges( tarp, chunk );
163 }
164 }

167void mdeath::splatter( monster &z )
168{
169 map &here = get_map();
170 const bool gibbable = !z.type->has_flag( MF_NOGIB );
171
172 const int max_hp = std::max( z.get_hp_max(), 1 );
173 const float overflow_damage = std::max( -z.get_hp(), 0 );
174 const float corpse_damage = 2.5 * overflow_damage / max_hp;
175 bool pulverized = corpse_damage > 5 && overflow_damage > z.get_hp_max();
176 // make sure that full splatter happens when this is a set death function, not part of normal
177 for( const auto &deathfunction : z.type->dies ) {
178 if( deathfunction == mdeath::splatter ) {
179 pulverized = true;
180 }
181 }
182
183 const field_type_id type_blood = z.bloodType();
184 const field_type_id type_gib = z.gibType();
185
186 if( gibbable ) {
187 const auto area = here.points_in_radius( z.pos(), 1 );
188 int number_of_gibs = std::min( std::floor( corpse_damage ) - 1, 1 + max_hp / 5.0f );
189
190 if( pulverized && z.type->size >= MS_MEDIUM ) {
191 number_of_gibs += rng( 1, 6 );
192 sfx::play_variant_sound( "mon_death", "zombie_gibbed", sfx::get_heard_volume( z.pos() ) );
193 }
194
195 for( int i = 0; i < number_of_gibs; ++i ) {
196 if( type_blood ) {
197 here.add_splatter( type_gib, random_entry( area ), rng( 1, i + 1 ) );
198 }
199 if( type_gib ) {
200 here.add_splatter( type_gib, random_entry( area ) );
201 }
202 }
203 }
204 // 1% of the weight of the monster is the base, with overflow damage as a multiplier
205 int gibbed_weight = rng( 0, std::round( to_gram( z.get_weight() ) / 100.0 *
206 ( overflow_damage / max_hp + 1 ) ) );
207 const int z_weight = to_gram( z.get_weight() );
208 // limit gibbing to 15%
209 gibbed_weight = std::min( gibbed_weight, z_weight * 15 / 100 );
210
211 if( pulverized && gibbable ) {
212 float overflow_ratio = overflow_damage / max_hp + 1;
213 int gib_distance = std::round( rng( 2, 4 ) );
214 for( const auto &entry : *z.type->harvest ) {
215 // only flesh and bones survive.
216 if( entry.type == "flesh" || entry.type == "bone" ) {
217 // the larger the overflow damage, the less you get
218 itype_id item_id( entry.drop );
219 const int chunk_amt = entry.mass_ratio / overflow_ratio / 10 * z_weight /
220 to_gram( item_id->weight );
221 scatter_chunks( item_id, chunk_amt, z, gib_distance,
222 chunk_amt / ( gib_distance - 1 ) );
223 gibbed_weight -= entry.mass_ratio / overflow_ratio / 20 * z_weight;
224 }
225 }
226 // add corpse with gib flag
227 item corpse = item::make_corpse( z.type->id, calendar::turn, z.unique_name, z.get_upgrade_time() );
228 // Set corpse to damage that aligns with being pulped
229 corpse.set_damage( 4000 );
230 corpse.set_flag( "GIBBED" );
231 if( z.has_effect( effect_no_ammo ) ) {
232 corpse.set_var( "no_ammo", "no_ammo" );
233 }
234 here.add_item_or_charges( z.pos(), corpse );
235 }

408void mdeath::guilt( monster &z )
409{
410 const int MAX_GUILT_DISTANCE = 5;
411 int kill_count = g->get_kill_tracker().kill_count( z.type->id );
412 int maxKills = 100; // this is when the player stop caring altogether.
413
414 // different message as we kill more of the same monster
415 std::string msg = _( "You feel guilty for killing %s." ); // default guilt message
416 game_message_type msgtype = m_bad; // default guilt message type
417 std::map<int, std::string> guilt_tresholds;
418 guilt_tresholds[75] = _( "You feel ashamed for killing %s." );
419 guilt_tresholds[50] = _( "You regret killing %s." );
420 guilt_tresholds[25] = _( "You feel remorse for killing %s." );
421
422 if( g->u.has_trait( trait_PSYCHOPATH ) || g->u.has_trait_flag( "PRED3" ) ||
423 g->u.has_trait_flag( "PRED4" ) || g->u.has_trait( trait_KILLER ) ) {
424 return;
425 }
426 if( rl_dist( z.pos(), g->u.pos() ) > MAX_GUILT_DISTANCE ) {
427 // Too far away, we can deal with it.
428 return;
429 }
430 if( z.get_hp() >= 0 ) {
431 // We probably didn't kill it
432 return;
433 }
434 if( kill_count >= maxKills ) {
435 // player no longer cares
436 if( kill_count == maxKills ) {
437 //~ Message after killing a lot of monsters which would normally affect the morale negatively. %s is the monster name, it will be pluralized with a number of 100.
438 add_msg( m_good, _( "After killing so many bloody %s you no longer care "
439 "about their deaths anymore." ), z.name( maxKills ) );
440 }
441 return;
442 } else if( ( g->u.has_trait_flag( "PRED1" ) ) || ( g->u.has_trait_flag( "PRED2" ) ) ) {
443 msg = ( _( "Culling the weak is distasteful, but necessary." ) );
444 msgtype = m_neutral;
445 } else {
446 msgtype = m_bad;
447 for( auto &guilt_treshold : guilt_tresholds ) {
448 if( kill_count >= guilt_treshold.first ) {
449 msg = guilt_treshold.second;
450 break;
451 }
452 }
453 }
454
455 add_msg( msgtype, msg, z.name() );
456
457 int moraleMalus = -50 * ( 1.0 - ( static_cast<float>( kill_count ) / maxKills ) );
458 int maxMalus = -250 * ( 1.0 - ( static_cast<float>( kill_count ) / maxKills ) );
459 time_duration duration = 30_minutes * ( 1.0 - ( static_cast<float>( kill_count ) / maxKills ) );
460 time_duration decayDelay = 3_minutes * ( 1.0 - ( static_cast<float>( kill_count ) / maxKills ) );
461 if( z.type->in_species( ZOMBIE ) ) {
462 moraleMalus /= 10;
463 if( g->u.has_trait( trait_PACIFIST ) ) {
464 moraleMalus *= 5;
465 } else if( g->u.has_trait_flag( "PRED1" ) ) {
466 moraleMalus /= 4;
467 } else if( g->u.has_trait_flag( "PRED2" ) ) {
468 moraleMalus /= 5;
469 }
470 }
471 g->u.add_morale( MORALE_KILLED_MONSTER, moraleMalus, maxMalus, duration, decayDelay );
472
473}

671 }

676 calendar::turn );

678 }
679 }

776void mdeath::detonate( monster &z )
777{
778 weighted_int_list<itype_id> amm_list;
779 for( const auto &amm : z.ammo ) {
780 amm_list.add( amm.first, amm.second );
781 }
782
783 std::vector<itype_id> pre_dets;
784 for( int i = 0; i < 3; i++ ) {
785 if( amm_list.get_weight() <= 0 ) {
786 break;
787 }
788 // Grab one item
789 itype_id tmp = *amm_list.pick();
790 // and reduce its weight by 1
791 amm_list.add_or_replace( tmp, amm_list.get_specific_weight( tmp ) - 1 );
792 // and stash it for use
793 pre_dets.push_back( tmp );
794 }
795
796 // Update any hardcoded explosion equivalencies
797 std::vector<std::pair<std::string, long>> dets;
798 for( const itype_id &bomb_id : pre_dets ) {
799 if( bomb_id.str() == "bot_grenade_hack" ) {
800 dets.push_back( std::make_pair( "grenade_act", 5 ) );
801 } else if( bomb_id.str() == "bot_flashbang_hack" ) {
802 dets.push_back( std::make_pair( "flashbang_act", 5 ) );
803 } else if( bomb_id.str() == "bot_gasbomb_hack" ) {
804 dets.push_back( std::make_pair( "gasbomb_act", 20 ) );
805 } else if( bomb_id.str() == "bot_c4_hack" ) {
806 dets.push_back( std::make_pair( "c4armed", 10 ) );
807 } else if( bomb_id.str() == "bot_mininuke_hack" ) {
808 dets.push_back( std::make_pair( "mininuke_act", 20 ) );
809 } else {
810 // Get the transformation item
811 const iuse_transform *actor = dynamic_cast<const iuse_transform *>(
812 bomb_id->get_use( "transform" )->get_actor_ptr() );
813 if( actor == nullptr ) {
814 // Invalid bomb item, move to the next ammo item
815 add_msg( m_debug, "Invalid bomb type in detonate mondeath for %s.", z.name() );
816 continue;
817 }
818 dets.emplace_back( actor->target.str(), actor->ammo_qty );
819 }
820 }
821
822 if( g->u.sees( z ) ) {
823 if( dets.empty() ) {
824 add_msg( m_info,
825 //~ %s is the possessive form of the monster's name
826 _( "The %s's hands fly to its pockets, but there's nothing left in them." ),
827 z.name() );
828 } else {
829 //~ %s is the possessive form of the monster's name
830 add_msg( m_bad, _( "The %s's hands fly to its remaining pockets, opening them!" ),
831 z.name() );
832 }
833 }
834 // HACK, used to stop them from having ammo on respawn
835 z.add_effect( effect_no_ammo, 1_turns, num_bp );
836
837 // First die normally
838 mdeath::normal( z );
839 // Then detonate our suicide bombs
840 for( const auto &bombs : dets ) {
841 item bomb_item( bombs.first, calendar::start_of_cataclysm );
842 bomb_item.charges = bombs.second;
843 bomb_item.active = true;
844 g->m.add_item_or_charges( z.pos(), bomb_item );
845 }
846}
