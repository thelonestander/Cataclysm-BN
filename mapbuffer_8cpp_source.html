103void mapbuffer::save( bool delete_after_save )
104{
105 assure_dir_exist( g->get_world_base_save_path() + "/maps" );
106
107 int num_saved_submaps = 0;
108 int num_total_submaps = submaps.size();
109
110 const tripoint map_origin = sm_to_omt_copy( g->m.get_abs_sub() );
111 const bool map_has_zlevels = g != nullptr && g->m.has_zlevels();
112
113 static_popup popup;
114
115 // A set of already-saved submaps, in global overmap coordinates.
116 std::set<tripoint> saved_submaps;
117 std::list<tripoint> submaps_to_delete;
118 static constexpr std::chrono::milliseconds update_interval( 500 );
119 auto last_update = std::chrono::steady_clock::now();
120
121 for( auto &elem : submaps ) {
122 auto now = std::chrono::steady_clock::now();
123 if( last_update + update_interval < now ) {
124 popup.message( _( "Please wait as the map saves [%d/%d]" ),
125 num_saved_submaps, num_total_submaps );
126 ui_manager::redraw();
127 refresh_display();
128 last_update = now;
129 }
130 // Whatever the coordinates of the current submap are,
131 // we're saving a 2x2 quad of submaps at a time.
132 // Submaps are generated in quads, so we know if we have one member of a quad,
133 // we have the rest of it, if that assumption is broken we have REAL problems.
134 const tripoint om_addr = sm_to_omt_copy( elem.first );
135 if( saved_submaps.count( om_addr ) != 0 ) {
136 // Already handled this one.
137 continue;
138 }
139 saved_submaps.insert( om_addr );
140
141 // A segment is a chunk of 32x32 submap quads.
142 // We're breaking them into subdirectories so there aren't too many files per directory.
143 // Might want to make a set for this one too so it's only checked once per save().
144 const std::string dirname = find_dirname( om_addr );
145 const std::string quad_path = find_quad_path( dirname, om_addr );
146
147 // delete_on_save deletes everything, otherwise delete submaps
148 // outside the current map.
149 const bool zlev_del = !map_has_zlevels && om_addr.z != g->get_levz();
150 save_quad( dirname, quad_path, om_addr, submaps_to_delete,
151 delete_after_save || zlev_del ||
152 om_addr.x < map_origin.x || om_addr.y < map_origin.y ||
153 om_addr.x > map_origin.x + HALF_MAPSIZE ||
154 om_addr.y > map_origin.y + HALF_MAPSIZE );
155 num_saved_submaps += 4;
156 }
157 for( auto &elem : submaps_to_delete ) {
158 remove_submap( elem );
159 }
160
161 get_distribution_grid_tracker().on_saved();
162}

164void mapbuffer::save_quad( const std::string &dirname, const std::string &filename,
165 const tripoint &om_addr, std::list<tripoint> &submaps_to_delete,
166 bool delete_after_save )
167{
168 std::vector<point> offsets;
169 std::vector<tripoint> submap_addrs;
170 offsets.push_back( point_zero );
171 offsets.push_back( point_south );
172 offsets.push_back( point_east );
173 offsets.push_back( point_south_east );
174
175 bool all_uniform = true;
176 for( auto &offsets_offset : offsets ) {
177 tripoint submap_addr = omt_to_sm_copy( om_addr );
178 submap_addr.x += offsets_offset.x;
179 submap_addr.y += offsets_offset.y;
180 submap_addrs.push_back( submap_addr );
181 submap *sm = submaps[submap_addr];
182 if( sm != nullptr && !sm->is_uniform ) {
183 all_uniform = false;
184 }
185 }
186
187 if( all_uniform ) {
188 // Nothing to save - this quad will be regenerated faster than it would be re-read
189 if( delete_after_save ) {
190 for( auto &submap_addr : submap_addrs ) {
191 if( submaps.count( submap_addr ) > 0 && submaps[submap_addr] != nullptr ) {
192 submaps_to_delete.push_back( submap_addr );
193 }
194 }
195 }
196
197 return;
198 }
199
200 // Don't create the directory if it would be empty
201 assure_dir_exist( dirname );
202 write_to_file( filename, [&]( std::ostream & fout ) {
203 JsonOut jsout( fout );
204 jsout.start_array();
205 for( auto &submap_addr : submap_addrs ) {
206 if( submaps.count( submap_addr ) == 0 ) {
207 continue;
208 }
209
210 submap *sm = submaps[submap_addr];
211
212 if( sm == nullptr ) {
213 continue;
214 }
215
216 jsout.start_object();
217
218 jsout.member( "version", savegame_version );
219 jsout.member( "coordinates" );
220
221 jsout.start_array();
222 jsout.write( submap_addr.x );
223 jsout.write( submap_addr.y );
224 jsout.write( submap_addr.z );
225 jsout.end_array();
226
227 sm->store( jsout );
228
229 jsout.end_object();
230
231 if( delete_after_save ) {
232 submaps_to_delete.push_back( submap_addr );
233 }
234 }
235
236 jsout.end_array();
237 } );
238}

242submap *mapbuffer::unserialize_submaps( const tripoint &p )
243{
244 // Map the tripoint to the submap quad that stores it.
245 const tripoint om_addr = sm_to_omt_copy( p );
246 const std::string dirname = find_dirname( om_addr );
247 std::string quad_path = find_quad_path( dirname, om_addr );
248
249 if( !file_exist( quad_path ) ) {
250 // Fix for old saves where the path was generated using std::stringstream, which
251 // did format the number using the current locale. That formatting may insert
252 // thousands separators, so the resulting path is "map/1,234.7.8.map" instead
253 // of "map/1234.7.8.map".
254 std::ostringstream buffer;
255 buffer << dirname << "/" << om_addr.x << "." << om_addr.y << "." << om_addr.z << ".map";
256 if( file_exist( buffer.str() ) ) {
257 quad_path = buffer.str();
258 }
259 }
260
261 using namespace std::placeholders;
262 if( !read_from_file_optional_json( quad_path, std::bind( &mapbuffer::deserialize, this, _1 ) ) ) {
263 // If it doesn't exist, trigger generating it.
264 return nullptr;
265 }
266 if( submaps.count( p ) == 0 ) {
267 debugmsg( "file %s did not contain the expected submap %d,%d,%d",
268 quad_path, p.x, p.y, p.z );
269 return nullptr;
270 }
271 return submaps[ p ];
272}

274void mapbuffer::deserialize( JsonIn &jsin )
275{
276 jsin.start_array();
277 while( !jsin.end_array() ) {
278 std::unique_ptr<submap> sm = std::make_unique<submap>();
279 tripoint submap_coordinates;
280 jsin.start_object();
281 int version = 0;
282 while( !jsin.end_object() ) {
283 std::string submap_member_name = jsin.get_member_name();
284 if( submap_member_name == "version" ) {
285 version = jsin.get_int();
286 } else if( submap_member_name == "coordinates" ) {
287 jsin.start_array();
288 tripoint loc{ jsin.get_int(), jsin.get_int(), jsin.get_int() };
289 jsin.end_array();
290 submap_coordinates = loc;
291 } else {
292 sm->load( jsin, submap_member_name, version );
293 }
294 }
295
296 if( !add_submap( submap_coordinates, sm ) ) {
297 debugmsg( "submap %d,%d,%d was already loaded", submap_coordinates.x, submap_coordinates.y,
298 submap_coordinates.z );
299 }
300 }
