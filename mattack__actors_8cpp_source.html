70bool leap_actor::call( monster &z ) const
71{
72 if( !z.can_act() || !z.move_effects( false ) ) {
73 return false;
74 }
75
76 std::vector<tripoint> options;
77 tripoint target = z.move_target();
78 float best_float = trigdist ? trig_dist( z.pos(), target ) : square_dist( z.pos(), target );
79 if( best_float < min_consider_range || best_float > max_consider_range ) {
80 return false;
81 }
82
83 // We wanted the float for range check
84 // int here will make the jumps more random
85 int best = std::numeric_limits<int>::max();
86 if( !allow_no_target && z.attack_target() == nullptr ) {
87 return false;
88 }
89 std::multimap<int, tripoint> candidates;
90 for( const tripoint &candidate : g->m.points_in_radius( z.pos(), max_range ) ) {
91 if( candidate == z.pos() ) {
92 continue;
93 }
94 float leap_dist = trigdist ? trig_dist( z.pos(), candidate ) :
95 square_dist( z.pos(), candidate );
96 if( leap_dist > max_range || leap_dist < min_range ) {
97 continue;
98 }
99 int candidate_dist = rl_dist( candidate, target );
100 if( candidate_dist >= best_float ) {
101 continue;
102 }
103 candidates.emplace( candidate_dist, candidate );
104 }
105 for( const auto &candidate : candidates ) {
106 const int &cur_dist = candidate.first;
107 const tripoint &dest = candidate.second;
108 if( cur_dist > best ) {
109 break;
110 }
111 if( !z.sees( dest ) ) {
112 continue;
113 }
114 if( !g->is_empty( dest ) ) {
115 continue;
116 }
117 bool blocked_path = false;
118 // check if monster has a clear path to the proposed point
119 std::vector<tripoint> line = g->m.find_clear_path( z.pos(), dest );
120 for( auto &i : line ) {
121 if( g->m.impassable( i ) ) {
122 blocked_path = true;
123 break;
124 }
125 }
126 // don't leap into water if you could drown (#38038)
127 if( z.is_aquatic_danger( dest ) ) {
128 blocked_path = true;
129 }
130 if( blocked_path ) {
131 continue;
132 }
133
134 options.push_back( dest );
135 best = cur_dist;
136 }
137
138 if( options.empty() ) {
139 return false; // Nowhere to leap!
140 }
141
142 z.moves -= move_cost;
143 const tripoint chosen = random_entry( options );
144 bool seen = g->u.sees( z ); // We can see them jump...
145 z.setpos( chosen );
146 seen |= g->u.sees( z ); // ... or we can see them land
147 if( seen ) {
148 add_msg( _( "The %s leaps!" ), z.name() );
149 }
150
151 return true;
152}

175bool mon_spellcasting_actor::call( monster &mon ) const
176{
177 if( !mon.can_act() ) {
178 return false;
179 }
180
181 if( !mon.attack_target() ) {
182 // this is an attack. there is no reason to attack if there isn't a real target.
183 return false;
184 }
185
186 const tripoint target = mon.attack_target()->pos();
187
188 std::string fx = spell_data.effect();
189 // is the spell an attack that needs to hit the target?
190 // examples of spells that don't: summons, teleport self
191 const bool targeted_attack = fx == "target_attack" || fx == "projectile_attack" ||
192 fx == "cone_attack" || fx == "line_attack";

195 return false;
196 }
197

200 target_name = target_monster->disp_name();
201 }
202
203 if( g->u.sees( target ) ) {
204 add_msg( spell_data.message(), mon.disp_name(), spell_data.name(), target_name );
