98{

101 aborted = true;
102 act.moves_left = 0;

104 }
105 avatar &you = get_avatar();
106
107 item *weapon = get_weapon();
108 if( !weapon || !avatar_action::can_fire_weapon( you, get_map(), *weapon ) ) {
109 aborted = true;
110 act.moves_left = 0;
111 return;
112 }
113
114 gun_mode gun = weapon->gun_current_mode();
115 if( first_turn && gun->has_flag( flag_RELOAD_AND_SHOOT ) && !gun->ammo_remaining() ) {
116 if( !load_RAS_weapon() ) {
117 aborted = true;
118 act.moves_left = 0;
119 return;
120 }
121 }
122 cata::optional<shape_factory> shape_gen;
123 if( weapon->ammo_current() && weapon->ammo_current()->ammo &&
124 weapon->ammo_current()->ammo->shape ) {
125 shape_gen = weapon->ammo_current()->ammo->shape;
126 }
127
128 g->temp_exit_fullscreen();
129 target_handler::trajectory trajectory;
130 if( !shape_gen ) {
131 trajectory = target_handler::mode_fire( you, *this );
132 } else {
133 trajectory = target_handler::mode_shaped( you, *shape_gen, *this );
134 }
135 g->reenter_fullscreen();
136
137 if( aborted ) {
138 act.moves_left = 0;
139 } else {
140 if( !trajectory.empty() ) {
141 fin_trajectory = trajectory;
142 act.moves_left = 0;
143 }
144 // If aborting on the first turn, keep 'first_turn' as 'true'.
145 // This allows refunding moves spent on unloading RELOAD_AND_SHOOT weapons
146 // to simulate avatar not loading them in the first place
147 first_turn = false;
148
149 // Allow interrupting activity only during 'aim and fire'.
150 // Prevents '.' key for 'aim for 10 turns' from conflicting with '.' key for 'interrupt activity'
151 // in case of high input lag (curses, sdl sometimes...), but allows to interrupt aiming
152 // if a bug happens / stars align to cause an endless aiming loop.
153 act.interruptable_with_kb = action != "AIM";

252{
253 // TODO: use activity for fetching ammo and loading weapon
254 player &you = get_avatar();
255 item *weapon = get_weapon();
256 gun_mode gun = weapon->gun_current_mode();
257 const auto ammo_location_is_valid = [&]() -> bool {
258 you.ammo_location.make_dirty();
259 if( !you.ammo_location )
260 {
261 return false;
262 }
263 if( !gun->can_reload_with( you.ammo_location->typeId() ) )
264 {
265 return false;
266 }
267 if( square_dist( you.pos(), you.ammo_location.position() ) > 1 )
268 {
269 return false;
270 }
271 return true;
272 };
273 item::reload_option opt = ammo_location_is_valid() ? item::reload_option( &you, weapon,
274 weapon, you.ammo_location ) : you.select_ammo( *gun );
275 if( !opt ) {
276 // Menu canceled
277 return false;
278 }
279 int reload_time = 0;
280 reload_time += opt.moves();
281 if( !gun->reload( you, std::move( opt.ammo ), 1 ) ) {
282 // Reload not allowed
283 return false;
284 }
285
286 // Burn 0.2% max base stamina x the strength required to fire.
287 you.mod_stamina( gun->get_min_str() * static_cast<int>( -0.002f *
288 get_option<int>( "PLAYER_MAX_STAMINA" ) ) );
289 // At low stamina levels, firing starts getting slow.
290 int sta_percent = ( 100 * you.get_stamina() ) / you.get_stamina_max();
291 reload_time += ( sta_percent < 25 ) ? ( ( 25 - sta_percent ) * 2 ) : 0;

764void move_items_activity_actor::do_turn( player_activity &act, Character &who )
765{
766 const tripoint dest = relative_destination + who.pos();
767
768 while( who.moves > 0 && !target_items.empty() ) {
769 item_location target = std::move( target_items.back() );
770 const int quantity = quantities.back();
771 target_items.pop_back();
772 quantities.pop_back();
773
774 if( !target ) {
775 debugmsg( "Lost target item of ACT_MOVE_ITEMS" );
776 continue;
777 }
778
779 // Don't need to make a copy here since movement can't be canceled
780 item &leftovers = *target;
781 // Make a copy to be put in the destination location
782 item newit = leftovers;
783
784 // Check that we can pick it up.
785 if( !newit.made_of( LIQUID ) ) {
786 // This is for hauling across zlevels, remove when going up and down stairs
787 // is no longer teleportation
788 if( newit.is_owned_by( who, true ) ) {
789 newit.set_owner( who );
790 } else {
791 continue;
792 }
793 // Handle charges, quantity == 0 means move all
794 if( quantity != 0 && newit.count_by_charges() ) {
795 newit.charges = std::min( newit.charges, quantity );
796 leftovers.charges -= quantity;
797 } else {
798 leftovers.charges = 0;
799 }
800 const tripoint src = target.position();
801 const int distance = src.z == dest.z ? std::max( rl_dist( src, dest ), 1 ) : 1;
802 who.mod_moves( -pickup::cost_to_move_item( who, newit ) * distance );
803 if( to_vehicle ) {
804 put_into_vehicle_or_drop( who, item_drop_reason::deliberate, { newit }, dest );
805 } else {
806 drop_on_map( who, item_drop_reason::deliberate, { newit }, dest );
807 }
808 // If we picked up a whole stack, remove the leftover item
809 if( leftovers.charges <= 0 ) {
810 target.remove_item();
811 }
812 }
813 }
814
815 if( target_items.empty() ) {
816 // Nuke the current activity, leaving the backlog alone.
817 act.set_to_null();
818 }
819}

847void pickup_activity_actor::do_turn( player_activity &act, Character &who )
848{
849 // If we don't have target items bail out
850 if( target_items.empty() ) {
851 who.cancel_activity();
852 return;
853 }
854
855 // If the player moves while picking up (i.e.: in a moving vehicle) cancel
856 // the activity, only populate starting_pos when grabbing from the ground
857 if( starting_pos && *starting_pos != who.pos() ) {
858 who.cancel_activity();
859 who.add_msg_if_player( _( "Moving canceled auto-pickup." ) );
860 return;
861 }
862
863 // Auto_resume implies autopickup.
864 const bool autopickup = who.activity.auto_resume;
865
866 // False indicates that the player canceled pickup when met with some prompt
867 const bool keep_going = pickup::do_pickup( target_items, autopickup );
868
869 // Check thievey witness
870 npc *witness = nullptr;
871 if( !act.str_values.empty() && act.str_values[0] == has_thievery_witness ) {
872 for( npc &guy : g->all_npcs() ) {
873 if( guy.get_attitude() == NPCATT_RECOVER_GOODS ) {
874 witness = &guy;
875 break;
876 }
877 }
878 }
879
880 // If there are items left we ran out of moves, so continue the activity
881 // Otherwise, we are done.
882 if( !keep_going || target_items.empty() || witness ) {
883 who.cancel_activity();
884
885 if( who.get_value( "THIEF_MODE_KEEP" ) != "YES" ) {
886 who.set_value( "THIEF_MODE", "THIEF_ASK" );
887 }
888
889 if( !keep_going ) {
890 // The user canceled the activity, so we're done
891 // AIM might have more pickup activities pending, also cancel them.
892 // TODO: Move this to advanced inventory instead of hacking it in here
893 cancel_aim_processing();
894 }
895
896 if( witness ) {
897 witness->talk_to_u();
898 // Then remove "has_thievery_witness" from the activity
899 act.str_values.clear();
900 }
901 }
