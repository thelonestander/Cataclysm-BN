633bool monexamine::give_items_to( monster &z )
634{
635 std::string pet_name = z.get_name();
636 if( !z.storage_item ) {
637 add_msg( _( "There is no container on your %s to put things in!" ), pet_name );
638 return true;
639 }
640
641 item &storage = *z.storage_item;
642 units::mass max_weight = z.weight_capacity() - z.get_carried_weight();
643 units::volume max_volume = storage.get_storage() - z.get_carried_volume();
644
645 drop_locations items = game_menus::inv::multidrop( g->u );
646 drop_locations to_move;
647 for( const drop_location &itq : items ) {
648 item it_copy = *itq.loc;
649 if( it_copy.count_by_charges() ) {
650 it_copy.charges = itq.count;
651 }
652 units::volume item_volume = it_copy.volume();
653 units::mass item_weight = it_copy.weight();
654 if( max_weight < item_weight ) {
655 add_msg( _( "The %1$s is too heavy for the %2$s to carry." ), it_copy.tname(), pet_name );
656 continue;
657 } else if( max_volume < item_volume ) {
658 add_msg( _( "The %1$s is too big to fit in the %2$s." ), it_copy.tname(), storage.tname() );
659 continue;
660 } else {
661 max_weight -= item_weight;
662 max_volume -= item_volume;
663 to_move.insert( to_move.end(), itq );
664 }
665 }
666 z.add_effect( effect_ai_waiting, 5_turns );
667 g->u.drop( to_move, z.pos(), true );
668
669 return false;
670}

755 return itm.has_flag( "TIE_UP" );

768 selection_menu.query();
769 auto index = selection_menu.ret;
770 if( index == 0 || index == UILIST_CANCEL || index < 0 ||
771 index > static_cast<int>( rope_inv.size() ) ) {

776 if( item_pos != INT_MIN ) {
777 z.tied_item = cata::make_value<item>( *rope_item );
