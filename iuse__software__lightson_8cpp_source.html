80void lightson_game::generate_change_coords( int changes )
81{
82 change_coords.resize( changes );
83 const int size = level_size.y * level_size.x;
84
85 point candidate;
86 for( int k = 0; k < changes; k++ ) {
87 do {
88 const int candidate_index = rng( 0, size - 1 );
89
90 candidate.x = candidate_index % level_size.x;
91 candidate.y = ( candidate_index - candidate.x ) / level_size.x;
92 // not accept repeatable coordinates
93 } while( !( k == 0 ||
94 std::find( change_coords.begin(), change_coords.end(), candidate ) == change_coords.end() ) );
95 change_coords[k] = candidate;
96 }
