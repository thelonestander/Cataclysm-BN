113int itype::invoke( player &p, item &it, const tripoint &pos, const std::string &iuse_name ) const
114{
115 const use_function *use = get_use( iuse_name );
116 if( use == nullptr ) {
117 debugmsg( "Tried to invoke %s on a %s, which doesn't have this use_function",
118 iuse_name, nname( 1 ) );
119 return 0;
120 }
121
122 const auto ret = use->can_call( p, it, false, pos );
123
124 if( !ret.success() ) {
125 p.add_msg_if_player( m_info, ret.str() );
126 return 0;

128
