215{

217

221 fselected = -1;
222
223 int f = 0;
224 int num_entries = entries.size();
225 for( int i = 0; i < num_entries; i++ ) {
226 if( filtering ) {
227 if( ignore_case ) {
228 if( !lcmatch( entries[i].txt, filter ) ) {
229 continue;
230 }
231 } else if( entries[i].txt.find( filter ) == std::string::npos ) {
232 continue;
233 }
234 }
235 fentries.push_back( i );
236 if( i == selected && ( hilight_disabled || entries[i].enabled ) ) {
237 fselected = f;
238 } else if( i > selected && fselected == -1 && ( hilight_disabled || entries[i].enabled ) ) {
239 // Past the previously selected entry, which has been filtered out,
240 // choose another nearby entry instead.
241 fselected = f;
242 }
243 f++;
244 }
245 if( fselected == -1 ) {
246 fselected = 0;
247 vshift = 0;
248 if( fentries.empty() ) {
249 selected = -1;
250 } else {
251 selected = fentries [ 0 ];
252 }

254 selected = fentries[fselected];
255 } else {
256 fselected = selected = -1;
257 }
258 // scroll to top of screen if all remaining entries fit the screen.
259 if( static_cast<int>( fentries.size() ) <= vmax ) {
260 vshift = 0;

330/**
331 * Find the minimum width between max( min_width, 1 ) and
332 * max( max_width, min_width, 1 ) to fold the string to no more than max_lines,
333 * or no more than the minimum number of lines possible, assuming that
334 * foldstring( width ).size() decreases monotonously with width.
335 **/
336static int find_minimum_fold_width( const std::string &str, int max_lines,
337 int min_width, int max_width )
338{
339 if( str.empty() ) {
340 return std::max( min_width, 1 );
341 }
342 min_width = std::max( min_width, 1 );
343 // max_width could be further limited by the string width, but utf8_width is
344 // not handling linebreaks properly.
345
346 if( min_width < max_width ) {
347 // If with max_width the string still folds to more than max_lines, find the
348 // minimum width that folds the string to such number of lines instead.
349 max_lines = std::max<int>( max_lines, foldstring( str, max_width ).size() );
350 while( min_width < max_width ) {
351 int width = ( min_width + max_width ) / 2;
352 // width may equal min_width, but will always be less than max_width.
353 int lines = foldstring( str, width ).size();
354 // If the current width folds the string to no more than max_lines
355 if( lines <= max_lines ) {
356 // The minimum width is between min_width and width.
357 max_width = width;
358 } else {
359 // The minimum width is between width + 1 and max_width.
360 min_width = width + 1;
361 }
362 // The new interval will always be smaller than the previous one,
363 // so the loop is guaranteed to end.
364 }
365 }
366 return min_width;
367}

751/**
752 * check for valid scrolling keypress and handle. return false if invalid keypress
753 */
754bool uilist::scrollby( const int scrollby )
755{
756 if( scrollby == 0 ) {
757 return false;
758 }
759
760 bool looparound = ( scrollby == -1 || scrollby == 1 );
761 bool backwards = ( scrollby < 0 );
762
763 fselected += scrollby;
764 if( !looparound ) {
765 if( backwards && fselected < 0 ) {
766 fselected = 0;
767 } else if( fselected >= static_cast<int>( fentries.size() ) ) {
768 fselected = fentries.size() - 1;
769 }
770 }
771
772 if( backwards ) {
773 if( fselected < 0 ) {
774 fselected = fentries.size() - 1;
775 }
776 for( size_t i = 0; i < fentries.size(); ++i ) {
777 if( hilight_disabled || entries[ fentries [ fselected ] ].enabled ) {
778 break;
779 }
780 --fselected;
781 if( fselected < 0 ) {
782 fselected = fentries.size() - 1;
783 }
784 }
785 } else {
786 if( fselected >= static_cast<int>( fentries.size() ) ) {
787 fselected = 0;
788 }
789 for( size_t i = 0; i < fentries.size(); ++i ) {
790 if( hilight_disabled || entries[ fentries [ fselected ] ].enabled ) {
791 break;
792 }
793 ++fselected;
794 if( fselected >= static_cast<int>( fentries.size() ) ) {
795 fselected = 0;
796 }
797 }
798 }
799 if( static_cast<size_t>( fselected ) < fentries.size() ) {
800 selected = fentries [ fselected ];
801 if( callback != nullptr ) {
802 callback->select( this );
803 }
804 }
805 return true;
806}

824/**
825 * Handle input and update display
826 *
827 */
828void uilist::query( bool loop, int timeout )
829{
830 if( test_mode ) {
831 debugmsg( "Tried to open UI in test mode" );
832 ret = UILIST_ERROR;
833 return;
834 }
835 keypress = 0;
836 if( entries.empty() ) {
837 ret = UILIST_ERROR;
838 return;
839 }
840 ret = UILIST_WAIT_INPUT;
841
842 input_context ctxt( input_category );
843 ctxt.register_updown();
844 ctxt.register_action( "PAGE_UP" );
845 ctxt.register_action( "PAGE_DOWN" );
846 ctxt.register_action( "SCROLL_UP" );
847 ctxt.register_action( "SCROLL_DOWN" );
848 if( allow_cancel ) {
849 ctxt.register_action( "QUIT" );
850 }
851 ctxt.register_action( "CONFIRM" );
852 ctxt.register_action( "FILTER" );
853 ctxt.register_action( "ANY_INPUT" );
854 ctxt.register_action( "HELP_KEYBINDINGS" );
855 for( const auto &additional_action : additional_actions ) {
856 ctxt.register_action( additional_action.first, additional_action.second );
857 }
858 hotkeys = ctxt.get_available_single_char_hotkeys( hotkeys );
859
860 shared_ptr_fast<ui_adaptor> ui = create_or_get_ui_adaptor();
861
862 ui_manager::redraw();
863
864#if defined(__ANDROID__)
865 for( const auto &entry : entries ) {
866 if( entry.hotkey > 0 && entry.enabled ) {
867 ctxt.register_manual_key( entry.hotkey, entry.txt );
868 }
869 }
870#endif
871
872 do {
873 ret_act = ctxt.handle_input( timeout );
874 const auto event = ctxt.get_raw_input();
875 keypress = event.get_first_input();
876 const auto iter = keymap.find( keypress );
877
878 if( scrollby( scroll_amount_from_action( ret_act ) ) ) {
879 /* nothing */
880 } else if( filtering && ret_act == "FILTER" ) {
881 inputfilter();
882 } else if( iter != keymap.end() ) {
883 selected = iter->second;
884 if( entries[ selected ].enabled ) {
885 ret = entries[ selected ].retval; // valid
886 } else if( allow_disabled ) {
887 ret = entries[selected].retval; // disabled
888 }
889 if( callback != nullptr ) {
890 callback->select( this );
891 }
892 } else if( !fentries.empty() && ret_act == "CONFIRM" ) {
893 if( entries[ selected ].enabled ) {
894 ret = entries[ selected ].retval; // valid
895 } else if( allow_disabled ) {
896 // disabled
897 ret = entries[selected].retval;
898 }
899 } else if( allow_cancel && ret_act == "QUIT" ) {
900 ret = UILIST_CANCEL;
901 } else if( ret_act == "TIMEOUT" ) {
902 ret = UILIST_TIMEOUT;
903 } else {
904 // including HELP_KEYBINDINGS, in case the caller wants to refresh their contents
905 bool unhandled = callback == nullptr || !callback->key( ctxt, event, selected, this );
906 if( unhandled && allow_anykey ) {
907 ret = UILIST_UNBOUND;
908 } else if( unhandled && allow_additional ) {
909 for( const auto &it : additional_actions ) {
910 if( it.first == ret_act ) {
911 ret = UILIST_ADDITIONAL;
912 break;
913 }
914 }
915 }
916 }
917
918 ui_manager::redraw();
919 } while( loop && ret == UILIST_WAIT_INPUT );
920}
