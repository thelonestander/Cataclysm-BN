35static std::string utf16_to_utf8( uint32_t ch )
36{
37 char out[5];
38 char *buf = out;
39 static const unsigned char utf8FirstByte[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
40 int utf8Bytes;
41 if( ch < 0x80 ) {
42 utf8Bytes = 1;
43 } else if( ch < 0x800 ) {
44 utf8Bytes = 2;
45 } else if( ch < 0x10000 ) {
46 utf8Bytes = 3;
47 } else if( ch <= 0x10FFFF ) {
48 utf8Bytes = 4;
49 } else {
50 std::stringstream err;
51 err << "unknown unicode: " << ch;
52 throw std::runtime_error( err.str() );
53 }
54
55 buf += utf8Bytes;
56 switch( utf8Bytes ) {
57 case 4:
58 *--buf = ( ch | 0x80 ) & 0xBF;
59 ch >>= 6;
60 /* fallthrough */
61 case 3:
62 *--buf = ( ch | 0x80 ) & 0xBF;
63 ch >>= 6;
64 /* fallthrough */
65 case 2:
66 *--buf = ( ch | 0x80 ) & 0xBF;
67 ch >>= 6;
68 /* fallthrough */
69 case 1:
70 *--buf = ch | utf8FirstByte[utf8Bytes];
71 }
72 out[utf8Bytes] = '\0';
73 return out;
74}

778void JsonIn::skip_separator()
779{
780 eat_whitespace();
781 signed char ch = peek();
782 if( ch == ',' ) {
783 if( ate_separator ) {
784 error( "duplicate comma" );
785 }
786 stream->get();
787 ate_separator = true;
788 } else if( ch == ']' || ch == '}' || ch == ':' ) {
789 // okay
790 if( ate_separator ) {
791 std::stringstream err;
792 err << "comma should not be found before '" << ch << "'";
793 uneat_whitespace();
794 error( err.str() );
795 }
796 ate_separator = false;
797 } else if( ch == EOF ) {
798 // that's okay too... probably
799 if( ate_separator ) {
800 uneat_whitespace();
801 error( "comma at end of file not allowed" );
802 }
803 ate_separator = false;
804 } else {
805 // not okay >:(
806 uneat_whitespace();
807 error( "missing comma", 1 );
808 }

851{

962static bool get_escaped_or_unicode( std::istream &stream, std::string &s, std::string &err )
963{
964 if( !stream.good() ) {
965 err = "stream not good";
966 return false;
967 }
968 char ch;
969 stream.get( ch );
970 if( !stream.good() ) {
971 err = "read operation failed";
972 return false;
973 }
974 if( ch == '\\' ) {
975 // converting \", \\, \/, \b, \f, \n, \r, \t and \uxxxx according to JSON spec.
976 stream.get( ch );
977 if( !stream.good() ) {
978 err = "read operation failed";
979 return false;
980 }
981 switch( ch ) {
982 case '\\':
983 s += '\\';
984 break;
985 case '"':
986 s += '"';
987 break;
988 case '/':
989 s += '/';
990 break;
991 case 'b':
992 s += '\b';
993 break;
994 case 'f':
995 s += '\f';
996 break;
997 case 'n':
998 s += '\n';
999 break;
1000 case 'r':
1001 s += '\r';
1002 break;
1003 case 't':
1004 s += '\t';
1005 break;
1006 case 'u': {
1007 uint32_t u = 0;
1008 for( int i = 0; i < 4; ++i ) {
1009 stream.get( ch );
1010 if( !stream.good() ) {
1011 err = "read operation failed";
1012 return false;
1013 }
1014 if( ch >= '0' && ch <= '9' ) {
1015 u = ( u << 4 ) | ( ch - '0' );
1016 } else if( ch >= 'a' && ch <= 'f' ) {
1017 u = ( u << 4 ) | ( ch - 'a' + 10 );
1018 } else if( ch >= 'A' && ch <= 'F' ) {
1019 u = ( u << 4 ) | ( ch - 'A' + 10 );
1020 } else {
1021 err = "expected hex digit";
1022 return false;
1023 }
1024 }
1025 try {
1026 s += utf16_to_utf8( u );
1027 } catch( const std::exception &e ) {
1028 err = e.what();
1029 return false;
1030 }
1031 }
1032 break;
1033 default:
1034 err = "invalid escape sequence";
1035 return false;
1036 }
1037 } else if( ch == '\r' || ch == '\n' ) {
1038 err = "reached end of line without closing string";
1039 return false;
1040 } else if( ch == '"' ) {
1041 // the caller is supposed to handle the ending quote
1042 err = "unexpected ending quote";
1043 return false;
1044 } else if( static_cast<unsigned char>( ch ) < 0x20 ) {
1045 err = "invalid character inside string";
1046 return false;
1047 } else {
1048 unsigned char uc = static_cast<unsigned char>( ch );
1049 uint32_t unicode = 0;
1050 int n = 0;
1051 if( uc >= 0xFC ) {
1052 unicode = uc & 0x01;
1053 n = 5;
1054 } else if( uc >= 0xF8 ) {
1055 unicode = uc & 0x03;
1056 n = 4;
1057 } else if( uc >= 0xF0 ) {
1058 unicode = uc & 0x07;
1059 n = 3;
1060 } else if( uc >= 0xE0 ) {
1061 unicode = uc & 0x0f;
1062 n = 2;
1063 } else if( uc >= 0xC0 ) {
1064 unicode = uc & 0x1f;
1065 n = 1;
1066 } else if( uc >= 0x80 ) {
1067 err = "invalid utf8 sequence";
1068 return false;
1069 } else {
1070 unicode = uc;
1071 n = 0;
1072 }
1073 s += ch;
1074 for( ; n > 0; --n ) {
1075 stream.get( ch );
1076 if( !stream.good() ) {
1077 err = "read operation failed";
1078 return false;
1079 }
1080 uc = static_cast<unsigned char>( ch );
1081 if( uc < 0x80 || uc >= 0xC0 ) {
1082 err = "invalid utf8 sequence";
1083 return false;
1084 }
1085 unicode = ( unicode << 6 ) | ( uc & 0x3f );
1086 s += ch;
1087 }
1088 if( unicode > 0x10FFFF ) {
1089 err = "invalid unicode codepoint";
1090 return false;
1091 }
1092 }
1093 return true;
1094}

1096std::string JsonIn::get_string()
1097{
1098 eat_whitespace();
1099 std::string s;
1100 char ch;
1101 std::string err;
1102 bool success = false;
1103 do {
1104 // the first character had better be a '"'
1105 stream->get( ch );
1106 if( !stream->good() ) {
1107 err = "read operation failed";
1108 break;
1109 }
1110 if( ch != '"' ) {
1111 err = "expected string but got '" + std::string( 1, ch ) + "'";
1112 break;
1113 }
1114 // add chars to the string, one at a time
1115 do {
1116 ch = stream->peek();
1117 if( !stream->good() ) {
1118 err = "read operation failed";
1119 break;
1120 }
1121 if( ch == '"' ) {
1122 stream->ignore();
1123 success = true;
1124 break;
1125 }
1126 if( !get_escaped_or_unicode( *stream, s, err ) ) {
1127 break;
1128 }
1129 } while( stream->good() );
1130 } while( false );
1131 if( success ) {
1132 end_value();
1133 return s;
1134 }
1135 if( stream->eof() ) {
1136 error( "couldn't find end of string, reached EOF." );
1137 } else if( stream->fail() ) {
1138 error( "stream failure while reading string." );
1139 } else {
1140 error( err, -1 );
1141 }
1142}

1270number_sci_notation JsonIn::get_any_number()
1271{
1272 // this could maybe be prettier?
1273 char ch;
1274 number_sci_notation ret;
1275 int mod_e = 0;
1276 eat_whitespace();
1277 stream->get( ch );
1278 if( ( ret.negative = ch == '-' ) ) {
1279 stream->get( ch );
1280 } else if( ch != '.' && ( ch < '0' || ch > '9' ) ) {
1281 // not a valid float
1282 std::stringstream err;
1283 err << "expecting number but found '" << ch << "'";
1284 error( err.str(), -1 );
1285 }
1286 if( ch == '0' ) {
1287 // allow a single leading zero in front of a '.' or 'e'/'E'
1288 stream->get( ch );
1289 if( ch >= '0' && ch <= '9' ) {
1290 error( "leading zeros not allowed", -1 );
1291 }
1292 }
1293 while( ch >= '0' && ch <= '9' ) {
1294 ret.number *= 10;
1295 ret.number += ( ch - '0' );
1296 stream->get( ch );
1297 }
1298 if( ch == '.' ) {
1299 stream->get( ch );
1300 while( ch >= '0' && ch <= '9' ) {
1301 ret.number *= 10;
1302 ret.number += ( ch - '0' );
1303 mod_e -= 1;
1304 stream->get( ch );
1305 }
1306 }
1307 if( ch == 'e' || ch == 'E' ) {
1308 stream->get( ch );
1309 bool neg;
1310 if( ( neg = ch == '-' ) ) {
1311 stream->get( ch );
1312 } else if( ch == '+' ) {
1313 stream->get( ch );
1314 }
1315 while( ch >= '0' && ch <= '9' ) {
1316 ret.exp *= 10;
1317 ret.exp += ( ch - '0' );
1318 stream->get( ch );
1319 }
1320 if( neg ) {
1321 ret.exp *= -1;
1322 }
1323 }
1324 // unget the final non-number character (probably a separator)
1325 stream->unget();
1326 end_value();
1327 ret.exp += mod_e;
1328 return ret;
1329}

1684 throw JsonError( err.str() );
1685 }
1686 // also print surrounding few lines of context, if not too large
1687 err << "\n\n";
1688 stream->seekg( offset, std::istream::cur );
1689 size_t pos = tell();
1690 rewind( 3, 240 );
1691 size_t startpos = tell();

1695 for( ; it < buffer.end() && ( *it == '\r' || *it == '\n' ); ++it ) {

1697 }

1703 }
1704 } else {
1705 err << *it;
1706 }
1707 }

1709 err << peek();
1710 }
1711 // display a pointer to the position
1712 rewind( 1, 240 );
1713 startpos = tell();
1714 err << '\n';
1715 if( pos > startpos ) {
1716 err << std::string( pos - startpos, ' ' );
1717 }
1718 err << "^\n";
1719 seek( pos );
1720 // if that wasn't the end of the line, continue underneath pointer
1721 char ch = stream->get();
1722 if( ch == '\r' ) {
1723 if( peek() == '\n' ) {
1724 stream->get();
1725 }
1726 } else if( ch == '\n' ) {
1727 // pass
1728 } else if( peek() != '\r' && peek() != '\n' && !stream->eof() ) {
1729 for( size_t i = 0; i < pos - startpos + 1; ++i ) {
1730 err << ' ';
1731 }
1732 }
1733 // print the next couple lines as well
1734 int line_count = 0;
1735 for( int i = 0; line_count < 3 && stream->good() && i < 240; ++i ) {
1736 stream->get( ch );
1737 if( !stream->good() ) {
1738 break;
1739 }

1744 stream->get( ch );
1745 }
