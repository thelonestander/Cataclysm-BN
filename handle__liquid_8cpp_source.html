53static void serialize_liquid_source( player_activity &act, const tripoint &pos, const item &liquid )
54{
55 const auto stack = g->m.i_at( pos );
56 // Need to store the *index* of the item on the ground, but it may be a virtual item from
57 // an infinite liquid source.
58 const auto iter = std::find_if( stack.begin(), stack.end(), [&]( const item & i ) {
59 return &i == &liquid;
60 } );
61 if( iter == stack.end() ) {
62 act.values.push_back( LST_INFINITE_MAP );
63 act.values.push_back( 0 ); // dummy
64 } else {
65 act.values.push_back( LST_MAP_ITEM );
66 act.values.push_back( std::distance( stack.begin(), iter ) );
67 }
68 act.coords.push_back( pos );

305static bool perform_liquid_transfer( item &liquid, const tripoint *const source_pos,
306 const vehicle *const source_veh, const int part_num,
307 const monster *const source_mon, liquid_dest_opt &target )
308{
309 bool transfer_ok = false;
310 if( !liquid.made_of( LIQUID ) ) {
311 debugmsg( "Tried to handle_liquid a non-liquid!" );
312 // "canceled by the user" because we *can* not handle it.
313 return transfer_ok;
314 }
315
316 const auto create_activity = [&]() {
317 if( source_veh != nullptr ) {
318 g->u.assign_activity( activity_id( "ACT_FILL_LIQUID" ) );
319 serialize_liquid_source( g->u.activity, *source_veh, part_num, liquid );
320 return true;
321 } else if( source_pos != nullptr ) {
322 g->u.assign_activity( activity_id( "ACT_FILL_LIQUID" ) );
323 serialize_liquid_source( g->u.activity, *source_pos, liquid );
324 return true;
325 } else if( source_mon != nullptr ) {
326 return false;
327 } else {
328 return false;
329 }
330 };
331
332 switch( target.dest_opt ) {
333 case LD_CONSUME:
334 g->u.consume_item( liquid );
335 transfer_ok = true;
336 break;
337 case LD_ITEM: {
338 item *const cont = target.item_loc.get_item();
339 const int item_index = g->u.get_item_position( cont );
340 // Currently activities can only store item position in the players inventory,
341 // not on ground or similar. TODO: implement storing arbitrary container locations.
342 if( item_index != INT_MIN && create_activity() ) {
343 serialize_liquid_target( g->u.activity, item_index );
344 } else if( g->u.pour_into( *cont, liquid ) ) {
345 if( cont->needs_processing() ) {
346 // Polymorphism fail, have to introspect into the type to set the target container as active.
347 switch( target.item_loc.where() ) {
348 case item_location::type::map:
349 g->m.make_active( target.item_loc );
350 break;
351 case item_location::type::vehicle:
352 g->m.veh_at( target.item_loc.position() )->vehicle().make_active( target.item_loc );
353 break;
354 case item_location::type::container:
355 case item_location::type::character:
356 case item_location::type::invalid:
357 break;
358 }
359 }
360 g->u.mod_moves( -100 );
361 }
362 transfer_ok = true;
363 break;
364 }
365 case LD_VEH:
366 if( target.veh == nullptr ) {
367 break;
368 }
369 if( create_activity() ) {
370 serialize_liquid_target( g->u.activity, *target.veh );
371 } else if( g->u.pour_into( *target.veh, liquid ) ) {
372 g->u.mod_moves( -1000 ); // consistent with veh_interact::do_refill activity
373 }
374 transfer_ok = true;
375 break;
376 case LD_KEG:
377 case LD_GROUND:
378 if( create_activity() ) {
379 serialize_liquid_target( g->u.activity, target.pos );
380 } else {
381 if( target.dest_opt == LD_KEG ) {
382 iexamine::pour_into_keg( target.pos, liquid );
383 } else {
384 g->m.add_item_or_charges( target.pos, liquid );
385 liquid.charges = 0;
386 }
387 g->u.mod_moves( -100 );
388 }
389 transfer_ok = true;
390 break;
391 case LD_NULL:
392 default:
393 break;
394 }
395 return transfer_ok;
396}

399 const tripoint *const source_pos,
400 const vehicle *const source_veh, const int part_num,
401 const monster *const source_mon )

403 if( liquid.made_of( SOLID ) ) {

411 }
412 struct liquid_dest_opt liquid_target;
413 if( get_liquid_target( liquid, source, radius, source_pos, source_veh, source_mon,
414 liquid_target ) ) {
415 return perform_liquid_transfer( liquid, source_pos, source_veh, part_num, source_mon,
416 liquid_target );
417 }
418 return false;
