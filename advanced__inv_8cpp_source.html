414struct advanced_inv_sorter {
415 advanced_inv_sortby sortby;
416 advanced_inv_sorter( advanced_inv_sortby sort ) {
417 sortby = sort;
418 }
419 bool operator()( const advanced_inv_listitem &d1, const advanced_inv_listitem &d2 ) {
420 // Note: the item pointer can only be null on sort by category, otherwise it is always valid.
421 switch( sortby ) {
422 case SORTBY_NONE:
423 if( d1.idx != d2.idx ) {
424 return d1.idx < d2.idx;
425 }
426 break;
427 case SORTBY_NAME:
428 // Fall through to code below the switch
429 break;
430 case SORTBY_WEIGHT:
431 if( d1.weight != d2.weight ) {
432 return d1.weight > d2.weight;
433 }
434 break;
435 case SORTBY_VOLUME:
436 if( d1.volume != d2.volume ) {
437 return d1.volume > d2.volume;
438 }
439 break;
440 case SORTBY_CHARGES:
441 if( d1.items.front()->charges != d2.items.front()->charges ) {
442 return d1.items.front()->charges > d2.items.front()->charges;
443 }
444 break;
445 case SORTBY_CATEGORY:
446 assert( d1.cat != nullptr );
447 assert( d2.cat != nullptr );
448 if( d1.cat != d2.cat ) {
449 return *d1.cat < *d2.cat;
450 } else if( d1.is_category_header() ) {
451 return true;
452 } else if( d2.is_category_header() ) {
453 return false;
454 }
455 break;
456 case SORTBY_DAMAGE:
457 if( d1.items.front()->damage() != d2.items.front()->damage() ) {
458 return d1.items.front()->damage() < d2.items.front()->damage();
459 }
460 break;
461 case SORTBY_AMMO: {
462 const std::string a1 = d1.items.front()->ammo_sort_name();
463 const std::string a2 = d2.items.front()->ammo_sort_name();
464 // There are many items with "false" ammo types (e.g.
465 // scrap metal has "components") that actually is not
466 // used as ammo, so we consider them as non-ammo.
467 const bool ammoish1 = !a1.empty() && a1 != "components" && a1 != "none" && a1 != "NULL";
468 const bool ammoish2 = !a2.empty() && a2 != "components" && a2 != "none" && a2 != "NULL";
469 if( ammoish1 != ammoish2 ) {
470 return ammoish1;
471 } else if( ammoish1 && ammoish2 ) {
472 if( a1 == a2 ) {
473 // For items with the same ammo type, we sort:
474 // guns > tools > magazines > ammunition
475 if( d1.items.front()->is_gun() && !d2.items.front()->is_gun() ) {
476 return true;
477 }
478 if( !d1.items.front()->is_gun() && d2.items.front()->is_gun() ) {
479 return false;
480 }
481 if( d1.items.front()->is_tool() && !d2.items.front()->is_tool() ) {
482 return true;
483 }
484 if( !d1.items.front()->is_tool() && d2.items.front()->is_tool() ) {
485 return false;
486 }
487 if( d1.items.front()->is_magazine() && d2.items.front()->is_ammo() ) {
488 return true;
489 }
490 if( d2.items.front()->is_magazine() && d1.items.front()->is_ammo() ) {
491 return false;
492 }
493 }
494 return localized_compare( a1, a2 );
495 }
496 }
497 break;
498 case SORTBY_SPOILAGE:
499 if( d1.items.front()->spoilage_sort_order() != d2.items.front()->spoilage_sort_order() ) {
500 return d1.items.front()->spoilage_sort_order() < d2.items.front()->spoilage_sort_order();
501 }
502 break;
503 case SORTBY_PRICE:
504 if( d1.items.front()->price( true ) != d2.items.front()->price( true ) ) {
505 return d1.items.front()->price( true ) > d2.items.front()->price( true );
506 }
507 break;
508 }
509 // secondary sort by name
510 const std::string *n1;
511 const std::string *n2;
512 if( d1.name_without_prefix == d2.name_without_prefix ) {
513 //if names without prefix equal, compare full name
514 n1 = &d1.name;
515 n2 = &d2.name;
516 } else {
517 //else compare name without prefix
518 n1 = &d1.name_without_prefix;
519 n2 = &d2.name_without_prefix;
520 }
521 return localized_compare( *n1, *n2 );
522 }

559void advanced_inventory::recalc_pane( side p )
560{
561 auto &pane = panes[p];
562 pane.recalc = false;
563 pane.items.clear();
564 // Add items from the source location or in case of all 9 surrounding squares,
565 // add items from several locations.
566 if( pane.get_area() == AIM_ALL ) {
567 auto &alls = squares[AIM_ALL];
568 auto &there = panes[-p + 1];
569 auto &other = squares[there.get_area()];
570 alls.volume = 0_ml;
571 alls.weight = 0_gram;
572 for( auto &s : squares ) {
573 // All the surrounding squares, nothing else
574 if( s.id < AIM_SOUTHWEST || s.id > AIM_NORTHEAST ) {
575 continue;
576 }
577
578 // To allow the user to transfer all items from all surrounding squares to
579 // a specific square, filter out items that are already on that square.
580 // e.g. left pane AIM_ALL, right pane AIM_NORTH. The user holds the
581 // enter key down in the left square and moves all items to the other side.
582 const bool same = other.is_same( s );
583
584 // Deal with squares with ground + vehicle storage
585 // Also handle the case when the other tile covers vehicle
586 // or the ground below the vehicle.
587 if( s.can_store_in_vehicle() && !( same && there.in_vehicle() ) ) {
588 bool do_vehicle = there.get_area() == s.id ? !there.in_vehicle() : true;
589 pane.add_items_from_area( s, do_vehicle );
590 alls.volume += s.volume;
591 alls.weight += s.weight;
592 }
593
594 // Add map items
595 if( !same || there.in_vehicle() ) {
596 pane.add_items_from_area( s );
597 alls.volume += s.volume;
598 alls.weight += s.weight;
599 }
600 }
601 } else {
602 pane.add_items_from_area( squares[pane.get_area()] );
603 }
604 // Insert category headers (only expected when sorting by category)
605 if( pane.sortby == SORTBY_CATEGORY ) {
606 std::set<const item_category *> categories;
607 for( auto &it : pane.items ) {
608 categories.insert( it.cat );
609 }
610 for( auto &cat : categories ) {
611 pane.items.push_back( advanced_inv_listitem( cat ) );
612 }
613 }
614 // Finally sort all items (category headers will now be moved to their proper position)
615 std::stable_sort( pane.items.begin(), pane.items.end(), advanced_inv_sorter( pane.sortby ) );
616 // itemsPerPage is 0 during processing
617 if( itemsPerPage > 0 ) {
618 pane.paginate( itemsPerPage );
619 }
620}

709bool advanced_inventory::move_all_items( bool nested_call )
710{
711 advanced_inventory_pane &spane = panes[src];
712 advanced_inventory_pane &dpane = panes[dest];
713
714 // AIM_ALL source area routine
715 if( spane.get_area() == AIM_ALL ) {
716 // move all to `AIM_WORN' doesn't make sense (see `MAX_WORN_PER_TYPE')
717 if( dpane.get_area() == AIM_WORN ) {
718 popup( _( "You look at the items, then your clothes, and scratch your head…" ) );
719 return false;
720 }
721 // if the source pane (AIM_ALL) is empty, then show a message and leave
722 if( !is_processing() && spane.items.empty() ) {
723 popup( _( "There are no items to be moved!" ) );
724 return false;
725 }
726
727 advanced_inv_area &sarea = squares[spane.get_area()];
728 advanced_inv_area &darea = squares[dpane.get_area()];
729
730 // Check first if the destination area still have enough room for moving all.
731 if( !is_processing() && sarea.volume > darea.free_volume( dpane.in_vehicle() ) &&
732 !query_yn( _( "There isn't enough room, do you really want to move all?" ) ) ) {
733 return false;
734 }
735
736 // make sure that there are items to be moved
737 bool done = false;
738 // copy the current pane, to be restored after the move is queued
739 advanced_inventory_pane shadow = panes[src];
740 // here we recursively call this function with each area in order to
741 // put all items in the proper destination area, with minimal fuss
742 int &loc = save_state->aim_all_location;
743 // re-entry nonsense
744 int &entry = save_state->re_enter_move_all;
745 // if we are just starting out, set entry to initial value
746 switch( static_cast<aim_entry>( entry++ ) ) {
747 case ENTRY_START:
748 ++entry;
749 /* fallthrough */
750 case ENTRY_VEHICLE:
751 if( squares[loc].can_store_in_vehicle() ) {
752 // either do the inverse of the pane (if it is the one we are transferring to),
753 // or just transfer the contents (if it is not the one we are transferring to)
754 spane.set_area( squares[loc], dpane.get_area() == loc ? !dpane.in_vehicle() : true );
755 // add items, calculate weights and volumes... the fun stuff
756 recalc_pane( src );
757 // then move the items to the destination area
758 move_all_items( true );
759 }
760 break;
761 case ENTRY_MAP:
762 spane.set_area( squares[loc++], false );
763 recalc_pane( src );
764 move_all_items( true );
765 break;
766 case ENTRY_RESET:
767 if( loc > AIM_AROUND_END ) {
768 loc = AIM_AROUND_BEGIN;
769 entry = ENTRY_START;
770 done = true;
771 } else {
772 entry = ENTRY_VEHICLE;
773 }
774 break;
775 default:
776 debugmsg( "Invalid `aim_entry' [%d] reached!", entry - 1 );
777 entry = ENTRY_START;
778 loc = AIM_AROUND_BEGIN;
779 return false;
780 }
781 // restore the pane to its former glory
782 panes[src] = shadow;
783 // make it auto loop back, if not already doing so
784 if( !done && !g->u.activity ) {
785 do_return_entry();
786 }
787 return true;
788 }
789
790 // Check some preconditions to quickly leave the function.
791 size_t liquid_items = 0;
792 for( const advanced_inv_listitem &elem : spane.items ) {
793 for( const item *elemit : elem.items ) {
794 if( elemit->made_of( LIQUID ) ) {
795 liquid_items++;
796 }
797 }
798 }
799 if( spane.items.empty() || liquid_items == spane.items.size() ) {
800 return false;
801 }
802 bool restore_area = false;
803 if( dpane.get_area() == AIM_ALL ) {
804 auto loc = dpane.get_area();
805 // ask where we want to store the item via the menu
806 if( !query_destination( loc ) ) {
807 return false;
808 }
809 restore_area = true;
810 }
811 if( !squares[dpane.get_area()].canputitems() ) {
812 popup( _( "You can't put items there!" ) );
813 return false;
814 }
815 advanced_inv_area &sarea = squares[spane.get_area()];
816 advanced_inv_area &darea = squares[dpane.get_area()];
817
818 // Make sure source and destination are different, otherwise items will disappear
819 // Need to check actual position to account for dragged vehicles
820 if( dpane.get_area() == AIM_DRAGGED && sarea.pos == darea.pos &&
821 spane.in_vehicle() == dpane.in_vehicle() ) {
822 return false;
823 } else if( spane.get_area() == dpane.get_area() && spane.in_vehicle() == dpane.in_vehicle() ) {
824 return false;
825 }
826
827 if( nested_call || !get_option<bool>( "CLOSE_ADV_INV" ) ) {
828 // Why is this here? It's because the activity backlog can act
829 // like a stack instead of a single deferred activity in order to
830 // accomplish some UI shenanigans. The inventory menu activity is
831 // added, then an activity to drop is pushed on the stack, then
832 // the drop activity is repeatedly popped and pushed on the stack
833 // until all its items are processed. When the drop activity runs out,
834 // the inventory menu activity is there waiting and seamlessly returns
835 // the player to the menu. If the activity is interrupted instead of
836 // completing, both activities are canceled.
837 // Thanks to kevingranade for the explanation.
838 do_return_entry();
839 }
840
841 if( spane.get_area() == AIM_INVENTORY || spane.get_area() == AIM_WORN ) {
842 drop_locations dropped;
843 // keep a list of favorites separated, only drop non-fav first if they exist
844 drop_locations dropped_favorite;
845
846 if( spane.get_area() == AIM_INVENTORY ) {
847 for( size_t index = 0; index < g->u.inv.size(); ++index ) {
848 const std::list<item> &stack = g->u.inv.const_stack( index );
849 const item &it = stack.front();
850 item_location indexed_item( g->u, const_cast<item *>( &it ) );
851
852 if( !spane.is_filtered( it ) ) {
853 int count;
854 if( it.count_by_charges() ) {
855 count = it.charges;
856 } else {
857 count = stack.size();
858 }
859 if( it.is_favorite ) {
860 dropped_favorite.emplace_back( indexed_item, count );
861 } else {
862 dropped.emplace_back( indexed_item, count );
863 }
864 }
865 }
866 } else if( spane.get_area() == AIM_WORN ) {
867 // do this in reverse, to account for vector item removal messing with future indices
868 auto iter = g->u.worn.rbegin();
869 for( size_t idx = 0; idx < g->u.worn.size(); ++idx, ++iter ) {
870 item &it = *iter;
871
872 if( !spane.is_filtered( it ) ) {
873 item_location loc( g->u, &it );
874 if( it.is_favorite ) {
875 dropped_favorite.emplace_back( loc, it.count() );
876 } else {
877 dropped.emplace_back( loc, it.count() );
878 }
879 }
880 }
881 }
882 if( dropped.empty() ) {
883 if( !query_yn( _( "Really drop all your favorite items?" ) ) ) {
884 return false;
885 }
886 dropped = dropped_favorite;
887 }
888
889 g->u.drop( dropped, g->u.pos() + darea.off );
890 } else {
891 if( dpane.get_area() == AIM_WORN ) {
892 // TODO: Start ACT_WEAR in this case
893 debugmsg( "Wearing clothes using move all is not yet implemented" );
894 } else {
895 // Vehicle and map destinations are handled the same.
896 // Check first if the destination area still have enough room for moving all.
897 if( !is_processing() && sarea.volume > darea.free_volume( dpane.in_vehicle() ) &&
898 !query_yn( _( "There isn't enough room, do you really want to move all?" ) ) ) {
899 return false;
900 }
901
902 // Stash the destination
903 const tripoint relative_destination = darea.off;
904
905 // Find target items and quantities thereof for the new activity
906 std::vector<item_location> target_items;
907 std::vector<int> quantities;
908
909 item_stack::iterator stack_begin, stack_end;
910 if( panes[src].in_vehicle() ) {
911 vehicle_stack targets = sarea.veh->get_items( sarea.vstor );
912 stack_begin = targets.begin();
913 stack_end = targets.end();
914 } else {
915 map_stack targets = g->m.i_at( sarea.pos );
916 stack_begin = targets.begin();
917 stack_end = targets.end();
918 }
919
920 // If moving to vehicle, silently filter buckets
921 // Moving them would cause tons of annoying prompts or spills
922 const bool filter_buckets = dpane.in_vehicle();
923 bool filtered_any_bucket = false;
924 // Push item_locations and item counts for all items at placement
925 for( item_stack::iterator it = stack_begin; it != stack_end; ++it ) {
926 if( spane.is_filtered( *it ) ) {
927 continue;
928 }
929 if( filter_buckets && it->is_bucket_nonempty() ) {
930 filtered_any_bucket = true;
931 continue;
932 }
933 if( spane.in_vehicle() ) {
934 target_items.emplace_back( vehicle_cursor( *sarea.veh, sarea.vstor ), &*it );
935 } else {
936 target_items.emplace_back( map_cursor( sarea.pos ), &*it );
937 }
938 // quantity of 0 means move all
939 quantities.push_back( 0 );
940 }
941
942 if( filtered_any_bucket ) {
943 add_msg( m_info, _( "Skipping filled buckets to avoid spilling their contents." ) );
944 }
945
946 if( dpane.get_area() == AIM_INVENTORY ) {
947 std::vector<pickup::pick_drop_selection> targets = pickup::optimize_pickup( target_items,
948 quantities );
949 g->u.assign_activity( player_activity( pickup_activity_actor(
950 targets,
951 panes[src].in_vehicle() ? cata::nullopt : cata::optional<tripoint>( g->u.pos() )
952 ) ) );
953 } else {
954 g->u.assign_activity( player_activity( move_items_activity_actor(
955 target_items,
956 quantities,
957 dpane.in_vehicle(),
958 relative_destination
959 ) ) );
960 }
961 }
962
963 }
964 // if dest was AIM_ALL then we used query_destination and should undo that
965 if( restore_area ) {
966 dpane.restore_area();
967 }
968 return true;
969}

1070{
1071 if( panes[left].get_area() == changeSquare || panes[right].get_area() == changeSquare ) {

1083 } else {
1084 swap_panes();
1085 }
1086 // we need to check the original area if we can place items in vehicle storage
1087 } else if( squares[changeSquare].canputitems( spane.get_cur_item_ptr() ) ) {
1088 bool in_vehicle_cargo = false;

1090 squares[changeSquare].set_container( spane.get_cur_item_ptr() );
1091 } else if( spane.get_area() == AIM_CONTAINER ) {

1094 } else if( squares[changeSquare].can_store_in_vehicle() && spane.get_area() != changeSquare ) {
1095 if( changeSquare == AIM_DRAGGED ) {
1096 in_vehicle_cargo = true;
1097 } else {
1098 // check item stacks in vehicle and map at said square
1099 auto sq = squares[changeSquare];
1100 auto map_stack = g->m.i_at( sq.pos );
1101 auto veh_stack = sq.veh->get_items( sq.vstor );
1102 // auto switch to vehicle storage if vehicle items are there, or neither are there
1103 if( !veh_stack.empty() || map_stack.empty() ) {
1104 in_vehicle_cargo = true;
1105 }
1106 }
1107 }
1108 spane.set_area( squares[changeSquare], in_vehicle_cargo );
1109 spane.index = 0;
1110 spane.recalc = true;
1111 if( dpane.get_area() == AIM_ALL ) {
1112 dpane.recalc = true;
1113 }
1114 } else {

1119void advanced_inventory::start_activity( const aim_location destarea, const aim_location srcarea,
1120 advanced_inv_listitem *sitem, int &amount_to_move,
1121 const bool from_vehicle, const bool to_vehicle ) const
1122{
1123
1124 const bool by_charges = sitem->items.front()->count_by_charges();
1125
1126 if( destarea == AIM_WORN ) {
1127 g->u.assign_activity( ACT_WEAR );
1128
1129 if( by_charges ) {
1130 if( from_vehicle ) {
1131 g->u.activity.targets.emplace_back( vehicle_cursor( *squares[srcarea].veh, squares[srcarea].vstor ),
1132 sitem->items.front() );
1133 } else {
1134 g->u.activity.targets.emplace_back( map_cursor( squares[srcarea].pos ), sitem->items.front() );
1135 }
1136 g->u.activity.values.push_back( amount_to_move );
1137 } else {
1138 for( std::list<item *>::iterator it = sitem->items.begin(); amount_to_move > 0 &&
1139 it != sitem->items.end(); ++it ) {
1140 if( from_vehicle ) {
1141 g->u.activity.targets.emplace_back( vehicle_cursor( *squares[srcarea].veh, squares[srcarea].vstor ),
1142 *it );
1143 } else {
1144 g->u.activity.targets.emplace_back( map_cursor( squares[srcarea].pos ), *it );
1145 }
1146 g->u.activity.values.push_back( 0 );
1147 --amount_to_move;
1148 }
1149 }
1150 } else {
1151 // Find target items and quantities thereof for the new activity
1152 std::vector<item_location> target_items;
1153 std::vector<int> quantities;
1154 if( by_charges ) {
1155 if( from_vehicle ) {
1156 target_items.emplace_back( vehicle_cursor( *squares[srcarea].veh, squares[srcarea].vstor ),
1157 sitem->items.front() );
1158 } else {
1159 target_items.emplace_back( map_cursor( squares[srcarea].pos ), sitem->items.front() );
1160 }
1161 quantities.push_back( amount_to_move );
1162 } else {

1164 it != sitem->items.end(); ++it ) {
1165 if( from_vehicle ) {
1166 target_items.emplace_back( vehicle_cursor( *squares[srcarea].veh, squares[srcarea].vstor ),
1167 *it );
1168 } else {
1169 target_items.emplace_back( map_cursor( squares[srcarea].pos ), *it );
1170 }
1171 quantities.push_back( 0 );

1173 }
1174 }
1175
1176 if( destarea == AIM_INVENTORY ) {
1177 std::vector<pickup::pick_drop_selection> targets = pickup::optimize_pickup( target_items,
1178 quantities );
1179 g->u.assign_activity( player_activity( pickup_activity_actor(
1180 targets,
1181 from_vehicle ? cata::nullopt : cata::optional<tripoint>( g->u.pos() )
1182 ) ) );
1183 } else {
1184 // Stash the destination
1185 const tripoint relative_destination = squares[destarea].off;
1186
1187 g->u.assign_activity( player_activity( move_items_activity_actor(
1188 target_items,
1189 quantities,
1190 to_vehicle,
1191 relative_destination
1192 ) ) );
1193 }

1197bool advanced_inventory::action_move_item( advanced_inv_listitem *sitem,
1198 advanced_inventory_pane &dpane, const advanced_inventory_pane &spane,
1199 const std::string &action )
1200{
1201 bool exit = false;
1202 if( sitem == nullptr || !sitem->is_item_entry() ) {
1203 return false;
1204 }
1205 aim_location destarea = dpane.get_area();
1206 aim_location srcarea = sitem->area;
1207 bool restore_area = destarea == AIM_ALL;
1208 if( !query_destination( destarea ) ) {
1209 return false;
1210 }
1211 // Not necessarily equivalent to spane.in_vehicle() if using AIM_ALL
1212 bool from_vehicle = sitem->from_vehicle;
1213 bool to_vehicle = dpane.in_vehicle();
1214
1215 // AIM_ALL should disable same area check and handle it with proper filtering instead.
1216 // This is a workaround around the lack of vehicle location info in
1217 // either aim_location or advanced_inv_listitem.
1218 if( squares[srcarea].is_same( squares[destarea] ) &&
1219 spane.get_area() != AIM_ALL &&
1220 spane.in_vehicle() == dpane.in_vehicle() ) {
1221 popup( _( "Source area is the same as destination (%s)." ), squares[destarea].name );
1222 return false;
1223 }
1224 assert( !sitem->items.empty() );
1225 int amount_to_move = 0;
1226 if( !query_charges( destarea, *sitem, action, amount_to_move ) ) {
1227 return false;
1228 }
1229 // This makes sure that all item references in the advanced_inventory_pane::items vector
1230 // are recalculated, even when they might not have changed, but they could (e.g. items
1231 // taken from inventory, but unable to put into the cargo trunk go back into the inventory,
1232 // but are potentially at a different place).
1233 recalc = true;
1234 assert( amount_to_move > 0 );
1235 if( destarea == AIM_CONTAINER ) {
1236 if( !move_content( *sitem->items.front(),
1237 *squares[destarea].get_container( to_vehicle ) ) ) {
1238 return false;
1239 }
1240 } else if( srcarea == AIM_INVENTORY && destarea == AIM_WORN ) {
1241
1242 // make sure advanced inventory is reopened after activity completion.
1243 do_return_entry();
1244
1245 g->u.assign_activity( ACT_WEAR );
1246
1247 g->u.activity.targets.emplace_back( g->u, sitem->items.front() );
1248 g->u.activity.values.push_back( amount_to_move );
1249
1250 // exit so that the activity can be carried out
1251 exit = true;
1252
1253 } else if( srcarea == AIM_INVENTORY || srcarea == AIM_WORN ) {
1254
1255 // make sure advanced inventory is reopened after activity completion.
1256 do_return_entry();
1257
1258 // if worn, we need to fix with the worn index number (starts at -2, as -1 is weapon)
1259 int idx = srcarea == AIM_INVENTORY ? sitem->idx : player::worn_position_to_index( sitem->idx );
1260
1261 if( srcarea == AIM_WORN && destarea == AIM_INVENTORY ) {
1262 // this is ok because worn items are never stacked (can't move more than 1).
1263 g->u.takeoff( idx );
1264
1265 // exit so that the action can be carried out
1266 exit = true;
1267 } else {
1268 // important if item is worn
1269 if( g->u.can_unwield( g->u.i_at( idx ) ).success() ) {
1270 drop_locations to_move = { drop_location( item_location( g->u, &g->u.i_at( idx ) ), amount_to_move ) };
1271 g->u.assign_activity( drop_activity_actor( g->u, to_move, !to_vehicle, squares[destarea].off ) );
1272 // exit so that the activity can be carried out
1273 exit = true;
1274 }
1275 }
1276 } else {
1277 // from map/vehicle: start ACT_PICKUP or ACT_MOVE_ITEMS as necessary
1278 // Make sure advanced inventory is reopened after activity completion.
1279 do_return_entry();
1280 start_activity( destarea, srcarea, sitem, amount_to_move, from_vehicle, to_vehicle );
1281
1282 // exit so that the activity can be carried out
1283 exit = true;
1284 }
1285
1286 // if dest was AIM_ALL then we used query_destination and should undo that
1287 if( restore_area ) {
1288 dpane.restore_area();
1289 }
1290 return exit;
1291}

1719 }
1720
1721 // Check volume, this should work the same for inventory, map and vehicles, but not for worn
1722 const int room_for = it.charges_per_volume( free_volume );
1723 if( amount > room_for && squares[destarea].id != AIM_WORN ) {
1724 if( room_for <= 0 ) {
1725 popup( _( "Destination area is full. Remove some items first." ) );
1726 return false;
1727 }
1728 amount = std::min( room_for, amount );
1729 }
1730 // Map and vehicles have a maximal item count, check that. Inventory does not have this.
1731 if( destarea != AIM_INVENTORY &&
1732 destarea != AIM_WORN &&
1733 destarea != AIM_CONTAINER ) {
1734 const int cntmax = p.max_size - p.get_item_count();
1735 // For items counted by charges, adding it adds 0 items if something there stacks with it.
1736 const bool adds0 = by_charges && std::any_of( panes[dest].items.begin(), panes[dest].items.end(),
1737 [&it]( const advanced_inv_listitem & li ) {
1738 return li.is_item_entry() && li.items.front()->stacks_with( it );
1739 } );
1740 if( cntmax <= 0 && !adds0 ) {
1741 popup( _( "Destination area has too many items. Remove some first." ) );
1742 return false;
1743 }
1744 // Items by charge count as a single item, regardless of the charges. As long as the
1745 // destination can hold another item, one can move all charges.
1746 if( !by_charges ) {
1747 amount = std::min( cntmax, amount );
1748 }
1749 }
1750 // Inventory has a weight capacity, map and vehicle don't have that
1751 if( destarea == AIM_INVENTORY || destarea == AIM_WORN ) {
1752 const units::mass unitweight = it.weight() / ( by_charges ? it.charges : 1 );
1753 const units::mass max_weight = g->u.has_trait( trait_DEBUG_STORAGE ) ?
1754 units::mass_max : g->u.weight_capacity() * 4 - g->u.weight_carried();
1755 if( unitweight > 0_gram && unitweight * amount > max_weight ) {
1756 const int weightmax = max_weight / unitweight;
1757 if( weightmax <= 0 ) {
1758 popup( _( "This is too heavy!" ) );
1759 return false;
1760 }
1761 amount = std::min( weightmax, amount );
1762 }
1763 }
1764 // handle how many of armor type we can equip (max of 2 per type)
1765 if( destarea == AIM_WORN ) {
1766 const auto &id = sitem.items.front()->typeId();
1767 // how many slots are available for the item?
1768 const int slots_available = MAX_WORN_PER_TYPE - g->u.amount_worn( id );
1769 // base the amount to equip on amount of slots available
1770 amount = std::min( slots_available, input_amount );
1771 }
1772 // Now we have the final amount. Query if requested or limited room left.
1773 if( action == "MOVE_VARIABLE_ITEM" || amount < input_amount ) {
1774 const int count = by_charges ? it.charges : sitem.stacks;
1775 const char *msg = nullptr;
1776 std::string popupmsg;
1777 if( amount >= input_amount ) {
1778 msg = _( "How many do you want to move? [Have %d] (0 to cancel)" );
1779 popupmsg = string_format( msg, count );
1780 } else {
1781 msg = _( "Destination can only hold %d! Move how many? [Have %d] (0 to cancel)" );
1782 popupmsg = string_format( msg, amount, count );
1783 }
1784 // At this point amount contains the maximal amount that the destination can hold.
1785 const int possible_max = std::min( input_amount, amount );
1786 if( amount <= 0 ) {
1787 popup( _( "The destination is already full!" ) );
1788 } else {
1789 amount = string_input_popup()
1790 .title( popupmsg )
1791 .width( 20 )
1792 .only_digits( true )
1793 .query_int();
1794 }
1795 if( amount <= 0 ) {
1796 return false;
1797 }
1798 if( amount > possible_max ) {
1799 amount = possible_max;
1800 }
1801 }
1802 return true;
1803}
