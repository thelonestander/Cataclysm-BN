226static void extract_effect(
227 const JsonObject &j,
228 std::unordered_map<std::tuple<std::string, bool, std::string, std::string>, double,
229 cata::tuple_hash> &data,
230 const std::string &mod_type, std::string data_key, std::string type_key, std::string arg_key )
231{
232 double val = 0;
233 double reduced_val = 0;
234 if( j.has_member( mod_type ) ) {
235 JsonArray jsarr = j.get_array( mod_type );
236 val = jsarr.get_float( 0 );
237 // If a second value exists use it, else reduced_val = val.
238 if( jsarr.size() >= 2 ) {
239 reduced_val = jsarr.get_float( 1 );
240 } else {
241 reduced_val = val;
242 }
243 }
244 // Store values if they aren't zero.
245 if( val != 0 ) {
246 data[std::make_tuple( data_key, false, type_key, arg_key )] = val;
247 }
248 if( reduced_val != 0 ) {
249 data[std::make_tuple( data_key, true, type_key, arg_key )] = reduced_val;
250 }

878 intensity = val;
879 if( old_intensity != intensity ) {
880 add_msg( m_debug, "%s intensity %d->%d", get_id().c_str(), old_intensity, intensity );
881 }

883 return intensity;

1031double effect::get_percentage( std::string arg, int val, bool reduced ) const
1032{
1033 auto &mod_data = eff_type->mod_data;
1034 auto found_top_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "chance_top" ) );
1035 auto found_top_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg,
1036 "chance_top" ) );
1037 // Convert to int or 0
1038 int top_base = 0;
1039 int top_scale = 0;
1040 if( found_top_base != mod_data.end() ) {
1041 top_base = found_top_base->second;
1042 }
1043 if( found_top_scale != mod_data.end() ) {
1044 top_scale = found_top_scale->second * ( intensity - 1 );
1045 }
1046 // Check chances if value is 0 (so we can check valueless effects like vomiting)
1047 // Else a nonzero value overrides a 0 chance for default purposes
1048 if( val == 0 ) {
1049 // If both top values <= 0 then it should never trigger
1050 if( top_base <= 0 && top_scale <= 0 ) {
1051 return 0;
1052 }
1053 // It will also never trigger if top_base + top_scale <= 0
1054 if( top_base + top_scale <= 0 ) {
1055 return 0;
1056 }
1057 }
1058
1059 // We only need to calculate these if we haven't already returned
1060 int bot_base = 0;
1061 int bot_scale = 0;
1062 int tick = 0;
1063 auto found_bot_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "chance_bot" ) );
1064 auto found_bot_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg,
1065 "chance_bot" ) );
1066 auto found_tick_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "tick" ) );
1067 auto found_tick_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg, "tick" ) );
1068 if( found_bot_base != mod_data.end() ) {
1069 bot_base = found_bot_base->second;
1070 }
1071 if( found_bot_scale != mod_data.end() ) {
1072 bot_scale = found_bot_scale->second * ( intensity - 1 );
1073 }
1074 if( found_tick_base != mod_data.end() ) {
1075 tick += found_tick_base->second;
1076 }
1077 if( found_tick_scale != mod_data.end() ) {
1078 tick += found_tick_scale->second * ( intensity - 1 );
1079 }
1080 // Tick is the exception where tick = 0 means tick = 1
1081 if( tick == 0 ) {
1082 tick = 1;
1083 }
1084
1085 double ret = 0;
1086 // If both bot values are zero the formula is one_in(top), else the formula is x_in_y(top, bot)
1087 if( bot_base != 0 && bot_scale != 0 ) {
1088 if( bot_base + bot_scale == 0 ) {
1089 // Special crash avoidance case, in most effect fields 0 = "nothing happens"
1090 // so assume false here for consistency
1091 ret = 0;
1092 } else {
1093 // Cast to double here to allow for partial percentages
1094 ret = 100 * static_cast<double>( top_base + top_scale ) / static_cast<double>
1095 ( bot_base + bot_scale );
1096 }
1097 } else {
1098 // Cast to double here to allow for partial percentages
1099 ret = 100 / static_cast<double>( top_base + top_scale );
1100 }
1101 // Divide by ticks between rolls
1102 if( tick > 1 ) {
1103 ret = ret / tick;
1104 }
1105 return ret;
1106}

1108bool effect::activated( const time_point &when, std::string arg, int val, bool reduced,
1109 double mod ) const
1110{
1111 auto &mod_data = eff_type->mod_data;
1112 auto found_top_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "chance_top" ) );
1113 auto found_top_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg,
1114 "chance_top" ) );
1115 // Convert to int or 0
1116 int top_base = 0;
1117 int top_scale = 0;
1118 if( found_top_base != mod_data.end() ) {
1119 top_base = found_top_base->second;
1120 }
1121 if( found_top_scale != mod_data.end() ) {
1122 top_scale = found_top_scale->second * ( intensity - 1 );
1123 }
1124 // Check chances if value is 0 (so we can check valueless effects like vomiting)
1125 // Else a nonzero value overrides a 0 chance for default purposes
1126 if( val == 0 ) {
1127 // If both top values <= 0 then it should never trigger
1128 if( top_base <= 0 && top_scale <= 0 ) {
1129 return false;
1130 }
1131 // It will also never trigger if top_base + top_scale <= 0
1132 if( top_base + top_scale <= 0 ) {
1133 return false;
1134 }
1135 }
1136
1137 // We only need to calculate these if we haven't already returned
1138 int bot_base = 0;
1139 int bot_scale = 0;
1140 int tick = 0;
1141 auto found_bot_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "chance_bot" ) );
1142 auto found_bot_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg,
1143 "chance_bot" ) );
1144 auto found_tick_base = mod_data.find( std::make_tuple( "base_mods", reduced, arg, "tick" ) );
1145 auto found_tick_scale = mod_data.find( std::make_tuple( "scaling_mods", reduced, arg, "tick" ) );
1146 if( found_bot_base != mod_data.end() ) {
1147 bot_base = found_bot_base->second;
1148 }
1149 if( found_bot_scale != mod_data.end() ) {
1150 bot_scale = found_bot_scale->second * ( intensity - 1 );
1151 }
1152 if( found_tick_base != mod_data.end() ) {
1153 tick += found_tick_base->second;
1154 }
1155 if( found_tick_scale != mod_data.end() ) {
1156 tick += found_tick_scale->second * ( intensity - 1 );
1157 }
1158 // Tick is the exception where tick = 0 means tick = 1
1159 if( tick == 0 ) {
1160 tick = 1;
1161 }
1162
1163 // Check if tick allows for triggering. If both bot values are zero the formula is
1164 // x_in_y(1, top) i.e. one_in(top), else the formula is x_in_y(top, bot),
1165 // mod multiplies the overall percentage chances
1166
1167 // has to be an && here to avoid undefined behavior of turn % 0
1168 if( tick > 0 &&
1169 ( when - calendar::turn_zero ) % time_duration::from_turns( tick ) == 0_turns ) {
1170 if( bot_base != 0 && bot_scale != 0 ) {
1171 if( bot_base + bot_scale == 0 ) {
1172 // Special crash avoidance case, in most effect fields 0 = "nothing happens"
1173 // so assume false here for consistency
1174 return false;
1175 } else {
1176 return x_in_y( ( top_base + top_scale ) * mod, bot_base + bot_scale );
1177 }
1178 } else {
1179 return x_in_y( mod, top_base + top_scale );
1180 }
1181 }
1182 return false;
1183}
