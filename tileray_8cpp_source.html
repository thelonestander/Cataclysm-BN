194void tileray::advance( int num )
195{
196 last_d = point_zero;
197 if( num == 0 ) {
198 return;
199 }
200 int anum = std::abs( num );
201 steps += anum;
202 const bool vertical = mostly_vertical();
203 if( abs_d.x && abs_d.y ) {
204 for( int i = 0; i < anum; i++ ) {
205 if( vertical ) {
206 // mostly vertical line
207 leftover += abs_d.x;
208 if( leftover >= abs_d.y ) {
209 last_d.x++;
210 leftover -= abs_d.y;
211 }
212 } else {
213 // mostly horizontal line
214 leftover += abs_d.y;
215 if( leftover >= abs_d.x ) {
216 last_d.y++;
217 leftover -= abs_d.x;
218 }
219 }
220 }
221 }
222 if( vertical ) {
223 last_d.y = anum;
224 } else {
225 last_d.x = anum;
226 }
227
228 // offset calculated for 0-90 deg quadrant, we need to adjust if direction is other
229 int quadr = quadrant();
230 last_d.x *= sx[quadr];
231 last_d.y *= sy[quadr];
232 if( num < 0 ) {
233 last_d = -last_d;
234 }
235}
