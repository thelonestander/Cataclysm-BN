130bool scent_map::inbounds( const tripoint &p ) const
131{
132 // HACK: This weird long check here is a hack around the fact that scentmap is 2D
133 // A z-level can access scentmap if it is within SCENT_MAP_Z_REACH flying z-level move from player's z-level
134 // That is, if a flying critter could move directly up or down (or stand still) and be on same z-level as player
135 const int levz = gm.get_levz();
136 const bool scent_map_z_level_inbounds = ( p.z == levz ) ||
137 ( std::abs( p.z - levz ) == SCENT_MAP_Z_REACH &&
138 get_map().valid_move( p, tripoint( p.xy(), levz ), false, true ) );
139 if( !scent_map_z_level_inbounds ) {
140 return false;
141 }
142 static constexpr point scent_map_boundary_min{};
143 static constexpr point scent_map_boundary_max( MAPSIZE_X, MAPSIZE_Y );
144
145 static constexpr half_open_rectangle<point> scent_map_boundaries(
146 scent_map_boundary_min, scent_map_boundary_max );
147
148 return scent_map_boundaries.contains( p.xy() );
149}

151void scent_map::update( const tripoint &center, map &m )
152{
153 // Stop updating scent after X turns of the player not moving.
154 // Once wind is added, need to reset this on wind shifts as well.
155 if( !player_last_position || center != *player_last_position ) {
156 player_last_position.emplace( center );
157 player_last_moved = calendar::turn;
158 } else if( player_last_moved + 1000_turns < calendar::turn ) {
159 return;
160 }
161
162 // note: the next four intermediate matrices need to be at least
163 // [2*SCENT_RADIUS+3][2*SCENT_RADIUS+1] in size to hold enough data
164 // The code I'm modifying used [MAPSIZE_X]. I'm staying with that to avoid new bugs.
165
166 // These two matrices are transposed so that x addresses are contiguous in memory
167 scent_array<int> sum_3_scent_y;
168 scent_array<int> squares_used_y;
169
170 // these are for caching flag lookups
171 scent_array<bool> blocks_scent; // currently only TFLAG_NO_SCENT blocks scent
172 scent_array<bool> reduces_scent;
173
174 // for loop constants
175 const int scentmap_minx = center.x - SCENT_RADIUS;
176 const int scentmap_maxx = center.x + SCENT_RADIUS;
177 const int scentmap_miny = center.y - SCENT_RADIUS;
178 const int scentmap_maxy = center.y + SCENT_RADIUS;
179
180 // decrease this to reduce gas spread. Keep it under 125 for
181 // stability. This is essentially a decimal number * 1000.
182 const int diffusivity = 100;
183
184 // The new scent flag searching function. Should be wayyy faster than the old one.
185 m.scent_blockers( blocks_scent, reduces_scent, point( scentmap_minx - 1, scentmap_miny - 1 ),
186 point( scentmap_maxx + 1, scentmap_maxy + 1 ) );
187 // Sum neighbors in the y direction. This way, each square gets called 3 times instead of 9
188 // times. This cost us an extra loop here, but it also eliminated a loop at the end, so there
189 // is a net performance improvement over the old code. Could probably still be better.
190 // note: this method needs an array that is one square larger on each side in the x direction
191 // than the final scent matrix. I think this is fine since SCENT_RADIUS is less than
192 // MAPSIZE_X, but if that changes, this may need tweaking.
193 for( int x = scentmap_minx - 1; x <= scentmap_maxx + 1; ++x ) {
194 for( int y = scentmap_miny; y <= scentmap_maxy; ++y ) {
195 // remember the sum of the scent val for the 3 neighboring squares that can defuse into
196 sum_3_scent_y[y][x] = 0;
197 squares_used_y[y][x] = 0;
198 for( int i = y - 1; i <= y + 1; ++i ) {
199 if( !blocks_scent[x][i] ) {
200 if( reduces_scent[x][i] ) {
201 // only 20% of scent can diffuse on REDUCE_SCENT squares
202 sum_3_scent_y[y][x] += 2 * grscent[x][i];
203 squares_used_y[y][x] += 2;
204 } else {
205 sum_3_scent_y[y][x] += 10 * grscent[x][i];
206 squares_used_y[y][x] += 10;
207 }
208 }
209 }
210 }
211 }
212
213 // Rest of the scent map
214 for( int x = scentmap_minx; x <= scentmap_maxx; ++x ) {
215 for( int y = scentmap_miny; y <= scentmap_maxy; ++y ) {
216 int &scent_here = grscent[x][y];
217 if( !blocks_scent[x][y] ) {
218 // to how many neighboring squares do we diffuse out? (include our own square
219 // since we also include our own square when diffusing in)
220 const int squares_used = squares_used_y[y][x - 1]
221 + squares_used_y[y][x]
222 + squares_used_y[y][x + 1];
223
224 int this_diffusivity;
225 if( !reduces_scent[x][y] ) {
226 this_diffusivity = diffusivity;
227 } else {
228 this_diffusivity = diffusivity / 5; //less air movement for REDUCE_SCENT square
229 }
230 // take the old scent and subtract what diffuses out
231 int temp_scent = scent_here * ( 10 * 1000 - squares_used * this_diffusivity );
232 // neighboring REDUCE_SCENT squares absorb some scent
233 temp_scent -= scent_here * this_diffusivity * ( 90 - squares_used ) / 5;
234 // we've already summed neighboring scent values in the y direction in the previous
235 // loop. Now we do it for the x direction, multiply by diffusion, and this is what
236 // diffuses into our current square.
237 scent_here =
238 ( temp_scent
239 + this_diffusivity * ( sum_3_scent_y[y][x - 1]
240 + sum_3_scent_y[y][x]
241 + sum_3_scent_y[y][x + 1] )
242 ) / ( 1000 * 10 );
243 } else {
244 // this cell blocks scent via NO_SCENT (in json)
245 scent_here = 0;
246 }
247 }
248 }
